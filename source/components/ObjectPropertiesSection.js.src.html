143/**
144 * @param {!WebInspector.RemoteObjectProperty} propertyA
145 * @param {!WebInspector.RemoteObjectProperty} propertyB
146 * @return {number}
147 */
148WebInspector.ObjectPropertiesSection.CompareProperties = function(propertyA, propertyB)
149{
150 var a = propertyA.name;
151 var b = propertyB.name;
152 if (a === "__proto__")
153 return 1;
154 if (b === "__proto__")
155 return -1;
156 if (propertyA.symbol && !propertyB.symbol)
157 return 1;
158 if (propertyB.symbol && !propertyA.symbol)
159 return -1;
160 return String.naturalOrderComparator(a, b);
161}

266 setSearchRegex: function(regex, additionalCssClassName)
267 {
268 var cssClasses = WebInspector.highlightedSearchResultClassName;

281 /**
282 * @param {!RegExp} regex
283 * @param {!Element} element
284 * @param {string} cssClassName
285 */
286 _applySearch: function(regex, element, cssClassName)
287 {
288 var ranges = [];
289 var content = element.textContent;
290 regex.lastIndex = 0;
291 var match = regex.exec(content);
292 while (match) {
293 ranges.push(new WebInspector.SourceRange(match.index, match[0].length));
294 match = regex.exec(content);
295 }
296 if (ranges.length)
297 WebInspector.highlightRangesWithStyleClass(element, ranges, cssClassName, this._highlightChanges);

586/**
587 * @param {!TreeElement} treeElement
588 * @param {!WebInspector.RemoteObject} value
589 * @param {boolean} skipProto
590 * @param {!WebInspector.Linkifier=} linkifier
591 * @param {?string=} emptyPlaceholder
592 * @param {boolean=} flattenProtoChain
593 * @param {!Array.<!WebInspector.RemoteObjectProperty>=} extraProperties
594 * @param {!WebInspector.RemoteObject=} targetValue
595 */
596WebInspector.ObjectPropertyTreeElement._populate = function(treeElement, value, skipProto, linkifier, emptyPlaceholder, flattenProtoChain, extraProperties, targetValue)
597{
598 if (value.arrayLength() > WebInspector.ObjectPropertiesSection._arrayLoadThreshold) {
599 treeElement.removeChildren();
600 WebInspector.ArrayGroupingTreeElement._populateArray(treeElement, value, 0, value.arrayLength() - 1, linkifier);
601 return;
602 }
603
604 /**
605 * @param {?Array.<!WebInspector.RemoteObjectProperty>} properties
606 * @param {?Array.<!WebInspector.RemoteObjectProperty>} internalProperties
607 */
608 function callback(properties, internalProperties)
609 {
610 treeElement.removeChildren();
611 if (!properties)
612 return;
613
614 extraProperties = extraProperties || [];
615 for (var i = 0; i < extraProperties.length; ++i)
616 properties.push(extraProperties[i]);
617
618 WebInspector.ObjectPropertyTreeElement.populateWithProperties(treeElement, properties, internalProperties,
619 skipProto, targetValue || value, linkifier, emptyPlaceholder);
620 }
621
622 if (flattenProtoChain)
623 value.getAllProperties(false, callback);
624 else
625 WebInspector.RemoteObject.loadFromObjectPerProto(value, callback);
626}

628/**
629 * @param {!TreeElement} treeNode
630 * @param {!Array.<!WebInspector.RemoteObjectProperty>} properties
631 * @param {?Array.<!WebInspector.RemoteObjectProperty>} internalProperties
632 * @param {boolean} skipProto
633 * @param {?WebInspector.RemoteObject} value
634 * @param {!WebInspector.Linkifier=} linkifier
635 * @param {?string=} emptyPlaceholder
636 */
637WebInspector.ObjectPropertyTreeElement.populateWithProperties = function(treeNode, properties, internalProperties, skipProto, value, linkifier, emptyPlaceholder) {
638 properties.sort(WebInspector.ObjectPropertiesSection.CompareProperties);
639
640 for (var i = 0; i < properties.length; ++i) {
641 var property = properties[i];
642 if (skipProto && property.name === "__proto__")
643 continue;
644 if (property.isAccessorProperty()) {
645 if (property.name !== "__proto__" && property.getter) {
646 property.parentObject = value;
647 treeNode.appendChild(new WebInspector.ObjectPropertyTreeElement(property, linkifier));
648 }
649 if (property.isOwn) {
650 if (property.getter) {
651 var getterProperty = new WebInspector.RemoteObjectProperty("get " + property.name, property.getter);
652 getterProperty.parentObject = value;
653 treeNode.appendChild(new WebInspector.ObjectPropertyTreeElement(getterProperty, linkifier));
654 }
655 if (property.setter) {
656 var setterProperty = new WebInspector.RemoteObjectProperty("set " + property.name, property.setter);
657 setterProperty.parentObject = value;
658 treeNode.appendChild(new WebInspector.ObjectPropertyTreeElement(setterProperty, linkifier));
659 }
660 }
661 } else {
662 property.parentObject = value;
663 treeNode.appendChild(new WebInspector.ObjectPropertyTreeElement(property, linkifier));
664 }
665 }
666 if (internalProperties) {
667 for (var i = 0; i < internalProperties.length; i++) {
668 internalProperties[i].parentObject = value;
669 var treeElement = new WebInspector.ObjectPropertyTreeElement(internalProperties[i], linkifier);
670 if (internalProperties[i].name === "[[Entries]]") {
671 treeElement.setExpandable(true);
672 treeElement.expand();
673 }
674 treeNode.appendChild(treeElement);
675 }
676 }
677 WebInspector.ObjectPropertyTreeElement._appendEmptyPlaceholderIfNeeded(treeNode, emptyPlaceholder);
678}

793 function* arrayIndexes(object)
794 {
795 if (toIndex - fromIndex < sparseIterationThreshold) {
796 for (var i = fromIndex; i <= toIndex; ++i) {
797 if (i in object)
798 yield i;
799 }
800 } else {

802 for (var i = 0; i < ownPropertyNames.length; ++i) {
803 var name = ownPropertyNames[i];
804 var index = name >>> 0;
805 if (("" + index) === name && fromIndex <= index && index <= toIndex)

815 for (var i of arrayIndexes(this))

817 }

819 var bucketSize = count;
820 if (count <= bucketThreshold)

836 var groupStart = -1;
837 var groupEnd = 0;

839 if (groupStart === -1)
840 groupStart = i;
841 groupEnd = i;
842 if (++count === bucketSize) {
843 ranges.push([groupStart, groupEnd, count]);
844 count = 0;
845 groupStart = -1;
846 }
847 }
848 if (count > 0)
849 ranges.push([groupStart, groupEnd, count]);
850 }
