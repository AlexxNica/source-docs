291 /**
292 * @suppressReceiverCheck
293 * @return {!{eventListeners:!Array<!WebInspector.EventListenerObjectInInspectedPage>, internalHandlers:?Array<function()>}}
294 * @this {Object}
295 */
296 function frameworkEventListeners()
297 {
298 var errorLines = [];
299 var eventListeners = [];
300 var internalHandlers = [];
301 var fetchers = [jQueryFetcher];
302 try {
303 if (self.devtoolsFrameworkEventListeners && isArrayLike(self.devtoolsFrameworkEventListeners))
304 fetchers = fetchers.concat(self.devtoolsFrameworkEventListeners);
305 } catch (e) {
306 errorLines.push("devtoolsFrameworkEventListeners call produced error: " + toString(e));
307 }
308
309 for (var i = 0; i < fetchers.length; ++i) {
310 try {
311 var fetcherResult = fetchers[i](this);
312 if (fetcherResult.eventListeners && isArrayLike(fetcherResult.eventListeners)) {
313 eventListeners = eventListeners.concat(fetcherResult.eventListeners.map(checkEventListener).filter(nonEmptyObject));
314 }
315 if (fetcherResult.internalHandlers && isArrayLike(fetcherResult.internalHandlers))
316 internalHandlers = internalHandlers.concat(fetcherResult.internalHandlers.map(checkInternalHandler).filter(nonEmptyObject));
317 } catch (e) {
318 errorLines.push("fetcher call produced error: " + toString(e));
319 }
320 }
321 var result = {eventListeners: eventListeners};
322 if (internalHandlers.length)
323 result.internalHandlers = internalHandlers;
324 if (errorLines.length) {
325 var errorString = "Framework Event Listeners API Errors:\n\t" + errorLines.join("\n\t");
326 errorString = errorString.substr(0, errorString.length - 1);
327 result.errorString = errorString;
328 }
329 return result;

349 /**
350 * @param {*} eventListener
351 * @return {?WebInspector.EventListenerObjectInInspectedPage}
352 */
353 function checkEventListener(eventListener)
354 {
355 try {
356 var errorString = "";
357 if (!eventListener)
358 errorString += "empty event listener, ";
359 var type = eventListener.type;
360 if (!type || (typeof type !== "string"))
361 errorString += "event listener's type isn't string or empty, ";
362 var useCapture = eventListener.useCapture;
363 if (typeof useCapture !== "boolean")
364 errorString += "event listener's useCapture isn't boolean or undefined, ";
365 var passive = eventListener.passive;
366 if (typeof passive !== "boolean")
367 errorString += "event listener's passive isn't boolean or undefined, ";
368 var handler = eventListener.handler;
369 if (!handler || (typeof handler !== "function"))
370 errorString += "event listener's handler isn't a function or empty, ";
371 var remove = eventListener.remove;
372 if (remove && (typeof remove !== "function"))
373 errorString += "event listener's remove isn't a function, ";
374 if (!errorString){
375 return {type: type, useCapture: useCapture, passive: passive, handler: handler, remove: remove};
376 } else {
377 errorLines.push(errorString.substr(0, errorString.length - 2));
378 return null;
379 }
380 } catch (e) {
381 errorLines.push(toString(e));
382 return null;
383 }
384 }

421 function jQueryFetcher(node)
422 {
423 if (!node || !(node instanceof Node))
424 return {eventListeners: []};
425 var jQuery = /** @type {?{fn,data,_data}}*/(window["jQuery"]);
426 if (!jQuery || !jQuery.fn)
427 return {eventListeners: []};
428 var jQueryFunction = /** @type {function(!Node)} */(jQuery);
429 var data = jQuery._data || jQuery.data;
430
431 var eventListeners = [];
432 var internalHandlers = [];
433
434 if (typeof data === "function") {
435 var events = data(node, "events");
436 for (var type in events) {
437 for (var key in events[type]) {
438 var frameworkListener = events[type][key];
439 if (typeof frameworkListener === "object" || typeof frameworkListener === "function") {
440 var listener = {
441 handler: frameworkListener.handler || frameworkListener,
442 useCapture: true,
443 passive: false,
444 type: type
445 };
446 listener.remove = jQueryRemove.bind(node, frameworkListener.selector);
447 eventListeners.push(listener);
448 }
449 }
450 }
451 var nodeData = data(node);
452 if (nodeData && typeof nodeData.handle === "function")
453 internalHandlers.push(nodeData.handle);
454 }
455 var entry = jQueryFunction(node)[0];
456 if (entry) {
457 var entryEvents = entry["$events"];
458 for (var type in entryEvents) {
459 var events = entryEvents[type];
460 for (var key in events) {
461 if (typeof events[key] === "function") {
462 var listener = {
463 handler: events[key],
464 useCapture: true,
465 passive: false,
466 type: type
467 };
468 // We don't support removing for old version < 1.4 of jQuery because it doesn't provide API for getting "selector".
469 eventListeners.push(listener);
470 }
471 }
472 }
473 if (entry && entry["$handle"])
474 internalHandlers.push(entry["$handle"]);
475 }
476 return {eventListeners: eventListeners, internalHandlers: internalHandlers};
477 }
