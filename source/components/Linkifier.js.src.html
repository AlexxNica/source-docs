84 * @param {string} text

91WebInspector.Linkifier.linkifyUsingRevealer = function(revealable, text, fallbackHref, fallbackLineNumber, title, classes)
92{
93 var a = createElement("a");
94 a.className = (classes || "") + " webkit-html-resource-link";
95 a.textContent = text.trimMiddle(WebInspector.Linkifier.MaxLengthForDisplayedURLs);
96 a.title = title || text;
97 if (fallbackHref) {
98 a.href = fallbackHref;
99 a.lineNumber = fallbackLineNumber;
100 }
101 /**
102 * @param {!Event} event
103 * @this {Object}
104 */
105 function clickHandler(event)
106 {
107 event.stopImmediatePropagation();
108 event.preventDefault();
109 if (fallbackHref && WebInspector.Linkifier.handleLink(fallbackHref, fallbackLineNumber))
110 return;
111
112 WebInspector.Revealer.reveal(this);
113 }
114 a.addEventListener("click", clickHandler.bind(revealable), false);
115 return a;
116}

361 /**
362 * @override
363 * @param {!Element} anchor
364 * @param {!WebInspector.UILocation} uiLocation
365 * @param {boolean} isBlackboxed
366 */
367 formatLiveAnchor: function(anchor, uiLocation, isBlackboxed)
368 {
369 var text = uiLocation.linkText();
370 text = text.replace(/([a-f0-9]{7})[a-f0-9]{13}[a-f0-9]*/g, "$1\u2026");
371 if (this._maxLength)
372 text = text.trimMiddle(this._maxLength);
373 anchor.textContent = text;
374
375 var titleText = uiLocation.uiSourceCode.url();
376 if (typeof uiLocation.lineNumber === "number")
377 titleText += ":" + (uiLocation.lineNumber + 1);
378 anchor.title = titleText;
379
380 anchor.classList.toggle("webkit-html-blackbox-link", isBlackboxed);
381 }

462/**
463 * @param {string} string
464 * @param {function(string,string,number=,number=):!Node} linkifier
465 * @return {!DocumentFragment}
466 */
467WebInspector.linkifyStringAsFragmentWithCustomLinkifier = function(string, linkifier)
468{
469 var container = createDocumentFragment();
470 var linkStringRegEx = /(?:[a-zA-Z][a-zA-Z0-9+.-]{2,}:\/\/|data:|www\.)[\w$\-_+*'=\|\/\\(){}[\]^%@&#~,:;.!?]{2,}[\w$\-_+*=\|\/\\({^%@&#~]/;
471
472 while (string && string.length < WebInspector.Linkifier.MaxLengthToIgnoreLinkifier) {
473 var linkString = linkStringRegEx.exec(string);
474 if (!linkString)
475 break;
476
477 linkString = linkString[0];
478 var linkIndex = string.indexOf(linkString);
479 var nonLink = string.substring(0, linkIndex);
480 container.appendChild(createTextNode(nonLink));
481
482 var title = linkString;
483 var realURL = (linkString.startsWith("www.") ? "http://" + linkString : linkString);
484 var splitResult = WebInspector.ParsedURL.splitLineAndColumn(realURL);
485 var linkNode;
486 if (splitResult)
487 linkNode = linkifier(title, splitResult.url, splitResult.lineNumber, splitResult.columnNumber);
488 else
489 linkNode = linkifier(title, realURL);
490
491 container.appendChild(linkNode);
492 string = string.substring(linkIndex + linkString.length, string.length);
493 }
494
495 if (string)
496 container.appendChild(createTextNode(string));
497
498 return container;
499}

501/**
502 * @param {string} string
503 * @return {!DocumentFragment}
504 */
505WebInspector.linkifyStringAsFragment = function(string)
506{
507 /**
508 * @param {string} title
509 * @param {string} url
510 * @param {number=} lineNumber
511 * @param {number=} columnNumber
512 * @return {!Node}
513 */
514 function linkifier(title, url, lineNumber, columnNumber)
515 {
516 var isExternal = !WebInspector.resourceForURL(url) && !WebInspector.networkMapping.uiSourceCodeForURLForAnyTarget(url);
517 var urlNode = WebInspector.linkifyURLAsNode(url, title, undefined, isExternal);
518 if (typeof lineNumber !== "undefined") {
519 urlNode.lineNumber = lineNumber;
520 if (typeof columnNumber !== "undefined")
521 urlNode.columnNumber = columnNumber;
522 }
523
524 return urlNode;
525 }
526
527 return WebInspector.linkifyStringAsFragmentWithCustomLinkifier(string, linkifier);
528}

531 * @param {string} url
532 * @param {number=} lineNumber
533 * @param {number=} columnNumber
534 * @param {string=} classes
535 * @param {string=} tooltipText
536 * @param {string=} urlDisplayName
537 * @return {!Element}
538 */
539WebInspector.linkifyResourceAsNode = function(url, lineNumber, columnNumber, classes, tooltipText, urlDisplayName)
540{
541 if (!url) {
542 var element = createElementWithClass("span", classes);
543 element.textContent = urlDisplayName || WebInspector.UIString("(unknown)");
544 return element;
545 }
546 var linkText = urlDisplayName || WebInspector.displayNameForURL(url);
547 if (typeof lineNumber === "number")
548 linkText += ":" + (lineNumber + 1);
549 var anchor = WebInspector.linkifyURLAsNode(url, linkText, classes, false, tooltipText);
550 anchor.lineNumber = lineNumber;
551 anchor.columnNumber = columnNumber;
552 return anchor;
553}
