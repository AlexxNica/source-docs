156 _updateImagePosition: function()
157 {
158 var width = this._imageElement.naturalWidth;
159 var height = this._imageElement.naturalHeight;
160 var clientWidth = this.element.clientWidth;
161 var clientHeight = this.element.clientHeight;
162
163 var paddingFraction = 0.1;
164 var paddingX = clientWidth * paddingFraction;
165 var paddingY = clientHeight * paddingFraction;
166 var scaleX = (clientWidth - paddingX) / width;
167 var scaleY = (clientHeight - paddingY) / height;
168 var scale = Math.min(scaleX, scaleY);
169
170 if (this._maskRectangle) {
171 var style = this._maskElement.style;
172 style.width = width + "px";
173 style.height = height + "px";
174 style.borderLeftWidth = this._maskRectangle.x + "px";
175 style.borderTopWidth = this._maskRectangle.y + "px";
176 style.borderRightWidth = (width - this._maskRectangle.x - this._maskRectangle.width) + "px";
177 style.borderBottomWidth = (height - this._maskRectangle.y - this._maskRectangle.height) + "px";
178 }
179 this._transformController.setScaleConstraints(0.5, 10 / scale);
180 var matrix = new WebKitCSSMatrix()
181 .scale(this._transformController.scale(), this._transformController.scale())
182 .translate(clientWidth / 2, clientHeight / 2)
183 .scale(scale, scale)
184 .translate(-width / 2, -height / 2);
185 var bounds = WebInspector.Geometry.boundsForTransformedPoints(matrix, [0, 0, 0, width, height, 0]);
186 this._transformController.clampOffsets(paddingX - bounds.maxX, clientWidth - paddingX - bounds.minX,
187 paddingY - bounds.maxY, clientHeight - paddingY - bounds.minY);
188 matrix = new WebKitCSSMatrix().translate(this._transformController.offsetX(), this._transformController.offsetY()).multiply(matrix);
189 this._imageContainer.style.webkitTransform = matrix.toString();
190 },
