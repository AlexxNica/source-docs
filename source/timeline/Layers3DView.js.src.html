287 _updateTransformAndConstraints: function()
288 {
289 var paddingFraction = 0.1;
290 var viewport = this._layerTree.viewportSize();
291 var root = this._layerTree.root();
292 var baseWidth = viewport ? viewport.width : this._dimensionsForAutoscale.width;
293 var baseHeight = viewport ? viewport.height : this._dimensionsForAutoscale.height;
294 var canvasWidth = this._canvasElement.width;
295 var canvasHeight = this._canvasElement.height;
296 var paddingX = canvasWidth * paddingFraction;
297 var paddingY = canvasHeight * paddingFraction;
298 var scaleX = (canvasWidth - 2 * paddingX) / baseWidth;
299 var scaleY = (canvasHeight - 2 * paddingY) / baseHeight;
300 var viewScale = Math.min(scaleX, scaleY);
301 var minScaleConstraint = Math.min(baseWidth / this._dimensionsForAutoscale.width, baseHeight / this._dimensionsForAutoscale.width) / 2;
302 this._transformController.setScaleConstraints(minScaleConstraint, 10 / viewScale); // 1/viewScale is 1:1 in terms of pixels, so allow zooming to 10x of native size
303 var scale = this._transformController.scale();
304 var rotateX = this._transformController.rotateX();
305 var rotateY = this._transformController.rotateY();
306 this._scale = scale * viewScale;
307 var scaleAndRotationMatrix = new WebKitCSSMatrix().scale(scale, scale, scale).translate(canvasWidth / 2, canvasHeight / 2, 0)
308 .rotate(rotateX, rotateY, 0).scale(viewScale, viewScale, viewScale).translate(-baseWidth / 2, -baseHeight / 2, 0);
309
310 var bounds;
311 for (var i = 0; i < this._rects.length; ++i)
312 bounds = WebInspector.Geometry.boundsForTransformedPoints(scaleAndRotationMatrix, this._rects[i].vertices, bounds);
313
314 this._transformController.clampOffsets((paddingX - bounds.maxX) / window.devicePixelRatio, (canvasWidth - paddingX - bounds.minX) / window.devicePixelRatio,
315 (paddingY - bounds.maxY) / window.devicePixelRatio, (canvasHeight - paddingY - bounds.minY) / window.devicePixelRatio);
316 var offsetX = this._transformController.offsetX() * window.devicePixelRatio;
317 var offsetY = this._transformController.offsetY() * window.devicePixelRatio;
318 // Multiply to translation matrix on the right rather than translate (which would implicitly multiply on the left).
319 this._projectionMatrix = new WebKitCSSMatrix().translate(offsetX, offsetY, 0).multiply(scaleAndRotationMatrix);
320
321 var glProjectionMatrix = new WebKitCSSMatrix().scale(1, -1, -1).translate(-1, -1, 0)
322 .scale(2 / this._canvasElement.width, 2 / this._canvasElement.height, 1 / 1000000).multiply(this._projectionMatrix);
323 this._gl.uniformMatrix4fv(this._shaderProgram.pMatrixUniform, false, this._arrayFromMatrix(glProjectionMatrix));
324 },

376 _calculateDepthsAndVisibility: function()
377 {
378 this._depthByLayerId = {};
379 var depth = 0;
380 var showInternalLayers = this._layerViewHost.showInternalLayersSetting().get();
381 var root = showInternalLayers ? this._layerTree.root() : (this._layerTree.contentRoot() || this._layerTree.root());
382 var queue = [root];
383 this._depthByLayerId[root.id()] = 0;
384 this._visibleLayers = {};
385 while (queue.length > 0) {
386 var layer = queue.shift();
387 this._visibleLayers[layer.id()] = showInternalLayers || layer.drawsContent();
388 var children = layer.children();
389 for (var i = 0; i < children.length; ++i) {
390 this._depthByLayerId[children[i].id()] = ++depth;
391 queue.push(children[i]);
392 }
393 }
394 this._maxDepth = depth;

420 {

443 /**
444 * @param {!WebInspector.Layers3DView.Rectangle} rect
445 */
446 _appendRect: function(rect)
447 {
448 var selection = rect.relatedObject;
449 var isSelected = WebInspector.LayerView.Selection.isEqual(this._lastSelection[WebInspector.Layers3DView.OutlineType.Selected], selection);
450 var isHovered = WebInspector.LayerView.Selection.isEqual(this._lastSelection[WebInspector.Layers3DView.OutlineType.Hovered], selection);
451 if (isSelected) {
452 rect.borderColor = WebInspector.Layers3DView.SelectedBorderColor;
453 } else if (isHovered) {
454 rect.borderColor = WebInspector.Layers3DView.HoveredBorderColor;
455 var fillColor = rect.fillColor || [255, 255, 255, 1];
456 var maskColor = WebInspector.Layers3DView.HoveredImageMaskColor;
457 rect.fillColor = [fillColor[0] * maskColor[0] / 255, fillColor[1] * maskColor[1] / 255, fillColor[2] * maskColor[2] / 255, fillColor[3] * maskColor[3]];
458 } else {
459 rect.borderColor = WebInspector.Layers3DView.BorderColor;
460 }
461 rect.lineWidth = isSelected ? WebInspector.Layers3DView.SelectedBorderWidth : WebInspector.Layers3DView.BorderWidth;
462 this._rects.push(rect);

594 _drawViewportAndChrome: function()
595 {
596 var viewport = this._layerTree.viewportSize();
597 if (!viewport)
598 return;
599
600 var drawChrome = !WebInspector.moduleSetting("frameViewerHideChromeWindow").get() && this._chromeTextures.length >= 3 && this._chromeTextures.indexOf(undefined) < 0;
601 var z = (this._maxDepth + 1) * WebInspector.Layers3DView.LayerSpacing;
602 var borderWidth = Math.ceil(WebInspector.Layers3DView.ViewportBorderWidth * this._scale);
603 var vertices = [viewport.width, 0, z, viewport.width, viewport.height, z, 0, viewport.height, z, 0, 0, z];
604 this._gl.lineWidth(borderWidth);
605 this._drawRectangle(vertices, drawChrome ? this._gl.LINE_STRIP : this._gl.LINE_LOOP, WebInspector.Layers3DView.ViewportBorderColor);
606
607 if (!drawChrome)
608 return;
609
610 var borderAdjustment = WebInspector.Layers3DView.ViewportBorderWidth / 2;
611 var viewportWidth = this._layerTree.viewportSize().width + 2 * borderAdjustment;
612 var chromeHeight = this._chromeTextures[0].image.naturalHeight;
613 var middleFragmentWidth = viewportWidth - this._chromeTextures[0].image.naturalWidth - this._chromeTextures[2].image.naturalWidth;
614 var x = -borderAdjustment;
615 var y = -chromeHeight;

1046 /**
1047 * Intersects quad with given transform matrix and line l(t) = (x0, y0, t)
1048 * @param {!CSSMatrix} matrix
1049 * @param {number} x0
1050 * @param {number} y0
1051 * @return {(number|undefined)}
1052 */
1053 intersectWithLine: function(matrix, x0, y0)
1054 {
1055 var i;
1056 // Vertices of the quad with transform matrix applied
1057 var points = [];
1058 for (i = 0; i < 4; ++i)
1059 points[i] = WebInspector.Geometry.multiplyVectorByMatrixAndNormalize(new WebInspector.Geometry.Vector(this.vertices[i * 3], this.vertices[i * 3 + 1], this.vertices[i * 3 + 2]), matrix);
1060 // Calculating quad plane normal
1061 var normal = WebInspector.Geometry.crossProduct(WebInspector.Geometry.subtract(points[1], points[0]), WebInspector.Geometry.subtract(points[2], points[1]));
1062 // General form of the equation of the quad plane: A * x + B * y + C * z + D = 0
1063 var A = normal.x;
1064 var B = normal.y;
1065 var C = normal.z;
1066 var D = -(A * points[0].x + B * points[0].y + C * points[0].z);
1067 // Finding t from the equation
1068 var t = -(D + A * x0 + B * y0) / C;
1069 // Point of the intersection
1070 var pt = new WebInspector.Geometry.Vector(x0, y0, t);
1071 // Vectors from the intersection point to vertices of the quad
1072 var tVects = points.map(WebInspector.Geometry.subtract.bind(null, pt));
1073 // Intersection point lies inside of the polygon if scalar products of normal of the plane and
1074 // cross products of successive tVects are all nonstrictly above or all nonstrictly below zero
1075 for (i = 0; i < tVects.length; ++i) {
