36WebInspector.TimelineLoader.loadFromFile = function(model, file, delegate)
37{
38 var loader = new WebInspector.TimelineLoader(model, delegate);
39 var fileReader = WebInspector.TimelineLoader._createFileReader(file, loader);

41 loader._totalSize = file.size;

78 cancel: function()
79 {
80 this._model.reset();
81 this._delegate.loadingComplete(false);
82 this._delegate = null;
83 if (this._canceledCallback)
84 this._canceledCallback();
85 },
86
87 /**
88 * @override
89 * @param {string} chunk
90 */
91 write: function(chunk)
92 {
93 if (!this._delegate)
94 return;
95 this._loadedBytes += chunk.length;
96 if (!this._firstChunk)
97 this._delegate.loadingProgress(this._totalSize ? this._loadedBytes / this._totalSize : undefined);
98
99 if (this._state === WebInspector.TimelineLoader.State.Initial) {
100 if (chunk[0] === "{")
101 this._state = WebInspector.TimelineLoader.State.LookingForEvents;
102 else if (chunk[0] === "[")
103 this._state = WebInspector.TimelineLoader.State.ReadingEvents;
104 else {
105 this._reportErrorAndCancelLoading(WebInspector.UIString("Malformed timeline data: Unknown JSON format"));
106 return;
107 }
108 }
109
110 if (this._state === WebInspector.TimelineLoader.State.LookingForEvents) {
111 var objectName = "\"traceEvents\":";
112 var startPos = this._buffer.length - objectName.length;
113 this._buffer += chunk;
114 var pos = this._buffer.indexOf(objectName, startPos);
115 if (pos === -1)
116 return;
117 chunk = this._buffer.slice(pos + objectName.length)
118 this._state = WebInspector.TimelineLoader.State.ReadingEvents;
119 }
120
121 this._jsonTokenizer.write(chunk);
122 },
123
124 /**
125 * @param {string} data
126 */
127 _writeBalancedJSON: function(data)
128 {
129 var json = data + "]";
130
131 if (this._firstChunk) {
132 this._delegate.loadingStarted();
133 } else {
134 var commaIndex = json.indexOf(",");
135 if (commaIndex !== -1)
136 json = json.slice(commaIndex + 1);
137 json = "[" + json;
138 }
139
140 var items;
141 try {
142 items = /** @type {!Array.<!WebInspector.TracingManager.EventPayload>} */ (JSON.parse(json));
143 } catch (e) {
144 this._reportErrorAndCancelLoading(WebInspector.UIString("Malformed timeline data: %s", e.toString()));
145 return;
146 }
147
148 if (this._firstChunk) {
149 this._firstChunk = false;
150 this._model.reset();
151 if (this._looksLikeAppVersion(items[0])) {
152 this._reportErrorAndCancelLoading(WebInspector.UIString("Legacy Timeline format is not supported."));
153 return;
154 }
155 }
156
157 try {
158 this._model.addEvents(items);
159 } catch (e) {
160 this._reportErrorAndCancelLoading(WebInspector.UIString("Malformed timeline data: %s", e.toString()));
161 return;
162 }
163 },
164
165 /**
166 * @param {string=} message
167 */
168 _reportErrorAndCancelLoading: function(message)
169 {
170 if (message)
171 WebInspector.console.error(message);
172 this.cancel();
173 },
174
175 /**
176 * @param {*} item
177 * @return {boolean}
178 */

180 {
181 return typeof item === "string" && item.indexOf("Chrome") !== -1;
182 },

185 * @override

187 close: function()
188 {
189 this._model.tracingComplete();
190 if (this._delegate)
191 this._delegate.loadingComplete(true);
192 },

228 }
