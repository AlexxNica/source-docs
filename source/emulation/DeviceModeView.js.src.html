391 /**

396 function screenshotCaptured(error, content)
397 {
398 this._model.deviceOutlineSetting().set(outlineVisible);
399 var dpr = window.devicePixelRatio;
400 var outlineRect = this._model.outlineRect().scale(dpr);
401 var screenRect = this._model.screenRect().scale(dpr);
402 screenRect.left -= outlineRect.left;
403 screenRect.top -= outlineRect.top;
404 var visiblePageRect = this._model.visiblePageRect().scale(dpr);
405 visiblePageRect.left += screenRect.left;
406 visiblePageRect.top += screenRect.top;
407 outlineRect.left = 0;
408 outlineRect.top = 0;

410 WebInspector.DOMModel.unmuteHighlight();
411 WebInspector.inspectorView.restore();
412
413 if (error) {
414 console.error(error);

416 }

419 var canvas = createElement("canvas");
420 var ctx = canvas.getContext("2d");
421 canvas.width = outlineRect.width;
422 canvas.height = outlineRect.height;
423
424 var promise = Promise.resolve();

530 _update: function()
531 {
532 var zoomFactor = WebInspector.zoomManager.zoomFactor();
533 var size = this._horizontal ? this._contentElement.offsetWidth : this._contentElement.offsetHeight;
534
535 if (this._scale !== this._renderedScale || zoomFactor !== this._renderedZoomFactor) {
536 this._contentElement.removeChildren();
537 this._count = 0;
538 this._renderedScale = this._scale;
539 this._renderedZoomFactor = zoomFactor;
540 }
541
542 var dipSize = size * zoomFactor / this._scale;
543 var count = Math.ceil(dipSize / 5);
544 var step = 1;
545 if (this._scale < 0.8)
546 step = 2;
547 if (this._scale < 0.6)
548 step = 4;
549 if (this._scale < 0.4)
550 step = 8;
551 if (this._scale < 0.2)
552 step = 16;
553 if (this._scale < 0.1)
554 step = 32;
555
556 for (var i = count; i < this._count; i++) {
557 if (!(i % step))
558 this._contentElement.lastChild.remove();
559 }
560
561 for (var i = this._count; i < count; i++) {
562 if (i % step)
563 continue;
564 var marker = this._contentElement.createChild("div", "device-mode-ruler-marker");
565 if (i) {
566 if (this._horizontal)
567 marker.style.left = (5 * i) * this._scale / zoomFactor + "px";
568 else
569 marker.style.top = (5 * i) * this._scale / zoomFactor + "px";
570 if (!(i % 20)) {
571 var text = marker.createChild("div", "device-mode-ruler-text");
572 text.textContent = i * 5;
573 text.addEventListener("click", this._onMarkerClick.bind(this, i * 5), false);
574 }
575 }
576 if (!(i % 10))
577 marker.classList.add("device-mode-ruler-marker-large");
578 else if (!(i % 5))
