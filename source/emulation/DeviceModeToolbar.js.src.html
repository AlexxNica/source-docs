11WebInspector.DeviceModeToolbar = function(model, showMediaInspectorSetting, showRulersSetting)
12{
13 this._model = model;
14 this._showMediaInspectorSetting = showMediaInspectorSetting;
15 this._showRulersSetting = showRulersSetting;
16
17 this._deviceOutlineSetting = this._model.deviceOutlineSetting();
18 this._showDeviceScaleFactorSetting = WebInspector.settings.createSetting("emulation.showDeviceScaleFactor", false);
19 this._showDeviceScaleFactorSetting.addChangeListener(this._updateDeviceScaleFactorVisibility, this);
20
21 this._showUserAgentTypeSetting = WebInspector.settings.createSetting("emulation.showUserAgentType", false);
22 this._showUserAgentTypeSetting.addChangeListener(this._updateUserAgentTypeVisibility, this);
23
24 this._showNetworkConditionsSetting = WebInspector.settings.createSetting("emulation.showNetworkConditions", false);
25 this._showNetworkConditionsSetting.addChangeListener(this._updateNetworkConditionsVisibility, this);
26
27 /** @type {!Map<!WebInspector.EmulatedDevice, !WebInspector.EmulatedDevice.Mode>} */
28 this._lastMode = new Map();
29
30 this._element = createElementWithClass("div", "device-mode-toolbar");
31
32 var leftContainer = this._element.createChild("div", "device-mode-toolbar-spacer");
33 leftContainer.createChild("div", "device-mode-toolbar-spacer");
34 var leftToolbar = new WebInspector.Toolbar("", leftContainer);
35 leftToolbar.makeWrappable();
36 this._fillLeftToolbar(leftToolbar);
37
38 var mainToolbar = new WebInspector.Toolbar("", this._element);
39 mainToolbar.makeWrappable();
40 this._fillMainToolbar(mainToolbar);
41
42 var rightContainer = this._element.createChild("div", "device-mode-toolbar-spacer");
43 var rightToolbar = new WebInspector.Toolbar("device-mode-toolbar-fixed-size", rightContainer);
44 rightToolbar.makeWrappable();
45 this._fillRightToolbar(rightToolbar);
46 var modeToolbar = new WebInspector.Toolbar("device-mode-toolbar-fixed-size", rightContainer);
47 modeToolbar.makeWrappable();
48 this._fillModeToolbar(modeToolbar);
49 rightContainer.createChild("div", "device-mode-toolbar-spacer");
50 var optionsToolbar = new WebInspector.Toolbar("", rightContainer);
51 optionsToolbar.makeWrappable(true);
52 this._fillOptionsToolbar(optionsToolbar);
53
54 this._emulatedDevicesList = WebInspector.EmulatedDevicesList.instance();
55 this._emulatedDevicesList.addEventListener(WebInspector.EmulatedDevicesList.Events.CustomDevicesUpdated, this._deviceListChanged, this);
56 this._emulatedDevicesList.addEventListener(WebInspector.EmulatedDevicesList.Events.StandardDevicesUpdated, this._deviceListChanged, this);
57
58 this._persistenceSetting = WebInspector.settings.createSetting("emulation.deviceModeValue", {device: "", orientation: "", mode: ""});

453 if (!modes.length)

489 update: function()
490 {
491 if (this._model.type() !== this._cachedModelType) {
492 this._cachedModelType = this._model.type();
493 this._widthInput.disabled = this._model.type() !== WebInspector.DeviceModeModel.Type.Responsive;
494 this._heightInput.disabled = this._model.type() !== WebInspector.DeviceModeModel.Type.Responsive;
495 this._deviceScaleItem.setEnabled(this._model.type() === WebInspector.DeviceModeModel.Type.Responsive);
496 this._uaItem.setEnabled(this._model.type() === WebInspector.DeviceModeModel.Type.Responsive);
497 }
498
499 var size = this._model.appliedDeviceSize();
500 this._updateHeightInput(this._model.type() === WebInspector.DeviceModeModel.Type.Responsive && this._model.isFullHeight() ? "" : String(size.height));
501 this._updateWidthInput(String(size.width));
502 this._heightInput.placeholder = size.height;
503
504 if (this._model.scale() !== this._cachedScale) {
505 this._scaleItem.setText(WebInspector.UIString("%.0f%%", this._model.scale() * 100));
506 this._cachedScale = this._model.scale();
507 }
508
509 var deviceScale = this._model.appliedDeviceScaleFactor();
510 if (deviceScale !== this._cachedDeviceScale) {
511 this._deviceScaleItem.setText(WebInspector.UIString("DPR: %.1f", deviceScale));
512 this._cachedDeviceScale = deviceScale;
513 }
514
515 var uaType = this._model.appliedUserAgentType();
516 if (uaType !== this._cachedUaType) {
517 this._uaItem.setText(uaType);
518 this._cachedUaType = uaType;
519 }
520
521 var deviceItemTitle = WebInspector.UIString("None");
522 if (this._model.type() === WebInspector.DeviceModeModel.Type.Responsive)
523 deviceItemTitle = WebInspector.UIString("Responsive");
524 if (this._model.type() === WebInspector.DeviceModeModel.Type.Device)
525 deviceItemTitle = this._model.device().title;
526 this._deviceSelectItem.setText(deviceItemTitle);
527
528 if (this._model.device() !== this._cachedModelDevice) {
529 var device = this._model.device();
530 this._modeButton.setVisible(!!device);
531 if (device) {
532 var modeCount = device ? device.modes.length : 0;
533 this._modeButton.setEnabled(modeCount >= 2);
534 this._modeButton.setTitle(modeCount === 2 ? WebInspector.UIString("Rotate") : WebInspector.UIString("Screen options"));
535 }
536 this._cachedModelDevice = device;
537 }
538
539 if (this._model.type() === WebInspector.DeviceModeModel.Type.Device)
540 this._lastMode.set(/** @type {!WebInspector.EmulatedDevice} */ (this._model.device()), /** @type {!WebInspector.EmulatedDevice.Mode} */ (this._model.mode()));
541
542 if (this._model.mode() !== this._cachedModelMode && this._model.type() !== WebInspector.DeviceModeModel.Type.None) {
543 this._cachedModelMode = this._model.mode();
544 var value = this._persistenceSetting.get();
545 if (this._model.device()) {
546 value.device = this._model.device().title;
547 value.orientation = this._model.mode() ? this._model.mode().orientation : "";
548 value.mode = this._model.mode() ? this._model.mode().title : "";
549 } else {
550 value.device = "";
551 value.orientation = "";
552 value.mode = "";
553 }
554 this._persistenceSetting.set(value);
555 }
556 },
557
558 restore: function()
559 {
560 for (var device of this._allDevices()) {
561 if (device.title === this._persistenceSetting.get().device) {
562 for (var mode of device.modes) {
563 if (mode.orientation === this._persistenceSetting.get().orientation && mode.title === this._persistenceSetting.get().mode) {
564 this._lastMode.set(device, mode);
565 this._emulateDevice(device);
566 return;
567 }
568 }
569 }
570 }
571
572 this._model.emulate(WebInspector.DeviceModeModel.Type.Responsive, null, null);
