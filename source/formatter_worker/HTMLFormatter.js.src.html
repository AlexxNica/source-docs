107 /**
108 * @param {!WebInspector.HTMLModel.Element} element
109 * @param {!WebInspector.HTMLModel.Token} token
110 */
111 _formatToken: function(element, token)
112 {
113 if (token.value.isWhitespace())
114 return;
115 if (token.type.has("comment") || token.type.has("meta")) {
116 this._builder.addNewLine();
117 this._builder.addToken(token.value.trim(), token.startOffset);
118 this._builder.addNewLine();
119 return;
120 }
121
122 var isBodyToken = element.openTag.endOffset <= token.startOffset && token.startOffset < element.closeTag.startOffset;
123 if (isBodyToken && element.name === "style") {
124 this._builder.addNewLine();
125 this._builder.increaseNestingLevel();
126 this._cssFormatter.format(this._text, this._lineEndings, token.startOffset, token.endOffset);
127 this._builder.decreaseNestingLevel();
128 return;
129 }
130 if (isBodyToken && element.name === "script") {
131 this._builder.addNewLine();
132 this._builder.increaseNestingLevel();
133 var mimeType = element.openTag.attributes.has("type") ? element.openTag.attributes.get("type").toLowerCase() : null;
134 if (!mimeType || WebInspector.HTMLFormatter.SupportedJavaScriptMimeTypes.has(mimeType)) {
135 this._jsFormatter.format(this._text, this._lineEndings, token.startOffset, token.endOffset);
136 } else {
137 this._builder.addToken(token.value, token.startOffset);
138 this._builder.addNewLine();
139 }
140 this._builder.decreaseNestingLevel();
141 return;
142 }
143
144 if (!isBodyToken && token.type.has("attribute"))
145 this._builder.addSoftSpace();
146
147 this._builder.addToken(token.value, token.startOffset);
148 }

218WebInspector.HTMLModel.prototype = {
219 /**
220 * @param {string} text
221 */
222 _build: function(text)
223 {
224 var tokenizer = WebInspector.createTokenizer("text/html");
225 var lastOffset = 0;
226 var lowerCaseText = text.toLowerCase();
227
228 while (true) {
229 tokenizer(text.substring(lastOffset), processToken.bind(this, lastOffset));
230 if (lastOffset >= text.length)
231 break;
232 var element = this._stack.peekLast();
233 lastOffset = lowerCaseText.indexOf("</" + element.name, lastOffset);
234 if (lastOffset === -1)
235 lastOffset = text.length;
236 var tokenStart = element.openTag.endOffset;
237 var tokenEnd = lastOffset;
238 var tokenValue = text.substring(tokenStart, tokenEnd);
239 this._tokens.push(new WebInspector.HTMLModel.Token(tokenValue, new Set(), tokenStart, tokenEnd));
240 }
241
242 while (this._stack.length > 1) {
243 var element = this._stack.peekLast();
244 this._popElement(new WebInspector.HTMLModel.Tag(element.name, text.length, text.length, new Map(), false, false));
245 }
246
247 /**
248 * @param {number} baseOffset
249 * @param {string} tokenValue
250 * @param {?string} type
251 * @param {number} tokenStart
252 * @param {number} tokenEnd
253 * @return {(!Object|undefined)}
254 * @this {WebInspector.HTMLModel}
255 */
256 function processToken(baseOffset, tokenValue, type, tokenStart, tokenEnd)
257 {
258 tokenStart += baseOffset;
259 tokenEnd += baseOffset;
260 lastOffset = tokenEnd;
261
262 var tokenType = type ? new Set(type.split(" ")) : new Set();
263 var token = new WebInspector.HTMLModel.Token(tokenValue, tokenType, tokenStart, tokenEnd);
264 this._tokens.push(token);
265 this._updateDOM(token);
266
267 var element = this._stack.peekLast();
268 if (element && (element.name === "script" || element.name === "style") && element.openTag.endOffset === lastOffset)
269 return WebInspector.AbortTokenization;
270 }

273 /**
274 * @param {!WebInspector.HTMLModel.Token} token
275 */
276 _updateDOM: function(token)
277 {
278 var S = WebInspector.HTMLModel.ParseState;
279 var value = token.value;
280 var type = token.type;
281 switch (this._state) {
282 case S.Initial:
283 if (type.has("bracket") && (value === "<" || value === "</")) {
284 this._onStartTag(token);
285 this._state = S.Tag;
286 }
287 return;
288 case S.Tag:
289 if (type.has("tag") && !type.has("bracket")) {
290 this._tagName = value.trim().toLowerCase();
291 } else if (type.has("attribute")) {
292 this._attributeName = value.trim().toLowerCase();
293 this._attributes.set(this._attributeName, "");
294 this._state = S.AttributeName;
295 } else if (type.has("bracket") && (value === ">" || value === "/>")) {
296 this._onEndTag(token);
297 this._state = S.Initial;
298 }
299 return;
300 case S.AttributeName:
301 if (!type.size && value === "=") {
302 this._state = S.AttributeValue;
303 } else if (type.has("bracket") && (value === ">" || value === "/>")) {
304 this._onEndTag(token);
305 this._state = S.Initial;
306 }
307 return;
308 case S.AttributeValue:
309 if (type.has("string")) {
310 this._attributes.set(this._attributeName, value);
311 this._state = S.Tag;
312 } else if (type.has("bracket") && (value === ">" || value === "/>")) {
313 this._onEndTag(token);
314 this._state = S.Initial;
315 }
316 return;
317 }

320 /**

323 _onStartTag: function(token)
324 {
325 this._tagName = "";
326 this._tagStartOffset = token.startOffset;
327 this._tagEndOffset = null;
328 this._attributes = new Map();

333 /**

336 _onEndTag: function(token)
337 {
338 this._tagEndOffset = token.endOffset;
339 var selfClosingTag = token.value === "/>" || WebInspector.HTMLModel.SelfClosingTags.has(this._tagName);
340 var tag = new WebInspector.HTMLModel.Tag(this._tagName, this._tagStartOffset, this._tagEndOffset, this._attributes, this._isOpenTag, selfClosingTag);
341 this._onTagComplete(tag);
342 },
343
344 /**
345 * @param {!WebInspector.HTMLModel.Tag} tag
346 */
347 _onTagComplete: function(tag)
348 {
349 if (tag.isOpenTag) {
350 var topElement = this._stack.peekLast();
351 if (topElement !== this._document && topElement.openTag.selfClosingTag)
352 this._popElement(autocloseTag(topElement, topElement.openTag.endOffset));
353 else if ((topElement.name in WebInspector.HTMLModel.AutoClosingTags) && WebInspector.HTMLModel.AutoClosingTags[topElement.name].has(tag.name))
354 this._popElement(autocloseTag(topElement, tag.startOffset));
355 this._pushElement(tag);
356 return;
357 }
358
359 while (this._stack.length > 1 && this._stack.peekLast().name !== tag.name)
360 this._popElement(autocloseTag(this._stack.peekLast(), tag.startOffset));
361 if (this._stack.length === 1)
362 return;
363 this._popElement(tag);
