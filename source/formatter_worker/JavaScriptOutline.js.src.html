5/**
6 * @param {string} content
7 */
8WebInspector.javaScriptOutline = function(content)
9{
10 var chunkSize = 100000; // characters per data chunk
11 var outlineChunk = [];
12 var previousIdentifier = null;
13 var previousToken = null;
14 var processedChunkCharacters = 0;
15 var addedFunction = false;
16 var isReadingArguments = false;
17 var argumentsText = "";
18 var currentFunction = null;
19 var tokenizer = new WebInspector.AcornTokenizer(content);
20 var AT = WebInspector.AcornTokenizer;
21
22 while (tokenizer.peekToken()) {
23 var token = /** @type {!Acorn.TokenOrComment} */(tokenizer.nextToken());
24 if (AT.lineComment(token) || AT.blockComment(token))
25 continue;
26
27 var tokenValue = content.substring(token.start, token.end);
28
29 if (AT.identifier(token) && previousToken && (AT.identifier(previousToken, "get") || AT.identifier(previousToken, "set"))) {
30 currentFunction = {
31 line: tokenizer.tokenLineStart(),
32 column: tokenizer.tokenColumnStart(),
33 name : previousToken.value + " " + tokenValue
34 };
35 addedFunction = true;
36 previousIdentifier = null;
37 } else if (AT.identifier(token)) {
38 previousIdentifier = tokenValue;
39 if (tokenValue && previousToken && AT.keyword(previousToken, "function")) {
40 // A named function: "function f...".
41 currentFunction = {
42 line: tokenizer.tokenLineStart(),
43 column: tokenizer.tokenColumnStart(),
44 name: tokenValue
45 };
46 addedFunction = true;
47 previousIdentifier = null;
48 }
49 } else if (AT.keyword(token, "function") && previousIdentifier && previousToken && AT.punctuator(previousToken, ":=")) {
50 // Anonymous function assigned to an identifier: "...f = function..."
51 // or "funcName: function...".
52 currentFunction = {
53 line: tokenizer.tokenLineStart(),
54 column: tokenizer.tokenColumnStart(),
55 name: previousIdentifier
56 };
57 addedFunction = true;
58 previousIdentifier = null;
59 } else if (AT.punctuator(token, ".") && previousToken && AT.identifier(previousToken))
60 previousIdentifier += ".";
61 else if (AT.punctuator(token, "(") && addedFunction)
62 isReadingArguments = true;
63 if (isReadingArguments && tokenValue)
64 argumentsText += tokenValue;
65
66 if (AT.punctuator(token, ")") && isReadingArguments) {
67 addedFunction = false;
68 isReadingArguments = false;
69 currentFunction.arguments = argumentsText.replace(/,[\r\n\s]*/g, ", ").replace(/([^,])[\r\n\s]+/g, "$1");
70 argumentsText = "";
71 outlineChunk.push(currentFunction);
72 }
73
74 previousToken = token;
75 processedChunkCharacters += token.end - token.start;
76
77 if (processedChunkCharacters >= chunkSize) {
78 postMessage({ chunk: outlineChunk, isLastChunk: false });
79 outlineChunk = [];
80 processedChunkCharacters = 0;
81 }
82 }
83
84 postMessage({ chunk: outlineChunk, isLastChunk: true });
85}
