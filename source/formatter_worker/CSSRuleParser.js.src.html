22/**
23 * @param {string} text
24 * @param {function(*)} chunkCallback
25 */
26WebInspector._innerParseCSS = function(text, chunkCallback)
27{
28 var chunkSize = 100000; // characters per data chunk
29 var lines = text.split("\n");
30 var rules = [];
31 var processedChunkCharacters = 0;
32
33 var state = WebInspector.CSSParserStates.Initial;
34 var rule;
35 var property;
36 var UndefTokenType = new Set();
37
38 var disabledRules = [];
39 function disabledRulesCallback(chunk)
40 {
41 disabledRules = disabledRules.concat(chunk.chunk);
42 }
43
44 /**
45 * @param {string} tokenValue
46 * @param {?string} tokenTypes
47 * @param {number} column
48 * @param {number} newColumn
49 */
50 function processToken(tokenValue, tokenTypes, column, newColumn)
51 {
52 var tokenType = tokenTypes ? new Set(tokenTypes.split(" ")) : UndefTokenType;
53 switch (state) {
54 case WebInspector.CSSParserStates.Initial:
55 if (tokenType.has("qualifier") || tokenType.has("builtin") || tokenType.has("tag")) {
56 rule = {
57 selectorText: tokenValue,
58 lineNumber: lineNumber,
59 columnNumber: column,
60 properties: [],
61 };
62 state = WebInspector.CSSParserStates.Selector;
63 } else if (tokenType.has("def")) {
64 rule = {
65 atRule: tokenValue,
66 lineNumber: lineNumber,
67 columnNumber: column,
68 };
69 state = WebInspector.CSSParserStates.AtRule;
70 }
71 break;
72 case WebInspector.CSSParserStates.Selector:
73 if (tokenValue === "{" && tokenType === UndefTokenType) {
74 rule.selectorText = rule.selectorText.trim();
75 rule.styleRange = createRange(lineNumber, newColumn);
76 state = WebInspector.CSSParserStates.Style;
77 } else {
78 rule.selectorText += tokenValue;
79 }
80 break;
81 case WebInspector.CSSParserStates.AtRule:
82 if ((tokenValue === ";" || tokenValue === "{") && tokenType === UndefTokenType) {
83 rule.atRule = rule.atRule.trim();
84 rules.push(rule);
85 state = WebInspector.CSSParserStates.Initial;
86 } else {
87 rule.atRule += tokenValue;
88 }
89 break;
90 case WebInspector.CSSParserStates.Style:
91 if (tokenType.has("meta") || tokenType.has("property")) {
92 property = {
93 name: tokenValue,
94 value: "",
95 range: createRange(lineNumber, column),
96 nameRange: createRange(lineNumber, column)
97 };
98 state = WebInspector.CSSParserStates.PropertyName;
99 } else if (tokenValue === "}" && tokenType === UndefTokenType) {
100 rule.styleRange.endLine = lineNumber;
101 rule.styleRange.endColumn = column;
102 rules.push(rule);
103 state = WebInspector.CSSParserStates.Initial;
104 } else if (tokenType.has("comment")) {
105 // The |processToken| is called per-line, so no token spans more than one line.
106 // Support only a one-line comments.
107 if (tokenValue.substring(0, 2) !== "/*" || tokenValue.substring(tokenValue.length - 2) !== "*/")
108 break;
109 var uncommentedText = tokenValue.substring(2, tokenValue.length - 2);
110 var fakeRule = "a{\n" + uncommentedText + "}";
111 disabledRules = [];
112 WebInspector._innerParseCSS(fakeRule, disabledRulesCallback);
113 if (disabledRules.length === 1 && disabledRules[0].properties.length === 1) {
114 var disabledProperty = disabledRules[0].properties[0];
115 disabledProperty.disabled = true;
116 disabledProperty.range = createRange(lineNumber, column);
117 disabledProperty.range.endColumn = newColumn;
118 var lineOffset = lineNumber - 1;
119 var columnOffset = column + 2;
120 disabledProperty.nameRange.startLine += lineOffset;
121 disabledProperty.nameRange.startColumn += columnOffset;
122 disabledProperty.nameRange.endLine += lineOffset;
123 disabledProperty.nameRange.endColumn += columnOffset;
124 disabledProperty.valueRange.startLine += lineOffset;
125 disabledProperty.valueRange.startColumn += columnOffset;
126 disabledProperty.valueRange.endLine += lineOffset;
127 disabledProperty.valueRange.endColumn += columnOffset;
128 rule.properties.push(disabledProperty);
129 }
130 }
131 break;
132 case WebInspector.CSSParserStates.PropertyName:
133 if (tokenValue === ":" && tokenType === UndefTokenType) {
134 property.name = property.name;
135 property.nameRange.endLine = lineNumber;
136 property.nameRange.endColumn = column;
137 property.valueRange = createRange(lineNumber, newColumn);
138 state = WebInspector.CSSParserStates.PropertyValue;
139 } else if (tokenType.has("property")) {
140 property.name += tokenValue;
141 }
142 break;
143 case WebInspector.CSSParserStates.PropertyValue:
144 if ((tokenValue === ";" || tokenValue === "}") && tokenType === UndefTokenType) {
145 property.value = property.value;
146 property.valueRange.endLine = lineNumber;
147 property.valueRange.endColumn = column;
148 property.range.endLine = lineNumber;
149 property.range.endColumn = tokenValue === ";" ? newColumn : column;
150 rule.properties.push(property);
151 if (tokenValue === "}") {
152 rule.styleRange.endLine = lineNumber;
153 rule.styleRange.endColumn = column;
154 rules.push(rule);
155 state = WebInspector.CSSParserStates.Initial;
156 } else {
157 state = WebInspector.CSSParserStates.Style;
158 }
159 } else if (!tokenType.has("comment")) {
160 property.value += tokenValue;
161 }
162 break;
163 default:
164 console.assert(false, "Unknown CSS parser state.");
165 }
166 processedChunkCharacters += newColumn - column;
167 if (processedChunkCharacters > chunkSize) {
168 chunkCallback({ chunk: rules, isLastChunk: false });
169 rules = [];
170 processedChunkCharacters = 0;
171 }
172 }
173 var tokenizer = WebInspector.createTokenizer("text/css");
174 var lineNumber;
175 for (lineNumber = 0; lineNumber < lines.length; ++lineNumber) {
176 var line = lines[lineNumber];
177 tokenizer(line, processToken);
178 processToken("\n", null, line.length, line.length + 1);
179 }
180 chunkCallback({ chunk: rules, isLastChunk: true });
181
182 /**
183 * @return {!{startLine: number, startColumn: number, endLine: number, endColumn: number}}
184 */
185 function createRange(lineNumber, columnNumber)
186 {
187 return {
188 startLine: lineNumber,
189 startColumn: columnNumber,
190 endLine: lineNumber,
191 endColumn: columnNumber
192 };
193 }
