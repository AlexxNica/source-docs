35WebInspector.FileManager = function()
36{
37 this._savedURLsSetting = WebInspector.settings.createLocalSetting("savedURLs", {});
38
39 /** @type {!Object.<string, ?function(boolean)>} */
40 this._saveCallbacks = {};
41 InspectorFrontendHost.events.addEventListener(InspectorFrontendHostAPI.Events.SavedURL, this._savedURL, this);
42 InspectorFrontendHost.events.addEventListener(InspectorFrontendHostAPI.Events.CanceledSaveURL, this._canceledSaveURL, this);
43 InspectorFrontendHost.events.addEventListener(InspectorFrontendHostAPI.Events.AppendedToURL, this._appendedToURL, this);
44}

46WebInspector.FileManager.EventTypes = {
47 SavedURL: "SavedURL",
48 AppendedToURL: "AppendedToURL"
49}

51WebInspector.FileManager.prototype = {
52 /**
53 * @param {string} url
54 * @param {string} content
55 * @param {boolean} forceSaveAs
56 * @param {function(boolean)=} callback
57 */
58 save: function(url, content, forceSaveAs, callback)
59 {
60 // Remove this url from the saved URLs while it is being saved.
61 var savedURLs = this._savedURLsSetting.get();
62 delete savedURLs[url];
63 this._savedURLsSetting.set(savedURLs);
64 this._saveCallbacks[url] = callback || null;
65 InspectorFrontendHost.save(url, content, forceSaveAs);
66 },
67
68 /**
69 * @param {!WebInspector.Event} event
70 */
71 _savedURL: function(event)
72 {
73 var url = /** @type {string} */ (event.data);
74 var savedURLs = this._savedURLsSetting.get();
75 savedURLs[url] = true;
76 this._savedURLsSetting.set(savedURLs);
77 this.dispatchEventToListeners(WebInspector.FileManager.EventTypes.SavedURL, url);
78 this._invokeSaveCallback(url, true);
79 },
80
81 /**
82 * @param {string} url
83 * @param {boolean} accepted
84 */
85 _invokeSaveCallback: function(url, accepted)
86 {
87 var callback = this._saveCallbacks[url];
88 delete this._saveCallbacks[url];
89 if (callback)
90 callback(accepted);
91 },
92
93 /**
94 * @param {!WebInspector.Event} event
95 */
96 _canceledSaveURL: function(event)
97 {
98 var url = /** @type {string} */ (event.data);
99 this._invokeSaveCallback(url, false);
100 },
101
102 /**
103 * @param {string} url
104 * @return {boolean}
105 */
106 isURLSaved: function(url)
107 {
108 var savedURLs = this._savedURLsSetting.get();
109 return savedURLs[url];
110 },
111
112 /**
113 * @param {string} url
114 * @param {string} content
115 */
116 append: function(url, content)
117 {
118 InspectorFrontendHost.append(url, content);
119 },
120
121 /**
122 * @param {string} url
123 */
124 close: function(url)
125 {
126 // Currently a no-op.
127 },
128
129 /**
130 * @param {!WebInspector.Event} event
131 */
132 _appendedToURL: function(event)
133 {
134 var url = /** @type {string} */ (event.data);
135 this.dispatchEventToListeners(WebInspector.FileManager.EventTypes.AppendedToURL, url);
136 },
