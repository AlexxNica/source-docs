50 _loadFromSettings: function()
51 {
52 var savedMapping = this._fileSystemMappingSetting.get();
53 this._fileSystemMappings = {};
54 for (var fileSystemPath in savedMapping) {
55 var savedFileSystemMappings = savedMapping[fileSystemPath];
56 fileSystemPath = WebInspector.IsolatedFileSystemManager.normalizePath(fileSystemPath);
57 this._fileSystemMappings[fileSystemPath] = [];
58 var fileSystemMappings = this._fileSystemMappings[fileSystemPath];
59
60 for (var i = 0; i < savedFileSystemMappings.length; ++i) {
61 var savedEntry = savedFileSystemMappings[i];
62 var entry = new WebInspector.FileSystemMapping.Entry(fileSystemPath, savedEntry.urlPrefix, savedEntry.pathPrefix, true);
63 fileSystemMappings.push(entry);
64 }
65 }
66
67 this._rebuildIndexes();
68 },
69
70 _saveToSettings: function()
71 {
72 var setting = {};
73 for (var fileSystemPath in this._fileSystemMappings) {
74 setting[fileSystemPath] = [];
75 var entries = this._fileSystemMappings[fileSystemPath];
76 for (var entry of entries) {
77 if (entry.configurable)
78 setting[fileSystemPath].push(entry);
79 }
80 }
81 this._fileSystemMappingSetting.set(setting);
82 },
83
84 _rebuildIndexes: function()
85 {
86 // We are building an index here to search for the longest url prefix match faster.
87 this._mappingForURLPrefix = {};
88 this._urlPrefixes = [];
89 for (var fileSystemPath in this._fileSystemMappings) {
90 var fileSystemMapping = this._fileSystemMappings[fileSystemPath];
91 for (var i = 0; i < fileSystemMapping.length; ++i) {
92 var entry = fileSystemMapping[i];
93 // Resolve conflict in favor of configurable mapping.
94 if (this._mappingForURLPrefix[entry.urlPrefix] && !entry.configurable)
95 continue;
96 this._mappingForURLPrefix[entry.urlPrefix] = entry;
97 if (this._urlPrefixes.indexOf(entry.urlPrefix) === -1)
98 this._urlPrefixes.push(entry.urlPrefix);
99 }
100 }
101 this._urlPrefixes.sort();

187 * @return {?WebInspector.FileSystemMapping.Entry}
188 */
189 _mappingEntryForURL: function(url)
190 {
191 for (var i = this._urlPrefixes.length - 1; i >= 0; --i) {
192 var urlPrefix = this._urlPrefixes[i];
193 if (url.startsWith(urlPrefix))
194 return this._mappingForURLPrefix[urlPrefix];
195 }
196 return null;

199 /**
200 * @param {string} fileSystemPath
201 * @param {string} filePath
202 * @return {?WebInspector.FileSystemMapping.Entry}
203 */
204 _mappingEntryForPath: function(fileSystemPath, filePath)
205 {
206 var entries = this._fileSystemMappings[fileSystemPath];
207 if (!entries)
208 return null;
209
210 var entry = null;
211 for (var i = 0; i < entries.length; ++i) {
212 var pathPrefix = entries[i].pathPrefix;
213 if (entry && entry.configurable && !entries[i].configurable)
214 continue;
215 // We are looking for the longest pathPrefix match.
216 if (entry && entry.pathPrefix.length > pathPrefix.length)
217 continue;
218 if (filePath.startsWith(pathPrefix))
219 entry = entries[i];
220 }
221 return entry;
222 },

229 _configurableMappingEntryForPathPrefix: function(fileSystemPath, pathPrefix)
230 {
231 var entries = this._fileSystemMappings[fileSystemPath];
232 for (var i = 0; i < entries.length; ++i) {
233 if (entries[i].configurable && pathPrefix === entries[i].pathPrefix)
234 return entries[i];

298 /**
299 * @param {string} url
300 * @param {string} fileSystemPath
301 * @param {string} filePath
302 */
303 addMappingForResource: function(url, fileSystemPath, filePath)
304 {
305 var commonPathSuffixLength = 0;
306 for (var i = 0; i < filePath.length; ++i) {
307 var filePathCharacter = filePath[filePath.length - 1 - i];
308 var urlCharacter = url[url.length - 1 - i];
309 if (filePathCharacter !== urlCharacter)
310 break;
311 if (filePathCharacter === "/")
312 commonPathSuffixLength = i;
313 }
314 var from = fileSystemPath.length;
315 var to = filePath.length - commonPathSuffixLength;
316 var pathPrefix = filePath.substring(from, to);
317 var urlPrefix = url.substr(0, url.length - commonPathSuffixLength);
318 if (to >= from)
319 this.addFileMapping(fileSystemPath, urlPrefix, pathPrefix);
320 else
321 this.addFileMapping(fileSystemPath, urlPrefix + pathPrefix, "/");
322 },
