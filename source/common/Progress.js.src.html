72WebInspector.CompositeProgress = function(parent)
73{
74 this._parent = parent;
75 this._children = [];
76 this._childrenDone = 0;
77 this._parent.setTotalWork(1);
78 this._parent.setWorked(0);
79}
80
81WebInspector.CompositeProgress.prototype = {
82 _childDone: function()
83 {
84 if (++this._childrenDone !== this._children.length)
85 return;
86 this._parent.done();
87 },
88
89 /**
90 * @param {number=} weight
91 * @return {!WebInspector.SubProgress}
92 */
93 createSubProgress: function(weight)
94 {
95 var child = new WebInspector.SubProgress(this, weight);
96 this._children.push(child);
97 return child;
98 },
99
100 _update: function()
101 {
102 var totalWeights = 0;
103 var done = 0;
104
105 for (var i = 0; i < this._children.length; ++i) {
106 var child = this._children[i];
107 if (child._totalWork)
108 done += child._weight * child._worked / child._totalWork;
109 totalWeights += child._weight;
110 }
111 this._parent.setWorked(done / totalWeights);
112 }
113}
