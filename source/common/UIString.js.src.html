42/**
43 * @param {string} string
44 * @param {...*} vararg
45 * @return {string}
46 */
47WebInspector.UIString.capitalize = function(string, vararg)
48{
49 if (WebInspector._useLowerCaseMenuTitles === undefined)
50 throw "WebInspector.setLocalizationPlatform() has not been called";
51
52 var localized = WebInspector.localize(string);
53 var capitalized;
54 if (WebInspector._useLowerCaseMenuTitles)
55 capitalized = localized.replace(/\^(.)/g, "$1");
56 else
57 capitalized = localized.replace(/\^(.)/g, function(str, char) { return char.toUpperCase(); });
58 return String.vsprintf(capitalized, Array.prototype.slice.call(arguments, 1));
59}
