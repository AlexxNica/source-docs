29/**
30 * @constructor
31 * @param {string} url
32 */
33WebInspector.ParsedURL = function(url)
34{
35 this.isValid = false;
36 this.url = url;
37 this.scheme = "";
38 this.host = "";
39 this.port = "";
40 this.path = "";
41 this.queryParams = "";
42 this.fragment = "";
43 this.folderPathComponents = "";
44 this.lastPathComponent = "";
45
46 // RegExp groups:
47 // 1 - scheme (using the RFC3986 grammar)
48 // 2 - hostname
49 // 3 - ?port
50 // 4 - ?path
51 // 5 - ?fragment
52 var match = url.match(/^([A-Za-z][A-Za-z0-9+.-]*):\/\/([^\s\/:]*)(?::([\d]+))?(?:(\/[^#]*)(?:#(.*))?)?$/i);
53 if (match) {
54 this.isValid = true;
55 this.scheme = match[1].toLowerCase();
56 this.host = match[2];
57 this.port = match[3];
58 this.path = match[4] || "/";
59 this.fragment = match[5];
60 } else {
61 if (this.url.startsWith("data:")) {
62 this.scheme = "data";
63 return;
64 }
65 if (this.url === "about:blank") {
66 this.scheme = "about";
67 return;
68 }
69 this.path = this.url;
70 }
71
72 // First cut the query params.
73 var path = this.path;
74 var indexOfQuery = path.indexOf("?");
75 if (indexOfQuery !== -1) {
76 this.queryParams = path.substring(indexOfQuery + 1);
77 path = path.substring(0, indexOfQuery);
78 }
79
80 // Then take last path component.
81 var lastSlashIndex = path.lastIndexOf("/");
82 if (lastSlashIndex !== -1) {
83 this.folderPathComponents = path.substring(0, lastSlashIndex);
84 this.lastPathComponent = path.substring(lastSlashIndex + 1);
85 } else
86 this.lastPathComponent = path;
87}

93WebInspector.ParsedURL.splitURLIntoPathComponents = function(url)
94{
95 if (url.startsWith("/"))
96 url = "file://" + url;
97 var parsedURL = new WebInspector.ParsedURL(url);
98 var origin;
99 var folderPath;
100 var name;
101 if (parsedURL.isValid) {
102 origin = parsedURL.scheme + "://" + parsedURL.host;
103 if (parsedURL.port)
104 origin += ":" + parsedURL.port;
105 folderPath = parsedURL.folderPathComponents;
106 name = parsedURL.lastPathComponent;
107 if (parsedURL.queryParams)
108 name += "?" + parsedURL.queryParams;
109 } else {
110 origin = "";
111 folderPath = "";
112 name = url;
113 }
114 var result = [origin];
115 var splittedPath = folderPath.split("/");
116 for (var i = 1; i < splittedPath.length; ++i) {
117 if (!splittedPath[i])
118 continue;
119 result.push(splittedPath[i]);
120 }
121 result.push(name);
122 return result;
123}

165/**
166 * @param {string} baseURL
167 * @param {string} href
168 * @return {?string}
169 */
170WebInspector.ParsedURL.completeURL = function(baseURL, href)
171{
172 if (href) {
173 // Return special URLs as-is.
174 var trimmedHref = href.trim();
175 if (trimmedHref.startsWith("data:") || trimmedHref.startsWith("blob:") || trimmedHref.startsWith("javascript:"))
176 return href;
177
178 // Return absolute URLs as-is.
179 var parsedHref = trimmedHref.asParsedURL();
180 if (parsedHref && parsedHref.scheme)
181 return trimmedHref;
182 } else {
183 return baseURL;
184 }
185
186 var parsedURL = baseURL.asParsedURL();
187 if (parsedURL) {
188 if (parsedURL.isDataURL())
189 return href;
190 var path = href;
191
192 var query = path.indexOf("?");
193 var postfix = "";
194 if (query !== -1) {
195 postfix = path.substring(query);
196 path = path.substring(0, query);
197 } else {
198 var fragment = path.indexOf("#");
199 if (fragment !== -1) {
200 postfix = path.substring(fragment);
201 path = path.substring(0, fragment);
202 }
203 }
204
205 if (!path) { // empty path, must be postfix
206 var basePath = parsedURL.path;
207 if (postfix.charAt(0) === "?") {
208 // A href of "?foo=bar" implies "basePath?foo=bar".
209 // With "basePath?a=b" and "?foo=bar" we should get "basePath?foo=bar".
210 var baseQuery = parsedURL.path.indexOf("?");
211 if (baseQuery !== -1)
212 basePath = basePath.substring(0, baseQuery);
213 } // else it must be a fragment
214 return parsedURL.scheme + "://" + parsedURL.host + (parsedURL.port ? (":" + parsedURL.port) : "") + basePath + postfix;
215 } else if (path.charAt(0) !== "/") { // relative path
216 var prefix = parsedURL.path;
217 var prefixQuery = prefix.indexOf("?");
218 if (prefixQuery !== -1)
219 prefix = prefix.substring(0, prefixQuery);
220 prefix = prefix.substring(0, prefix.lastIndexOf("/")) + "/";
221 path = prefix + path;
222 } else if (path.length > 1 && path.charAt(1) === "/") {
223 // href starts with "//" which is a full URL with the protocol dropped (use the baseURL protocol).
224 return parsedURL.scheme + ":" + path + postfix;
225 } // else absolute path
226 return parsedURL.scheme + "://" + parsedURL.host + (parsedURL.port ? (":" + parsedURL.port) : "") + normalizePath(path) + postfix;
227 }
228 return null;
229}

318/**
319 * @param {string} string
320 * @return {!{url: string, lineNumber: (number|undefined), columnNumber: (number|undefined)}}
321 */
322WebInspector.ParsedURL.splitLineAndColumn = function(string)
323{
324 var lineColumnRegEx = /(?::(\d+))?(?::(\d+))?$/;
325 var lineColumnMatch = lineColumnRegEx.exec(string);
326 var lineNumber;
327 var columnNumber;
328 console.assert(lineColumnMatch);
329
330 if (typeof(lineColumnMatch[1]) === "string") {
331 lineNumber = parseInt(lineColumnMatch[1], 10);
332 // Immediately convert line and column to 0-based numbers.
333 lineNumber = isNaN(lineNumber) ? undefined : lineNumber - 1;
334 }
335 if (typeof(lineColumnMatch[2]) === "string") {
336 columnNumber = parseInt(lineColumnMatch[2], 10);
337 columnNumber = isNaN(columnNumber) ? undefined : columnNumber - 1;
338 }
339
340 return {url: string.substring(0, string.length - lineColumnMatch[0].length), lineNumber: lineNumber, columnNumber: columnNumber};
341}
