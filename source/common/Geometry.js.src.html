123/**
124 * @param {string} text
125 * @return {?WebInspector.Geometry.CubicBezier}
126 */
127WebInspector.Geometry.CubicBezier.parse = function(text)
128{
129 var keywordValues = WebInspector.Geometry.CubicBezier.KeywordValues;
130 var value = text.toLowerCase().replace(/\s+/g, "");
131 if (Object.keys(keywordValues).indexOf(value) !== -1)
132 return WebInspector.Geometry.CubicBezier.parse(keywordValues[value]);
133 var bezierRegex = /^cubic-bezier\(([^,]+),([^,]+),([^,]+),([^,]+)\)$/;
134 var match = value.match(bezierRegex);
135 if (match) {
136 var control1 = new WebInspector.Geometry.Point(parseFloat(match[1]), parseFloat(match[2]));
137 var control2 = new WebInspector.Geometry.Point(parseFloat(match[3]), parseFloat(match[4]));
138 return new WebInspector.Geometry.CubicBezier(control1, control2);
139 }
140 return null;
141}

144WebInspector.Geometry.CubicBezier.prototype = {
145 /**
146 * @param {number} t
147 * @return {!WebInspector.Geometry.Point}
148 */
149 evaluateAt: function(t)
150 {
151 /**
152 * @param {number} v1
153 * @param {number} v2
154 * @param {number} t
155 */
156 function evaluate(v1, v2, t)
157 {
158 return 3 * (1 - t) * (1 - t) * t * v1 + 3 * (1 - t) * t * t * v2 + Math.pow(t, 3);
159 }
160
161 var x = evaluate(this.controlPoints[0].x, this.controlPoints[1].x, t);
162 var y = evaluate(this.controlPoints[0].y, this.controlPoints[1].y, t);
163 return new WebInspector.Geometry.Point(x, y);
164 },

173 for (var keyword in keywordValues) {
174 if (raw === keywordValues[keyword])
175 return keyword;
176 }
177 return raw;
178 }

194/**

196 * @return {!WebInspector.Geometry.EulerAngles}

198WebInspector.Geometry.EulerAngles.fromRotationMatrix = function(rotationMatrix)
199{
200 var beta = Math.atan2(rotationMatrix.m23, rotationMatrix.m33);
201 var gamma = Math.atan2(-rotationMatrix.m13, Math.sqrt(rotationMatrix.m11 * rotationMatrix.m11 + rotationMatrix.m12 * rotationMatrix.m12));
202 var alpha = Math.atan2(rotationMatrix.m12, rotationMatrix.m11);
203 return new WebInspector.Geometry.EulerAngles(WebInspector.Geometry.radiansToDegrees(alpha), WebInspector.Geometry.radiansToDegrees(beta), WebInspector.Geometry.radiansToDegrees(gamma));
204}

261/**
262 * @param {!WebInspector.Geometry.Vector} v
263 * @param {!CSSMatrix} m
264 * @return {!WebInspector.Geometry.Vector}
265 */
266WebInspector.Geometry.multiplyVectorByMatrixAndNormalize = function(v, m)
267{
268 var t = v.x * m.m14 + v.y * m.m24 + v.z * m.m34 + m.m44;
269 var x = (v.x * m.m11 + v.y * m.m21 + v.z * m.m31 + m.m41) / t;
270 var y = (v.x * m.m12 + v.y * m.m22 + v.z * m.m32 + m.m42) / t;
271 var z = (v.x * m.m13 + v.y * m.m23 + v.z * m.m33 + m.m43) / t;
272 return new WebInspector.Geometry.Vector(x, y, z);
273}

275/**
276 * @param {!WebInspector.Geometry.Vector} u
277 * @param {!WebInspector.Geometry.Vector} v
278 * @return {number}
279 */
280WebInspector.Geometry.calculateAngle = function(u, v)
281{
282 var uLength = u.length();
283 var vLength = v.length();
284 if (uLength <= WebInspector.Geometry._Eps || vLength <= WebInspector.Geometry._Eps)
285 return 0;
286 var cos = WebInspector.Geometry.scalarProduct(u, v) / uLength / vLength;
287 if (Math.abs(cos) > 1)
288 return 0;
289 return WebInspector.Geometry.radiansToDegrees(Math.acos(cos));
290}

310/**
311 * @param {!CSSMatrix} matrix
312 * @param {!Array.<number>} points
313 * @param {{minX: number, maxX: number, minY: number, maxY: number}=} aggregateBounds
314 * @return {!{minX: number, maxX: number, minY: number, maxY: number}}
315 */
316WebInspector.Geometry.boundsForTransformedPoints = function(matrix, points, aggregateBounds)
317{
318 if (!aggregateBounds)
319 aggregateBounds = {minX: Infinity, maxX: -Infinity, minY: Infinity, maxY: -Infinity};
320 if (points.length % 3)
321 console.assert("Invalid size of points array");
322 for (var p = 0; p < points.length; p += 3) {
323 var vector = new WebInspector.Geometry.Vector(points[p], points[p + 1], points[p + 2]);
324 vector = WebInspector.Geometry.multiplyVectorByMatrixAndNormalize(vector, matrix);
325 aggregateBounds.minX = Math.min(aggregateBounds.minX, vector.x);
326 aggregateBounds.maxX = Math.max(aggregateBounds.maxX, vector.x);
327 aggregateBounds.minY = Math.min(aggregateBounds.minY, vector.y);
328 aggregateBounds.maxY = Math.max(aggregateBounds.maxY, vector.y);
329 }
330 return aggregateBounds;
331}
