635 _onForwardKeyboardEvent: function(message)
636 {
637 message.entries.forEach(handleEventEntry);
638
639 /**
640 * @param {*} entry
641 * @suppressGlobalPropertiesCheck
642 */
643 function handleEventEntry(entry)
644 {
645 if (!entry.ctrlKey && !entry.altKey && !entry.metaKey && !/^F\d+$/.test(entry.key) && entry.key !== "Escape")
646 return;
647 // Fool around closure compiler -- it has its own notion of both KeyboardEvent constructor
648 // and initKeyboardEvent methods and overriding these in externs.js does not have effect.
649 var event = new window.KeyboardEvent(entry.eventType, {
650 key: entry.key,
651 code: entry.code,
652 keyCode: entry.keyCode,
653 location: entry.location,
654 ctrlKey: entry.ctrlKey,
655 altKey: entry.altKey,
656 shiftKey: entry.shiftKey,
657 metaKey: entry.metaKey
658 });
659 event.__keyCode = keyCodeForEntry(entry);
660 document.dispatchEvent(event);
661 }
662
663 function keyCodeForEntry(entry)
664 {
665 var keyCode = entry.keyCode;
666 if (!keyCode) {
667 // This is required only for synthetic events (e.g. dispatched in tests).
668 if (entry.key === "Escape")
669 keyCode = 27;
670 }
671 return keyCode || 0;
672 }
