118/**
119 * @constructor
120 */
121function EventSinkImpl(type, customDispatch)
122{
123 this._type = type;
124 this._listeners = [];
125 this._customDispatch = customDispatch;
126}
127
128EventSinkImpl.prototype = {
129 addListener: function(callback)
130 {
131 if (typeof callback !== "function")
132 throw "addListener: callback is not a function";
133 if (this._listeners.length === 0)
134 extensionServer.sendRequest({ command: commands.Subscribe, type: this._type });
135 this._listeners.push(callback);
136 extensionServer.registerHandler("notify-" + this._type, this._dispatch.bind(this));
137 },
138
139 removeListener: function(callback)
140 {
141 var listeners = this._listeners;
142
143 for (var i = 0; i < listeners.length; ++i) {
144 if (listeners[i] === callback) {
145 listeners.splice(i, 1);
146 break;
147 }
148 }
149 if (this._listeners.length === 0)
150 extensionServer.sendRequest({ command: commands.Unsubscribe, type: this._type });
151 },
152
153 /**
154 * @param {...} vararg
155 */
156 _fire: function(vararg)
157 {
158 var listeners = this._listeners.slice();
159 for (var i = 0; i < listeners.length; ++i)
160 listeners[i].apply(null, arguments);
161 },
162
163 _dispatch: function(request)
164 {
165 if (this._customDispatch)
166 this._customDispatch.call(this, request);
167 else
168 this._fire.apply(this, request.arguments);
169 }
170}

369function declareInterfaceClass(implConstructor)
370{
371 return function()
372 {
373 var impl = { __proto__: implConstructor.prototype };
374 implConstructor.apply(impl, arguments);
375 populateInterfaceClass(this, impl);
376 };

379function defineDeprecatedProperty(object, className, oldName, newName)
380{
381 var warningGiven = false;
382 function getter()
383 {
384 if (!warningGiven) {
385 console.warn(className + "." + oldName + " is deprecated. Use " + className + "." + newName + " instead");
386 warningGiven = true;
387 }
388 return object[newName];
389 }
390 object.__defineGetter__(oldName, getter);
391}

393function extractCallbackArgument(args)
394{
395 var lastArgument = args[args.length - 1];
396 return typeof lastArgument === "function" ? lastArgument : undefined;

802var keyboardEventRequestQueue = [];
803var forwardTimer = null;
804
805function forwardKeyboardEvent(event)
806{
807 // We only care about global hotkeys, not about random text
808 if (!event.ctrlKey && !event.altKey && !event.metaKey && !/^F\d+$/.test(event.key) && event.key !== "Escape")
809 return;
810 var requestPayload = {
811 eventType: event.type,
812 ctrlKey: event.ctrlKey,
813 altKey: event.altKey,
814 metaKey: event.metaKey,
815 keyIdentifier: event.keyIdentifier,
816 key: event.key,
817 code: event.code,
818 location: event.location,
819 keyCode: event.keyCode
820 };
821 keyboardEventRequestQueue.push(requestPayload);
822 if (!forwardTimer)
823 forwardTimer = setTimeout(forwardEventQueue, 0);
824}
825
826function forwardEventQueue()
827{
828 forwardTimer = null;
829 var request = {
830 command: commands.ForwardKeyboardEvent,
831 entries: keyboardEventRequestQueue
832 };
833 extensionServer.sendRequest(request);
834 keyboardEventRequestQueue = [];
835}

840/**
841 * @constructor
842 */
843function ExtensionServerClient()
844{
845 this._callbacks = {};
846 this._handlers = {};
847 this._lastRequestId = 0;
848 this._lastObjectId = 0;
849
850 this.registerHandler("callback", this._onCallback.bind(this));
851
852 var channel = new MessageChannel();
853 this._port = channel.port1;
854 this._port.addEventListener("message", this._onMessage.bind(this), false);
855 this._port.start();
856
857 window.parent.postMessage("registerExtension", [ channel.port2 ], "*");
858}
859
860ExtensionServerClient.prototype = {
861 /**
862 * @param {!Object} message
863 * @param {function()=} callback
864 */
865 sendRequest: function(message, callback)
866 {
867 if (typeof callback === "function")
868 message.requestId = this._registerCallback(callback);
869 this._port.postMessage(message);
870 },
871
872 /**
873 * @return {boolean}
874 */
875 hasHandler: function(command)
876 {
877 return !!this._handlers[command];
878 },
879
880 registerHandler: function(command, handler)
881 {
882 this._handlers[command] = handler;
883 },
884
885 unregisterHandler: function(command)
886 {
887 delete this._handlers[command];
888 },
889
890 /**
891 * @return {string}
892 */
893 nextObjectId: function()
894 {
895 return injectedScriptId.toString() + "_" + ++this._lastObjectId;
896 },
897
898 _registerCallback: function(callback)
899 {
900 var id = ++this._lastRequestId;
901 this._callbacks[id] = callback;
902 return id;
903 },
904
905 _onCallback: function(request)
906 {
907 if (request.requestId in this._callbacks) {
908 var callback = this._callbacks[request.requestId];
909 delete this._callbacks[request.requestId];
910 callback(request.result);
911 }
912 },
913
914 _onMessage: function(event)
