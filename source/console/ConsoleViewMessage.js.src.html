144 /**
145 * @param {string} title
146 * @return {!Element}
147 * @this {WebInspector.ConsoleMessage}
148 */
149 function linkifyRequest(title)
150 {
151 return WebInspector.Linkifier.linkifyUsingRevealer(/** @type {!WebInspector.NetworkRequest} */ (this.request), title, this.request.url);
152 }
153
154 var consoleMessage = this._message;
155 if (!this._messageElement) {
156 if (consoleMessage.source === WebInspector.ConsoleMessage.MessageSource.ConsoleAPI) {
157 switch (consoleMessage.type) {
158 case WebInspector.ConsoleMessage.MessageType.Trace:
159 this._messageElement = this._format(consoleMessage.parameters || ["console.trace"]);
160 break;
161 case WebInspector.ConsoleMessage.MessageType.Clear:
162 this._messageElement = createTextNode(WebInspector.UIString("Console was cleared"));
163 this._formattedMessage.classList.add("console-info");
164 break;
165 case WebInspector.ConsoleMessage.MessageType.Assert:
166 var args = [WebInspector.UIString("Assertion failed:")];
167 if (consoleMessage.parameters)
168 args = args.concat(consoleMessage.parameters);
169 this._messageElement = this._format(args);
170 break;
171 case WebInspector.ConsoleMessage.MessageType.Dir:
172 var obj = consoleMessage.parameters ? consoleMessage.parameters[0] : undefined;
173 var args = ["%O", obj];
174 this._messageElement = this._format(args);
175 break;
176 case WebInspector.ConsoleMessage.MessageType.Profile:
177 case WebInspector.ConsoleMessage.MessageType.ProfileEnd:
178 this._messageElement = this._format([consoleMessage.messageText]);
179 break;
180 default:
181 if (consoleMessage.parameters && consoleMessage.parameters.length === 1 && consoleMessage.parameters[0].type === "string")
182 this._messageElement = this._tryFormatAsError(/** @type {string} */(consoleMessage.parameters[0].value));
183
184 var args = consoleMessage.parameters || [consoleMessage.messageText];
185 this._messageElement = this._messageElement || this._format(args);
186 }
187 } else if (consoleMessage.source === WebInspector.ConsoleMessage.MessageSource.Network) {
188 if (consoleMessage.request) {
189 this._messageElement = createElement("span");
190 if (consoleMessage.level === WebInspector.ConsoleMessage.MessageLevel.Error || consoleMessage.level === WebInspector.ConsoleMessage.MessageLevel.RevokedError) {
191 this._messageElement.createTextChildren(consoleMessage.request.requestMethod, " ");
192 this._messageElement.appendChild(WebInspector.Linkifier.linkifyUsingRevealer(consoleMessage.request, consoleMessage.request.url, consoleMessage.request.url));
193 if (consoleMessage.request.failed)
194 this._messageElement.createTextChildren(" ", consoleMessage.request.localizedFailDescription);
195 else
196 this._messageElement.createTextChildren(" ", String(consoleMessage.request.statusCode), " (", consoleMessage.request.statusText, ")");
197 } else {
198 var fragment = WebInspector.linkifyStringAsFragmentWithCustomLinkifier(consoleMessage.messageText, linkifyRequest.bind(consoleMessage));
199 this._messageElement.appendChild(fragment);
200 }
201 } else {
202 var url = consoleMessage.url;
203 if (url) {
204 var isExternal = !WebInspector.resourceForURL(url) && !WebInspector.networkMapping.uiSourceCodeForURLForAnyTarget(url);
205 this._anchorElement = WebInspector.linkifyURLAsNode(url, url, "console-message-url", isExternal);
206 }
207 this._messageElement = this._format([consoleMessage.messageText]);
208 }
209 } else {
210 var args = consoleMessage.parameters || [consoleMessage.messageText];
211 this._messageElement = this._format(args);
212 }
213 }
214
215 if (consoleMessage.source !== WebInspector.ConsoleMessage.MessageSource.Network || consoleMessage.request) {
216 if (consoleMessage.scriptId) {
217 this._anchorElement = this._linkifyScriptId(consoleMessage.scriptId, consoleMessage.url || "", consoleMessage.line, consoleMessage.column);
218 } else {
219 if (consoleMessage.stackTrace && consoleMessage.stackTrace.callFrames.length)
220 this._anchorElement = this._linkifyStackTraceTopFrame(consoleMessage.stackTrace);
221 else if (consoleMessage.url && consoleMessage.url !== "undefined")
222 this._anchorElement = this._linkifyLocation(consoleMessage.url, consoleMessage.line, consoleMessage.column);
223 }
224 }
225
226 this._formattedMessage.appendChild(this._messageElement);
227 if (this._anchorElement) {
228 // Append a space to prevent the anchor text from being glued to the console message when the user selects and copies the console messages.
229 this._anchorElement.appendChild(createTextNode(" "));
230 this._formattedMessage.insertBefore(this._anchorElement, this._formattedMessage.firstChild);
231 }
232
233 var dumpStackTrace = !!consoleMessage.stackTrace && (consoleMessage.source === WebInspector.ConsoleMessage.MessageSource.Network || consoleMessage.level === WebInspector.ConsoleMessage.MessageLevel.Error || consoleMessage.level === WebInspector.ConsoleMessage.MessageLevel.RevokedError || consoleMessage.type === WebInspector.ConsoleMessage.MessageType.Trace || consoleMessage.level === WebInspector.ConsoleMessage.MessageLevel.Warning);
234 if (dumpStackTrace) {
235 var treeOutline = new TreeOutline();
236 treeOutline.element.classList.add("outline-disclosure", "outline-disclosure-no-padding");
237 var content = this._formattedMessage;
238 var root = new TreeElement(content);
239 root.toggleOnClick = true;
240 root.selectable = false;
241 content.treeElementForTest = root;
242 treeOutline.appendChild(root);
243 if (consoleMessage.type === WebInspector.ConsoleMessage.MessageType.Trace)
244 root.expand();
245
246 this._populateStackTraceTreeElement(root);
247 this._formattedMessage = treeOutline.element;

302 _format: function(parameters)
303 {
304 // This node is used like a Builder. Values are continually appended onto it.
305 var formattedResult = createElement("span");
306 if (!parameters.length)
307 return formattedResult;
308
309 var target = this._target();
310
311 // Formatting code below assumes that parameters are all wrappers whereas frontend console
312 // API allows passing arbitrary values as messages (strings, numbers, etc.). Wrap them here.
313 for (var i = 0; i < parameters.length; ++i) {
314 // FIXME: Only pass runtime wrappers here.
315 if (parameters[i] instanceof WebInspector.RemoteObject)
316 continue;
317
318 if (!target) {
319 parameters[i] = WebInspector.RemoteObject.fromLocalObject(parameters[i]);
320 continue;
321 }
322
323 if (typeof parameters[i] === "object")
324 parameters[i] = target.runtimeModel.createRemoteObject(parameters[i]);
325 else
326 parameters[i] = target.runtimeModel.createRemoteObjectFromPrimitiveValue(parameters[i]);
327 }
328
329 // There can be string log and string eval result. We distinguish between them based on message type.
330 var shouldFormatMessage = WebInspector.RemoteObject.type(parameters[0]) === "string" && (this._message.type !== WebInspector.ConsoleMessage.MessageType.Result || this._message.level === WebInspector.ConsoleMessage.MessageLevel.Error || this._message.level === WebInspector.ConsoleMessage.MessageLevel.RevokedError);
331
332 // Multiple parameters with the first being a format string. Save unused substitutions.
333 if (shouldFormatMessage) {
334 // Multiple parameters with the first being a format string. Save unused substitutions.
335 var result = this._formatWithSubstitutionString(parameters[0].description, parameters.slice(1), formattedResult);
336 parameters = result.unusedSubstitutions;
337 if (parameters.length)
338 formattedResult.createTextChild(" ");
339 }
340
341 if (this._message.type === WebInspector.ConsoleMessage.MessageType.Table) {
342 formattedResult.appendChild(this._formatParameterAsTable(parameters));
343 return formattedResult;
344 }
345
346 // Single parameter, or unused substitutions from above.
347 for (var i = 0; i < parameters.length; ++i) {
348 // Inline strings when formatting.
349 if (shouldFormatMessage && parameters[i].type === "string")
350 formattedResult.appendChild(WebInspector.linkifyStringAsFragment(parameters[i].description));
351 else
352 formattedResult.appendChild(this._formatParameter(parameters[i], false, true));
353 if (i < parameters.length - 1)
354 formattedResult.createTextChild(" ");
355 }
356 return formattedResult;

528 /**
529 * @param {!Array.<!WebInspector.RemoteObject>} parameters
530 * @return {!Element}
531 */
532 _formatParameterAsTable: function(parameters)
533 {
534 var element = createElementWithClass("div", "console-message-formatted-table");
535 var table = parameters[0];
536 if (!table || !table.preview)
537 return element;
538
539 var columnNames = [];
540 var preview = table.preview;
541 var rows = [];
542 for (var i = 0; i < preview.properties.length; ++i) {
543 var rowProperty = preview.properties[i];
544 var rowPreview = rowProperty.valuePreview;
545 if (!rowPreview)
546 continue;
547
548 var rowValue = {};
549 const maxColumnsToRender = 20;
550 for (var j = 0; j < rowPreview.properties.length; ++j) {
551 var cellProperty = rowPreview.properties[j];
552 var columnRendered = columnNames.indexOf(cellProperty.name) !== -1;
553 if (!columnRendered) {
554 if (columnNames.length === maxColumnsToRender)
555 continue;
556 columnRendered = true;
557 columnNames.push(cellProperty.name);
558 }
559
560 if (columnRendered) {
561 var cellElement = this._renderPropertyPreviewOrAccessor(table, [rowProperty, cellProperty]);
562 cellElement.classList.add("console-message-nowrap-below");
563 rowValue[cellProperty.name] = cellElement;
564 }
565 }
566 rows.push([rowProperty.name, rowValue]);
567 }
568
569 var flatValues = [];
570 for (var i = 0; i < rows.length; ++i) {
571 var rowName = rows[i][0];
572 var rowValue = rows[i][1];
573 flatValues.push(rowName);
574 for (var j = 0; j < columnNames.length; ++j)
575 flatValues.push(rowValue[columnNames[j]]);
576 }
577
578 var dataGridContainer = element.createChild("span");
579 element.appendChild(this._formatParameter(table, true, false));
580 if (!flatValues.length)
581 return element;
582
583 columnNames.unshift(WebInspector.UIString("(index)"));
584 var dataGrid = WebInspector.SortableDataGrid.create(columnNames, flatValues);
585 dataGrid.renderInline();
586 dataGridContainer.appendChild(dataGrid.element);
587 this._dataGrids.push(dataGrid);
588 return element;

638 _printArrayResult: function(array, elem, properties)
639 {
640 if (!properties) {
641 this._formatParameterAsObject(array, elem, false);
642 return;
643 }
644
645 var titleElement = createElement("span");
646 var elements = {};
647 for (var i = 0; i < properties.length; ++i) {
648 var property = properties[i];
649 var name = property.name;
650 if (isNaN(name))
651 continue;
652 if (property.getter)
653 elements[name] = this._formatAsAccessorProperty(array, [name], true);
654 else if (property.value)
655 elements[name] = this._formatAsArrayEntry(property.value);
656 }
657
658 titleElement.createTextChild("[");
659 var lastNonEmptyIndex = -1;
660
661 function appendUndefined(titleElement, index)
662 {
663 if (index - lastNonEmptyIndex <= 1)
664 return;
665 var span = titleElement.createChild("span", "object-value-undefined");
666 span.textContent = WebInspector.UIString("undefined × %d", index - lastNonEmptyIndex - 1);
667 }
668
669 var length = array.arrayLength();
670 for (var i = 0; i < length; ++i) {
671 var element = elements[i];
672 if (!element)
673 continue;
674
675 if (i - lastNonEmptyIndex > 1) {
676 appendUndefined(titleElement, i);
677 titleElement.createTextChild(", ");
678 }
679
680 titleElement.appendChild(element);
681 lastNonEmptyIndex = i;
682 if (i < length - 1)
683 titleElement.createTextChild(", ");
684 }
685 appendUndefined(titleElement, length);
686
687 titleElement.createTextChild("]");
688
689 var section = new WebInspector.ObjectPropertiesSection(array, titleElement, this._linkifier);
690 section.element.classList.add("console-view-object-properties-section");
691 section.enableContextMenu();
