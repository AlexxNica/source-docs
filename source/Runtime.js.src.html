41/**
42 * @param {string} url
43 * @return {!Promise.<string>}
44 */
45function loadResourcePromise(url)
46{
47 return new Promise(load);
48
49 /**
50 * @param {function(?)} fulfill
51 * @param {function(*)} reject
52 */
53 function load(fulfill, reject)
54 {
55 var xhr = new XMLHttpRequest();
56 xhr.open("GET", url, true);
57 xhr.onreadystatechange = onreadystatechange;
58
59 /**
60 * @param {Event} e
61 */
62 function onreadystatechange(e)
63 {
64 if (xhr.readyState !== 4)
65 return;
66
67 if ([0, 200, 304].indexOf(xhr.status) === -1) // Testing harness file:/// results in 0.
68 reject(new Error("While loading from url " + url + " server responded with a status of " + xhr.status));
69 else
70 fulfill(e.target.response);
71 }
72 xhr.send(null);
73 }
74}

76/**
77 * http://tools.ietf.org/html/rfc3986#section-5.2.4
78 * @param {string} path
79 * @return {string}
80 */
81function normalizePath(path)
82{
83 if (path.indexOf("..") === -1 && path.indexOf(".") === -1)
84 return path;
85
86 var normalizedSegments = [];
87 var segments = path.split("/");
88 for (var i = 0; i < segments.length; i++) {
89 var segment = segments[i];
90 if (segment === ".")
91 continue;
92 else if (segment === "..")
93 normalizedSegments.pop();
94 else if (segment)
95 normalizedSegments.push(segment);
96 }
97 var normalizedPath = normalizedSegments.join("/");
98 if (normalizedPath[normalizedPath.length - 1] === "/")
99 return normalizedPath;
100 if (path[0] === "/" && normalizedPath)
101 normalizedPath = "/" + normalizedPath;
102 if ((path[path.length - 1] === "/") || (segments[segments.length - 1] === ".") || (segments[segments.length - 1] === ".."))
103 normalizedPath = normalizedPath + "/";
104
105 return normalizedPath;
106}

108/**
109 * @param {!Array.<string>} scriptNames
110 * @param {string=} base
111 * @return {!Promise.<undefined>}
112 */
113function loadScriptsPromise(scriptNames, base)
114{
115 /** @type {!Array<!Promise<undefined>>} */
116 var promises = [];
117 /** @type {!Array<string>} */
118 var urls = [];
119 var sources = new Array(scriptNames.length);
120 var scriptToEval = 0;
121 for (var i = 0; i < scriptNames.length; ++i) {
122 var scriptName = scriptNames[i];
123 var sourceURL = (base || self._importScriptPathPrefix) + scriptName;
124
125 var schemaIndex = sourceURL.indexOf("://") + 3;
126 var pathIndex = sourceURL.indexOf("/", schemaIndex);
127 if (pathIndex === -1)
128 pathIndex = sourceURL.length;
129 sourceURL = sourceURL.substring(0, pathIndex) + normalizePath(sourceURL.substring(pathIndex));
130
131 if (_loadedScripts[sourceURL])
132 continue;
133 urls.push(sourceURL);
134 promises.push(loadResourcePromise(sourceURL).then(scriptSourceLoaded.bind(null, i), scriptSourceLoaded.bind(null, i, undefined)));
135 }
136 return Promise.all(promises).then(undefined);
137
138 /**
139 * @param {number} scriptNumber
140 * @param {string=} scriptSource
141 */
142 function scriptSourceLoaded(scriptNumber, scriptSource)
143 {
144 sources[scriptNumber] = scriptSource || "";
145 // Eval scripts as fast as possible.
146 while (typeof sources[scriptToEval] !== "undefined") {
147 evaluateScript(urls[scriptToEval], sources[scriptToEval]);
148 ++scriptToEval;
149 }
150 }

152 /**
153 * @param {string} sourceURL
154 * @param {string=} scriptSource
155 */
156 function evaluateScript(sourceURL, scriptSource)
157 {
158 _loadedScripts[sourceURL] = true;
159 if (!scriptSource) {
160 // Do not reject, as this is normal in the hosted mode.
161 console.error("Empty response arrived for script '" + sourceURL + "'");
162 return;
163 }
164 self.eval(scriptSource + "\n//# sourceURL=" + sourceURL);
165 }
166}

169 var baseUrl = self.location ? self.location.origin + self.location.pathname : "";
170 self._importScriptPathPrefix = baseUrl.substring(0, baseUrl.lastIndexOf("/") + 1);

236{
237 console.timeStamp("Runtime.startApplication");
238
239 var allDescriptorsByName = {};
240 for (var i = 0; i < allDescriptors.length; ++i) {
241 var d = allDescriptors[i];

243 }

246 if (applicationDescriptor)

253 /**
254 * @param {!{modules: !Array.<!Object>, has_html: boolean}} appDescriptor
255 * @return {!Promise.<undefined>}
256 */
257 function parseModuleDescriptors(appDescriptor)
258 {
259 var configuration = appDescriptor.modules;
260 var moduleJSONPromises = [];
261 var coreModuleNames = [];
262 for (var i = 0; i < configuration.length; ++i) {
263 var descriptor = configuration[i];
264 var name = descriptor["name"];
265 var moduleJSON = allDescriptorsByName[name];
266 if (moduleJSON)
267 moduleJSONPromises.push(Promise.resolve(moduleJSON));
268 else
269 moduleJSONPromises.push(loadResourcePromise(name + "/module.json").then(JSON.parse.bind(JSON)));
270 if (descriptor["type"] === "autostart")
271 coreModuleNames.push(name);
272 }
273
274 return Promise.all(moduleJSONPromises).then(instantiateRuntime);

276 /**
277 * @param {!Array.<!Object>} moduleDescriptors
278 * @return {!Promise.<undefined>}
279 */
280 function instantiateRuntime(moduleDescriptors)
281 {
282 for (var i = 0; i < moduleDescriptors.length; ++i) {
283 moduleDescriptors[i].name = configuration[i]["name"];
284 moduleDescriptors[i].condition = configuration[i]["condition"];
285 moduleDescriptors[i].remote = configuration[i]["type"] === "remote";
286 }
287 self.runtime = new Runtime(moduleDescriptors);
288 if (coreModuleNames)
289 return /** @type {!Promise<undefined>} */ (self.runtime._loadAutoStartModules(coreModuleNames));
290 return Promise.resolve();
291 }
292 }

806/**
807 * @param {!Object} descriptor
808 * @return {boolean}
809 */
810Runtime._isDescriptorEnabled = function(descriptor)
811{
812 var activatorExperiment = descriptor["experiment"];
813 if (activatorExperiment === "*")
814 return Runtime.experiments.supportEnabled();
815 if (activatorExperiment && activatorExperiment.startsWith("!") && Runtime.experiments.isEnabled(activatorExperiment.substring(1)))
816 return false;
817 if (activatorExperiment && !activatorExperiment.startsWith("!") && !Runtime.experiments.isEnabled(activatorExperiment))
818 return false;
819 var condition = descriptor["condition"];
820 if (condition && !condition.startsWith("!") && !Runtime.queryParam(condition))
821 return false;
822 if (condition && condition.startsWith("!") && Runtime.queryParam(condition.substring(1)))
823 return false;
824 return true;
825}
