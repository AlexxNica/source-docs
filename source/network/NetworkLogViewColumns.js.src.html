473 if (!selectedIndex)
474 selectedIndex = 1; // Sort by first item by default.

477 var value = selectedOption.value;

479 this._dataGrid.sortNodes(selectedItemConfig.sortingFunction);
480 this._dataGrid.markColumnAsSortedBy(columnConfig.id, /** @type {!WebInspector.DataGrid.Order} */ (selectedItemConfig.sort));

596 for (var columnConfig of responseHeaders)
