24WebInspector.ResourceWebSocketFrameView = function(request)
25{
26 WebInspector.VBox.call(this);
27 this.registerRequiredCSS("network/webSocketFrameView.css");
28 this.element.classList.add("websocket-frame-view");
29 this._request = request;
30
31 this._splitWidget = new WebInspector.SplitWidget(false, true, "resourceWebSocketFrameSplitViewState");
32 this._splitWidget.show(this.element);
33
34 var columns = [
35 {id: "data", title: WebInspector.UIString("Data"), sortable: false, weight: 88},
36 {id: "length", title: WebInspector.UIString("Length"), sortable: false, align: WebInspector.DataGrid.Align.Right, weight: 5},
37 {id: "time", title: WebInspector.UIString("Time"), sortable: true, weight: 7}
38 ];
39
40 this._dataGrid = new WebInspector.SortableDataGrid(columns, undefined, undefined, undefined, this._onContextMenu.bind(this));

43 this._timeComparator = /** @type {!WebInspector.SortableDataGrid.NodeComparator} */ (WebInspector.ResourceWebSocketFrameNodeTimeComparator);
44 this._dataGrid.sortNodes(this._timeComparator, false);
45 this._dataGrid.markColumnAsSortedBy("time", WebInspector.DataGrid.Order.Ascending);
46 this._dataGrid.addEventListener(WebInspector.DataGrid.Events.SortingChanged, this._sortItems, this);
47
48 this._dataGrid.setName("ResourceWebSocketFrameView");
49 this._dataGrid.addEventListener(WebInspector.DataGrid.Events.SelectedNode, this._onFrameSelected, this);
50 this._dataGrid.addEventListener(WebInspector.DataGrid.Events.DeselectedNode, this._onFrameDeselected, this);
51 this._splitWidget.setMainWidget(this._dataGrid.asWidget());
52
53 var view = new WebInspector.EmptyWidget("Select frame to browse its content.");
54 this._splitWidget.setSidebarWidget(view);

56 /** @type {?WebInspector.ResourceWebSocketFrameNode} */
57 this._selectedNode = null;

197 * @param {!WebInspector.NetworkRequest.WebSocketFrame} frame
198 */
199WebInspector.ResourceWebSocketFrameNode = function(url, frame)
200{
201 this._frame = frame;
202 this._dataText = frame.text;
203 this._url = url;
204 var length = frame.text.length;
205 var time = new Date(frame.time * 1000);
206 var timeText = ("0" + time.getHours()).substr(-2) + ":" + ("0" + time.getMinutes()).substr(-2) + ":" + ("0" + time.getSeconds()).substr(-2) + "." + ("00" + time.getMilliseconds()).substr(-3);
207 var timeNode = createElement("div");
208 timeNode.createTextChild(timeText);
209 timeNode.title = time.toLocaleString();
210
211 this._isTextFrame = frame.opCode === WebInspector.ResourceWebSocketFrameView.OpCodes.TextFrame;
212 if (!this._isTextFrame)
213 this._dataText = WebInspector.ResourceWebSocketFrameView.opCodeDescription(frame.opCode, frame.mask);
214
215 WebInspector.SortableDataGridNode.call(this, {data: this._dataText, length: length, time: timeNode});
216}
