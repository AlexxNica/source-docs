35WebInspector.ImageView = function(mimeType, contentProvider)
36{
37 WebInspector.SimpleView.call(this, WebInspector.UIString("Image"));
38 this.registerRequiredCSS("source_frame/imageView.css");
39 this.element.classList.add("image-view");
40 this._url = contentProvider.contentURL();
41 this._parsedURL = new WebInspector.ParsedURL(this._url);
42 this._mimeType = mimeType;
43 this._contentProvider = contentProvider;
44 this._sizeLabel = new WebInspector.ToolbarText();
45 this._dimensionsLabel = new WebInspector.ToolbarText();
46 this._mimeTypeLabel = new WebInspector.ToolbarText(mimeType);

88 /**
89 * @param {?string} content
90 * @return {number}

92 _base64ToSize: function(content)
93 {
94 if (!content || !content.length)
95 return 0;
96 var size = (content.length || 0) * 3 / 4;
97 if (content.length > 0 && content[content.length - 1] === "=")
98 size--;
99 if (content.length > 1 && content[content.length - 2] === "=")
100 size--;
101 return size;
