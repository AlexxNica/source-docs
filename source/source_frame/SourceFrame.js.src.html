477 replaceSelectionWith: function(searchConfig, replacement)
478 {

484 var oldText = this._textEditor.copyRange(range);

488 text = oldText.replace(regex, replacement);
489 else

496 /**
497 * @override
498 * @param {!WebInspector.SearchableView.SearchConfig} searchConfig
499 * @param {string} replacement
500 */
501 replaceAllWith: function(searchConfig, replacement)
502 {
503 this._resetCurrentSearchResultIndex();
504
505 var text = this._textEditor.text();
506 var range = this._textEditor.range();
507
508 var regex = searchConfig.toSearchRegex(true);
509 if (regex.__fromRegExpQuery)
510 text = text.replace(regex, replacement);
511 else
512 text = text.replace(regex, function() { return replacement; });
513
514 var ranges = this._collectRegexMatches(regex);
515 if (!ranges.length)
516 return;
517
518 // Calculate the position of the end of the last range to be edited.
519 var currentRangeIndex = ranges.lowerBound(this._textEditor.selection(), WebInspector.TextRange.comparator);
520 var lastRangeIndex = mod(currentRangeIndex - 1, ranges.length);
521 var lastRange = ranges[lastRangeIndex];
522 var replacementLineEndings = replacement.computeLineEndings();
523 var replacementLineCount = replacementLineEndings.length;
524 var lastLineNumber = lastRange.startLine + replacementLineEndings.length - 1;
525 var lastColumnNumber = lastRange.startColumn;
526 if (replacementLineEndings.length > 1)
527 lastColumnNumber = replacementLineEndings[replacementLineCount - 1] - replacementLineEndings[replacementLineCount - 2] - 1;
528
529 this._textEditor.editRange(range, text);
530 this._textEditor.revealPosition(lastLineNumber, lastColumnNumber);
531 this._textEditor.setSelection(WebInspector.TextRange.createFromLocation(lastLineNumber, lastColumnNumber));
532 },
533
534 _collectRegexMatches: function(regexObject)
535 {
536 var ranges = [];
537 for (var i = 0; i < this._textEditor.linesCount; ++i) {
538 var line = this._textEditor.line(i);
539 var offset = 0;
540 do {
541 var match = regexObject.exec(line);
542 if (match) {
543 var matchEndIndex = match.index + Math.max(match[0].length, 1);
544 if (match[0].length)
545 ranges.push(new WebInspector.TextRange(i, offset + match.index, i, offset + matchEndIndex));
546 offset += matchEndIndex;
547 line = line.substring(matchEndIndex);
548 }
549 } while (match && line);
550 }
551 return ranges;

600 _updateSourcePosition: function()
601 {
602 var selections = this._textEditor.selections();
603 if (!selections.length)
604 return;
605 if (selections.length > 1) {
606 this._sourcePosition.setText(WebInspector.UIString("%d selection regions", selections.length));
607 return;
608 }
609 var textRange = selections[0];
610 if (textRange.isEmpty()) {
611 this._sourcePosition.setText(WebInspector.UIString("Line %d, Column %d", textRange.endLine + 1, textRange.endColumn + 1));
612 return;
613 }
614 textRange = textRange.normalize();
615
616 var selectedText = this._textEditor.copyRange(textRange);
617 if (textRange.startLine === textRange.endLine)
618 this._sourcePosition.setText(WebInspector.UIString("%d characters selected", selectedText.length));
619 else
620 this._sourcePosition.setText(WebInspector.UIString("%d lines, %d characters selected", textRange.endLine - textRange.startLine + 1, selectedText.length));
