56WebInspector.FilteredListWidget.filterRegex = function(query)
57{
58 const toEscape = String.regexSpecialCharacters();
59 var regexString = "";
60 for (var i = 0; i < query.length; ++i) {
61 var c = query.charAt(i);
62 if (toEscape.indexOf(c) !== -1)
63 c = "\\" + c;
64 if (i)
65 regexString += "[^\\0" + c + "]*";
66 regexString += c;
67 }
68 return new RegExp(regexString, "i");
69}

170 _filterItems: function()
171 {
172 delete this._filterTimer;
173 if (this._scoringTimer) {
174 clearTimeout(this._scoringTimer);
175 delete this._scoringTimer;
176 }
177
178 var query = this._delegate.rewriteQuery(this._value());
179 this._query = query;
180 var filterRegex = query ? WebInspector.FilteredListWidget.filterRegex(query) : null;
181
182 var oldSelectedAbsoluteIndex = this._selectedIndexInFiltered ? this._filteredItems[this._selectedIndexInFiltered] : null;
183 var filteredItems = [];
184 this._selectedIndexInFiltered = 0;
185
186 var bestScores = [];
187 var bestItems = [];
188 var bestItemsToCollect = 100;
189 var minBestScore = 0;
190 var overflowItems = [];
191
192 scoreItems.call(this, 0);

204 /**
205 * @param {number} fromIndex
206 * @this {WebInspector.FilteredListWidget}
207 */
208 function scoreItems(fromIndex)
209 {
210 var maxWorkItems = 1000;
211 var workDone = 0;
212 for (var i = fromIndex; i < this._delegate.itemCount() && workDone < maxWorkItems; ++i) {
213 // Filter out non-matching items quickly.
214 if (filterRegex && !filterRegex.test(this._delegate.itemKeyAt(i)))
215 continue;
216
217 // Score item.
218 var score = this._delegate.itemScoreAt(i, query);
219 if (query)
220 workDone++;
221
222 // Find its index in the scores array (earlier elements have bigger scores).
223 if (score > minBestScore || bestScores.length < bestItemsToCollect) {
224 var index = bestScores.upperBound(score, compareIntegers);
225 bestScores.splice(index, 0, score);
226 bestItems.splice(index, 0, i);
227 if (bestScores.length > bestItemsToCollect) {
228 // Best list is too large -> drop last elements.
229 overflowItems.push(bestItems.peekLast());
230 bestScores.length = bestItemsToCollect;
231 bestItems.length = bestItemsToCollect;
232 }
233 minBestScore = bestScores.peekLast();
234 } else
235 filteredItems.push(i);
236 }
237
238 // Process everything in chunks.
239 if (i < this._delegate.itemCount()) {
240 this._scoringTimer = setTimeout(scoreItems.bind(this, i), 0);
241 return;
242 }
243 delete this._scoringTimer;
244
245 this._filteredItems = bestItems.concat(overflowItems).concat(filteredItems);
246 for (var i = 0; i < this._filteredItems.length; ++i) {
247 if (this._filteredItems[i] === oldSelectedAbsoluteIndex) {
248 this._selectedIndexInFiltered = i;
249 break;
250 }
251 }
252 this._viewportControl.invalidate();
253 if (!query)
254 this._selectedIndexInFiltered = 0;
255 this._updateSelection(this._selectedIndexInFiltered, false);
256 }

293 _onKeyDown: function(event)
294 {
295 var newSelectedIndex = this._selectedIndexInFiltered;
296
297 switch (event.keyCode) {
298 case WebInspector.KeyboardShortcut.Keys.Down.code:
299 if (++newSelectedIndex >= this._filteredItems.length)
300 newSelectedIndex = 0;
301 this._updateSelection(newSelectedIndex, true);
302 event.consume(true);
303 break;
304 case WebInspector.KeyboardShortcut.Keys.Up.code:
305 if (--newSelectedIndex < 0)
306 newSelectedIndex = this._filteredItems.length - 1;
307 this._updateSelection(newSelectedIndex, false);
308 event.consume(true);
309 break;
310 case WebInspector.KeyboardShortcut.Keys.PageDown.code:
311 newSelectedIndex = Math.min(newSelectedIndex + this._rowsPerViewport(), this._filteredItems.length - 1);
312 this._updateSelection(newSelectedIndex, true);
313 event.consume(true);
314 break;
315 case WebInspector.KeyboardShortcut.Keys.PageUp.code:
316 newSelectedIndex = Math.max(newSelectedIndex - this._rowsPerViewport(), 0);
317 this._updateSelection(newSelectedIndex, false);
318 event.consume(true);
319 break;
320 case WebInspector.KeyboardShortcut.Keys.Enter.code:
321 this._onEnter(event);
322 break;
323 default:
324 }

485 /**
486 * @param {string} text
487 * @param {string} query
488 * @return {?Array.<!WebInspector.SourceRange>}
489 */
490 function rangesForMatch(text, query)
491 {
492 var opcodes = WebInspector.Diff.charDiff(query, text);
493 var offset = 0;
494 var ranges = [];
495 for (var i = 0; i < opcodes.length; ++i) {
496 var opcode = opcodes[i];
497 if (opcode[0] === WebInspector.Diff.Operation.Equal)
498 ranges.push(new WebInspector.SourceRange(offset, opcode[1].length));
499 else if (opcode[0] !== WebInspector.Diff.Operation.Insert)
500 return null;
501 offset += opcode[1].length;
502 }
503 return ranges;
504 }
505
506 var text = element.textContent;
507 var ranges = rangesForMatch(text, query);
508 if (!ranges || !this.caseSensitive())
509 ranges = rangesForMatch(text.toUpperCase(), query.toUpperCase());
510 if (ranges) {
511 WebInspector.highlightRangesWithStyleClass(element, ranges, "highlight");
512 return true;
513 }
514 return false;
