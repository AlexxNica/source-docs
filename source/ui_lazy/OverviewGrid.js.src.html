157WebInspector.OverviewGrid.Window = function(parentElement, dividersLabelBarElement)
158{
159 this._parentElement = parentElement;
160
161 WebInspector.installDragHandle(this._parentElement, this._startWindowSelectorDragging.bind(this), this._windowSelectorDragging.bind(this), this._endWindowSelectorDragging.bind(this), "text", null);
162 if (dividersLabelBarElement)
163 WebInspector.installDragHandle(dividersLabelBarElement, this._startWindowDragging.bind(this), this._windowDragging.bind(this), null, "-webkit-grabbing", "-webkit-grab");
164
165 this.windowLeft = 0.0;
166 this.windowRight = 1.0;
167
168 this._parentElement.addEventListener("mousewheel", this._onMouseWheel.bind(this), true);
169 this._parentElement.addEventListener("dblclick", this._resizeWindowMaximum.bind(this), true);
170 WebInspector.appendStyle(this._parentElement, "ui_lazy/overviewGrid.css");
171
172 this._leftResizeElement = parentElement.createChild("div", "overview-grid-window-resizer");
173 this._leftResizeElement.style.left = "0";
174 WebInspector.installDragHandle(this._leftResizeElement, this._resizerElementStartDragging.bind(this), this._leftResizeElementDragging.bind(this), null, "ew-resize");
175
176 this._rightResizeElement = parentElement.createChild("div", "overview-grid-window-resizer");
177 this._rightResizeElement.style.right = "0";
178 WebInspector.installDragHandle(this._rightResizeElement, this._resizerElementStartDragging.bind(this), this._rightResizeElementDragging.bind(this), null, "ew-resize");
179 this.setEnabled(true);
180}

261 _endWindowSelectorDragging: function(event)
262 {
263 var window = this._overviewWindowSelector._close(event.x - this._offsetLeft);
264 delete this._overviewWindowSelector;
265 var clickThreshold = 3;
266 if (window.end - window.start < clickThreshold) {
267 if (this.dispatchEventToListeners(WebInspector.OverviewGrid.Events.Click, event))
268 return;
269 var middle = window.end;
270 window.start = Math.max(0, middle - WebInspector.OverviewGrid.MinSelectableSize / 2);
271 window.end = Math.min(this._parentElement.clientWidth, middle + WebInspector.OverviewGrid.MinSelectableSize / 2);
272 } else if (window.end - window.start < WebInspector.OverviewGrid.MinSelectableSize) {
273 if (this._parentElement.clientWidth - window.end > WebInspector.OverviewGrid.MinSelectableSize)
274 window.end = window.start + WebInspector.OverviewGrid.MinSelectableSize;
275 else
276 window.start = window.end - WebInspector.OverviewGrid.MinSelectableSize;
277 }

340 /**
341 * @param {number} windowLeft
342 * @param {number} windowRight
343 */
344 _setWindow: function(windowLeft, windowRight)
345 {
346 var left = windowLeft;
347 var right = windowRight;
348 var width = windowRight - windowLeft;
349
350 // We allow actual time window to be arbitrarily small but don't want the UI window to be too small.
351 var widthInPixels = width * this._parentElement.clientWidth;
352 var minWidthInPixels = WebInspector.OverviewGrid.MinSelectableSize / 2;
353 if (widthInPixels < minWidthInPixels) {
354 var factor = minWidthInPixels / widthInPixels;
355 left = ((windowRight + windowLeft) - width * factor) / 2;
356 right = ((windowRight + windowLeft) + width * factor) / 2;
357 }
358
359 this.windowLeft = windowLeft;
360 this._leftResizeElement.style.left = left * 100 + "%";
361 this.windowRight = windowRight;
362 this._rightResizeElement.style.left = right * 100 + "%";
363
364 this.dispatchEventToListeners(WebInspector.OverviewGrid.Events.WindowChanged);
365 },

372 {
373 var clientWidth = this._parentElement.clientWidth;
374 var windowLeft = typeof start === "number" ? start / clientWidth : this.windowLeft;
375 var windowRight = typeof end === "number" ? end / clientWidth : this.windowRight;

379 /**
380 * @param {!Event} event
381 */
382 _onMouseWheel: function(event)
383 {
384 if (!this._enabled)
385 return;
386 if (typeof event.wheelDeltaY === "number" && event.wheelDeltaY) {
387 const zoomFactor = 1.1;
388 const mouseWheelZoomSpeed = 1 / 120;
389
390 var reference = event.offsetX / event.target.clientWidth;
391 this._zoom(Math.pow(zoomFactor, -event.wheelDeltaY * mouseWheelZoomSpeed), reference);
392 }
393 if (typeof event.wheelDeltaX === "number" && event.wheelDeltaX) {
394 var offset = Math.round(event.wheelDeltaX * WebInspector.OverviewGrid.WindowScrollSpeedFactor);
395 var windowLeft = this._leftResizeElement.offsetLeft + WebInspector.OverviewGrid.ResizerOffset;
396 var windowRight = this._rightResizeElement.offsetLeft + WebInspector.OverviewGrid.ResizerOffset;
397
398 if (windowLeft - offset < 0)
399 offset = windowLeft;
400
401 if (windowRight - offset > this._parentElement.clientWidth)
402 offset = windowRight - this._parentElement.clientWidth;
403
404 this._setWindowPosition(windowLeft - offset, windowRight - offset);
405
406 event.preventDefault();
407 }
408 },

410 /**
411 * @param {number} factor
412 * @param {number} reference
413 */
414 _zoom: function(factor, reference)
415 {
416 var left = this.windowLeft;
417 var right = this.windowRight;
418 var windowSize = right - left;
419 var newWindowSize = factor * windowSize;
420 if (newWindowSize > 1) {
421 newWindowSize = 1;
422 factor = newWindowSize / windowSize;
423 }
424 left = reference + (left - reference) * factor;
425 left = Number.constrain(left, 0, 1 - newWindowSize);
426
427 right = reference + (right - reference) * factor;
428 right = Number.constrain(right, newWindowSize, 1);
429 this._setWindow(left, right);

450 _close: function(position)
451 {
452 position = Math.max(0, Math.min(position, this._width));
