453 /**
454 * @override
455 * @param {!WebInspector.SearchableView.SearchConfig} searchConfig
456 * @param {boolean} shouldJump
457 * @param {boolean=} jumpBackwards
458 */
459 performSearch: function(searchConfig, shouldJump, jumpBackwards)
460 {
461 var query = searchConfig.query;
462 // Call searchCanceled since it will reset everything we need before doing a new search.
463 this.searchCanceled();
464
465 const whitespaceTrimmedQuery = query.trim();
466 if (!whitespaceTrimmedQuery.length)
467 return;
468
469 this._searchQuery = query;
470
471 var promises = [];
472 var domModels = WebInspector.DOMModel.instances();
473 for (var domModel of domModels)
474 promises.push(domModel.performSearchPromise(whitespaceTrimmedQuery, WebInspector.moduleSetting("showUAShadowDOM").get()));
475 Promise.all(promises).then(resultCountCallback.bind(this));
476
477 /**
478 * @param {!Array.<number>} resultCounts
479 * @this {WebInspector.ElementsPanel}
480 */
481 function resultCountCallback(resultCounts)
482 {
483 /**
484 * @type {!Array.<{domModel: !WebInspector.DOMModel, index: number, node: (?WebInspector.DOMNode|undefined)}>}
485 */
486 this._searchResults = [];
487 for (var i = 0; i < resultCounts.length; ++i) {
488 var resultCount = resultCounts[i];
489 for (var j = 0; j < resultCount; ++j)
490 this._searchResults.push({domModel: domModels[i], index: j, node: undefined});
491 }
492 this._searchableView.updateSearchMatchesCount(this._searchResults.length);
493 if (!this._searchResults.length)
494 return;
495 this._currentSearchResultIndex = -1;
496
497 if (shouldJump)
498 this._jumpToSearchResult(jumpBackwards ? -1 : 0);
499 }

698 * @override
699 * @param {!KeyboardEvent} event
700 */
701 handleShortcut: function(event)
702 {
703 /**
704 * @param {!WebInspector.ElementsTreeOutline} treeOutline
705 */
706 function handleUndoRedo(treeOutline)
707 {
708 if (WebInspector.KeyboardShortcut.eventHasCtrlOrMeta(event) && !event.shiftKey && (event.key === "Z" || event.key === "z")) { // Z key
709 treeOutline.domModel().undo();
710 event.handled = true;
711 return;
712 }
713
714 var isRedoKey = WebInspector.isMac() ? event.metaKey && event.shiftKey && (event.key === "Z" || event.key === "z") : // Z key
715 event.ctrlKey && (event.key === "Y" || event.key === "y"); // Y key
716 if (isRedoKey) {
717 treeOutline.domModel().redo();
718 event.handled = true;
719 }
720 }
721
722 if (WebInspector.isEditing() && event.keyCode !== WebInspector.KeyboardShortcut.Keys.F2.code)
723 return;
724
725 var treeOutline = null;
726 for (var i = 0; i < this._treeOutlines.length; ++i) {
727 if (this._treeOutlines[i].selectedDOMNode() === this._lastValidSelectedNode)
728 treeOutline = this._treeOutlines[i];
729 }
730 if (!treeOutline)
731 return;
732
733 if (!treeOutline.editing()) {
734 handleUndoRedo.call(null, treeOutline);
735 if (event.handled) {
736 this._stylesWidget.forceUpdate();
737 return;
738 }
739 }
740
741 treeOutline.handleShortcut(event);
742 if (event.handled)
743 return;
744
745 WebInspector.Panel.prototype.handleShortcut.call(this, event);
