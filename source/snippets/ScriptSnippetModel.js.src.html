240 /**
241 * @param {!RuntimeAgent.ScriptId=} scriptId
242 * @param {?RuntimeAgent.ExceptionDetails=} exceptionDetails
243 * @this {WebInspector.ScriptSnippetModel}
244 */
245 function compileCallback(scriptId, exceptionDetails)
246 {
247 var mapping = this._mappingForTarget.get(target);
248 if (mapping.evaluationIndex(uiSourceCode) !== evaluationIndex)
249 return;
250
251 var script = /** @type {!WebInspector.Script} */(executionContext.debuggerModel.scriptForId(scriptId || exceptionDetails.scriptId));
252 mapping._addScript(script, uiSourceCode);
253 if (!scriptId) {
254 this._printRunOrCompileScriptResultFailure(target, exceptionDetails, evaluationUrl);
255 return;
256 }
257
258 var breakpointLocations = this._removeBreakpoints(uiSourceCode);
259 this._restoreBreakpoints(uiSourceCode, breakpointLocations);
260
261 this._runScript(scriptId, executionContext, evaluationUrl);
262 }
