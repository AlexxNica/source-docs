337/**
338 * @param {string} quoteCharacter
339 * @param {!CodeMirror} codeMirror
340 * @return {*}
341 */
342WebInspector.CodeMirrorTextEditor._maybeAvoidSmartQuotes = function(quoteCharacter, codeMirror)
343{
344 var textEditor = codeMirror._codeMirrorTextEditor;
345 if (!WebInspector.moduleSetting("textEditorBracketMatching").get())
346 return CodeMirror.Pass;
347 var selections = textEditor.selections();
348 if (selections.length !== 1 || !selections[0].isEmpty())
349 return CodeMirror.Pass;
350
351 var selection = selections[0];
352 var token = textEditor.tokenAtTextPosition(selection.startLine, selection.startColumn);
353 if (!token || !token.type || token.type.indexOf("string") === -1)
354 return CodeMirror.Pass;
355 var line = textEditor.line(selection.startLine);
356 var tokenValue = line.substring(token.startColumn, token.endColumn);
357 if (tokenValue[0] === tokenValue[tokenValue.length - 1] && (tokenValue[0] === "'" || tokenValue[0] === "\""))
358 return CodeMirror.Pass;
359 codeMirror.replaceSelection(quoteCharacter);
360}

370/**
371 * @param {!Array.<string>} lines
372 * @return {string}
373 */
374WebInspector.CodeMirrorTextEditor._guessIndentationLevel = function(lines)
375{
376 var tabRegex = /^\t+/;
377 var tabLines = 0;
378 var indents = {};
379
380 for (var lineNumber = 0; lineNumber < lines.length; ++lineNumber) {
381 var text = lines[lineNumber];
382 if (text.length === 0 || !WebInspector.TextUtils.isSpaceChar(text[0]))
383 continue;
384 if (tabRegex.test(text)) {
385 ++tabLines;
386 continue;
387 }
388 var i = 0;
389 while (i < text.length && WebInspector.TextUtils.isSpaceChar(text[i]))
390 ++i;
391 if (i % 2 !== 0)
392 continue;
393 indents[i] = 1 + (indents[i] || 0);
394 }
395
396 var linesCountPerIndentThreshold = 3 * lines.length / 100;
397 if (tabLines && tabLines > linesCountPerIndentThreshold)
398 return "\t";
399 var minimumIndent = Infinity;
400 for (var i in indents) {
401 if (indents[i] < linesCountPerIndentThreshold)
402 continue;
403 var indent = parseInt(i, 10);
404 if (minimumIndent > indent)
405 minimumIndent = indent;
406 }
407 if (minimumIndent === Infinity)
408 return WebInspector.moduleSetting("textEditorIndent").get();
409 return " ".repeat(minimumIndent);
410}

469 _camelCaseMoveFromPosition: function(lineNumber, columnNumber, direction)
470 {
471 /**
472 * @param {number} charNumber
473 * @param {number} length
474 * @return {boolean}
475 */
476 function valid(charNumber, length)
477 {
478 return charNumber >= 0 && charNumber < length;
479 }

481 /**
482 * @param {string} text
483 * @param {number} charNumber
484 * @return {boolean}
485 */
486 function isWordStart(text, charNumber)
487 {
488 var position = charNumber;
489 var nextPosition = charNumber + 1;
490 return valid(position, text.length) && valid(nextPosition, text.length)
491 && WebInspector.TextUtils.isWordChar(text[position]) && WebInspector.TextUtils.isWordChar(text[nextPosition])
492 && WebInspector.TextUtils.isUpperCase(text[position]) && WebInspector.TextUtils.isLowerCase(text[nextPosition]);
493 }
494
495 /**
496 * @param {string} text
497 * @param {number} charNumber
498 * @return {boolean}
499 */
500 function isWordEnd(text, charNumber)
501 {
502 var position = charNumber;
503 var prevPosition = charNumber - 1;
504 return valid(position, text.length) && valid(prevPosition, text.length)
505 && WebInspector.TextUtils.isWordChar(text[position]) && WebInspector.TextUtils.isWordChar(text[prevPosition])
506 && WebInspector.TextUtils.isUpperCase(text[position]) && WebInspector.TextUtils.isLowerCase(text[prevPosition]);
507 }

509 /**
510 * @param {number} lineNumber
511 * @param {number} lineLength
512 * @param {number} columnNumber
513 * @return {{lineNumber: number, columnNumber: number}}
514 */
515 function constrainPosition(lineNumber, lineLength, columnNumber)
516 {
517 return {
518 lineNumber: lineNumber,
519 columnNumber: Number.constrain(columnNumber, 0, lineLength)
520 };
521 }
522
523 var text = this.line(lineNumber);
524 var length = text.length;
525
526 if ((columnNumber === length && direction === 1)
527 || (columnNumber === 0 && direction === -1))
528 return this._normalizePositionForOverlappingColumn(lineNumber, length, columnNumber + direction);
529
530 var charNumber = direction === 1 ? columnNumber : columnNumber - 1;
531
532 // Move through initial spaces if any.
533 while (valid(charNumber, length) && WebInspector.TextUtils.isSpaceChar(text[charNumber]))
534 charNumber += direction;
535 if (!valid(charNumber, length))
536 return constrainPosition(lineNumber, length, charNumber);
537
538 if (WebInspector.TextUtils.isStopChar(text[charNumber])) {
539 while (valid(charNumber, length) && WebInspector.TextUtils.isStopChar(text[charNumber]))
540 charNumber += direction;
541 if (!valid(charNumber, length))
542 return constrainPosition(lineNumber, length, charNumber);
543 return {
544 lineNumber: lineNumber,
545 columnNumber: direction === -1 ? charNumber + 1 : charNumber
546 };
547 }
548
549 charNumber += direction;
550 while (valid(charNumber, length) && !isWordStart(text, charNumber) && !isWordEnd(text, charNumber) && WebInspector.TextUtils.isWordChar(text[charNumber]))
551 charNumber += direction;
552
553 if (!valid(charNumber, length))
554 return constrainPosition(lineNumber, length, charNumber);
555 if (isWordStart(text, charNumber) || isWordEnd(text, charNumber)) {
556 return {
557 lineNumber: lineNumber,
558 columnNumber: charNumber
559 };
560 }
561
562 return {
563 lineNumber: lineNumber,
564 columnNumber: direction === -1 ? charNumber + 1 : charNumber
565 };
