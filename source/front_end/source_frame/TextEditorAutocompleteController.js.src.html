113 * @param {!WebInspector.TextRange} prefixRange
114 * @param {!WebInspector.TextRange} substituteRange
115 * @return {!Promise.<!WebInspector.SuggestBox.Suggestions>}
116 */
117 _wordsWithPrefix: function(prefixRange, substituteRange)
118 {
119 var external = this._config.suggestionsCallback ? this._config.suggestionsCallback(prefixRange, substituteRange) : null;
120 if (external)
121 return external;
122
123 if (!this._dictionary || prefixRange.startColumn === prefixRange.endColumn)
124 return Promise.resolve([]);
125
126 var completions = this._dictionary.wordsWithPrefix(this._textEditor.copyRange(prefixRange));
127 var substituteWord = this._textEditor.copyRange(substituteRange);
128 if (this._dictionary.wordCount(substituteWord) === 1)
129 completions = completions.filter((word) => word !== substituteWord);
130
131 completions.sort((a, b) => this._dictionary.wordCount(b) - this._dictionary.wordCount(a) || a.length - b.length);
132 return Promise.resolve(completions.map(item => ({ title: item })));

135 /**
136 * @param {!CodeMirror} codeMirror
137 * @param {!Array.<!CodeMirror.ChangeObject>} changes
138 */
139 _changes: function(codeMirror, changes)
140 {
141 if (!changes.length)
142 return;
143
144 if (this._dictionary && this._updatedLines) {
145 for (var lineNumber in this._updatedLines)
146 this._removeWordsFromText(this._updatedLines[lineNumber]);
147 delete this._updatedLines;
148
149 var linesToUpdate = {};
150 for (var changeIndex = 0; changeIndex < changes.length; ++changeIndex) {
151 var changeObject = changes[changeIndex];
152 var editInfo = WebInspector.CodeMirrorUtils.changeObjectToEditOperation(changeObject);
153 for (var i = editInfo.newRange.startLine; i <= editInfo.newRange.endLine; ++i)
154 linesToUpdate[i] = this._codeMirror.getLine(i);
155 }
156 for (var lineNumber in linesToUpdate)
157 this._addWordsFromText(linesToUpdate[lineNumber]);
158 }
159
160 var singleCharInput = false;
161 var singleCharDelete = false;
162 var cursor = this._codeMirror.getCursor("head");
163 for (var changeIndex = 0; changeIndex < changes.length; ++changeIndex) {
164 var changeObject = changes[changeIndex];
165 if (changeObject.origin === "+input"
166 && changeObject.text.length === 1
167 && changeObject.text[0].length === 1
168 && changeObject.to.line === cursor.line
169 && changeObject.to.ch + 1 === cursor.ch) {
170 singleCharInput = true;
171 break;
172 }
173 if (this._suggestBox
174 && changeObject.origin === "+delete"
175 && changeObject.removed.length === 1
176 && changeObject.removed[0].length === 1
177 && changeObject.to.line === cursor.line
178 && changeObject.to.ch - 1 === cursor.ch) {
179 singleCharDelete = true;
180 break;
181 }
182 }
183 if (singleCharInput && this._hintMarker)
184 this._hintElement.textContent = this._hintElement.textContent.substring(1);
185
186 if (singleCharDelete && this._hintMarker && this._lastPrefix) {
187 this._hintElement.textContent = this._lastPrefix.charAt(this._lastPrefix.length - 1) + this._hintElement.textContent;
188 this._lastPrefix = this._lastPrefix.substring(0, this._lastPrefix.length - 1);
189 }
190 if (singleCharInput || singleCharDelete)
191 setImmediate(this.autocomplete.bind(this));
192 else
193 this.finishAutocomplete();
194 },

222 autocomplete: function()
223 {
224 this._initializeIfNeeded();
225 if (this._codeMirror.somethingSelected()) {
226 this.finishAutocomplete();
227 return;
228 }
229
230 var cursor = this._codeMirror.getCursor("head");
231 var substituteRange = this._substituteRange(cursor.line, cursor.ch);
232 if (!substituteRange || !this._validateSelectionsContexts(substituteRange)) {
233 this.finishAutocomplete();
234 return;
235 }
236
237 var prefixRange = substituteRange.clone();
238 prefixRange.endColumn = cursor.ch;
239 var prefix = this._textEditor.copyRange(prefixRange);
240 var hadSuggestBox = false;
241 if (this._suggestBox)
242 hadSuggestBox = true;
243
244 this._wordsWithPrefix(prefixRange, substituteRange).then(wordsAcquired.bind(this));

246 /**
247 * @param {!WebInspector.SuggestBox.Suggestions} wordsWithPrefix
248 * @this {WebInspector.TextEditorAutocompleteController}
249 */
250 function wordsAcquired(wordsWithPrefix)
251 {
252 if (!wordsWithPrefix.length || (wordsWithPrefix.length === 1 && prefix === wordsWithPrefix[0].title) || (!this._suggestBox && hadSuggestBox)) {
253 this.finishAutocomplete();
254 this._onSuggestionsShownForTest([]);
255 return;
256 }
257 if (!this._suggestBox)
258 this._suggestBox = new WebInspector.SuggestBox(this, 6);
259
260 var oldPrefixRange = this._prefixRange;
261 this._prefixRange = prefixRange;
262 if (!oldPrefixRange || prefixRange.startLine !== oldPrefixRange.startLine || prefixRange.startColumn !== oldPrefixRange.startColumn)
263 this._updateAnchorBox();
264 this._suggestBox.updateSuggestions(this._anchorBox, wordsWithPrefix, 0, !this._isCursorAtEndOfLine(), prefix);
265 this._onSuggestionsShownForTest(wordsWithPrefix);
266 this._addHintMarker(wordsWithPrefix[0].title);
267 }

313 keyDown: function(event)
314 {
315 if (!this._suggestBox)
316 return false;
317 switch (event.keyCode) {
318 case WebInspector.KeyboardShortcut.Keys.Tab.code:
319 this._suggestBox.acceptSuggestion();
320 this.finishAutocomplete();
321 return true;
322 case WebInspector.KeyboardShortcut.Keys.End.code:
323 case WebInspector.KeyboardShortcut.Keys.Right.code:
324 if (this._isCursorAtEndOfLine()) {
325 this._suggestBox.acceptSuggestion();
326 this.finishAutocomplete();
327 return true;
328 } else {
329 this.finishAutocomplete();
330 return false;
331 }
332 case WebInspector.KeyboardShortcut.Keys.Left.code:
333 case WebInspector.KeyboardShortcut.Keys.Home.code:
334 this.finishAutocomplete();
335 return false;
336 case WebInspector.KeyboardShortcut.Keys.Esc.code:
337 this.finishAutocomplete();
338 return true;
339 }
