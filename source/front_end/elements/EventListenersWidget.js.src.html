57 /**
58 * @param {string} name
59 * @param {string} value
60 * @this {WebInspector.EventListenersWidget}
61 */
62 function addDispatchFilterOption(name, value)
63 {
64 var option = dispatchFilter.createOption(name, "", value);
65 if (value === this._dispatchFilterBySetting.get())
66 dispatchFilter.select(option);
67 }
68 addDispatchFilterOption.call(this, WebInspector.UIString("All"), WebInspector.EventListenersWidget.DispatchFilterBy.All);
69 addDispatchFilterOption.call(this, WebInspector.UIString("Passive"), WebInspector.EventListenersWidget.DispatchFilterBy.Passive);
70 addDispatchFilterOption.call(this, WebInspector.UIString("Blocking"), WebInspector.EventListenersWidget.DispatchFilterBy.Blocking);
71 dispatchFilter.setMaxWidth(200);
72 this._toolbarItems.push(dispatchFilter);
73 this._toolbarItems.push(new WebInspector.ToolbarCheckbox(WebInspector.UIString("Framework listeners"), WebInspector.UIString("Resolve event listeners bound with framework"), this._showFrameworkListenersSetting));

75 WebInspector.context.addFlavorChangeListener(WebInspector.DOMNode, this.update, this);

93 doUpdate: function()
94 {
95 if (this._lastRequestedNode) {
96 this._lastRequestedNode.target().runtimeAgent().releaseObjectGroup(WebInspector.EventListenersWidget._objectGroupName);
97 delete this._lastRequestedNode;
98 }
99 var node = WebInspector.context.flavor(WebInspector.DOMNode);
100 if (!node) {
101 this._eventListenersView.reset();
102 this._eventListenersView.addEmptyHolderIfNeeded();
103 return Promise.resolve();
104 }
105 this._lastRequestedNode = node;
106 var selectedNodeOnly = !this._showForAncestorsSetting.get();
107 var promises = [];
108 var listenersView = this._eventListenersView;
109 promises.push(node.resolveToObjectPromise(WebInspector.EventListenersWidget._objectGroupName));
110 if (!selectedNodeOnly) {
111 var currentNode = node.parentNode;
112 while (currentNode) {
113 promises.push(currentNode.resolveToObjectPromise(WebInspector.EventListenersWidget._objectGroupName));
114 currentNode = currentNode.parentNode;
115 }
116 promises.push(this._windowObjectInNodeContext(node));
117 }
118 return Promise.all(promises).then(this._eventListenersView.addObjects.bind(this._eventListenersView)).then(this._showFrameworkListenersChanged.bind(this));
119 },

154 /**
155 * @param {function(?)} fulfill
156 * @param {function(*)} reject
157 */
158 function windowObjectInNodeContext(fulfill, reject)
159 {
160 var executionContexts = node.target().runtimeModel.executionContexts();
161 var context = null;
162 if (node.frameId()) {
163 for (var i = 0; i < executionContexts.length; ++i) {
164 var executionContext = executionContexts[i];
165 if (executionContext.frameId === node.frameId() && executionContext.isDefault)
166 context = executionContext;
167 }
168 } else {
169 context = executionContexts[0];
170 }
171 context.evaluate("self", WebInspector.EventListenersWidget._objectGroupName, false, true, false, false, false, fulfill);
172 }
