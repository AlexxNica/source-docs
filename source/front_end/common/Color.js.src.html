30/**
31 * @param {!Array.<number>} rgba
32 * @param {!WebInspector.Color.Format} format
33 * @param {string=} originalText
34 * @constructor
35 */
36WebInspector.Color = function(rgba, format, originalText)
37{
38 this._rgba = rgba;
39 this._originalText = originalText || null;
40 this._originalTextIsValid = !!this._originalText;
41 this._format = format;
42 if (typeof this._rgba[3] === "undefined")
43 this._rgba[3] = 1;
44
45 for (var i = 0; i < 4; ++i) {
46 if (this._rgba[i] < 0) {
47 this._rgba[i] = 0;
48 this._originalTextIsValid = false;
49 }
50 if (this._rgba[i] > 1) {
51 this._rgba[i] = 1;
52 this._originalTextIsValid = false;
53 }
54 }
55}

74/**
75 * @param {string} text
76 * @return {?WebInspector.Color}
77 */
78WebInspector.Color.parse = function(text)
79{
80 // Simple - #hex, rgb(), nickname, hsl()
81 var value = text.toLowerCase().replace(/\s+/g, "");
82 var simple = /^(?:#([0-9a-f]{3}|[0-9a-f]{6})|rgb\(((?:-?\d+%?,){2}-?\d+%?)\)|(\w+)|hsl\((-?\d+\.?\d*(?:,-?\d+\.?\d*%){2})\))$/i;
83 var match = value.match(simple);
84 if (match) {
85 if (match[1]) { // hex
86 var hex = match[1].toLowerCase();
87 var format;
88 if (hex.length === 3) {
89 format = WebInspector.Color.Format.ShortHEX;
90 hex = hex.charAt(0) + hex.charAt(0) + hex.charAt(1) + hex.charAt(1) + hex.charAt(2) + hex.charAt(2);
91 } else
92 format = WebInspector.Color.Format.HEX;
93 var r = parseInt(hex.substring(0,2), 16);
94 var g = parseInt(hex.substring(2,4), 16);
95 var b = parseInt(hex.substring(4,6), 16);
96 return new WebInspector.Color([r / 255, g / 255, b / 255, 1], format, text);
97 }
98
99 if (match[2]) { // rgb
100 var rgbString = match[2].split(/\s*,\s*/);
101 var rgba = [ WebInspector.Color._parseRgbNumeric(rgbString[0]),
102 WebInspector.Color._parseRgbNumeric(rgbString[1]),
103 WebInspector.Color._parseRgbNumeric(rgbString[2]), 1 ];
104 return new WebInspector.Color(rgba, WebInspector.Color.Format.RGB, text);
105 }
106
107 if (match[3]) { // nickname
108 var nickname = match[3].toLowerCase();
109 if (nickname in WebInspector.Color.Nicknames) {
110 var rgba = WebInspector.Color.Nicknames[nickname];
111 var color = WebInspector.Color.fromRGBA(rgba);
112 color._format = WebInspector.Color.Format.Nickname;
113 color._originalText = text;
114 return color;
115 }
116 return null;
117 }
118
119 if (match[4]) { // hsl
120 var hslString = match[4].replace(/%/g, "").split(/\s*,\s*/);
121 var hsla = [ WebInspector.Color._parseHueNumeric(hslString[0]),
122 WebInspector.Color._parseSatLightNumeric(hslString[1]),
123 WebInspector.Color._parseSatLightNumeric(hslString[2]), 1 ];
124 var rgba = [];
125 WebInspector.Color.hsl2rgb(hsla, rgba);
126 return new WebInspector.Color(rgba, WebInspector.Color.Format.HSL, text);
127 }
128
129 return null;
130 }
131
132 // Advanced - rgba(), hsla()
133 var advanced = /^(?:rgba\(((?:-?\d+%?,){3}-?(?:\d+|\d*\.\d+))\)|hsla\((-?(?:\d+|\d*\.\d+)(?:,-?(?:\d+|\d*\.\d+)*%){2},-?(?:\d+|\d*\.\d+))\))$/;
134 match = value.match(advanced);
135 if (match) {
136 if (match[1]) { // rgba
137 var rgbaString = match[1].split(/\s*,\s*/);
138 var rgba = [ WebInspector.Color._parseRgbNumeric(rgbaString[0]),
139 WebInspector.Color._parseRgbNumeric(rgbaString[1]),
140 WebInspector.Color._parseRgbNumeric(rgbaString[2]),
141 WebInspector.Color._parseAlphaNumeric(rgbaString[3]) ];
142 return new WebInspector.Color(rgba, WebInspector.Color.Format.RGBA, text);
143 }
144
145 if (match[2]) { // hsla
146 var hslaString = match[2].replace(/%/g, "").split(/\s*,\s*/);
147 var hsla = [ WebInspector.Color._parseHueNumeric(hslaString[0]),
148 WebInspector.Color._parseSatLightNumeric(hslaString[1]),
149 WebInspector.Color._parseSatLightNumeric(hslaString[2]),
150 WebInspector.Color._parseAlphaNumeric(hslaString[3]) ];
151 var rgba = [];
152 WebInspector.Color.hsl2rgb(hsla, rgba);
153 return new WebInspector.Color(rgba, WebInspector.Color.Format.HSLA, text);
154 }
155 }
156
157 return null;
158}

189 /**
190 * @return {!Array.<number>} HSLA with components within [0..1]
191 */
192 hsla: function()
193 {
194 if (this._hsla)
195 return this._hsla;
196 var r = this._rgba[0];
197 var g = this._rgba[1];
198 var b = this._rgba[2];
199 var max = Math.max(r, g, b);
200 var min = Math.min(r, g, b);
201 var diff = max - min;
202 var add = max + min;
203
204 if (min === max)
205 var h = 0;
206 else if (r === max)
207 var h = ((1 / 6 * (g - b) / diff) + 1) % 1;
208 else if (g === max)
209 var h = (1 / 6 * (b - r) / diff) + 1 / 3;
210 else
211 var h = (1 / 6 * (r - g) / diff) + 2 / 3;
212
213 var l = 0.5 * add;
214
215 if (l === 0)
216 var s = 0;
217 else if (l === 1)
218 var s = 0;
219 else if (l <= 0.5)
220 var s = diff / add;
221 else
222 var s = diff / (2 - add);
223
224 this._hsla = [h, s, l, this._rgba[3]];
225 return this._hsla;
226 },

462/**
463 * @param {!Array.<number>} hsva
464 * @param {!Array.<number>} out_hsla
465 */
466WebInspector.Color._hsva2hsla = function(hsva, out_hsla)
467{
468 var h = hsva[0];
469 var s = hsva[1];
470 var v = hsva[2];
471
472 var t = (2 - s) * v;
473 if (v === 0 || s === 0)
474 s = 0;
475 else
476 s *= v / (t < 1 ? t : 2 - t);
477
478 out_hsla[0] = h;
479 out_hsla[1] = s;
480 out_hsla[2] = t / 2;
481 out_hsla[3] = hsva[3];
482}

484/**
485 * @param {!Array.<number>} hsl
486 * @param {!Array.<number>} out_rgb
487 */
488WebInspector.Color.hsl2rgb = function(hsl, out_rgb)
489{
490 var h = hsl[0];
491 var s = hsl[1];
492 var l = hsl[2];
493
494 function hue2rgb(p, q, h)
495 {
496 if (h < 0)
497 h += 1;
498 else if (h > 1)
499 h -= 1;
500
501 if ((h * 6) < 1)
502 return p + (q - p) * h * 6;
503 else if ((h * 2) < 1)
504 return q;
505 else if ((h * 3) < 2)
506 return p + (q - p) * ((2 / 3) - h) * 6;
507 else
508 return p;
509 }
510
511 if (s < 0)
512 s = 0;
513
514 if (l <= 0.5)
515 var q = l * (1 + s);
516 else
517 var q = l + s - (l * s);
518
519 var p = 2 * l - q;
520
521 var tr = h + (1 / 3);
522 var tg = h;
523 var tb = h - (1 / 3);
524
525 out_rgb[0] = hue2rgb(p, q, tr);
526 out_rgb[1] = hue2rgb(p, q, tg);
527 out_rgb[2] = hue2rgb(p, q, tb);
528 out_rgb[3] = hsl[3];
529}

548/**
549 * Calculate the luminance of this color using the WCAG algorithm.
550 * See http://www.w3.org/TR/2008/REC-WCAG20-20081211/#relativeluminancedef
551 * @param {!Array<number>} rgba
552 * @return {number}
553 */
554WebInspector.Color.luminance = function(rgba)
555{
556 var rSRGB = rgba[0];
557 var gSRGB = rgba[1];
558 var bSRGB = rgba[2];
559
560 var r = rSRGB <= 0.03928 ? rSRGB / 12.92 : Math.pow(((rSRGB + 0.055) / 1.055), 2.4);
561 var g = gSRGB <= 0.03928 ? gSRGB / 12.92 : Math.pow(((gSRGB + 0.055) / 1.055), 2.4);
562 var b = bSRGB <= 0.03928 ? bSRGB / 12.92 : Math.pow(((bSRGB + 0.055) / 1.055), 2.4);
563
564 return 0.2126 * r + 0.7152 * g + 0.0722 * b;
565}

567/**
568 * Combine the two given color according to alpha blending.
569 * @param {!Array<number>} fgRGBA
570 * @param {!Array<number>} bgRGBA
571 * @param {!Array<number>} out_blended
572 */
573WebInspector.Color.blendColors = function(fgRGBA, bgRGBA, out_blended)
574{
575 var alpha = fgRGBA[3];
576
577 out_blended[0] = ((1 - alpha) * bgRGBA[0]) + (alpha * fgRGBA[0]);
578 out_blended[1] = ((1 - alpha) * bgRGBA[1]) + (alpha * fgRGBA[1]);
579 out_blended[2] = ((1 - alpha) * bgRGBA[2]) + (alpha * fgRGBA[2]);
580 out_blended[3] = alpha + (bgRGBA[3] * (1 - alpha));
581}

583/**
584 * Calculate the contrast ratio between a foreground and a background color.
585 * Returns the ratio to 1, for example for two two colors with a contrast ratio of 21:1, this function will return 21.
586 * See http://www.w3.org/TR/2008/REC-WCAG20-20081211/#contrast-ratiodef
587 * @param {!Array<number>} fgRGBA
588 * @param {!Array<number>} bgRGBA
589 * @return {number}
590 */
591WebInspector.Color.calculateContrastRatio = function(fgRGBA, bgRGBA)
592{
593 WebInspector.Color.blendColors(fgRGBA, bgRGBA, WebInspector.Color.calculateContrastRatio._blendedFg);
594
595 var fgLuminance = WebInspector.Color.luminance(WebInspector.Color.calculateContrastRatio._blendedFg);
596 var bgLuminance = WebInspector.Color.luminance(bgRGBA);
597 var contrastRatio = (Math.max(fgLuminance, bgLuminance) + 0.05) /
598 (Math.min(fgLuminance, bgLuminance) + 0.05);
599
600 for (var i = 0; i < WebInspector.Color.calculateContrastRatio._blendedFg.length; i++)
601 WebInspector.Color.calculateContrastRatio._blendedFg[i] = 0;
602
603 return contrastRatio;
604}
605
606WebInspector.Color.calculateContrastRatio._blendedFg = [0, 0, 0, 0];

608/**
609 * Compute a desired luminance given a given luminance and a desired contrast
610 * ratio.
611 * @param {number} luminance The given luminance.
612 * @param {number} contrast The desired contrast ratio.
613 * @param {boolean} lighter Whether the desired luminance is lighter or darker
614 * than the given luminance. If no luminance can be found which meets this
615 * requirement, a luminance which meets the inverse requirement will be
616 * returned.
617 * @return {number} The desired luminance.
618 */
619WebInspector.Color.desiredLuminance = function(luminance, contrast, lighter)
620{
621 function computeLuminance()
622 {
623 if (lighter)
624 return (luminance + 0.05) * contrast - 0.05;
625 else
626 return (luminance + 0.05) / contrast - 0.05;
627 }
628 var desiredLuminance = computeLuminance();
629 if (desiredLuminance < 0 || desiredLuminance > 1) {
630 lighter = !lighter;
631 desiredLuminance = computeLuminance();
632 }
633 return desiredLuminance;
634};

808WebInspector.Color.detectColorFormat = function(color)
809{
810 const cf = WebInspector.Color.Format;
811 var format;
812 var formatSetting = WebInspector.moduleSetting("colorFormat").get();
813 if (formatSetting === cf.Original)
814 format = cf.Original;
815 else if (formatSetting === cf.RGB)
816 format = (color.hasAlpha() ? cf.RGBA : cf.RGB);
817 else if (formatSetting === cf.HSL)
818 format = (color.hasAlpha() ? cf.HSLA : cf.HSL);
819 else if (!color.hasAlpha())
820 format = (color.canBeShortHex() ? cf.ShortHEX : cf.HEX);
821 else
822 format = cf.RGBA;
823
824 return format;
