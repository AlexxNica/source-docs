141 /**
142 * @param {!WebInspector.Script} script
143 * @param {?WebInspector.TextSourceMap} sourceMap
144 * @return {!Promise<undefined>}
145 */
146 sourceMapLoaded: function(script, sourceMap)
147 {
148 if (!sourceMap)
149 return Promise.resolve();
150 var previousScriptState = this._scriptPositions(script);
151 if (!previousScriptState)
152 return Promise.resolve();
153
154 var mappings = sourceMap.mappings().slice();
155 mappings.sort(mappingComparator);
156
157 if (!mappings.length) {
158 if (previousScriptState.length > 0)
159 return this._setScriptState(script, []);
160 return Promise.resolve();
161 }
162
163 var currentBlackboxed = false;
164 var isBlackboxed = false;
165 var positions = [];
166 // If content in script file begin is not mapped and one or more ranges are blackboxed then blackbox it.
167 if (mappings[0].lineNumber !== 0 || mappings[0].columnNumber !== 0) {
168 positions.push({ lineNumber: 0, columnNumber: 0});
169 currentBlackboxed = true;
170 }
171 for (var mapping of mappings) {
172 if (mapping.sourceURL && currentBlackboxed !== this.isBlackboxedURL(mapping.sourceURL)) {
173 positions.push({ lineNumber: mapping.lineNumber, columnNumber: mapping.columnNumber });
174 currentBlackboxed = !currentBlackboxed;
175 }
176 isBlackboxed = currentBlackboxed || isBlackboxed;
177 }
178 return this._setScriptState(script, !isBlackboxed ? [] : positions);

246 /**
247 * @param {string} url
248 */
249 _blackboxURL: function(url)
250 {
251 var regexPatterns = WebInspector.moduleSetting("skipStackFramesPattern").getAsArray();
252 var regexValue = this._urlToRegExpString(url);
253 if (!regexValue)
254 return;
255 var found = false;
256 for (var i = 0; i < regexPatterns.length; ++i) {
257 var item = regexPatterns[i];
258 if (item.pattern === regexValue) {
259 item.disabled = false;
260 found = true;
261 break;
262 }
263 }
264 if (!found)
265 regexPatterns.push({ pattern: regexValue });
266 WebInspector.moduleSetting("skipStackFramesPattern").setAsArray(regexPatterns);
267 },

269 /**
270 * @param {string} url
271 */
272 _unblackboxURL: function(url)
273 {
274 var regexPatterns = WebInspector.moduleSetting("skipStackFramesPattern").getAsArray();
275 var regexValue = WebInspector.blackboxManager._urlToRegExpString(url);
276 if (!regexValue)
277 return;
278 regexPatterns = regexPatterns.filter(function(item) {
279 return item.pattern !== regexValue;
280 });
281 for (var i = 0; i < regexPatterns.length; ++i) {
282 var item = regexPatterns[i];
283 if (item.disabled)
284 continue;
285 try {
286 var regex = new RegExp(item.pattern);
287 if (regex.test(url))
288 item.disabled = true;
289 } catch (e) {
290 }
291 }
292 WebInspector.moduleSetting("skipStackFramesPattern").setAsArray(regexPatterns);

296 {
297 this._isBlackboxedURLCache.clear();

299 var promises = [];
300 for (var debuggerModel of WebInspector.DebuggerModel.instances()) {
301 promises.push(this._setBlackboxPatterns.bind(this, debuggerModel));
302 for (var scriptId in debuggerModel.scripts) {

387 /**
388 * @param {!WebInspector.Script} script
389 * @param {!Array<!DebuggerAgent.ScriptPosition>} positions
390 * @return {!Promise<undefined>}
391 */
392 _setScriptState: function(script, positions)
393 {
394 var previousScriptState = this._scriptPositions(script);
395 if (previousScriptState) {
396 var hasChanged = false;
397 hasChanged = previousScriptState.length !== positions.length;
398 for (var i = 0; !hasChanged && i < positions.length; ++i)
399 hasChanged = positions[i].lineNumber !== previousScriptState[i].lineNumber || positions[i].columnNumber !== previousScriptState[i].columnNumber;
400 if (!hasChanged)
401 return Promise.resolve();
402 } else {
403 if (positions.length === 0)
404 return Promise.resolve().then(updateState.bind(this, false));
405 }
406
407 return script.setBlackboxedRanges(positions).then(updateState.bind(this));
408
409 /**
410 * @param {boolean} success
411 * @this {WebInspector.BlackboxManager}
412 */
413 function updateState(success)
414 {
415 if (success) {
416 this._setScriptPositions(script, positions);
417 this._debuggerWorkspaceBinding.updateLocations(script);
418 var isBlackboxed = positions.length !== 0;
419 if (!isBlackboxed && script.sourceMapURL)
420 this._debuggerWorkspaceBinding.maybeLoadSourceMap(script);
421 } else {
422 var hasPositions = !!this._scriptPositions(script);
423 if (!hasPositions)
424 this._setScriptPositions(script, []);
425 }

429 /**
430 * @param {string} url
431 * @return {string}
432 */
433 _urlToRegExpString: function(url)
434 {
435 var parsedURL = new WebInspector.ParsedURL(url);
436 if (parsedURL.isAboutBlank() || parsedURL.isDataURL())
437 return "";
438 if (!parsedURL.isValid)
439 return "^" + url.escapeForRegExp() + "$";
440 var name = parsedURL.lastPathComponent;
441 if (name)
442 name = "/" + name;
443 else if (parsedURL.folderPathComponents)
444 name = parsedURL.folderPathComponents + "/";
445 if (!name)
446 name = parsedURL.host;
447 if (!name)
448 return "";
449 var scheme = parsedURL.scheme;
450 var prefix = "";
451 if (scheme && scheme !== "http" && scheme !== "https") {
452 prefix = "^" + scheme + "://";
453 if (scheme === "chrome-extension")
454 prefix += parsedURL.host + "\\b";
455 prefix += ".*";
456 }
457 return prefix + name.escapeForRegExp() + (url.endsWith(name) ? "$" : "\\b");
458 }
