38WebInspector.WatchExpressionsSidebarPane = function()

40 WebInspector.ThrottledWidget.call(this);
41 this.registerRequiredCSS("components/objectValue.css");
42
43 /** @type {!Array.<!WebInspector.WatchExpression>} */
44 this._watchExpressions = [];

48 this._addButton.addEventListener("click", this._addButtonClicked.bind(this));

353 /**
354 * @param {?WebInspector.RemoteObject} result

357 _createWatchExpression: function(result, exceptionDetails)
358 {
359 this._result = result;
360
361 var headerElement = createElementWithClass("div", "watch-expression-header");
362 var deleteButton = headerElement.createChild("button", "watch-expression-delete-button");
363 deleteButton.title = WebInspector.UIString("Delete watch expression");
364 deleteButton.addEventListener("click", this._deleteWatchExpression.bind(this), false);
365
366 var titleElement = headerElement.createChild("div", "watch-expression-title");
367 this._nameElement = WebInspector.ObjectPropertiesSection.createNameElement(this._expression);
368 if (!!exceptionDetails || !result) {
369 this._valueElement = createElementWithClass("span", "error-message value");
370 titleElement.classList.add("dimmed");
371 this._valueElement.textContent = WebInspector.UIString("<not available>");
372 } else {
373 this._valueElement = WebInspector.ObjectPropertiesSection.createValueElementWithCustomSupport(result, !!exceptionDetails, titleElement, this._linkifier);
374 }
375 var separatorElement = createElementWithClass("span", "watch-expressions-separator");
376 separatorElement.textContent = ": ";
377 titleElement.appendChildren(this._nameElement, separatorElement, this._valueElement);
378
379 this._element.removeChildren();
380 this._objectPropertiesSection = null;
381 if (!exceptionDetails && result && result.hasChildren && !result.customPreview()) {
382 headerElement.classList.add("watch-expression-object-header");
383 this._objectPropertiesSection = new WebInspector.ObjectPropertiesSection(result, headerElement, this._linkifier);
384 this._objectPresentationElement = this._objectPropertiesSection.element;
385 this._expandController.watchSection(/** @type {string} */ (this._expression), this._objectPropertiesSection);
386 var objectTreeElement = this._objectPropertiesSection.objectTreeElement();
387 objectTreeElement.toggleOnClick = false;
388 objectTreeElement.listItemElement.addEventListener("click", this._onSectionClick.bind(this), false);
389 objectTreeElement.listItemElement.addEventListener("dblclick", this._dblClickOnWatchExpression.bind(this));
390 } else {
391 this._objectPresentationElement = headerElement;
392 this._objectPresentationElement.addEventListener("dblclick", this._dblClickOnWatchExpression.bind(this));
393 }
394
395 this._element.appendChild(this._objectPresentationElement);
