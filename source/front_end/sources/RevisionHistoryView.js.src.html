46 /**

51 {

56 WebInspector.workspace.uiSourceCodes().forEach(populateRevisions.bind(this));

76 _createUISourceCodeItem: function(uiSourceCode)
77 {
78 var uiSourceCodeItem = new TreeElement(uiSourceCode.displayName(), true);
79 uiSourceCodeItem.selectable = false;
80
81 // Insert in sorted order
82 var rootElement = this._treeOutline.rootElement();
83 for (var i = 0; i < rootElement.childCount(); ++i) {
84 if (rootElement.childAt(i).title.localeCompare(uiSourceCode.displayName()) > 0) {
85 rootElement.insertChild(uiSourceCodeItem, i);
86 break;
87 }
88 }
89 if (i === rootElement.childCount())
90 rootElement.appendChild(uiSourceCodeItem);
91
92 this._uiSourceCodeItems.set(uiSourceCode, uiSourceCodeItem);
93
94 var revisionCount = uiSourceCode.history.length;
95 for (var i = revisionCount - 1; i >= 0; --i) {
96 var revision = uiSourceCode.history[i];
97 var historyItem = new WebInspector.RevisionHistoryTreeElement(revision, uiSourceCode.history[i - 1], i !== revisionCount - 1);
98 uiSourceCodeItem.appendChild(historyItem);
99 }

101 var linkItem = new TreeElement();

103 uiSourceCodeItem.appendChild(linkItem);

105 var revertToOriginal = linkItem.listItemElement.createChild("span", "revision-history-link revision-history-link-row");
106 revertToOriginal.textContent = WebInspector.UIString("apply original content");
107 revertToOriginal.addEventListener("click", this._revertToOriginal.bind(this, uiSourceCode));
108
109 var clearHistoryElement = uiSourceCodeItem.listItemElement.createChild("span", "revision-history-link");
110 clearHistoryElement.textContent = WebInspector.UIString("revert");
111 clearHistoryElement.addEventListener("click", this._clearHistory.bind(this, uiSourceCode));

225 /**
226 * @param {?string} baseContent
227 * @param {?string} newContent
228 * @this {WebInspector.RevisionHistoryTreeElement}
229 */
230 function diff(baseContent, newContent)
231 {
232 var baseLines = baseContent.split("\n");
233 var newLines = newContent.split("\n");
234 var opcodes = WebInspector.Diff.lineDiff(baseLines, newLines);
235 var lastWasSeparator = false;
236
237 var baseLineNumber = 0;
238 var newLineNumber = 0;
239 for (var idx = 0; idx < opcodes.length; idx++) {
240 var code = opcodes[idx][0];
241 var rowCount = opcodes[idx][1].length;
242 if (code === WebInspector.Diff.Operation.Equal) {
243 baseLineNumber += rowCount;
244 newLineNumber += rowCount;
245 if (!lastWasSeparator)
246 this._createLine(null, null, " \u2026", "separator");
247 lastWasSeparator = true;
248 } else if (code === WebInspector.Diff.Operation.Delete) {
249 lastWasSeparator = false;
250 for (var i = 0; i < rowCount; ++i)
251 this._createLine(baseLineNumber + i, null, baseLines[baseLineNumber + i], "removed");
252 baseLineNumber += rowCount;
253 } else if (code === WebInspector.Diff.Operation.Insert) {
254 lastWasSeparator = false;
255 for (var i = 0; i < rowCount; ++i)
256 this._createLine(null, newLineNumber + i, newLines[newLineNumber + i], "added");
257 newLineNumber += rowCount;
258 }
259 }
260 }
