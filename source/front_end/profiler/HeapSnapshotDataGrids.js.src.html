210 sortingChanged: function()
211 {
212 var sortAscending = this.isSortOrderAscending();
213 var sortColumnIdentifier = this.sortColumnIdentifier();
214 if (this._lastSortColumnIdentifier === sortColumnIdentifier && this._lastSortAscending === sortAscending)
215 return;
216 this._lastSortColumnIdentifier = sortColumnIdentifier;
217 this._lastSortAscending = sortAscending;
218 var sortFields = this._sortFields(sortColumnIdentifier, sortAscending);
219
220 function SortByTwoFields(nodeA, nodeB)
221 {
222 var field1 = nodeA[sortFields[0]];
223 var field2 = nodeB[sortFields[0]];
224 var result = field1 < field2 ? -1 : (field1 > field2 ? 1 : 0);
225 if (!sortFields[1])
226 result = -result;
227 if (result !== 0)
228 return result;
229 field1 = nodeA[sortFields[2]];
230 field2 = nodeB[sortFields[2]];
231 result = field1 < field2 ? -1 : (field1 > field2 ? 1 : 0);
232 if (!sortFields[3])
233 result = -result;
234 return result;
235 }
236 this._performSorting(SortByTwoFields);
237 },

245 for (var i = 0, l = children.length; i < l; ++i) {
246 var child = children[i];
247 this.appendChildAfterSorting(child);
248 if (child.expanded)
249 child.sort();
250 }
251 this.recursiveSortingLeave();

352 /**
353 * @override
354 * @param {boolean} force
355 */
356 updateVisibleNodes: function(force)
357 {
358 // Guard zone is used to ensure there are always some extra items
359 // above and below the viewport to support keyboard navigation.
360 var guardZoneHeight = 40;
361 var scrollHeight = this.scrollContainer.scrollHeight;
362 var scrollTop = this.scrollContainer.scrollTop;
363 var scrollBottom = scrollHeight - scrollTop - this.scrollContainer.offsetHeight;
364 scrollTop = Math.max(0, scrollTop - guardZoneHeight);
365 scrollBottom = Math.max(0, scrollBottom - guardZoneHeight);
366 var viewPortHeight = scrollHeight - scrollTop - scrollBottom;
367 // Do nothing if populated nodes still fit the viewport.
368 if (!force && scrollTop >= this._topPaddingHeight && scrollBottom >= this._bottomPaddingHeight)
369 return;
370 var hysteresisHeight = 500;
371 scrollTop -= hysteresisHeight;
372 viewPortHeight += 2 * hysteresisHeight;
373 var selectedNode = this.selectedNode;
374 this.rootNode().removeChildren();
375
376 this._topPaddingHeight = 0;
377 this._bottomPaddingHeight = 0;
378
379 this._addVisibleNodes(this.rootNode(), scrollTop, scrollTop + viewPortHeight);
380
381 this.setVerticalPadding(this._topPaddingHeight, this._bottomPaddingHeight);
382
383 if (selectedNode) {
384 // Keep selection even if the node is not in the current viewport.
385 if (selectedNode.parent)
386 selectedNode.select(true);
387 else
388 this.selectedNode = selectedNode;
389 }

392 /**
393 * @param {!WebInspector.DataGridNode} parentNode
394 * @param {number} topBound
395 * @param {number} bottomBound
396 * @return {number}
397 */
398 _addVisibleNodes: function(parentNode, topBound, bottomBound)
399 {
400 if (!parentNode.expanded)
401 return 0;
402
403 var children = this.allChildren(parentNode);
404 var topPadding = 0;
405 var nameFilterValue = this._nameFilter ? this._nameFilter.value().toLowerCase() : "";
406 // Iterate over invisible nodes beyond the upper bound of viewport.
407 // Do not insert them into the grid, but count their total height.
408 for (var i = 0; i < children.length; ++i) {
409 var child = children[i];
410 if (nameFilterValue && child.filteredOut && child.filteredOut(nameFilterValue))
411 continue;
412 var newTop = topPadding + this._nodeHeight(child);
413 if (newTop > topBound)
414 break;
415 topPadding = newTop;
416 }
417
418 // Put visible nodes into the data grid.
419 var position = topPadding;
420 for (; i < children.length && position < bottomBound; ++i) {
421 var child = children[i];
422 if (nameFilterValue && child.filteredOut && child.filteredOut(nameFilterValue))
423 continue;
424 var hasChildren = child.hasChildren;
425 child.removeChildren();
426 child.hasChildren = hasChildren;
427 child.revealed = true;
428 parentNode.appendChild(child);
429 position += child.nodeSelfHeight();
430 position += this._addVisibleNodes(child, topBound - position, bottomBound - position);
431 }
432
433 // Count the invisible nodes beyond the bottom bound of the viewport.
434 var bottomPadding = 0;
435 for (; i < children.length; ++i) {
436 var child = children[i];
437 if (nameFilterValue && child.filteredOut && child.filteredOut(nameFilterValue))
438 continue;
439 bottomPadding += this._nodeHeight(child);
440 }
441
442 this._topPaddingHeight += topPadding;
443 this._bottomPaddingHeight += bottomPadding;
444 return position + bottomPadding;
445 },
446
447 /**
448 * @param {!WebInspector.HeapSnapshotGridNode} node
449 * @return {number}
450 */
451 _nodeHeight: function(node)
452 {
453 if (!node.revealed)
454 return 0;
455 var result = node.nodeSelfHeight();
456 if (!node.expanded)
457 return result;
458 var children = this.allChildren(node);
459 for (var i = 0; i < children.length; i++)
460 result += this._nodeHeight(children[i]);
461 return result;

465 * @param {!Array.<!WebInspector.HeapSnapshotGridNode>} pathToReveal
466 * @return {!Promise.<!WebInspector.HeapSnapshotGridNode>}

468 revealTreeNode: function(pathToReveal)
469 {
470 var height = this._calculateOffset(pathToReveal);

492 /**
493 * @param {!Array.<!WebInspector.HeapSnapshotGridNode>} pathToReveal
494 * @return {number}
495 */
496 _calculateOffset: function(pathToReveal)
497 {
498 var parentNode = this.rootNode();
499 var height = 0;
500 for (var i = 0; i < pathToReveal.length; ++i) {
501 var node = pathToReveal[i];
502 var children = this.allChildren(parentNode);
503 for (var j = 0; j < children.length; ++j) {
504 var child = children[j];
505 if (node === child) {
506 height += node.nodeSelfHeight();
507 break;
508 }
509 height += this._nodeHeight(child);
510 }
511 parentNode = node;
512 }
513 return height - pathToReveal.peekLast().nodeSelfHeight();

753 /**
754 * @param {?string} className
755 * @return {?Promise<?WebInspector.HeapSnapshotGridNode>}
756 * @this {WebInspector.HeapSnapshotConstructorsDataGrid}
757 */
758 function didGetClassName(className)
759 {
760 if (!className)
761 return null;
762 var constructorNodes = this.topLevelNodes();
763 for (var i = 0; i < constructorNodes.length; i++) {
