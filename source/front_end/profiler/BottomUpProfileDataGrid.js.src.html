26// Bottom Up Profiling shows the entire callstack backwards:
27// The root node is a representation of each individual function called, and each child of that node represents
28// a reverse-callstack showing how many of those calls came from it. So, unlike top-down, the statistics in
29// each child still represent the root node. We have to be particularly careful of recursion with this mode
30// because a root node can represent itself AND an ancestor.
31

129/**
130 * @param {!WebInspector.BottomUpProfileDataGridNode|!WebInspector.BottomUpProfileDataGridTree} container
131 */
132WebInspector.BottomUpProfileDataGridNode._sharedPopulate = function(container)
133{
134 var remainingNodeInfos = container._remainingNodeInfos;
135 var count = remainingNodeInfos.length;
136
137 for (var index = 0; index < count; ++index) {
138 var nodeInfo = remainingNodeInfos[index];
139 var ancestor = nodeInfo.ancestor;
140 var focusNode = nodeInfo.focusNode;
141 var child = container.findChild(ancestor);
142
143 // If we already have this child, then merge the data together.
144 if (child) {
145 var totalAccountedFor = nodeInfo.totalAccountedFor;
146
147 child.self += focusNode.self;
148
149 if (!totalAccountedFor)
150 child.total += focusNode.total;
151 } else {
152 // If not, add it as a true ancestor.
153 // In heavy mode, we take our visual identity from ancestor node...
154 child = new WebInspector.BottomUpProfileDataGridNode(ancestor, /** @type {!WebInspector.TopDownProfileDataGridTree} */ (container.tree));
155
156 if (ancestor !== focusNode) {
157 // But the actual statistics from the "root" node (bottom of the callstack).
158 child.self = focusNode.self;
159 child.total = focusNode.total;
160 }
161
162 container.appendChild(child);
163 }
164
165 var parent = ancestor.parent;
166 if (parent && parent.parent) {
167 nodeInfo.ancestor = parent;
168 child._remainingNodeInfos.push(nodeInfo);
169 }
170 }
171
172 delete container._remainingNodeInfos;
173}

175/**
176 * @constructor
177 * @extends {WebInspector.ProfileDataGridTree}
178 * @param {!WebInspector.ProfileDataGridNode.Formatter} formatter
179 * @param {!WebInspector.SearchableView} searchableView
180 * @param {!WebInspector.ProfileNode} rootProfileNode
181 * @param {number} total
182 */
183WebInspector.BottomUpProfileDataGridTree = function(formatter, searchableView, rootProfileNode, total)
184{
185 WebInspector.ProfileDataGridTree.call(this, formatter, searchableView, total);
186
187 // Iterate each node in pre-order.
188 var profileNodeUIDs = 0;
189 var profileNodeGroups = [[], [rootProfileNode]];
190 /** @type {!Map<string, !Set<number>>} */
191 var visitedProfileNodesForCallUID = new Map();
192
193 this._remainingNodeInfos = [];
194
195 for (var profileNodeGroupIndex = 0; profileNodeGroupIndex < profileNodeGroups.length; ++profileNodeGroupIndex) {
196 var parentProfileNodes = profileNodeGroups[profileNodeGroupIndex];
197 var profileNodes = profileNodeGroups[++profileNodeGroupIndex];
198 var count = profileNodes.length;
199
200 for (var index = 0; index < count; ++index) {
201 var profileNode = profileNodes[index];
202
203 if (!profileNode.UID)
204 profileNode.UID = ++profileNodeUIDs;
205
206 if (profileNode.parent) {
207 // The total time of this ancestor is accounted for if we're in any form of recursive cycle.
208 var visitedNodes = visitedProfileNodesForCallUID.get(profileNode.callUID);
209 var totalAccountedFor = false;
210
211 if (!visitedNodes) {
212 visitedNodes = new Set();
213 visitedProfileNodesForCallUID.set(profileNode.callUID, visitedNodes);
214 } else {
215 // The total time for this node has already been accounted for iff one of it's parents has already been visited.
216 // We can do this check in this style because we are traversing the tree in pre-order.
217 var parentCount = parentProfileNodes.length;
218 for (var parentIndex = 0; parentIndex < parentCount; ++parentIndex) {
219 if (visitedNodes.has(parentProfileNodes[parentIndex].UID)) {
220 totalAccountedFor = true;
221 break;
222 }
223 }
224 }
225
226 visitedNodes.add(profileNode.UID);
227
228 this._remainingNodeInfos.push({ ancestor: profileNode, focusNode: profileNode, totalAccountedFor: totalAccountedFor });
229 }
230
231 var children = profileNode.children;
232 if (children.length) {
233 profileNodeGroups.push(parentProfileNodes.concat([profileNode]));
234 profileNodeGroups.push(children);
235 }
236 }
237 }
238
239 // Populate the top level nodes.
240 WebInspector.ProfileDataGridNode.populate(this);
241
242 return this;
243}

246 /**
247 * When focusing, we keep the entire callstack up to this ancestor.
248 * @param {!WebInspector.ProfileDataGridNode} profileDataGridNode
249 */
250 focus: function(profileDataGridNode)
251 {
252 if (!profileDataGridNode)
253 return;
254
255 this.save();
256
257 var currentNode = profileDataGridNode;
258 var focusNode = profileDataGridNode;
259
260 while (currentNode.parent && (currentNode instanceof WebInspector.ProfileDataGridNode)) {
261 currentNode._takePropertiesFromProfileDataGridNode(profileDataGridNode);
262
263 focusNode = currentNode;
264 currentNode = currentNode.parent;
265
266 if (currentNode instanceof WebInspector.ProfileDataGridNode)
267 currentNode._keepOnlyChild(focusNode);
268 }
269
270 this.children = [focusNode];
271 this.total = profileDataGridNode.total;
272 },

274 /**
275 * @param {!WebInspector.ProfileDataGridNode} profileDataGridNode
276 */
277 exclude: function(profileDataGridNode)
278 {
279 if (!profileDataGridNode)
280 return;
281
282 this.save();
283
284 var excludedCallUID = profileDataGridNode.callUID;
285 var excludedTopLevelChild = this.childrenByCallUID.get(excludedCallUID);
286
287 // If we have a top level node that is excluded, get rid of it completely (not keeping children),
288 // since bottom up data relies entirely on the root node.
289 if (excludedTopLevelChild)
290 this.children.remove(excludedTopLevelChild);
291
292 var children = this.children;
293 var count = children.length;
294
295 for (var index = 0; index < count; ++index)
296 children[index]._exclude(excludedCallUID);
297
298 if (this.lastComparator)
299 this.sort(this.lastComparator, true);

302 /**
303 * @override
304 * @param {!WebInspector.SearchableView.SearchConfig} searchConfig
305 * @param {boolean} shouldJump
306 * @param {boolean=} jumpBackwards
307 */
308 performSearch: function(searchConfig, shouldJump, jumpBackwards)
309 {
310 this.searchCanceled();
311 var matchesQuery = this._matchFunction(searchConfig);
312 if (!matchesQuery)
313 return;
314
315 this._searchResults = [];
316 for (var current = this.children[0]; current; current = current.traverseNextNode(true, null, true)) {
317 if (matchesQuery(current))
318 this._searchResults.push({ profileNode: current });
319 }
320 this._searchResultIndex = jumpBackwards ? 0 : this._searchResults.length - 1;
321 this._searchableView.updateSearchMatchesCount(this._searchResults.length);
322 this._searchableView.updateCurrentMatchIndex(this._searchResultIndex);
323 },
