110 /**
111 * @param {function(!T, !T)} comparator
112 * @param {boolean} force
113 * @template T
114 */
115 sort: function(comparator, force)
116 {
117 var gridNodeGroups = [[this]];
118
119 for (var gridNodeGroupIndex = 0; gridNodeGroupIndex < gridNodeGroups.length; ++gridNodeGroupIndex) {
120 var gridNodes = gridNodeGroups[gridNodeGroupIndex];
121 var count = gridNodes.length;
122
123 for (var index = 0; index < count; ++index) {
124 var gridNode = gridNodes[index];
125
126 // If the grid node is collapsed, then don't sort children (save operation for later).
127 // If the grid node has the same sorting as previously, then there is no point in sorting it again.
128 if (!force && (!gridNode.expanded || gridNode.lastComparator === comparator)) {
129 if (gridNode.children.length)
130 gridNode.shouldRefreshChildren = true;
131 continue;
132 }
133
134 gridNode.lastComparator = comparator;
135
136 var children = gridNode.children;
137 var childCount = children.length;
138
139 if (childCount) {
140 children.sort(comparator);
141
142 for (var childIndex = 0; childIndex < childCount; ++childIndex)
143 children[childIndex].recalculateSiblings(childIndex);
144
145 gridNodeGroups.push(children);
146 }
147 }
148 }

267 * @param {boolean} shouldAbsorb
268 */
269WebInspector.ProfileDataGridNode.merge = function(container, child, shouldAbsorb)
270{
271 container.self += child.self;
272
273 if (!shouldAbsorb)
274 container.total += child.total;
275
276 var children = container.children.slice();
277
278 container.removeChildren();
279
280 var count = children.length;
281
282 for (var index = 0; index < count; ++index) {
283 if (!shouldAbsorb || children[index] !== child)
284 container.appendChild(children[index]);
285 }
286
287 children = child.children.slice();
288 count = children.length;
289
290 for (var index = 0; index < count; ++index) {
291 var orphanedChild = children[index];
292 var existingChild = container.childrenByCallUID.get(orphanedChild.callUID);
293
294 if (existingChild)
295 existingChild.merge(/** @type{!WebInspector.ProfileDataGridNode} */(orphanedChild), false);
296 else
297 container.appendChild(orphanedChild);
298 }

395 /**
396 * @param {!WebInspector.SearchableView.SearchConfig} searchConfig
397 * @return {?function(!WebInspector.ProfileDataGridNode):boolean}
398 */
399 _matchFunction: function(searchConfig)
400 {
401 var query = searchConfig.query.trim();
402 if (!query.length)
403 return null;
404
405 var greaterThan = (query.startsWith(">"));
406 var lessThan = (query.startsWith("<"));
407 var equalTo = (query.startsWith("=") || ((greaterThan || lessThan) && query.indexOf("=") === 1));
408 var percentUnits = (query.endsWith("%"));
409 var millisecondsUnits = (query.length > 2 && query.endsWith("ms"));
410 var secondsUnits = (!millisecondsUnits && query.endsWith("s"));
411
412 var queryNumber = parseFloat(query);
413 if (greaterThan || lessThan || equalTo) {
414 if (equalTo && (greaterThan || lessThan))
415 queryNumber = parseFloat(query.substring(2));
416 else
417 queryNumber = parseFloat(query.substring(1));
418 }
419
420 var queryNumberMilliseconds = (secondsUnits ? (queryNumber * 1000) : queryNumber);
421
422 // Make equalTo implicitly true if it wasn't specified there is no other operator.
423 if (!isNaN(queryNumber) && !(greaterThan || lessThan))
424 equalTo = true;
425
426 var matcher = createPlainTextSearchRegex(query, "i");
427
428 /**
429 * @param {!WebInspector.ProfileDataGridNode} profileDataGridNode
430 * @return {boolean}
431 */
432 function matchesQuery(profileDataGridNode)
433 {
434 delete profileDataGridNode._searchMatchedSelfColumn;
435 delete profileDataGridNode._searchMatchedTotalColumn;
436 delete profileDataGridNode._searchMatchedFunctionColumn;
437
438 if (percentUnits) {
439 if (lessThan) {
440 if (profileDataGridNode.selfPercent < queryNumber)
441 profileDataGridNode._searchMatchedSelfColumn = true;
442 if (profileDataGridNode.totalPercent < queryNumber)
443 profileDataGridNode._searchMatchedTotalColumn = true;
444 } else if (greaterThan) {
445 if (profileDataGridNode.selfPercent > queryNumber)
446 profileDataGridNode._searchMatchedSelfColumn = true;
447 if (profileDataGridNode.totalPercent > queryNumber)
448 profileDataGridNode._searchMatchedTotalColumn = true;
449 }
450
451 if (equalTo) {
452 if (profileDataGridNode.selfPercent === queryNumber)
453 profileDataGridNode._searchMatchedSelfColumn = true;
454 if (profileDataGridNode.totalPercent === queryNumber)
455 profileDataGridNode._searchMatchedTotalColumn = true;
456 }
457 } else if (millisecondsUnits || secondsUnits) {
458 if (lessThan) {
459 if (profileDataGridNode.self < queryNumberMilliseconds)
460 profileDataGridNode._searchMatchedSelfColumn = true;
461 if (profileDataGridNode.total < queryNumberMilliseconds)
462 profileDataGridNode._searchMatchedTotalColumn = true;
463 } else if (greaterThan) {
464 if (profileDataGridNode.self > queryNumberMilliseconds)
465 profileDataGridNode._searchMatchedSelfColumn = true;
466 if (profileDataGridNode.total > queryNumberMilliseconds)
467 profileDataGridNode._searchMatchedTotalColumn = true;
468 }
469
470 if (equalTo) {
471 if (profileDataGridNode.self === queryNumberMilliseconds)
472 profileDataGridNode._searchMatchedSelfColumn = true;
473 if (profileDataGridNode.total === queryNumberMilliseconds)
474 profileDataGridNode._searchMatchedTotalColumn = true;
475 }
476 }
477
478 if (profileDataGridNode.functionName.match(matcher) || (profileDataGridNode.url && profileDataGridNode.url.match(matcher)))
479 profileDataGridNode._searchMatchedFunctionColumn = true;
480
481 if (profileDataGridNode._searchMatchedSelfColumn ||
482 profileDataGridNode._searchMatchedTotalColumn ||
483 profileDataGridNode._searchMatchedFunctionColumn) {
484 profileDataGridNode.refresh();
485 return true;
486 }
487
488 return false;
489 }
490 return matchesQuery;

495 * @param {!WebInspector.SearchableView.SearchConfig} searchConfig
496 * @param {boolean} shouldJump
497 * @param {boolean=} jumpBackwards
498 */
499 performSearch: function(searchConfig, shouldJump, jumpBackwards)
500 {
501 this.searchCanceled();
502 var matchesQuery = this._matchFunction(searchConfig);
503 if (!matchesQuery)
504 return;
505
506 this._searchResults = [];
507 for (var current = this.children[0]; current; current = current.traverseNextNode(false, null, false)) {
508 if (matchesQuery(current))
509 this._searchResults.push({ profileNode: current });
510 }
511 this._searchResultIndex = jumpBackwards ? 0 : this._searchResults.length - 1;
512 this._searchableView.updateSearchMatchesCount(this._searchResults.length);
513 this._searchableView.updateCurrentMatchIndex(this._searchResultIndex);
