308 /**
309 * @param {!Event} event
310 */
311 _simulateTouchForMouseEvent: function(event)
312 {
313 const buttons = {0: "none", 1: "left", 2: "middle", 3: "right"};
314 const types = {"mousedown" : "mousePressed", "mouseup": "mouseReleased", "mousemove": "mouseMoved", "mousewheel": "mouseWheel"};
315 if (!(event.type in types) || !(event.which in buttons))
316 return;
317 if (event.type !== "mousewheel" && buttons[event.which] === "none")
318 return;
319
320 if (event.type === "mousedown" || typeof this._eventScreenOffsetTop === "undefined")
321 this._eventScreenOffsetTop = this._screenOffsetTop;
322
323 var modifiers = (event.altKey ? 1 : 0) | (event.ctrlKey ? 2 : 0) | (event.metaKey ? 4 : 0) | (event.shiftKey ? 8 : 0);
324
325 var convertedPosition = this._zoomIntoScreenSpace(event);
326 convertedPosition.y = Math.round(convertedPosition.y - this._eventScreenOffsetTop);
327 var params = {type: types[event.type], x: convertedPosition.x, y: convertedPosition.y, modifiers: modifiers, timestamp: event.timeStamp / 1000, button: buttons[event.which], clickCount: 0};
328 if (event.type === "mousewheel") {
329 params.deltaX = event.wheelDeltaX / this._screenZoom;
330 params.deltaY = event.wheelDeltaY / this._screenZoom;
331 } else {
332 this._eventParams = params;
333 }
334 if (event.type === "mouseup")
335 delete this._eventScreenOffsetTop;
336 this._target.inputAgent().invoke_emulateTouchFromMouseEvent(params);
337 },

349 }

449 /**
450 * @param {!DOMAgent.Quad} quad
451 * @this {WebInspector.ScreencastView}
452 */
453 function scaleQuad(quad)
454 {
455 for (var i = 0; i < quad.length; i += 2) {
456 quad[i] = quad[i] * this._screenZoom;
457 quad[i + 1] = (quad[i + 1] + this._screenOffsetTop) * this._screenZoom;
458 }
459 }
460
461 scaleQuad.call(this, model.content);
462 scaleQuad.call(this, model.padding);
463 scaleQuad.call(this, model.border);
464 scaleQuad.call(this, model.margin);

468 _repaint: function()
469 {
470 var model = this._model;
471 var config = this._config;
472
473 var canvasWidth = this._canvasElement.getBoundingClientRect().width;
474 var canvasHeight = this._canvasElement.getBoundingClientRect().height;
475 this._canvasElement.width = window.devicePixelRatio * canvasWidth;
476 this._canvasElement.height = window.devicePixelRatio * canvasHeight;
477
478 this._context.save();
479 this._context.scale(window.devicePixelRatio, window.devicePixelRatio);
480
481 // Paint top and bottom gutter.
482 this._context.save();
483 this._context.fillStyle = this._checkerboardPattern;
484 this._context.fillRect(0, 0, canvasWidth, this._screenOffsetTop * this._screenZoom);
485 this._context.fillRect(0, this._screenOffsetTop * this._screenZoom + this._imageElement.naturalHeight * this._imageZoom, canvasWidth, canvasHeight);
486 this._context.restore();
487
488 if (model && config) {
489 this._context.save();
490 const transparentColor = "rgba(0, 0, 0, 0)";
491 var quads = [];
492 if (model.content && config.contentColor !== transparentColor)
493 quads.push({quad: model.content, color: config.contentColor});
494 if (model.padding && config.paddingColor !== transparentColor)
495 quads.push({quad: model.padding, color: config.paddingColor});
496 if (model.border && config.borderColor !== transparentColor)
497 quads.push({quad: model.border, color: config.borderColor});
498 if (model.margin && config.marginColor !== transparentColor)
499 quads.push({quad: model.margin, color: config.marginColor});
500
501 for (var i = quads.length - 1; i > 0; --i)
502 this._drawOutlinedQuadWithClip(quads[i].quad, quads[i - 1].quad, quads[i].color);
503 if (quads.length > 0)
504 this._drawOutlinedQuad(quads[0].quad, quads[0].color);
505 this._context.restore();
506
507 this._drawElementTitle();
508
509 this._context.globalCompositeOperation = "destination-over";
510 }
511
512 this._context.drawImage(this._imageElement, 0, this._screenOffsetTop * this._screenZoom, this._imageElement.naturalWidth * this._imageZoom, this._imageElement.naturalHeight * this._imageZoom);
513 this._context.restore();
514

574 _drawElementTitle: function()
575 {
576 if (!this._node)
577 return;
578
579 var canvasWidth = this._canvasElement.getBoundingClientRect().width;
580 var canvasHeight = this._canvasElement.getBoundingClientRect().height;
581
582 var lowerCaseName = this._node.localName() || this._node.nodeName().toLowerCase();
583 this._tagNameElement.textContent = lowerCaseName;
584 this._nodeIdElement.textContent = this._node.getAttribute("id") ? "#" + this._node.getAttribute("id") : "";
585 this._nodeIdElement.textContent = this._node.getAttribute("id") ? "#" + this._node.getAttribute("id") : "";
586 var className = this._node.getAttribute("class");
587 if (className && className.length > 50)
588 className = className.substring(0, 50) + "\u2026";
589 this._classNameElement.textContent = className || "";
590 this._nodeWidthElement.textContent = this._model.width;
591 this._nodeHeightElement.textContent = this._model.height;
592
593 this._titleElement.classList.remove("hidden");
594 var titleWidth = this._titleElement.offsetWidth + 6;
595 var titleHeight = this._titleElement.offsetHeight + 4;
596
597 var anchorTop = this._model.margin[1];
598 var anchorBottom = this._model.margin[7];
599
600 const arrowHeight = 7;
601 var renderArrowUp = false;
602 var renderArrowDown = false;
603
604 var boxX = Math.max(2, this._model.margin[0]);
605 if (boxX + titleWidth > canvasWidth)
606 boxX = canvasWidth - titleWidth - 2;
607
608 var boxY;
609 if (anchorTop > canvasHeight) {
610 boxY = canvasHeight - titleHeight - arrowHeight;
611 renderArrowDown = true;
612 } else if (anchorBottom < 0) {
613 boxY = arrowHeight;
614 renderArrowUp = true;
615 } else if (anchorBottom + titleHeight + arrowHeight < canvasHeight) {
616 boxY = anchorBottom + arrowHeight - 4;
617 renderArrowUp = true;
618 } else if (anchorTop - titleHeight - arrowHeight > 0) {
619 boxY = anchorTop - titleHeight - arrowHeight + 3;
620 renderArrowDown = true;
621 } else
622 boxY = arrowHeight;
623
624 this._context.save();
625 this._context.translate(0.5, 0.5);
626 this._context.beginPath();
627 this._context.moveTo(boxX, boxY);
628 if (renderArrowUp) {
629 this._context.lineTo(boxX + 2 * arrowHeight, boxY);
630 this._context.lineTo(boxX + 3 * arrowHeight, boxY - arrowHeight);
631 this._context.lineTo(boxX + 4 * arrowHeight, boxY);
632 }
633 this._context.lineTo(boxX + titleWidth, boxY);
634 this._context.lineTo(boxX + titleWidth, boxY + titleHeight);
635 if (renderArrowDown) {
636 this._context.lineTo(boxX + 4 * arrowHeight, boxY + titleHeight);
637 this._context.lineTo(boxX + 3 * arrowHeight, boxY + titleHeight + arrowHeight);
638 this._context.lineTo(boxX + 2 * arrowHeight, boxY + titleHeight);
639 }
640 this._context.lineTo(boxX, boxY + titleHeight);
641 this._context.closePath();
642 this._context.fillStyle = "rgb(255, 255, 194)";
643 this._context.fill();
644 this._context.strokeStyle = "rgb(128, 128, 128)";
645 this._context.stroke();
646
647 this._context.restore();
648
649 this._titleElement.style.top = (boxY + 3) + "px";
650 this._titleElement.style.left = (boxX + 3) + "px";
