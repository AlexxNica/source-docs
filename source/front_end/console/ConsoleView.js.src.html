438 _addConsoleMessage: function(message)
439 {
440 /**
441 * @param {!WebInspector.ConsoleViewMessage} viewMessage1
442 * @param {!WebInspector.ConsoleViewMessage} viewMessage2
443 * @return {number}
444 */
445 function compareTimestamps(viewMessage1, viewMessage2)
446 {
447 return WebInspector.ConsoleMessage.timestampComparator(viewMessage1.consoleMessage(), viewMessage2.consoleMessage());
448 }
449
450 if (message.type === WebInspector.ConsoleMessage.MessageType.Command || message.type === WebInspector.ConsoleMessage.MessageType.Result)
451 message.timestamp = this._consoleMessages.length ? this._consoleMessages.peekLast().consoleMessage().timestamp : 0;
452 var viewMessage = this._createViewMessage(message);
453 message[this._viewMessageSymbol] = viewMessage;
454 var insertAt = this._consoleMessages.upperBound(viewMessage, compareTimestamps)
455 var insertedInMiddle = insertAt < this._consoleMessages.length;
456 this._consoleMessages.splice(insertAt, 0, viewMessage);
457
458 if (this._urlToMessageCount[message.url])
459 ++this._urlToMessageCount[message.url];
460 else
461 this._urlToMessageCount[message.url] = 1;
462
463 if (!insertedInMiddle) {
464 this._appendMessageToEnd(viewMessage);
465 this._updateFilterStatus();
466 this._searchableView.updateSearchMatchesCount(this._regexMatchRanges.length);
467 } else {
468 this._needsFullUpdate = true;
469 }
470
471 this._scheduleViewportRefresh();
472 this._consoleMessageAddedForTest(viewMessage);

493 /**
494 * @param {!WebInspector.ConsoleViewMessage} viewMessage
495 */
496 _appendMessageToEnd: function(viewMessage)
497 {
498 if (!this._filter.shouldBeVisible(viewMessage)) {
499 this._hiddenByFilterCount++;
500 return;
501 }
502
503 if (this._tryToCollapseMessages(viewMessage, this._visibleViewMessages.peekLast()))
504 return;
505
506 var lastMessage = this._visibleViewMessages.peekLast();
507 if (viewMessage.consoleMessage().type === WebInspector.ConsoleMessage.MessageType.EndGroup) {
508 if (lastMessage && !this._currentGroup.messagesHidden())
509 lastMessage.incrementCloseGroupDecorationCount();
510 this._currentGroup = this._currentGroup.parentGroup();
511 return;
512 }
513 if (!this._currentGroup.messagesHidden()) {
514 var originatingMessage = viewMessage.consoleMessage().originatingMessage();
515 if (lastMessage && originatingMessage && lastMessage.consoleMessage() === originatingMessage)
516 lastMessage.toMessageElement().classList.add("console-adjacent-user-command-result");
517
518 this._visibleViewMessages.push(viewMessage);
519 this._searchMessage(this._visibleViewMessages.length - 1);
520 }
521
522 if (viewMessage.consoleMessage().isGroupStartMessage())
523 this._currentGroup = new WebInspector.ConsoleGroup(this._currentGroup, viewMessage);
524
525 this._messageAppendedForTests();

617 _saveConsole: function()
618 {
619 var url = WebInspector.targetManager.mainTarget().inspectedURL();
620 var filename = String.sprintf("%s-%d.log", url.asParsedURL().host, Date.now());
621 var stream = new WebInspector.FileOutputStream();
622
623 var progressIndicator = new WebInspector.ProgressIndicator();
624 progressIndicator.setTitle(WebInspector.UIString("Writing file"));
625 progressIndicator.setTotalWork(this.itemCount());
626
627 /** @const */
628 var chunkSize = 350;
629 var messageIndex = 0;
630
631 stream.open(filename, openCallback.bind(this));
632
633 /**
634 * @param {boolean} accepted
635 * @this {WebInspector.ConsoleView}
636 */
637 function openCallback(accepted)
638 {
639 if (!accepted)
640 return;
641 this._progressToolbarItem.element.appendChild(progressIndicator.element);
642 writeNextChunk.call(this, stream);
643 }
644
645 /**
646 * @param {!WebInspector.OutputStream} stream
647 * @param {string=} error
648 * @this {WebInspector.ConsoleView}
649 */
650 function writeNextChunk(stream, error)
651 {
652 if (messageIndex >= this.itemCount() || error) {
653 stream.close();
654 progressIndicator.done();
655 return;
656 }
657 var lines = [];
658 for (var i = 0; i < chunkSize && i + messageIndex < this.itemCount(); ++i) {
659 var message = this.itemElement(messageIndex + i);
660 lines.push(message.searchableElement().deepTextContent());
661 }
662 messageIndex += i;
663 stream.write(lines.join("\n") + "\n", writeNextChunk.bind(this));
664 progressIndicator.setWorked(messageIndex);
665 }

684 _updateMessageList: function()
685 {
686 this._topGroup = WebInspector.ConsoleGroup.createTopGroup();
687 this._currentGroup = this._topGroup;
688 this._regexMatchRanges = [];
689 this._hiddenByFilterCount = 0;
690 for (var i = 0; i < this._visibleViewMessages.length; ++i) {
691 this._visibleViewMessages[i].resetCloseGroupDecorationCount();
692 this._visibleViewMessages[i].resetIncrementRepeatCount();
693 }
694 this._visibleViewMessages = [];
695 for (var i = 0; i < this._consoleMessages.length; ++i)
696 this._appendMessageToEnd(this._consoleMessages[i]);
