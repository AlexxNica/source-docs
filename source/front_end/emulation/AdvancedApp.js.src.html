44 /**
45 * @param {!WebInspector.Event} event

47 _openToolboxWindow: function(event)
48 {
49 if (/** @type {string} */ (event.data.to) !== WebInspector.DockController.State.Undocked)
50 return;

65 WebInspector.installComponentRootStyles(/** @type {!Element} */ (toolboxDocument.body));
66 WebInspector.ContextMenu.installHandler(toolboxDocument);
67 WebInspector.Tooltip.installHandler(toolboxDocument);

70 this._toolboxRootView.attachToDocument(toolboxDocument);

72 this._updateDeviceModeView();
73 },

75 _updateDeviceModeView: function()
76 {
77 if (this._isDocked())

79 else if (this._toolboxRootView)

83 /**
84 * @param {!WebInspector.Event} event
85 */
86 _onBeforeDockSideChange: function(event)
87 {
88 if (/** @type {string} */ (event.data.to) === WebInspector.DockController.State.Undocked && this._toolboxRootView) {
89 // Hide inspectorView and force layout to mimic the undocked state.
90 this._rootSplitWidget.hideSidebar();
91 this._inspectedPagePlaceholder.update();
92 }
93
94 this._changingDockSide = true;
95 },
96
97 /**
98 * @param {!WebInspector.Event=} event
99 */
100 _onDockSideChange: function(event)
101 {
102 this._updateDeviceModeView();
103
104 var toDockSide = event ? /** @type {string} */ (event.data.to) : WebInspector.dockController.dockSide();
105 if (toDockSide === WebInspector.DockController.State.Undocked) {
106 this._updateForUndocked();
107 } else if (this._toolboxRootView && event && /** @type {string} */ (event.data.from) === WebInspector.DockController.State.Undocked) {
108 // Don't update yet for smooth transition.
109 this._rootSplitWidget.hideSidebar();
110 } else {
111 this._updateForDocked(toDockSide);
112 }
113 },
114
115 /**
116 * @param {!WebInspector.Event} event
117 */
118 _onAfterDockSideChange: function(event)
119 {
120 // We may get here on the first dock side change while loading without BeforeDockSideChange.
121 if (!this._changingDockSide)
122 return;
123 if (/** @type {string} */ (event.data.from) === WebInspector.DockController.State.Undocked) {
124 // Restore docked layout in case of smooth transition.
125 this._updateForDocked(/** @type {string} */ (event.data.to));
126 }
127 this._changingDockSide = false;
128 this._inspectedPagePlaceholder.update();
129 },
130
131 /**
132 * @param {string} dockSide
133 */
134 _updateForDocked: function(dockSide)
135 {
136 this._rootSplitWidget.setVertical(dockSide === WebInspector.DockController.State.DockedToRight);
137 this._rootSplitWidget.setSecondIsSidebar(dockSide === WebInspector.DockController.State.DockedToRight || dockSide === WebInspector.DockController.State.DockedToBottom);
138 this._rootSplitWidget.toggleResizer(this._rootSplitWidget.resizerElement(), true);
139 this._rootSplitWidget.toggleResizer(WebInspector.inspectorView.topResizerElement(), dockSide === WebInspector.DockController.State.DockedToBottom);
140 this._rootSplitWidget.showBoth();
141 },
142
143 _updateForUndocked: function()
144 {
145 this._rootSplitWidget.toggleResizer(this._rootSplitWidget.resizerElement(), false);
146 this._rootSplitWidget.toggleResizer(WebInspector.inspectorView.topResizerElement(), false);
147 this._rootSplitWidget.hideMain();
148 },
149
150 _isDocked: function()
151 {
152 return WebInspector.dockController.dockSide() !== WebInspector.DockController.State.Undocked;
153 },
154
155 /**
156 * @param {!WebInspector.Event} event
157 */
158 _onSetInspectedPageBounds: function(event)
159 {
160 if (this._changingDockSide)
161 return;
162 var window = this._inspectedPagePlaceholder.element.window();
163 if (!window.innerWidth || !window.innerHeight)
164 return;
165 if (!this._inspectedPagePlaceholder.isShowing())
166 return;
167 var bounds = /** @type {{x: number, y: number, width: number, height: number}} */ (event.data);
168 console.timeStamp("AdvancedApp.setInspectedPageBounds");
169 InspectorFrontendHost.setInspectedPageBounds(bounds);
170 }
