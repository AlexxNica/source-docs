209WebInspector.GenericSettingsTab.prototype = {
210 /**
211 * @param {!Runtime.Extension} extension
212 */
213 _addSetting: function(extension)
214 {
215 if (!WebInspector.GenericSettingsTab.isSettingVisible(extension))
216 return;
217 var descriptor = extension.descriptor();
218 var sectionName = descriptor["category"];
219 var settingName = descriptor["settingName"];
220 var setting = WebInspector.moduleSetting(settingName);
221 var uiTitle = WebInspector.UIString(extension.title());
222
223 var sectionElement = this._sectionElement(sectionName);
224 var settingControl;
225
226 switch (descriptor["settingType"]) {
227 case "boolean":
228 settingControl = WebInspector.SettingsUI.createSettingCheckbox(uiTitle, setting);
229 break;
230 case "enum":
231 var descriptorOptions = descriptor["options"];
232 var options = new Array(descriptorOptions.length);
233 for (var i = 0; i < options.length; ++i) {
234 // The "raw" flag indicates text is non-i18n-izable.
235 var optionName = descriptorOptions[i]["raw"] ? descriptorOptions[i]["text"] : WebInspector.UIString(descriptorOptions[i]["text"]);
236 options[i] = [optionName, descriptorOptions[i]["value"]];
237 }
238 settingControl = this._createSelectSetting(uiTitle, options, setting);
239 break;
240 default:
241 console.error("Invalid setting type: " + descriptor["settingType"]);
242 return;
243 }
244 this._nameToSettingElement.set(settingName, settingControl);
245 sectionElement.appendChild(/** @type {!Element} */ (settingControl));
246 },
247
248 /**
249 * @param {!Runtime.Extension} extension
250 */
251 _addSettingUI: function(extension)
252 {
253 var descriptor = extension.descriptor();
254 var sectionName = descriptor["category"] || "";
255 extension.instance().then(appendCustomSetting.bind(this));
256
257 /**
258 * @param {!Object} object
259 * @this {WebInspector.GenericSettingsTab}
260 */
261 function appendCustomSetting(object)
262 {
263 var settingUI = /** @type {!WebInspector.SettingUI} */ (object);
264 var element = settingUI.settingElement();
265 if (element)
266 this._sectionElement(sectionName).appendChild(element);
267 }
268 },
269
270 /**
271 * @param {string} sectionName
272 * @return {!Element}
273 */
274 _sectionElement: function(sectionName)
275 {
276 var sectionElement = this._nameToSection.get(sectionName);
277 if (!sectionElement) {
278 var uiSectionName = sectionName && WebInspector.UIString(sectionName);
279 sectionElement = this._appendSection(uiSectionName);
280 this._nameToSection.set(sectionName, sectionElement);
281 }
282 return sectionElement;

539 /**
540 * @param {!Runtime.Extension} extension
541 */
542 function revealModuleSetting(extension)
543 {
544 if (!WebInspector.GenericSettingsTab.isSettingVisible(extension))
545 return;
546 if (extension.descriptor()["settingName"] === setting.name) {
547 InspectorFrontendHost.bringToFront();
548 WebInspector.SettingsScreen._showSettingsScreen();
549 success = true;
550 }
551 }

553 /**
554 * @param {!Runtime.Extension} extension

556 function revealSettingUI(extension)

561 WebInspector.SettingsScreen._showSettingsScreen();
562 success = true;
563 }
564 }
