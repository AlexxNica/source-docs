9WebInspector.FormattedContentBuilder = function(indentString)
10{
11 this._lastOriginalPosition = 0;
12
13 this._formattedContent = [];
14 this._formattedContentLength = 0;
15 this._lastFormattedPosition = 0;
16
17 /** @type {!{original: !Array.<number>, formatted: !Array.<number>}} */
18 this._mapping = { original: [0], formatted: [0] };
19
20 this._nestingLevel = 0;
21 this._indentString = indentString;
22 /** @type {!Map<number, string>} */
23 this._cachedIndents = new Map();
24
25 this._newLines = 0;
26 this._softSpace = false;
27 this._hardSpaces = 0;
28 this._enforceSpaceBetweenWords = true;
29}

31WebInspector.FormattedContentBuilder.prototype = {
32 /**
33 * @param {boolean} value
34 * @return {boolean}
35 */
36 setEnforceSpaceBetweenWords: function(value)
37 {
38 var oldValue = this._enforceSpaceBetweenWords;
39 this._enforceSpaceBetweenWords = value;
40 return oldValue;
41 },
42
43 /**
44 * @param {string} token
45 * @param {number} offset
46 */
47 addToken: function(token, offset)
48 {
49 var last = this._formattedContent.peekLast();
50 if (this._enforceSpaceBetweenWords && last && /\w/.test(last[last.length - 1]) && /\w/.test(token))
51 this.addSoftSpace();
52
53 this._appendFormatting();
54
55 // Insert token.
56 this._addMappingIfNeeded(offset);
57 this._addText(token);
58 },
59
60 addSoftSpace: function()
61 {
62 if (!this._hardSpaces)
63 this._softSpace = true;
64 },
65
66 addHardSpace: function()
67 {
68 this._softSpace = false;
69 ++this._hardSpaces;
70 },
71
72 /**
73 * @param {boolean=} noSquash
74 */
75 addNewLine: function(noSquash)
76 {
77 // Avoid leading newlines.
78 if (!this._formattedContentLength)
79 return;
80 if (noSquash)
81 ++this._newLines;
82 else
83 this._newLines = this._newLines || 1;
84 },
85
86 increaseNestingLevel: function()
87 {
88 this._nestingLevel += 1;
89 },
90
91 decreaseNestingLevel: function()
92 {
93 if (this._nestingLevel > 0)
94 this._nestingLevel -= 1;
95 },
96
97 _appendFormatting: function()
98 {
99 if (this._newLines) {
100 for (var i = 0; i < this._newLines; ++i)
101 this._addText("\n");
102 this._addText(this._indent());
103 } else if (this._softSpace) {
104 this._addText(" ");
105 }
106 if (this._hardSpaces) {
107 for (var i = 0; i < this._hardSpaces; ++i)
108 this._addText(" ");
109 }
110 this._newLines = 0;
111 this._softSpace = false;
112 this._hardSpaces = 0;
113 },
114
115 /**
116 * @return {string}
117 */
118 content: function()
119 {
120 return this._formattedContent.join("") + (this._newLines ? "\n" : "");
121 },
122
123 /**
124 * @return {!{original: !Array.<number>, formatted: !Array.<number>}}
125 */
126 mapping: function()
127 {
128 return this._mapping;
129 },
130
131 /**
132 * @return {string}
133 */
134 _indent: function()
135 {
136 var cachedValue = this._cachedIndents.get(this._nestingLevel)
137 if (cachedValue)
138 return cachedValue;
139
140 var fullIndent = "";
141 for (var i = 0; i < this._nestingLevel; ++i)
142 fullIndent += this._indentString;
143
144 // Cache a maximum of 20 nesting level indents.
145 if (this._nestingLevel <= 20)
146 this._cachedIndents.set(this._nestingLevel, fullIndent);
147 return fullIndent;
148 },
149
150 /**
151 * @param {string} text
152 */
153 _addText: function(text)
154 {
155 this._formattedContent.push(text);
156 this._formattedContentLength += text.length;
157 },
158
159 /**
160 * @param {number} originalPosition
161 */
162 _addMappingIfNeeded: function(originalPosition)
163 {
164 if (originalPosition - this._lastOriginalPosition === this._formattedContentLength - this._lastFormattedPosition)
165 return;
166 this._mapping.original.push(originalPosition);
167 this._lastOriginalPosition = originalPosition;
168 this._mapping.formatted.push(this._formattedContentLength);
169 this._lastFormattedPosition = this._formattedContentLength;
170 }
