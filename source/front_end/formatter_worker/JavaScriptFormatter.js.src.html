48 {
49 this._fromOffset = fromOffset;
50 this._toOffset = toOffset;
51 this._content = text.substring(this._fromOffset, this._toOffset);
52 this._tokenizer = new WebInspector.AcornTokenizer(this._content);
53 var ast = acorn.parse(this._content, { ranges: false, ecmaVersion: 6 });
54 var walker = new WebInspector.ESTreeWalker(this._beforeVisit.bind(this), this._afterVisit.bind(this));

58 /**
59 * @param {?Acorn.TokenOrComment} token
60 * @param {string} format
61 */
62 _push: function(token, format)
63 {
64 for (var i = 0; i < format.length; ++i) {
65 if (format[i] === "s")
66 this._builder.addSoftSpace();
67 else if (format[i] === "S")
68 this._builder.addHardSpace();
69 else if (format[i] === "n")
70 this._builder.addNewLine();
71 else if (format[i] === ">")
72 this._builder.increaseNestingLevel();
73 else if (format[i] === "<")
74 this._builder.decreaseNestingLevel();
75 else if (format[i] === "t")
76 this._builder.addToken(this._content.substring(token.start, token.end), this._fromOffset + token.start);
77 }

80 /**
81 * @param {!ESTree.Node} node

83 _beforeVisit: function(node)
84 {
85 if (!node.parent)
86 return;
87 while (this._tokenizer.peekToken() && this._tokenizer.peekToken().start < node.start) {
88 var token = /** @type {!Acorn.TokenOrComment} */(this._tokenizer.nextToken());
89 var format = this._formatToken(node.parent, token);

95 * @param {!ESTree.Node} node
96 */
97 _afterVisit: function(node)
98 {
99 while (this._tokenizer.peekToken() && this._tokenizer.peekToken().start < node.end) {
100 var token = /** @type {!Acorn.TokenOrComment} */(this._tokenizer.nextToken());
101 var format = this._formatToken(node, token);
102 this._push(token, format);
103 }
104 this._push(null, this._finishNode(node));

107 /**
108 * @param {!ESTree.Node} node
109 * @return {boolean}
110 */
111 _inForLoopHeader: function(node)
112 {
113 var parent = node.parent;
114 if (!parent)
115 return false;
116 if (parent.type === "ForStatement")
117 return node === parent.init || node === parent.test || node === parent.update;
118 if (parent.type === "ForInStatement" || parent.type === "ForOfStatement")
119 return node === parent.left || parent.right;
120 return false;

123 /**
124 * @param {!ESTree.Node} node
125 * @param {!Acorn.TokenOrComment} token
126 * @return {string}
127 */
128 _formatToken: function(node, token)
129 {
130 var AT = WebInspector.AcornTokenizer;
131 if (AT.lineComment(token))
132 return "tn";
133 if (AT.blockComment(token))
134 return "tn";
135 if (node.type === "ContinueStatement" || node.type === "BreakStatement") {
136 return node.label && AT.keyword(token) ? "ts" : "t";
137 } else if (node.type === "Identifier") {
138 return "t";
139 } else if (node.type === "ReturnStatement") {
140 if (AT.punctuator(token, ";"))
141 return "t";
142 return node.argument ? "ts" : "t";
143 } else if (node.type === "Property") {
144 if (AT.punctuator(token, ":"))
145 return "ts";
146 return "t";
147 } else if (node.type === "ArrayExpression") {
148 if (AT.punctuator(token, ","))
149 return "ts";
150 return "t";
151 } else if (node.type === "LabeledStatement") {
152 if (AT.punctuator(token, ":"))
153 return "ts";
154 } else if (node.type === "LogicalExpression" || node.type === "AssignmentExpression" || node.type === "BinaryExpression") {
155 if (AT.punctuator(token) && !AT.punctuator(token, "()"))
156 return "sts";
157 } else if (node.type === "ConditionalExpression") {
158 if (AT.punctuator(token, "?:"))
159 return "sts";
160 } else if (node.type === "VariableDeclarator") {
161 if (AT.punctuator(token, "="))
162 return "sts";
163 } else if (node.type === "FunctionDeclaration") {
164 if (AT.punctuator(token, ",)"))
165 return "ts";
166 } else if (node.type === "FunctionExpression") {
167 if (AT.punctuator(token, ",)"))
168 return "ts";
169 if (AT.keyword(token, "function"))
170 return node.id ? "ts" : "t";
171 } else if (node.type === "WithStatement") {
172 if (AT.punctuator(token, ")"))
173 return node.body && node.body.type === "BlockStatement" ? "ts" : "tn>";
174 } else if (node.type === "SwitchStatement") {
175 if (AT.punctuator(token, "{"))
176 return "tn>";
177 if (AT.punctuator(token, "}"))
178 return "n<tn";
179 if (AT.punctuator(token, ")"))
180 return "ts";
181 } else if (node.type === "SwitchCase") {
182 if (AT.keyword(token, "case"))
183 return "n<ts";
184 if (AT.keyword(token, "default"))
185 return "n<t";
186 if (AT.punctuator(token, ":"))
187 return "tn>";
188 } else if (node.type === "VariableDeclaration") {
189 if (AT.punctuator(token, ",")) {
190 var allVariablesInitialized = true;
191 var declarations = /** @type {!Array.<!ESTree.Node>} */(node.declarations);
192 for (var i = 0; i < declarations.length; ++i)
193 allVariablesInitialized = allVariablesInitialized && !!declarations[i].init;
194 return !this._inForLoopHeader(node) && allVariablesInitialized ? "nSSts" : "ts";
195 }
196 } else if (node.type === "BlockStatement") {
197 if (AT.punctuator(token, "{"))
198 return node.body.length ? "tn>" : "t";
199 if (AT.punctuator(token, "}"))
200 return node.body.length ? "n<t" : "t";
201 } else if (node.type === "CatchClause") {
202 if (AT.punctuator(token, ")"))
203 return "ts";
204 } else if (node.type === "ObjectExpression") {
205 if (!node.properties.length)
206 return "t";
207 if (AT.punctuator(token, "{"))
208 return "tn>";
209 if (AT.punctuator(token, "}"))
210 return "n<t";
211 if (AT.punctuator(token, ","))
212 return "tn";
213 } else if (node.type === "IfStatement") {
214 if (AT.punctuator(token, ")"))
215 return node.consequent && node.consequent.type === "BlockStatement" ? "ts" : "tn>";
216
217 if (AT.keyword(token, "else")) {
218 var preFormat = node.consequent && node.consequent.type === "BlockStatement" ? "st" : "n<t";
219 var postFormat = "n>";
220 if (node.alternate && (node.alternate.type === "BlockStatement" || node.alternate.type === "IfStatement"))
221 postFormat = "s";
222 return preFormat + postFormat;
223 }
224 } else if (node.type === "CallExpression") {
225 if (AT.punctuator(token, ","))
226 return "ts";
227 } else if (node.type === "SequenceExpression" && AT.punctuator(token, ",")) {
228 return node.parent && node.parent.type === "SwitchCase" ? "ts" : "tn";
229 } else if (node.type === "ForStatement" || node.type === "ForOfStatement" || node.type === "ForInStatement") {
230 if (AT.punctuator(token, ";"))
231 return "ts";
232 if (AT.keyword(token, "in") || AT.identifier(token, "of"))
233 return "sts";
234
235 if (AT.punctuator(token, ")"))
236 return node.body && node.body.type === "BlockStatement" ? "ts" : "tn>";
237 } else if (node.type === "WhileStatement") {
238 if (AT.punctuator(token, ")"))
239 return node.body && node.body.type === "BlockStatement" ? "ts" : "tn>";
240 } else if (node.type === "DoWhileStatement") {
241 var blockBody = node.body && node.body.type === "BlockStatement";
242 if (AT.keyword(token, "do"))
243 return blockBody ? "ts" : "tn>";
244 if (AT.keyword(token, "while"))
245 return blockBody ? "sts" : "n<ts";
246 } else if (node.type === "ClassBody") {
247 if (AT.punctuator(token, "{"))
248 return "stn>";
249 if (AT.punctuator(token, "}"))
250 return "<ntn";
251 return "t";
252 } else if (node.type === "YieldExpression") {
253 return "t";
254 } else if (node.type === "Super") {
255 return "t";
256 }
257 return AT.keyword(token) && !AT.keyword(token, "this") ? "ts" : "t";
258 },

260 /**
261 * @param {!ESTree.Node} node
262 * @return {string}
263 */
264 _finishNode: function(node)
265 {
266 if (node.type === "WithStatement") {
267 if (node.body && node.body.type !== "BlockStatement")
268 return "n<";
269 } else if (node.type === "VariableDeclaration") {
270 if (!this._inForLoopHeader(node))
271 return "n";
272 } else if (node.type === "ForStatement" || node.type === "ForOfStatement" || node.type === "ForInStatement") {
273 if (node.body && node.body.type !== "BlockStatement")
274 return "n<";
275 } else if (node.type === "BlockStatement") {
276 if (node.parent && node.parent.type === "IfStatement" && node.parent.alternate && node.parent.consequent === node)
277 return "";
278 if (node.parent && node.parent.type === "FunctionExpression" && node.parent.parent && node.parent.parent.type === "Property")
279 return "";
280 if (node.parent && node.parent.type === "FunctionExpression" && node.parent.parent && node.parent.parent.type === "VariableDeclarator")
281 return "";
282 if (node.parent && node.parent.type === "FunctionExpression" && node.parent.parent && node.parent.parent.type === "CallExpression")
283 return "";
284 if (node.parent && node.parent.type === "DoWhileStatement")
285 return "";
286 if (node.parent && node.parent.type === "TryStatement" && node.parent.block === node)
287 return "s";
288 if (node.parent && node.parent.type === "CatchClause" && node.parent.parent.finalizer)
289 return "s";
290 return "n";
291 } else if (node.type === "WhileStatement") {
292 if (node.body && node.body.type !== "BlockStatement")
293 return "n<";
294 } else if (node.type === "IfStatement") {
295 if (node.alternate) {
296 if (node.alternate.type !== "BlockStatement" && node.alternate.type !== "IfStatement")
297 return "<";
298 } else if (node.consequent) {
299 if (node.consequent.type !== "BlockStatement")
300 return "<";
301 }
302 } else if (node.type === "BreakStatement" || node.type === "ContinueStatement" || node.type === "ThrowStatement" || node.type === "ReturnStatement" || node.type === "ExpressionStatement") {
303 return "n";
304 }
305 return "";
306 }
