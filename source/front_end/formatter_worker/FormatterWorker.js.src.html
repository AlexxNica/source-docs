35WebInspector.createTokenizer = function(mimeType)
36{
37 var mode = CodeMirror.getMode({indentUnit: 2}, mimeType);
38 var state = CodeMirror.startState(mode);
39 /**
40 * @param {string} line
41 * @param {function(string, ?string, number, number):?} callback
42 */
43 function tokenize(line, callback)
44 {
45 var stream = new CodeMirror.StringStream(line);
46 while (!stream.eol()) {
47 var style = mode.token(stream, state);
48 var value = stream.current();
49 if (callback(value, style, stream.start, stream.start + value.length) === WebInspector.AbortTokenization)
50 return;
51 stream.start = stream.pos;
52 }
53 }
54 return tokenize;
55}

59self.onmessage = function(event) {
60 var method = /** @type {string} */(event.data.method);
61 var params = /** @type !{indentString: string, content: string, mimeType: string} */ (event.data.params);
62 if (!method)
63 return;
64
65 switch (method) {
66 case "format":
67 WebInspector.format(params.mimeType, params.content, params.indentString);
68 break;
69 case "parseCSS":
70 WebInspector.parseCSS(params.content);
71 break;
72 case "parseSCSS":
73 WebInspector.FormatterWorkerContentParser.parse(params.content, "text/x-scss");
74 break;
75 case "javaScriptOutline":
76 WebInspector.javaScriptOutline(params.content);
77 break;
78 case "javaScriptIdentifiers":
79 WebInspector.javaScriptIdentifiers(params.content);
80 break;
81 case "evaluatableJavaScriptSubstring":
82 WebInspector.evaluatableJavaScriptSubstring(params.content);
83 break;
84 case "relaxedJSONParser":
85 WebInspector.relaxedJSONParser(params.content);
86 break;
87 default:
88 console.error("Unsupport method name: " + method);
89 }
90};

103WebInspector.evaluatableJavaScriptSubstring = function(content)
104{
105 var tokenizer = acorn.tokenizer(content, {ecmaVersion: 6});
106 var result = "";
107 try {
108 var token = tokenizer.getToken();
109 while (token.type !== acorn.tokTypes.eof && WebInspector.AcornTokenizer.punctuator(token))
110 token = tokenizer.getToken();
111
112 var startIndex = token.start;
113 var endIndex = token.end;
114 var openBracketsCounter = 0;
115 while (token.type !== acorn.tokTypes.eof) {
116 var isIdentifier = WebInspector.AcornTokenizer.identifier(token);
117 var isThis = WebInspector.AcornTokenizer.keyword(token, "this");
118 var isString = token.type === acorn.tokTypes.string;
119 if (!isThis && !isIdentifier && !isString)
120 break;
121
122 endIndex = token.end;
123 token = tokenizer.getToken();
124 while (WebInspector.AcornTokenizer.punctuator(token, ".[]")) {
125 if (WebInspector.AcornTokenizer.punctuator(token, "["))
126 openBracketsCounter++;
127
128 if (WebInspector.AcornTokenizer.punctuator(token, "]")) {
129 endIndex = openBracketsCounter > 0 ? token.end : endIndex;
130 openBracketsCounter--;
131 }
132
133 token = tokenizer.getToken();
134 }
135 }
136 result = content.substring(startIndex, endIndex);
137 } catch (e) {
138 console.error(e);
139 }
140 postMessage(result);
141}

146WebInspector.javaScriptIdentifiers = function(content)
147{
148 var root = acorn.parse(content, { ranges: false, ecmaVersion: 6 });
149
150 /** @type {!Array<!ESTree.Node>} */
151 var identifiers = [];
152 var walker = new WebInspector.ESTreeWalker(beforeVisit);
153
154 /**
155 * @param {!ESTree.Node} node
156 * @return {boolean}
157 */
158 function isFunction(node)
159 {
160 return node.type === "FunctionDeclaration" || node.type === "FunctionExpression" || node.type === "ArrowFunctionExpression";
161 }
162
163 /**
164 * @param {!ESTree.Node} node
165 */
166 function beforeVisit(node)
167 {
168 if (isFunction(node)) {
169 if (node.id)
170 identifiers.push(node.id);
171 return WebInspector.ESTreeWalker.SkipSubtree;
172 }
173
174 if (node.type !== "Identifier")
175 return;
176
177 if (node.parent && node.parent.type === "MemberExpression" && node.parent.property === node && !node.parent.computed)
178 return;
179 identifiers.push(node);
180 }
181
182 if (!root || root.type !== "Program" || root.body.length !== 1 || !isFunction(root.body[0])) {
183 postMessage([]);
184 return;
185 }
186
187 var functionNode = root.body[0];
188 for (var param of functionNode.params)
189 walker.walk(param);
190 walker.walk(functionNode.body);
191 var reduced = identifiers.map(id => ({name: id.name, offset: id.start}));
192 postMessage(reduced);
193}

200WebInspector.format = function(mimeType, text, indentString)
201{
202 // Default to a 4-space indent.
203 indentString = indentString || " ";
204 var result = {};
205 var builder = new WebInspector.FormattedContentBuilder(indentString);
206 var lineEndings = text.computeLineEndings();
207 try {
208 switch (mimeType) {
209 case "text/html":
210 var formatter = new WebInspector.HTMLFormatter(builder);
211 formatter.format(text, lineEndings);
212 break;
213 case "text/css":
214 var formatter = new WebInspector.CSSFormatter(builder);
215 formatter.format(text, lineEndings, 0, text.length);
216 break;
217 case "text/javascript":
218 var formatter = new WebInspector.JavaScriptFormatter(builder);
219 formatter.format(text, lineEndings, 0, text.length);
220 break;
221 default:
222 var formatter = new WebInspector.IdentityFormatter(builder);
223 formatter.format(text, lineEndings, 0, text.length);
224 }
225 result.mapping = builder.mapping();
226 result.content = builder.content();
227 } catch (e) {
228 console.error(e);
229 result.mapping = { original: [0], formatted: [0] };
230 result.content = text;
231 }
232 postMessage(result);
233}

252WebInspector.FormatterWorkerContentParser.parse = function(content, mimeType)
253{
254 var extension = self.runtime.extensions(WebInspector.FormatterWorkerContentParser).find(findExtension);
255 console.assert(extension);
256 extension.instance()
257 .then(instance => instance.parse(content))
258 .catchException(null)
259 .then(postMessage);
