31/**
32 * @constructor
33 * @extends {WebInspector.Object}
34 * @param {boolean} canDock
35 */
36WebInspector.DockController = function(canDock)
37{
38 this._canDock = canDock;
39
40 this._closeButton = new WebInspector.ToolbarButton(WebInspector.UIString("Close"), "delete-toolbar-item");
41 this._closeButton.addEventListener("click", InspectorFrontendHost.closeWindow.bind(InspectorFrontendHost));
42
43 if (!canDock) {
44 this._dockSide = WebInspector.DockController.State.Undocked;
45 this._updateUI();
46 return;
47 }
48
49 this._states = [WebInspector.DockController.State.DockedToRight, WebInspector.DockController.State.DockedToBottom, WebInspector.DockController.State.Undocked];
50 this._currentDockStateSetting = WebInspector.settings.moduleSetting("currentDockState");
51 this._currentDockStateSetting.addChangeListener(this._dockSideChanged, this);
52 this._lastDockStateSetting = WebInspector.settings.createSetting("lastDockState", "bottom");
53 if (this._states.indexOf(this._currentDockStateSetting.get()) === -1)
54 this._currentDockStateSetting.set("right");
55 if (this._states.indexOf(this._lastDockStateSetting.get()) === -1)
56 this._currentDockStateSetting.set("bottom");
57}
58
59WebInspector.DockController.State = {
60 DockedToBottom: "bottom",
61 DockedToRight: "right",
62 Undocked: "undocked"
63}
64
65// Use BeforeDockSideChanged to do something before all the UI bits are updated,
66// DockSideChanged to update UI, and AfterDockSideChanged to perform actions
67// after frontend is docked/undocked in the browser.
68WebInspector.DockController.Events = {
69 BeforeDockSideChanged: "BeforeDockSideChanged",
70 DockSideChanged: "DockSideChanged",
71 AfterDockSideChanged: "AfterDockSideChanged"
72}

74WebInspector.DockController.prototype = {
75 initialize: function()
76 {
77 if (!this._canDock)
78 return;
79
80 this._titles = [WebInspector.UIString("Dock to right"), WebInspector.UIString("Dock to bottom"), WebInspector.UIString("Undock into separate window")];

109 {
110 return this._dockSide === WebInspector.DockController.State.DockedToRight;
111 },

113 /**
114 * @param {string} dockSide
115 */
116 setDockSide: function(dockSide)
117 {
118 if (this._states.indexOf(dockSide) === -1)
119 dockSide = this._states[0];
120
121 if (this._dockSide === dockSide)
122 return;
123
124 if (this._dockSide)
125 this._lastDockStateSetting.set(this._dockSide);
126
127 WebInspector.DockController._previousFocusedElement = WebInspector.currentFocusElement();
128 var eventData = { from: this._dockSide, to: dockSide };
129 this.dispatchEventToListeners(WebInspector.DockController.Events.BeforeDockSideChanged, eventData);
130 console.timeStamp("DockController.setIsDocked");
131 this._dockSide = dockSide;
132 this._currentDockStateSetting.set(dockSide);
133 InspectorFrontendHost.setIsDocked(dockSide !== WebInspector.DockController.State.Undocked, this._setIsDockedResponse.bind(this, eventData));
134 this._updateUI();
135 this.dispatchEventToListeners(WebInspector.DockController.Events.DockSideChanged, eventData);
136 },
137
138 /**
139 * @param {{from: string, to: string}} eventData
140 */
141 _setIsDockedResponse: function(eventData)
142 {
143 this.dispatchEventToListeners(WebInspector.DockController.Events.AfterDockSideChanged, eventData);
144
145 if (WebInspector.DockController._previousFocusedElement)
146 WebInspector.DockController._previousFocusedElement.focus();
147 delete WebInspector.DockController._previousFocusedElement;
148 },
149
150 /**
151 * @suppressGlobalPropertiesCheck
152 */
153 _updateUI: function()
154 {
155 var body = document.body; // Only for main window.
156 switch (this._dockSide) {
157 case WebInspector.DockController.State.DockedToBottom:
158 body.classList.remove("undocked");
159 body.classList.remove("dock-to-right");
160 body.classList.add("dock-to-bottom");
161 break;
162 case WebInspector.DockController.State.DockedToRight:
163 body.classList.remove("undocked");
164 body.classList.add("dock-to-right");
165 body.classList.remove("dock-to-bottom");
166 break;
167 case WebInspector.DockController.State.Undocked:
168 body.classList.add("undocked");
169 body.classList.remove("dock-to-right");
170 body.classList.remove("dock-to-bottom");
171 break;
172 }
173 this._closeButton.setVisible(this._dockSide !== WebInspector.DockController.State.Undocked);
174 },
175
176 _toggleDockSide: function()
177 {
178 if (this._lastDockStateSetting.get() === this._currentDockStateSetting.get()) {
179 var index = this._states.indexOf(this._currentDockStateSetting.get()) || 0;
180 this._lastDockStateSetting.set(this._states[(index + 1) % this._states.length]);
181 }
182 this.setDockSide(this._lastDockStateSetting.get());
183 },
