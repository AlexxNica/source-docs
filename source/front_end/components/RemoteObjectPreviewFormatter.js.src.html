12WebInspector.RemoteObjectPreviewFormatter.prototype = {
13 /**
14 * @param {!Element} parentElement
15 * @param {!RuntimeAgent.ObjectPreview} preview
16 */
17 appendObjectPreview: function(parentElement, preview)
18 {
19 var description = preview.description;
20 if (preview.type !== "object" || preview.subtype === "null") {
21 parentElement.appendChild(this.renderPropertyPreview(preview.type, preview.subtype, description));
22 return;
23 }
24 if (description && preview.subtype !== "array" && preview.subtype !== "typedarray") {
25 var text = preview.subtype ? description : this._abbreviateFullQualifiedClassName(description);
26 parentElement.createTextChildren(text, " ");
27 }
28 if (preview.entries)
29 this._appendEntriesPreview(parentElement, preview);
30 else
31 this._appendPropertiesPreview(parentElement, preview);
32 },
33
34 /**
35 * @param {string} description
36 * @return {string}
37 */
38 _abbreviateFullQualifiedClassName: function(description)
39 {
40 var abbreviatedDescription = description.split(".");
41 for (var i = 0; i < abbreviatedDescription.length - 1; ++i)
42 abbreviatedDescription[i] = abbreviatedDescription[i].trimMiddle(3);
43 return abbreviatedDescription.join(".");
44 },
45
46 /**
47 * @param {!Element} parentElement
48 * @param {!RuntimeAgent.ObjectPreview} preview
49 */
50 _appendPropertiesPreview: function(parentElement, preview)
51 {
52 var isArray = preview.subtype === "array" || preview.subtype === "typedarray";
53 var arrayLength = WebInspector.RemoteObject.arrayLength(preview);
54 var properties = preview.properties;
55 if (isArray)
56 properties = properties.slice().stableSort(compareIndexesFirst);
57
58 /**
59 * @param {!RuntimeAgent.PropertyPreview} a
60 * @param {!RuntimeAgent.PropertyPreview} b
61 */
62 function compareIndexesFirst(a, b)
63 {
64 var index1 = toArrayIndex(a.name);
65 var index2 = toArrayIndex(b.name);
66 if (index1 < 0)
67 return index2 < 0 ? 0 : 1;
68 return index2 < 0 ? -1 : index1 - index2;
69 }
70
71 /**
72 * @param {string} name
73 * @return {number}
74 */
75 function toArrayIndex(name)
76 {
77 var index = name >>> 0;
78 if (String(index) === name && index < arrayLength)
79 return index;
80 return -1;
81 }
82
83 parentElement.createTextChild(isArray ? "[" : "{");
84 for (var i = 0; i < properties.length; ++i) {
85 if (i > 0)
86 parentElement.createTextChild(", ");
87
88 var property = properties[i];
89 var name = property.name;
90 if (!isArray || name !== String(i) || i >= arrayLength) {
91 if (/^\s|\s$|^$|\n/.test(name))
92 parentElement.createChild("span", "name").createTextChildren("\"", name.replace(/\n/g, "\u21B5"), "\"");
93 else
94 parentElement.createChild("span", "name").textContent = name;
95 parentElement.createTextChild(": ");
96 }
97
98 parentElement.appendChild(this._renderPropertyPreviewOrAccessor([property]));
99 }
100 if (preview.overflow)
101 parentElement.createChild("span").textContent = "\u2026";
102 parentElement.createTextChild(isArray ? "]" : "}");
103 },
104
105
106 /**
107 * @param {!Element} parentElement
108 * @param {!RuntimeAgent.ObjectPreview} preview
109 */
110 _appendEntriesPreview: function(parentElement, preview)
111 {
112 parentElement.createTextChild("{");
113 for (var i = 0; i < preview.entries.length; ++i) {
114 if (i > 0)
115 parentElement.createTextChild(", ");
116
117 var entry = preview.entries[i];
118 if (entry.key) {
119 this.appendObjectPreview(parentElement, entry.key);
120 parentElement.createTextChild(" => ");
121 }
122 this.appendObjectPreview(parentElement, entry.value);
123 }
124 if (preview.overflow)
125 parentElement.createChild("span").textContent = "\u2026";
126 parentElement.createTextChild("}");
127 },
128
129
130 /**
131 * @param {!Array.<!RuntimeAgent.PropertyPreview>} propertyPath
132 * @return {!Element}
133 */
134 _renderPropertyPreviewOrAccessor: function(propertyPath)
135 {
136 var property = propertyPath.peekLast();
137 return this.renderPropertyPreview(property.type, /** @type {string} */ (property.subtype), property.value);
138 },
139
140 /**
141 * @param {string} type
142 * @param {string=} subtype
143 * @param {string=} description
144 * @return {!Element}
145 */
146 renderPropertyPreview: function(type, subtype, description)
147 {
148 var span = createElementWithClass("span", "object-value-" + (subtype || type));
149 description = description || "";
150
151 if (type === "function") {
152 span.textContent = "function";
153 return span;
154 }
155
156 if (type === "object" && subtype === "node" && description) {
157 span.classList.add("object-value-preview-node");
158 WebInspector.DOMPresentationUtils.createSpansForNodeTitle(span, description);
159 return span;
160 }
161
162 if (type === "string") {
163 span.createTextChildren("\"", description.replace(/\n/g, "\u21B5"), "\"");
164 return span;
165 }
166
167 if (type === "object" && !subtype) {
168 span.textContent = this._abbreviateFullQualifiedClassName(description);
169 return span;
170 }
171
172 span.textContent = description;
173 return span;
174 }
