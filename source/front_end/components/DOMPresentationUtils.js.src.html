34/**
35 * @param {!WebInspector.DOMNode} node
36 * @param {!Element} parentElement
37 */
38WebInspector.DOMPresentationUtils.decorateNodeLabel = function(node, parentElement)
39{
40 var originalNode = node;
41 var isPseudo = node.nodeType() === Node.ELEMENT_NODE && node.pseudoType();
42 if (isPseudo && node.parentNode)
43 node = node.parentNode;
44
45 var title = node.nodeNameInCorrectCase();
46
47 var nameElement = parentElement.createChild("span", "node-label-name");
48 nameElement.textContent = title;
49
50 var idAttribute = node.getAttribute("id");
51 if (idAttribute) {
52 var idElement = parentElement.createChild("span", "node-label-id");
53 var part = "#" + idAttribute;
54 title += part;
55 idElement.createTextChild(part);
56
57 // Mark the name as extra, since the ID is more important.
58 nameElement.classList.add("extra");
59 }
60
61 var classAttribute = node.getAttribute("class");
62 if (classAttribute) {
63 var classes = classAttribute.split(/\s+/);
64 var foundClasses = {};
65
66 if (classes.length) {
67 var classesElement = parentElement.createChild("span", "extra node-label-class");
68 for (var i = 0; i < classes.length; ++i) {
69 var className = classes[i];
70 if (className && !(className in foundClasses)) {
71 var part = "." + className;
72 title += part;
73 classesElement.createTextChild(part);
74 foundClasses[className] = true;
75 }
76 }
77 }
78 }
79
80 if (isPseudo) {
81 var pseudoElement = parentElement.createChild("span", "extra node-label-pseudo");
82 var pseudoText = "::" + originalNode.pseudoType();
83 pseudoElement.createTextChild(pseudoText);
84 title += pseudoText;
85 }
86 parentElement.title = title;
87}

153/**
154 * @param {!WebInspector.Target} target
155 * @param {string} originalImageURL
156 * @param {boolean} showDimensions
157 * @param {function(!Element=)} userCallback
158 * @param {!Object=} precomputedFeatures

160WebInspector.DOMPresentationUtils.buildImagePreviewContents = function(target, originalImageURL, showDimensions, userCallback, precomputedFeatures)
161{
162 var resource = target.resourceTreeModel.resourceForURL(originalImageURL);
163 var imageURL = originalImageURL;
164 if (!isImageResource(resource) && precomputedFeatures && precomputedFeatures.currentSrc) {
165 imageURL = precomputedFeatures.currentSrc;
166 resource = target.resourceTreeModel.resourceForURL(imageURL);
167 }
168 if (!isImageResource(resource)) {
169 userCallback();
170 return;
171 }
172
173 var imageElement = createElement("img");
174 imageElement.addEventListener("load", buildContent, false);
175 imageElement.addEventListener("error", errorCallback, false);
176 resource.populateImageSource(imageElement);
177
178 function errorCallback()
179 {
180 // Drop the event parameter when invoking userCallback.
181 userCallback();
182 }

184 /**
185 * @param {?WebInspector.Resource} resource
186 * @return {boolean}
187 */
188 function isImageResource(resource)
189 {
190 return !!resource && resource.resourceType() === WebInspector.resourceTypes.Image;
191 }
192
193 function buildContent()
194 {
195 var container = createElement("table");
196 container.className = "image-preview-container";
197 var naturalWidth = precomputedFeatures ? precomputedFeatures.naturalWidth : imageElement.naturalWidth;
198 var naturalHeight = precomputedFeatures ? precomputedFeatures.naturalHeight : imageElement.naturalHeight;
199 var offsetWidth = precomputedFeatures ? precomputedFeatures.offsetWidth : naturalWidth;
200 var offsetHeight = precomputedFeatures ? precomputedFeatures.offsetHeight : naturalHeight;
201 var description;
202 if (showDimensions) {
203 if (offsetHeight === naturalHeight && offsetWidth === naturalWidth)
204 description = WebInspector.UIString("%d \xd7 %d pixels", offsetWidth, offsetHeight);
205 else
206 description = WebInspector.UIString("%d \xd7 %d pixels (Natural: %d \xd7 %d pixels)", offsetWidth, offsetHeight, naturalWidth, naturalHeight);
207 }
208
209 container.createChild("tr").createChild("td", "image-container").appendChild(imageElement);
210 if (description)
211 container.createChild("tr").createChild("td").createChild("span", "description").textContent = description;
212 if (imageURL !== originalImageURL)
213 container.createChild("tr").createChild("td").createChild("span", "description").textContent = String.sprintf("currentSrc: %s", imageURL.trimMiddle(100));
214 userCallback(container);
215 }
216}

224WebInspector.DOMPresentationUtils.buildStackTracePreviewContents = function(target, linkifier, stackTrace)
225{
226 var element = createElement("span");
227 element.style.display = "inline-block";
228 var shadowRoot = WebInspector.createShadowRootWithCoreStyles(element, "components/domUtils.css");
229 var contentElement = shadowRoot.createChild("table", "stack-preview-container");
230
231 /**
232 * @param {!RuntimeAgent.StackTrace} stackTrace
233 */
234 function appendStackTrace(stackTrace)
235 {
236 for (var stackFrame of stackTrace.callFrames) {
237 var row = createElement("tr");
238 row.createChild("td", "function-name").textContent = WebInspector.beautifyFunctionName(stackFrame.functionName);
239 var link = linkifier.maybeLinkifyConsoleCallFrame(target, stackFrame);
240 if (link) {
241 row.createChild("td").textContent = " @ ";
242 row.createChild("td").appendChild(link);
243 }
244 contentElement.appendChild(row);
245 }
246 }
247
248 if (!stackTrace)
249 return element;
250
251 appendStackTrace(stackTrace);
252
253 var asyncStackTrace = stackTrace.parent;
254 while (asyncStackTrace) {
255 if (!asyncStackTrace.callFrames.length) {
256 asyncStackTrace = asyncStackTrace.parent;
257 continue;
258 }
259 var row = contentElement.createChild("tr");
260 row.createChild("td", "stack-preview-async-description").textContent = WebInspector.asyncStackTraceLabel(asyncStackTrace.description);
261 row.createChild("td");
262 row.createChild("td");
263 appendStackTrace(asyncStackTrace);
264 asyncStackTrace = asyncStackTrace.parent;
265 }
266
267 return element;
268}

287{
288 var lowerCaseName = node.localName() || node.nodeName().toLowerCase();
289 if (node.nodeType() !== Node.ELEMENT_NODE)
290 return lowerCaseName;
291 if (lowerCaseName === "input" && node.getAttribute("type") && !node.getAttribute("id") && !node.getAttribute("class"))
292 return lowerCaseName + "[type=\"" + node.getAttribute("type") + "\"]";
293 if (node.getAttribute("id"))
294 return lowerCaseName + "#" + node.getAttribute("id");
295 if (node.getAttribute("class"))

297 return lowerCaseName;

300/**
301 * @param {!WebInspector.DOMNode} node
302 * @param {boolean=} optimized
303 * @return {string}
304 */
305WebInspector.DOMPresentationUtils.cssPath = function(node, optimized)
306{
307 if (node.nodeType() !== Node.ELEMENT_NODE)
308 return "";
309
310 var steps = [];
311 var contextNode = node;
312 while (contextNode) {
313 var step = WebInspector.DOMPresentationUtils._cssPathStep(contextNode, !!optimized, contextNode === node);
314 if (!step)
315 break; // Error - bail out early.
316 steps.push(step);
317 if (step.optimized)
318 break;
319 contextNode = contextNode.parentNode;
320 }
321
322 steps.reverse();
323 return steps.join(" > ");
324}

327 * @param {!WebInspector.DOMNode} node
328 * @param {boolean} optimized
329 * @param {boolean} isTargetNode
330 * @return {?WebInspector.DOMNodePathStep}
331 */
332WebInspector.DOMPresentationUtils._cssPathStep = function(node, optimized, isTargetNode)
333{
334 if (node.nodeType() !== Node.ELEMENT_NODE)
335 return null;
336
337 var id = node.getAttribute("id");
338 if (optimized) {
339 if (id)
340 return new WebInspector.DOMNodePathStep(idSelector(id), true);
341 var nodeNameLower = node.nodeName().toLowerCase();
342 if (nodeNameLower === "body" || nodeNameLower === "head" || nodeNameLower === "html")
343 return new WebInspector.DOMNodePathStep(node.nodeNameInCorrectCase(), true);
344 }
345 var nodeName = node.nodeNameInCorrectCase();
346
347 if (id)
348 return new WebInspector.DOMNodePathStep(nodeName + idSelector(id), true);
349 var parent = node.parentNode;
350 if (!parent || parent.nodeType() === Node.DOCUMENT_NODE)
351 return new WebInspector.DOMNodePathStep(nodeName, true);

358 {
359 var classAttribute = node.getAttribute("class");
360 if (!classAttribute)
361 return [];
362
363 return classAttribute.split(/\s+/g).filter(Boolean).map(function(name) {

378 /**
379 * @param {string} ident
380 * @return {string}
381 */
382 function escapeIdentifierIfNeeded(ident)
383 {
384 if (isCSSIdentifier(ident))
385 return ident;
386 var shouldEscapeFirst = /^(?:[0-9]|-[0-9-]?)/.test(ident);
387 var lastIndex = ident.length - 1;
388 return ident.replace(/./g, function(c, i) {
389 return ((shouldEscapeFirst && i === 0) || !isCSSIdentChar(c)) ? escapeAsciiChar(c, i === lastIndex) : c;
390 });
391 }

393 /**
394 * @param {string} c
395 * @param {boolean} isLast
396 * @return {string}
397 */
398 function escapeAsciiChar(c, isLast)
399 {
400 return "\\" + toHexByte(c) + (isLast ? "" : " ");
401 }

403 /**
404 * @param {string} c
405 */
406 function toHexByte(c)
407 {
408 var hexByte = c.charCodeAt(0).toString(16);
409 if (hexByte.length === 1)
410 hexByte = "0" + hexByte;
411 return hexByte;
412 }

414 /**
415 * @param {string} c
416 * @return {boolean}
417 */
418 function isCSSIdentChar(c)
419 {
420 if (/[a-zA-Z0-9_-]/.test(c))
421 return true;
422 return c.charCodeAt(0) >= 0xA0;
423 }

425 /**
426 * @param {string} value
427 * @return {boolean}
428 */
429 function isCSSIdentifier(value)
430 {
431 return /^-?[a-zA-Z_][a-zA-Z0-9_-]*$/.test(value);
432 }
433
434 var prefixedOwnClassNamesArray = prefixedElementClassNames(node);
435 var needsClassNames = false;
436 var needsNthChild = false;
437 var ownIndex = -1;
438 var elementIndex = -1;
439 var siblings = parent.children();
440 for (var i = 0; (ownIndex === -1 || !needsNthChild) && i < siblings.length; ++i) {
441 var sibling = siblings[i];
442 if (sibling.nodeType() !== Node.ELEMENT_NODE)
443 continue;
444 elementIndex += 1;
445 if (sibling === node) {
446 ownIndex = elementIndex;
447 continue;
448 }
449 if (needsNthChild)
450 continue;
451 if (sibling.nodeNameInCorrectCase() !== nodeName)
452 continue;
453
454 needsClassNames = true;
455 var ownClassNames = new Set(prefixedOwnClassNamesArray);
456 if (!ownClassNames.size) {
457 needsNthChild = true;
458 continue;
459 }
460 var siblingClassNamesArray = prefixedElementClassNames(sibling);
461 for (var j = 0; j < siblingClassNamesArray.length; ++j) {
462 var siblingClass = siblingClassNamesArray[j];
463 if (!ownClassNames.has(siblingClass))
464 continue;
465 ownClassNames.delete(siblingClass);
466 if (!ownClassNames.size) {
467 needsNthChild = true;
468 break;
469 }
470 }
471 }
472
473 var result = nodeName;
474 if (isTargetNode && nodeName.toLowerCase() === "input" && node.getAttribute("type") && !node.getAttribute("id") && !node.getAttribute("class"))
475 result += "[type=\"" + node.getAttribute("type") + "\"]";
476 if (needsNthChild) {
477 result += ":nth-child(" + (ownIndex + 1) + ")";
478 } else if (needsClassNames) {
479 for (var prefixedName of prefixedOwnClassNamesArray)
480 result += "." + escapeIdentifierIfNeeded(prefixedName.substr(1));
481 }
482
483 return new WebInspector.DOMNodePathStep(result, false);
484}

491WebInspector.DOMPresentationUtils.xPath = function(node, optimized)
492{
493 if (node.nodeType() === Node.DOCUMENT_NODE)
494 return "/";
495
496 var steps = [];
497 var contextNode = node;
498 while (contextNode) {
499 var step = WebInspector.DOMPresentationUtils._xPathValue(contextNode, optimized);
500 if (!step)
501 break; // Error - bail out early.
502 steps.push(step);
503 if (step.optimized)
504 break;
505 contextNode = contextNode.parentNode;
506 }
507
508 steps.reverse();
509 return (steps.length && steps[0].optimized ? "" : "/") + steps.join("/");
510}

512/**
513 * @param {!WebInspector.DOMNode} node
514 * @param {boolean=} optimized
515 * @return {?WebInspector.DOMNodePathStep}
516 */
517WebInspector.DOMPresentationUtils._xPathValue = function(node, optimized)
518{
519 var ownValue;
520 var ownIndex = WebInspector.DOMPresentationUtils._xPathIndex(node);
521 if (ownIndex === -1)
522 return null; // Error.
523
524 switch (node.nodeType()) {
525 case Node.ELEMENT_NODE:
526 if (optimized && node.getAttribute("id"))
527 return new WebInspector.DOMNodePathStep("//*[@id=\"" + node.getAttribute("id") + "\"]", true);
528 ownValue = node.localName();
529 break;
530 case Node.ATTRIBUTE_NODE:
531 ownValue = "@" + node.nodeName();
532 break;
533 case Node.TEXT_NODE:
534 case Node.CDATA_SECTION_NODE:
535 ownValue = "text()";
536 break;
537 case Node.PROCESSING_INSTRUCTION_NODE:
538 ownValue = "processing-instruction()";
539 break;
540 case Node.COMMENT_NODE:
541 ownValue = "comment()";
542 break;
543 case Node.DOCUMENT_NODE:
544 ownValue = "";
545 break;
546 default:
547 ownValue = "";
548 break;
549 }
550
551 if (ownIndex > 0)
552 ownValue += "[" + ownIndex + "]";
553
554 return new WebInspector.DOMNodePathStep(ownValue, node.nodeType() === Node.DOCUMENT_NODE);
555}

557/**
558 * @param {!WebInspector.DOMNode} node
559 * @return {number}
560 */
561WebInspector.DOMPresentationUtils._xPathIndex = function(node)
562{
563 // Returns -1 in case of error, 0 if no siblings matching the same expression, <XPath index among the same expression-matching sibling nodes> otherwise.
564 function areNodesSimilar(left, right)
565 {
566 if (left === right)
567 return true;
568
569 if (left.nodeType() === Node.ELEMENT_NODE && right.nodeType() === Node.ELEMENT_NODE)
570 return left.localName() === right.localName();
571
572 if (left.nodeType() === right.nodeType())
573 return true;
574
575 // XPath treats CDATA as text nodes.
576 var leftType = left.nodeType() === Node.CDATA_SECTION_NODE ? Node.TEXT_NODE : left.nodeType();
577 var rightType = right.nodeType() === Node.CDATA_SECTION_NODE ? Node.TEXT_NODE : right.nodeType();
578 return leftType === rightType;
579 }
580
581 var siblings = node.parentNode ? node.parentNode.children() : null;
582 if (!siblings)
583 return 0; // Root node - no siblings.
584 var hasSameNamedElements;
585 for (var i = 0; i < siblings.length; ++i) {
586 if (areNodesSimilar(node, siblings[i]) && siblings[i] !== node) {
587 hasSameNamedElements = true;
588 break;
589 }
590 }
591 if (!hasSameNamedElements)
592 return 0;
593 var ownIndex = 1; // XPath indices start with 1.
594 for (var i = 0; i < siblings.length; ++i) {
595 if (areNodesSimilar(node, siblings[i])) {
596 if (siblings[i] === node)
597 return ownIndex;
598 ++ownIndex;
599 }
600 }
601 return -1; // An error occurred: |node| not found in parent's children.
602}
