150 * @param {boolean} showBlocking
151 */
152 showFrameworkListeners: function(showFramework, showPassive, showBlocking)
153 {
154 var eventTypes = this._treeOutline.rootElement().children();
155 for (var eventType of eventTypes) {
156 var hiddenEventType = true;
157 for (var listenerElement of eventType.children()) {
158 var listenerType = listenerElement.eventListener().listenerType();
159 var hidden = false;
160 if (listenerType === "frameworkUser" && !showFramework)
161 hidden = true;
162 if (listenerType === "frameworkInternal" && showFramework)
163 hidden = true;
164 if (!showPassive && listenerElement.eventListener().passive())
165 hidden = true;
166 if (!showBlocking && !listenerElement.eventListener().passive())
167 hidden = true;
168 listenerElement.hidden = hidden;
169 hiddenEventType = hiddenEventType && hidden;
170 }
171 eventType.hidden = hiddenEventType;
