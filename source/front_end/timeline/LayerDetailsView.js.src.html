123 /**

127 _onScrollRectClicked: function(index, event)

131 this._layerViewHost.selectObject(new WebInspector.LayerView.ScrollRectSelection(this._selection.layer(), index));

134 _onPaintProfilerButtonClicked: function()
135 {
136 var traceEvent = this._selection.type() === WebInspector.LayerView.Selection.Type.Tile ? /** @type {!WebInspector.LayerView.TileSelection} */ (this._selection).traceEvent() : null;
137 this.dispatchEventToListeners(WebInspector.LayerDetailsView.Events.PaintProfilerRequested, traceEvent);
138 },
139
140 /**
141 * @param {!LayerTreeAgent.ScrollRect} scrollRect
142 * @param {number} index
143 */
144 _createScrollRectElement: function(scrollRect, index)
145 {
146 if (index)
147 this._scrollRectsCell.createTextChild(", ");
148 var element = this._scrollRectsCell.createChild("span", "scroll-rect");
149 if (this._selection.scrollRectIndex === index)
150 element.classList.add("active");
151 element.textContent = WebInspector.LayerTreeModel.ScrollRectType[scrollRect.type].description + " (" + scrollRect.rect.x + ", " + scrollRect.rect.y +
152 ", " + scrollRect.rect.width + ", " + scrollRect.rect.height + ")";
153 element.addEventListener("click", this._onScrollRectClicked.bind(this, index), false);
154 },
155
156 update: function()
157 {
158 var layer = this._selection && this._selection.layer();
159 if (!layer) {
160 this._tableElement.remove();
161 this._paintProfilerButton.remove();
162 this._emptyWidget.show(this.element);
163 return;
164 }
165 this._emptyWidget.detach();
166 this.element.appendChild(this._tableElement);
167 this.element.appendChild(this._paintProfilerButton);
168 this._sizeCell.textContent = WebInspector.UIString("%d × %d (at %d,%d)", layer.width(), layer.height(), layer.offsetX(), layer.offsetY());
169 this._paintCountCell.parentElement.classList.toggle("hidden", !layer.paintCount());
170 this._paintCountCell.textContent = layer.paintCount();
171 this._memoryEstimateCell.textContent = Number.bytesToString(layer.gpuMemoryUsage());
172 layer.requestCompositingReasons(this._updateCompositingReasons.bind(this));
173 this._scrollRectsCell.removeChildren();
174 layer.scrollRects().forEach(this._createScrollRectElement.bind(this));
175 var traceEvent = this._selection.type() === WebInspector.LayerView.Selection.Type.Tile ? /** @type {!WebInspector.LayerView.TileSelection} */ (this._selection).traceEvent() : null;
176 this._paintProfilerButton.classList.toggle("hidden", !traceEvent);
177 },
178
179 _buildContent: function()
180 {
181 this._tableElement = this.element.createChild("table");
182 this._tbodyElement = this._tableElement.createChild("tbody");
183 this._sizeCell = this._createRow(WebInspector.UIString("Size"));
184 this._compositingReasonsCell = this._createRow(WebInspector.UIString("Compositing Reasons"));
185 this._memoryEstimateCell = this._createRow(WebInspector.UIString("Memory estimate"));
186 this._paintCountCell = this._createRow(WebInspector.UIString("Paint count"));
187 this._scrollRectsCell = this._createRow(WebInspector.UIString("Slow scroll regions"));
188 this._paintProfilerButton = this.element.createChild("a", "hidden link");
189 this._paintProfilerButton.textContent = WebInspector.UIString("Paint Profiler");
190 this._paintProfilerButton.addEventListener("click", this._onPaintProfilerButtonClicked.bind(this));
191 },
192
193 /**
194 * @param {string} title
195 */
196 _createRow: function(title)
197 {
198 var tr = this._tbodyElement.createChild("tr");
199 var titleCell = tr.createChild("td");
200 titleCell.textContent = title;
201 return tr.createChild("td");
202 },

204 /**
205 * @param {!Array.<string>} compositingReasons
206 */
207 _updateCompositingReasons: function(compositingReasons)
208 {
209 if (!compositingReasons || !compositingReasons.length) {
210 this._compositingReasonsCell.textContent = "n/a";
211 return;
212 }
213 this._compositingReasonsCell.removeChildren();
214 var list = this._compositingReasonsCell.createChild("ul");
215 for (var i = 0; i < compositingReasons.length; ++i) {
216 var text = WebInspector.LayerDetailsView.CompositingReasonDetail[compositingReasons[i]] || compositingReasons[i];
217 // If the text is more than one word but does not terminate with period, add the period.
218 if (/\s.*[^.]$/.test(text))
219 text += ".";
220 list.createChild("li").textContent = text;
221 }
