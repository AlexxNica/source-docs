264 /**
265 * @param {boolean} force
266 */
267 _updateData: function(force)
268 {
269 var key = this._parseKey(this._keyInputElement.value);
270 var pageSize = this._pageSize;
271 var skipCount = this._skipCount;
272 this._refreshButton.setEnabled(false);
273 this._clearButton.setEnabled(!this._isIndex);
274
275 if (!force && this._lastKey === key && this._lastPageSize === pageSize && this._lastSkipCount === skipCount)
276 return;
277
278 if (this._lastKey !== key || this._lastPageSize !== pageSize) {
279 skipCount = 0;
280 this._skipCount = 0;
281 }
282 this._lastKey = key;
283 this._lastPageSize = pageSize;
284 this._lastSkipCount = skipCount;

286 /**
287 * @param {!Array.<!WebInspector.IndexedDBModel.Entry>} entries
288 * @param {boolean} hasMore
289 * @this {WebInspector.IDBDataView}
290 */
291 function callback(entries, hasMore)
292 {
293 this._refreshButton.setEnabled(true);
294 this.clear();
295 this._entries = entries;
296 for (var i = 0; i < entries.length; ++i) {
297 var data = {};
298 data["number"] = i + skipCount;
299 data["key"] = entries[i].key;
300 data["primaryKey"] = entries[i].primaryKey;
301 data["value"] = entries[i].value;
302
303 var node = new WebInspector.IDBDataGridNode(data);
304 this._dataGrid.rootNode().appendChild(node);
305 }
306
307 this._pageBackButton.setEnabled(!!skipCount);
308 this._pageForwardButton.setEnabled(hasMore);
309 }
310
311 var idbKeyRange = key ? window.IDBKeyRange.lowerBound(key) : null;
312 if (this._isIndex)
313 this._model.loadIndexData(this._databaseId, this._objectStore.name, this._index.name, idbKeyRange, skipCount, pageSize, callback.bind(this));
314 else
315 this._model.loadObjectStoreData(this._databaseId, this._objectStore.name, idbKeyRange, skipCount, pageSize, callback.bind(this));
316 },
