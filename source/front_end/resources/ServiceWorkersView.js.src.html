190 {
191 var fingerprint = this._registration.fingerprint();
192 if (fingerprint === this._fingerprint)

194 this._fingerprint = fingerprint;
195
196 this._toolbar.setEnabled(!this._registration.isDeleted);

198 var versions = this._registration.versionsByMode();
199 var title = this._registration.isDeleted ? WebInspector.UIString("%s - deleted", this._registration.scopeURL) : this._registration.scopeURL;
200 this._section.setTitle(title);
201
202 var active = versions.get(WebInspector.ServiceWorkerVersion.Modes.Active);
203 var waiting = versions.get(WebInspector.ServiceWorkerVersion.Modes.Waiting);
204 var installing = versions.get(WebInspector.ServiceWorkerVersion.Modes.Installing);
205
206 var statusValue = this._wrapWidget(this._section.appendField(WebInspector.UIString("Status")));
207 statusValue.removeChildren();
208 var versionsStack = statusValue.createChild("div", "service-worker-version-stack");
209 versionsStack.createChild("div", "service-worker-version-stack-bar");
210
211 if (active) {
212 var scriptElement = this._section.appendField(WebInspector.UIString("Source"));
213 scriptElement.removeChildren();
214 var fileName = WebInspector.ParsedURL.extractName(active.scriptURL);
215 scriptElement.appendChild(WebInspector.linkifyURLAsNode(active.scriptURL, fileName));
216 scriptElement.createChild("div", "report-field-value-subtitle").textContent = WebInspector.UIString("Received %s", new Date(active.scriptResponseTime * 1000).toLocaleString());
217
218 var activeEntry = versionsStack.createChild("div", "service-worker-version");
219 activeEntry.createChild("div", "service-worker-active-circle");
220 activeEntry.createChild("span").textContent = WebInspector.UIString("#%s activated and is %s", active.id, active.runningStatus);
221
222 if (active.isRunning() || active.isStarting()) {
223 createLink(activeEntry, WebInspector.UIString("stop"), this._stopButtonClicked.bind(this, active.id));
224 if (!this._manager.targetForVersionId(active.id))
225 createLink(activeEntry, WebInspector.UIString("inspect"), this._inspectButtonClicked.bind(this, active.id));
226 } else if (active.isStartable()) {
227 createLink(activeEntry, WebInspector.UIString("start"), this._startButtonClicked.bind(this));
228 }
229
230 var clientsList = this._wrapWidget(this._section.appendField(WebInspector.UIString("Clients")));
231 clientsList.removeChildren();
232 this._section.setFieldVisible(WebInspector.UIString("Clients"), active.controlledClients.length);

235 if (this._clientInfoCache.has(client))
236 this._updateClientInfo(clientLabelText, /** @type {!WebInspector.TargetInfo} */(this._clientInfoCache.get(client)));
237 this._manager.getTargetInfo(client, this._onClientInfo.bind(this, clientLabelText));
238 }

243 waitingEntry.createChild("div", "service-worker-waiting-circle");
244 waitingEntry.createChild("span").textContent = WebInspector.UIString("#%s waiting to activate", waiting.id);
245 createLink(waitingEntry, WebInspector.UIString("skipWaiting"), this._skipButtonClicked.bind(this));
246 waitingEntry.createChild("div", "service-worker-subtitle").textContent = new Date(waiting.scriptResponseTime * 1000).toLocaleString();
247 if (!this._manager.targetForVersionId(waiting.id) && (waiting.isRunning() || waiting.isStarting()))
248 createLink(waitingEntry, WebInspector.UIString("inspect"), this._inspectButtonClicked.bind(this, waiting.id));
249 }
250 if (installing) {

255 if (!this._manager.targetForVersionId(installing.id) && (installing.isRunning() || installing.isStarting()))
