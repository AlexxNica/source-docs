275 _renderStatusCell: function(cell)
276 {
277 cell.classList.toggle("network-dim-cell", !this._isFailed() && (this._request.cached() || !this._request.statusCode));
278
279 if (this._request.failed && !this._request.canceled && !this._request.wasBlocked()) {
280 var failText = WebInspector.UIString("(failed)");
281 if (this._request.localizedFailDescription) {
282 cell.createTextChild(failText);
283 this._appendSubtitle(cell, this._request.localizedFailDescription);
284 cell.title = failText + " " + this._request.localizedFailDescription;
285 } else
286 cell.setTextAndTitle(failText);
287 } else if (this._request.statusCode) {
288 cell.createTextChild("" + this._request.statusCode);
289 this._appendSubtitle(cell, this._request.statusText);
290 cell.title = this._request.statusCode + " " + this._request.statusText;
291 } else if (this._request.parsedURL.isDataURL()) {
292 cell.setTextAndTitle(WebInspector.UIString("(data)"));
293 } else if (this._request.canceled) {
294 cell.setTextAndTitle(WebInspector.UIString("(canceled)"));
295 } else if (this._request.wasBlocked()) {
296 var reason = WebInspector.UIString("other");
297 switch (this._request.blockedReason()) {
298 case NetworkAgent.BlockedReason.Csp:
299 reason = WebInspector.UIString("csp");
300 break;
301 case NetworkAgent.BlockedReason.MixedContent:
302 reason = WebInspector.UIString("mixed-content");
303 break;
304 case NetworkAgent.BlockedReason.Origin:
305 reason = WebInspector.UIString("origin");
306 break;
307 case NetworkAgent.BlockedReason.Inspector:
308 reason = WebInspector.UIString("devtools");
309 break;
310 case NetworkAgent.BlockedReason.Other:
311 reason = WebInspector.UIString("other");
312 break;
313 }
314 cell.setTextAndTitle(WebInspector.UIString("(blocked:%s)", reason));
315 } else if (this._request.finished) {
316 cell.setTextAndTitle(WebInspector.UIString("Finished"));
317 } else {
318 cell.setTextAndTitle(WebInspector.UIString("(pending)"));
319 }

333 _renderInitiatorCell: function(cell)
334 {
335 this._initiatorCell = cell;
336 var request = this._request;
337 var initiator = request.initiatorInfo();
338
339 if (request.timing && request.timing.pushStart)
340 cell.appendChild(createTextNode(WebInspector.UIString("Push / ")));
341 switch (initiator.type) {
342 case WebInspector.NetworkRequest.InitiatorType.Parser:
343 cell.title = initiator.url + ":" + (initiator.lineNumber + 1);
344 var uiSourceCode = WebInspector.networkMapping.uiSourceCodeForURLForAnyTarget(initiator.url);
345 cell.appendChild(WebInspector.linkifyResourceAsNode(initiator.url, initiator.lineNumber, initiator.columnNumber, undefined, undefined, uiSourceCode ? uiSourceCode.displayName() : undefined));
346 this._appendSubtitle(cell, WebInspector.UIString("Parser"));
347 break;
348
349 case WebInspector.NetworkRequest.InitiatorType.Redirect:
350 cell.title = initiator.url;
351 console.assert(request.redirectSource);
352 var redirectSource = /** @type {!WebInspector.NetworkRequest} */ (request.redirectSource);
353 cell.appendChild(WebInspector.linkifyRequestAsNode(redirectSource));
354 this._appendSubtitle(cell, WebInspector.UIString("Redirect"));
355 break;
356
357 case WebInspector.NetworkRequest.InitiatorType.Script:
358 if (!this._linkifiedInitiatorAnchor) {
359 this._linkifiedInitiatorAnchor = this._parentView.linkifier.linkifyScriptLocation(request.target(), initiator.scriptId, initiator.url, initiator.lineNumber, initiator.columnNumber);
360 this._linkifiedInitiatorAnchor.title = "";
361 }
362 cell.appendChild(this._linkifiedInitiatorAnchor);
363 this._appendSubtitle(cell, WebInspector.UIString("Script"));
364 cell.classList.add("network-script-initiated");
365 cell.request = request;
366 break;
367
368 default:
369 cell.title = WebInspector.UIString("Other");
370 cell.classList.add("network-dim-cell");
371 cell.appendChild(createTextNode(WebInspector.UIString("Other")));
372 }

432 _updateTimingGraph: function()
433 {
434 var calculator = this._parentView.calculator();
435 var timeRanges = WebInspector.RequestTimingView.calculateRequestTimeRanges(this._request, calculator.minimumBoundary());
436 var right = timeRanges[0].end;
437
438 var container = this._barAreaElement;
439 var nextBar = container.firstChild;
440 for (var i = 0; i < timeRanges.length; ++i) {
441 var range = timeRanges[i];
442 var start = calculator.computePercentageFromEventTime(range.start);
443 var end = (range.end !== Number.MAX_VALUE) ? calculator.computePercentageFromEventTime(range.end) : 100;
444 if (!nextBar)
445 nextBar = container.createChild("div");
446 nextBar.className = "network-graph-bar request-timing";
447 nextBar.classList.add(range.name);
448 nextBar.style.setProperty("left", start + "%");
449 nextBar.style.setProperty("right", (100 - end) + "%");
450 nextBar = nextBar.nextSibling;
451 }
452 while (nextBar) {
453 var nextSibling = nextBar.nextSibling;
454 nextBar.remove();
455 nextBar = nextSibling;
456 }

459 _updateGraph: function()
460 {
461 this._staleGraph = false;
462 if (!this._timelineCell)
463 return;
464 if (this._showTiming) {
465 this._updateTimingGraph();
466 return;
467 }
468
469 var calculator = this._parentView.calculator();
470 var percentages = calculator.computeBarGraphPercentages(this._request);
471 this._percentages = percentages;
472
473 this._barAreaElement.classList.remove("hidden");
474
475 this._barLeftElement.style.setProperty("left", percentages.start + "%");
476 this._barLeftElement.style.setProperty("right", (100 - percentages.middle) + "%");
477
478 this._barRightElement.style.setProperty("left", percentages.middle + "%");
479 this._barRightElement.style.setProperty("right", (100 - percentages.end) + "%");
480
481 var labels = calculator.computeBarGraphLabels(this._request);
482 this._labelLeftElement.textContent = labels.left;
483 this._labelRightElement.textContent = labels.right;
484
485 var tooltip = (labels.tooltip || "");
486 this._barLeftElement.title = tooltip;
487 this._labelLeftElement.title = tooltip;
488 this._labelRightElement.title = tooltip;
489 this._barRightElement.title = tooltip;
490
491 if (this._parentView[WebInspector.NetworkDataGridNode._hoveredRowSymbol] === this)
492 this._refreshLabelPositions();
493 },
494
495 _refreshLabelPositions: function()
496 {
497 if (!this._percentages)
498 return;
499 this._labelLeftElement.style.removeProperty("left");
500 this._labelLeftElement.style.removeProperty("right");
501 this._labelLeftElement.classList.remove("before");
502 this._labelLeftElement.classList.remove("hidden");
503
504 this._labelRightElement.style.removeProperty("left");
505 this._labelRightElement.style.removeProperty("right");
506 this._labelRightElement.classList.remove("after");
507 this._labelRightElement.classList.remove("hidden");
508
509 const labelPadding = 10;
510 const barRightElementOffsetWidth = this._barRightElement.offsetWidth;
511 const barLeftElementOffsetWidth = this._barLeftElement.offsetWidth;
512
513 if (this._barLeftElement) {
514 var leftBarWidth = barLeftElementOffsetWidth - labelPadding;
515 var rightBarWidth = (barRightElementOffsetWidth - barLeftElementOffsetWidth) - labelPadding;
516 } else {
517 var leftBarWidth = (barLeftElementOffsetWidth - barRightElementOffsetWidth) - labelPadding;
518 var rightBarWidth = barRightElementOffsetWidth - labelPadding;
519 }
520
521 const labelLeftElementOffsetWidth = this._labelLeftElement.offsetWidth;
522 const labelRightElementOffsetWidth = this._labelRightElement.offsetWidth;
523
524 const labelBefore = (labelLeftElementOffsetWidth > leftBarWidth);
525 const labelAfter = (labelRightElementOffsetWidth > rightBarWidth);
526 const graphElementOffsetWidth = this._timelineCell.offsetWidth;
527
528 if (labelBefore && (graphElementOffsetWidth * (this._percentages.start / 100)) < (labelLeftElementOffsetWidth + 10))
529 var leftHidden = true;
530
531 if (labelAfter && (graphElementOffsetWidth * ((100 - this._percentages.end) / 100)) < (labelRightElementOffsetWidth + 10))
532 var rightHidden = true;
533
534 if (barLeftElementOffsetWidth === barRightElementOffsetWidth) {
535 // The left/right label data are the same, so a before/after label can be replaced by an on-bar label.
536 if (labelBefore && !labelAfter)
537 leftHidden = true;
538 else if (labelAfter && !labelBefore)
539 rightHidden = true;
540 }
541
542 if (labelBefore) {
543 if (leftHidden)
544 this._labelLeftElement.classList.add("hidden");
545 this._labelLeftElement.style.setProperty("right", (100 - this._percentages.start) + "%");
546 this._labelLeftElement.classList.add("before");
547 } else {
548 this._labelLeftElement.style.setProperty("left", this._percentages.start + "%");
549 this._labelLeftElement.style.setProperty("right", (100 - this._percentages.middle) + "%");
550 }
551
552 if (labelAfter) {
553 if (rightHidden)
554 this._labelRightElement.classList.add("hidden");
555 this._labelRightElement.style.setProperty("left", this._percentages.end + "%");
556 this._labelRightElement.classList.add("after");
557 } else {
558 this._labelRightElement.style.setProperty("left", this._percentages.middle + "%");
559 this._labelRightElement.style.setProperty("right", (100 - this._percentages.end) + "%");
560 }

635{
636 var aInitiator = a._request.initiatorInfo();
637 var bInitiator = b._request.initiatorInfo();
638
639 if (aInitiator.type < bInitiator.type)
640 return -1;
641 if (aInitiator.type > bInitiator.type)
642 return 1;
643
644 if (typeof aInitiator.__source === "undefined")
645 aInitiator.__source = WebInspector.displayNameForURL(aInitiator.url);
646 if (typeof bInitiator.__source === "undefined")
647 bInitiator.__source = WebInspector.displayNameForURL(bInitiator.url);
648
649 if (aInitiator.__source < bInitiator.__source)
650 return -1;
651 if (aInitiator.__source > bInitiator.__source)
652 return 1;
653
654 if (aInitiator.lineNumber < bInitiator.lineNumber)
655 return -1;
656 if (aInitiator.lineNumber > bInitiator.lineNumber)
657 return 1;
658
659 if (aInitiator.columnNumber < bInitiator.columnNumber)
660 return -1;
661 if (aInitiator.columnNumber > bInitiator.columnNumber)
662 return 1;
663
664 return a._request.indentityCompare(b._request);
665}
