11{

13
14 this._mainView = new WebInspector.SecurityMainView(this);

17 this._sidebarTree = new WebInspector.SecurityPanelSidebarTree(this._sidebarMainViewElement, this.showOrigin.bind(this));
18 this.panelSidebarElement().appendChild(this._sidebarTree.element);

26
27 /** @type {!Map<!WebInspector.NetworkLogView.MixedContentFilterValues, number>} */

139 /**
140 * @param {!WebInspector.NetworkRequest} request
141 */
142 _processRequest: function(request)
143 {
144 var origin = WebInspector.ParsedURL.extractOrigin(request.url);
145
146 if (!origin) {
147 // We don't handle resources like data: URIs. Most of them don't affect the lock icon.
148 return;
149 }
150
151 var securityState = /** @type {!SecurityAgent.SecurityState} */ (request.securityState());
152
153 if (request.mixedContentType === NetworkAgent.RequestMixedContentType.Blockable && this._ranInsecureContentStyle)
154 securityState = this._ranInsecureContentStyle;
155 else if (request.mixedContentType === NetworkAgent.RequestMixedContentType.OptionallyBlockable && this._displayedInsecureContentStyle)
156 securityState = this._displayedInsecureContentStyle;
157
158 if (this._origins.has(origin)) {
159 var originState = this._origins.get(origin);
160 var oldSecurityState = originState.securityState;
161 originState.securityState = this._securityStateMin(oldSecurityState, securityState);
162 if (oldSecurityState !== originState.securityState) {
163 this._sidebarTree.updateOrigin(origin, securityState);
164 if (originState.originView)
165 originState.originView.setSecurityState(securityState);
166 }
167 } else {
168 // TODO(lgarron): Store a (deduplicated) list of different security details we have seen. https://crbug.com/503170
169 var originState = {};
170 originState.securityState = securityState;
171
172 var securityDetails = request.securityDetails();
173 if (securityDetails) {
174 originState.securityDetails = securityDetails;
175 originState.certificateDetailsPromise = request.networkManager().certificateDetailsPromise(securityDetails.certificateId);
176 }
177
178 this._origins.set(origin, originState);
179
180 this._sidebarTree.addOrigin(origin, securityState);
181
182 // Don't construct the origin view yet (let it happen lazily).
183 }
184 },
185
186 /**
187 * @param {!WebInspector.Event} event
188 */
189 _onRequestFinished: function(event)
190 {
191 var request = /** @type {!WebInspector.NetworkRequest} */ (event.data);
192 this._updateFilterRequestCounts(request);
193 this._processRequest(request);
194 },
195
196 /**
197 * @param {!WebInspector.NetworkRequest} request
198 */
199 _updateFilterRequestCounts: function(request)
200 {
201 if (request.mixedContentType === NetworkAgent.RequestMixedContentType.None)
202 return;
203
204 /** @type {!WebInspector.NetworkLogView.MixedContentFilterValues} */
205 var filterKey = WebInspector.NetworkLogView.MixedContentFilterValues.All;
206 if (request.wasBlocked())
207 filterKey = WebInspector.NetworkLogView.MixedContentFilterValues.Blocked;
208 else if (request.mixedContentType === NetworkAgent.RequestMixedContentType.Blockable)
209 filterKey = WebInspector.NetworkLogView.MixedContentFilterValues.BlockOverridden;
210 else if (request.mixedContentType === NetworkAgent.RequestMixedContentType.OptionallyBlockable)
211 filterKey = WebInspector.NetworkLogView.MixedContentFilterValues.Displayed;
212
213 if (!this._filterRequestCounts.has(filterKey))
214 this._filterRequestCounts.set(filterKey, 1);
215 else
216 this._filterRequestCounts.set(filterKey, this._filterRequestCounts.get(filterKey) + 1);
217
218 this._mainView.refreshExplanations();
219 },
220
221 /**
222 * @param {!WebInspector.NetworkLogView.MixedContentFilterValues} filterKey
223 * @return {number}
224 */
225 filterRequestCount: function(filterKey)
226 {
227 return this._filterRequestCounts.get(filterKey) || 0;
228 },
229
230 /**
231 * @param {!SecurityAgent.SecurityState} stateA
232 * @param {!SecurityAgent.SecurityState} stateB
233 * @return {!SecurityAgent.SecurityState}
234 */
235 _securityStateMin: function(stateA, stateB)
236 {
237 return WebInspector.SecurityModel.SecurityStateComparator(stateA, stateB) < 0 ? stateA : stateB;

240 /**

258 networkManager.addEventListener(WebInspector.NetworkManager.EventTypes.ResponseReceived, this._onResponseReceived, this);
259 networkManager.addEventListener(WebInspector.NetworkManager.EventTypes.RequestFinished, this._onRequestFinished, this);
260
261 var securityModel = WebInspector.SecurityModel.fromTarget(target);
262 securityModel.addEventListener(WebInspector.SecurityModel.EventTypes.SecurityStateChanged, this._onSecurityStateChanged, this);
263 },

292 this._sidebarTree.setMainOrigin(origin);
293 this._processRequest(request);
294 }

345 this._showOriginInPanel = showOriginInPanel;
346
347 this._mainOrigin = null;

410
411 /**
412 * @param {!WebInspector.SecurityPanel.Origin} origin
413 * @param {!SecurityAgent.SecurityState} securityState
414 */
415 updateOrigin: function(origin, securityState)
416 {
417 var originElement = /** @type {!WebInspector.SecurityPanelSidebarTreeElement} */ (this._elementsByOrigin.get(origin));
418 originElement.setSecurityState(securityState);
419
420 var newParent;
421 if (origin === this._mainOrigin) {
422 newParent = this._originGroups.get(WebInspector.SecurityPanelSidebarTree.OriginGroupName.MainOrigin);
423 } else {
424 switch (securityState) {
425 case SecurityAgent.SecurityState.Secure:
426 newParent = this._originGroups.get(WebInspector.SecurityPanelSidebarTree.OriginGroupName.Secure);
427 break;
428 case SecurityAgent.SecurityState.Unknown:
429 newParent = this._originGroups.get(WebInspector.SecurityPanelSidebarTree.OriginGroupName.Unknown);
430 break;
431 default:
432 newParent = this._originGroups.get(WebInspector.SecurityPanelSidebarTree.OriginGroupName.NonSecure);
433 break;
434 }
435 }
436
437 var oldParent = originElement.parent;
438 if (oldParent !== newParent) {
439 if (oldParent) {
440 oldParent.removeChild(originElement);
441 if (oldParent.childCount() === 0)
442 oldParent.hidden = true;
443 }
444 newParent.appendChild(originElement);
445 newParent.hidden = false;
446 }
447
448 },
449
450 _clearOriginGroups: function()
451 {
452 for (var originGroup of this._originGroups.values()) {
453 originGroup.removeChildren();
454 originGroup.hidden = true;
455 }
456 this._originGroups.get(WebInspector.SecurityPanelSidebarTree.OriginGroupName.MainOrigin).hidden = false;
457 },
458
459 clearOrigins: function()
460 {
461 this._clearOriginGroups();
462 this._elementsByOrigin.clear();
463 },

602
603 /**
604 * @param {!SecurityAgent.SecurityState} newSecurityState
605 * @param {!Array<!SecurityAgent.SecurityStateExplanation>} explanations
606 * @param {?SecurityAgent.MixedContentStatus} mixedContentStatus
607 * @param {boolean} schemeIsCryptographic
608 */
609 updateSecurityState: function(newSecurityState, explanations, mixedContentStatus, schemeIsCryptographic)
610 {
611 // Remove old state.
612 // It's safe to call this even when this._securityState is undefined.
613 this._summarySection.classList.remove("security-summary-" + this._securityState);
614
615 // Add new state.
616 this._securityState = newSecurityState;
617 this._summarySection.classList.add("security-summary-" + this._securityState);
618 var summaryExplanationStrings = {
619 "unknown": WebInspector.UIString("The security of this page is unknown."),
620 "insecure": WebInspector.UIString("This page is insecure (broken HTTPS)."),
621 "neutral": WebInspector.UIString("This page is not secure."),
622 "secure": WebInspector.UIString("This page is secure (valid HTTPS).")
623 }
624 this._summaryText.textContent = summaryExplanationStrings[this._securityState];
625
626 this._explanations = explanations,
627 this._mixedContentStatus = mixedContentStatus;
628 this._schemeIsCryptographic = schemeIsCryptographic;
629
630 this._panel.setRanInsecureContentStyle(mixedContentStatus.ranInsecureContentStyle);
631 this._panel.setDisplayedInsecureContentStyle(mixedContentStatus.displayedInsecureContentStyle);
632
633 this.refreshExplanations();
634 },
635
636 refreshExplanations: function()
637 {
638 this._securityExplanationsMain.removeChildren();
639 this._securityExplanationsExtra.removeChildren();
640 for (var explanation of this._explanations) {
641 if (explanation.securityState === SecurityAgent.SecurityState.Info) {
642 this._addExplanation(this._securityExplanationsExtra, explanation);
643 } else {
644 this._addExplanation(this._securityExplanationsMain, explanation);
645 }
646 }
647
648 this._addMixedContentExplanations();
649 },
650
651 _addMixedContentExplanations: function()
652 {
653 if (!this._schemeIsCryptographic)
654 return;
655
656 if (this._mixedContentStatus && (this._mixedContentStatus.ranInsecureContent || this._mixedContentStatus.displayedInsecureContent)) {
657 if (this._mixedContentStatus.ranInsecureContent)
658 this._addMixedContentExplanation(this._securityExplanationsMain, this._mixedContentStatus.ranInsecureContentStyle, WebInspector.UIString("Active Mixed Content"), WebInspector.UIString("You have recently allowed insecure content (such as scripts or iframes) to run on this site."), WebInspector.NetworkLogView.MixedContentFilterValues.BlockOverridden, showBlockOverriddenMixedContentInNetworkPanel);
659 if (this._mixedContentStatus.displayedInsecureContent)
660 this._addMixedContentExplanation(this._securityExplanationsMain, this._mixedContentStatus.displayedInsecureContentStyle, WebInspector.UIString("Mixed Content"), WebInspector.UIString("The site includes HTTP resources."), WebInspector.NetworkLogView.MixedContentFilterValues.Displayed, showDisplayedMixedContentInNetworkPanel);
661 }
662
663 if (this._mixedContentStatus && (!this._mixedContentStatus.displayedInsecureContent && !this._mixedContentStatus.ranInsecureContent)) {
664 this._addExplanation(this._securityExplanationsMain, /** @type {!SecurityAgent.SecurityStateExplanation} */ ({
665 "securityState": SecurityAgent.SecurityState.Secure,
666 "summary": WebInspector.UIString("Secure Resources"),
667 "description": WebInspector.UIString("All resources on this page are served securely.")
668 }));
669 }
670
671 if (this._panel.filterRequestCount(WebInspector.NetworkLogView.MixedContentFilterValues.Blocked) > 0)
672 this._addMixedContentExplanation(this._securityExplanationsExtra, SecurityAgent.SecurityState.Info, WebInspector.UIString("Blocked mixed content"), WebInspector.UIString("Your page requested insecure resources that were blocked."), WebInspector.NetworkLogView.MixedContentFilterValues.Blocked, showBlockedMixedContentInNetworkPanel);
673

685 ]);
686 }

688 /**
