169 const base64Digits = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";

190WebInspector.TextSourceMap.load = function(sourceMapURL, compiledURL)
191{
192 var callback;
193 var promise = new Promise(fulfill => callback = fulfill);
194 WebInspector.multitargetNetworkManager.loadResource(sourceMapURL, contentLoaded);
195 return promise;

197 /**
198 * @param {number} statusCode
199 * @param {!Object.<string, string>} headers
200 * @param {string} content
201 */
202 function contentLoaded(statusCode, headers, content)
203 {
204 if (!content || statusCode >= 400) {
205 callback(null);
206 return;
207 }
208
209 if (content.slice(0, 3) === ")]}")
210 content = content.substring(content.indexOf("\n"));
211 try {
212 var payload = /** @type {!SourceMapV3} */ (JSON.parse(content));
213 var baseURL = sourceMapURL.startsWith("data:") ? compiledURL : sourceMapURL;
214
215 callback(new WebInspector.TextSourceMap(compiledURL, baseURL, payload));
216 } catch (e) {
217 console.error(e);
218 WebInspector.console.warn("DevTools failed to parse SourceMap: " + sourceMapURL);
219 callback(null);
220 }
221 }

309 * @override
310 * @param {number} lineNumber in compiled resource
311 * @param {number} columnNumber in compiled resource
312 * @return {?WebInspector.SourceMapEntry}
313 */
314 findEntry: function(lineNumber, columnNumber)
315 {
316 var first = 0;
317 var count = this._mappings.length;
318 while (count > 1) {
319 var step = count >> 1;
320 var middle = first + step;
321 var mapping = this._mappings[middle];
322 if (lineNumber < mapping.lineNumber || (lineNumber === mapping.lineNumber && columnNumber < mapping.columnNumber)) {
323 count = step;
324 } else {
325 first = middle;
326 count -= step;
327 }
328 }
329 var entry = this._mappings[first];
330 if (!first && entry && (lineNumber < entry.lineNumber || (lineNumber === entry.lineNumber && columnNumber < entry.columnNumber)))
331 return null;
332 return entry;

401 /**
402 * @param {!SourceMapV3} map
403 * @param {number} lineNumber
404 * @param {number} columnNumber
405 */
406 _parseMap: function(map, lineNumber, columnNumber)
407 {
408 var sourceIndex = 0;
409 var sourceLineNumber = 0;
410 var sourceColumnNumber = 0;
411 var nameIndex = 0;
412
413 var sources = [];
414 var names = map.names || [];
415 var sourceRoot = map.sourceRoot || "";
416 if (sourceRoot && !sourceRoot.endsWith("/"))
417 sourceRoot += "/";
418 for (var i = 0; i < map.sources.length; ++i) {
419 var href = sourceRoot + map.sources[i];
420 var url = WebInspector.ParsedURL.completeURL(this._sourceMappingURL, href) || href;
421 var hasSource = map.sourcesContent && map.sourcesContent[i];
422 if (url === this._compiledURL && hasSource)
423 url += WebInspector.UIString(" [sm]");
424 sources.push(url);
425 this._sources[url] = true;
426
427 if (hasSource)
428 this._sourceContentByURL[url] = map.sourcesContent[i];
429 }
430
431 var stringCharIterator = new WebInspector.TextSourceMap.StringCharIterator(map.mappings);
432 var sourceURL = sources[sourceIndex];
433
434 while (true) {
435 if (stringCharIterator.peek() === ",")
436 stringCharIterator.next();
437 else {
438 while (stringCharIterator.peek() === ";") {
439 lineNumber += 1;
440 columnNumber = 0;
441 stringCharIterator.next();
442 }
443 if (!stringCharIterator.hasNext())
444 break;
445 }
446
447 columnNumber += this._decodeVLQ(stringCharIterator);
448 if (!stringCharIterator.hasNext() || this._isSeparator(stringCharIterator.peek())) {
449 this._mappings.push(new WebInspector.SourceMapEntry(lineNumber, columnNumber));
450 continue;
451 }
452
453 var sourceIndexDelta = this._decodeVLQ(stringCharIterator);
454 if (sourceIndexDelta) {
455 sourceIndex += sourceIndexDelta;
456 sourceURL = sources[sourceIndex];
457 }
458 sourceLineNumber += this._decodeVLQ(stringCharIterator);
459 sourceColumnNumber += this._decodeVLQ(stringCharIterator);
460 if (!this._isSeparator(stringCharIterator.peek()))
461 nameIndex += this._decodeVLQ(stringCharIterator);
462
463 this._mappings.push(new WebInspector.SourceMapEntry(lineNumber, columnNumber, sourceURL, sourceLineNumber, sourceColumnNumber, names[nameIndex]));
464 }
465
466 for (var i = 0; i < this._mappings.length; ++i) {
467 var mapping = this._mappings[i];
468 var url = mapping.sourceURL;
469 if (!url)
470 continue;
471 if (!this._reverseMappingsBySourceURL.has(url))
472 this._reverseMappingsBySourceURL.set(url, []);
473 var reverseMappings = this._reverseMappingsBySourceURL.get(url);
474 reverseMappings.push(mapping);
475 }

487 /**
488 * @param {!WebInspector.TextSourceMap.StringCharIterator} stringCharIterator
489 * @return {number}
490 */
491 _decodeVLQ: function(stringCharIterator)
492 {
493 // Read unsigned value.
494 var result = 0;
495 var shift = 0;
496 do {
497 var digit = this._base64Map[stringCharIterator.next()];
498 result += (digit & this._VLQ_BASE_MASK) << shift;
499 shift += this._VLQ_BASE_SHIFT;
500 } while (digit & this._VLQ_CONTINUATION_MASK);
501
502 // Fix the sign.
503 var negative = result & 1;
504 result >>= 1;
505 return negative ? -result : result;

508 /**
509 * @param {string} url
510 * @param {!WebInspector.TextRange} textRange
511 * @return {!WebInspector.TextRange}
512 */
513 reverseMapTextRange: function(url, textRange)
514 {
515 /**
516 * @param {!{lineNumber: number, columnNumber: number}} position
517 * @param {!WebInspector.SourceMapEntry} mapping
518 * @return {number}
519 */
520 function comparator(position, mapping)
521 {
522 if (position.lineNumber !== mapping.sourceLineNumber)
523 return position.lineNumber - mapping.sourceLineNumber;
524
525 return position.columnNumber - mapping.sourceColumnNumber;
526 }
527
528 var mappings = this._reversedMappings(url);
529 var startIndex = mappings.lowerBound({lineNumber: textRange.startLine, columnNumber: textRange.startColumn}, comparator);
530 var endIndex = mappings.upperBound({lineNumber: textRange.endLine, columnNumber: textRange.endColumn}, comparator);
531
532 var startMapping = mappings[startIndex];
533 var endMapping = mappings[endIndex];
534 return new WebInspector.TextRange(startMapping.lineNumber, startMapping.columnNumber, endMapping.lineNumber, endMapping.columnNumber);
535 },

538 _VLQ_BASE_MASK: (1 << 5) - 1,
