76 /**
77 * @param {number=} usage
78 * @param {number=} eventCount
79 * @param {number=} percentFull
80 */
81 _bufferUsage: function(usage, eventCount, percentFull)
82 {
83 this._eventBufferSize = eventCount;
84 this._activeClient.tracingBufferUsage(usage || percentFull || 0);
85 },
86
87 /**
88 * @param {!Array.<!WebInspector.TracingManager.EventPayload>} events
89 */
90 _eventsCollected: function(events)
91 {
92 this._activeClient.traceEventsCollected(events);
93 this._eventsRetrieved += events.length;
94 if (!this._eventBufferSize)
95 return;
96 if (this._eventsRetrieved > this._eventBufferSize)
97 this._eventsRetrieved = this._eventBufferSize;
98 this._activeClient.eventsRetrievalProgress(this._eventsRetrieved / this._eventBufferSize);
99 },
100
101 _tracingComplete: function()
102 {
103 this._eventBufferSize = 0;
104 this._eventsRetrieved = 0;
105 this._activeClient.tracingComplete();
106 this._activeClient = null;
107 this._finishing = false;
108 },
109
110 /**
111 * @param {!WebInspector.TracingManagerClient} client
112 * @param {string} categoryFilter
113 * @param {string} options
114 * @param {function(?string)=} callback
115 */
116 start: function(client, categoryFilter, options, callback)
117 {
118 if (this._activeClient)
119 throw new Error("Tracing is already started");
120 var bufferUsageReportingIntervalMs = 500;
121 this._activeClient = client;
122 this._target.tracingAgent().start(categoryFilter, options, bufferUsageReportingIntervalMs, WebInspector.TracingManager.TransferMode.ReportEvents, callback);
123 this._activeClient.tracingStarted();
124 },
125
126 stop: function()
127 {
128 if (!this._activeClient)
129 throw new Error("Tracing is not started");
130 if (this._finishing)
131 throw new Error("Tracing is already being stopped");
132 this._finishing = true;
133 this._target.tracingAgent().end();
134 }
135}
