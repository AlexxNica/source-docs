27 /**
28 * @this {WebInspector.CSSMatchedStyles}
29 */
30 function addAttributesStyle()
31 {
32 if (!attributesPayload)
33 return;
34 var style = new WebInspector.CSSStyleDeclaration(cssModel, null, attributesPayload, WebInspector.CSSStyleDeclaration.Type.Attributes);
35 this._nodeForStyle.set(style, this._node);
36 this._nodeStyles.push(style);
37 }
38
39 // Inline style has the greatest specificity.
40 if (inlinePayload && this._node.nodeType() === Node.ELEMENT_NODE) {
41 var style = new WebInspector.CSSStyleDeclaration(cssModel, null, inlinePayload, WebInspector.CSSStyleDeclaration.Type.Inline);
42 this._nodeForStyle.set(style, this._node);
43 this._nodeStyles.push(style);
44 }
45
46 // Add rules in reverse order to match the cascade order.
47 var addedAttributesStyle;
48 for (var i = matchedPayload.length - 1; i >= 0; --i) {
49 var rule = new WebInspector.CSSStyleRule(cssModel, matchedPayload[i].rule);
50 if ((rule.isInjected() || rule.isUserAgent()) && !addedAttributesStyle) {
51 // Show element's Style Attributes after all author rules.
52 addedAttributesStyle = true;
53 addAttributesStyle.call(this);
54 }
55 this._nodeForStyle.set(rule.style, this._node);
56 this._nodeStyles.push(rule.style);
57 addMatchingSelectors.call(this, this._node, rule, matchedPayload[i].matchingSelectors);
58 }
59
60 if (!addedAttributesStyle)
61 addAttributesStyle.call(this);
62
63 // Walk the node structure and identify styles with inherited properties.
64 var parentNode = this._node.parentNode;
65 for (var i = 0; parentNode && inheritedPayload && i < inheritedPayload.length; ++i) {
66 var entryPayload = inheritedPayload[i];
67 var inheritedInlineStyle = entryPayload.inlineStyle ? new WebInspector.CSSStyleDeclaration(cssModel, null, entryPayload.inlineStyle, WebInspector.CSSStyleDeclaration.Type.Inline) : null;
68 if (inheritedInlineStyle && this._containsInherited(inheritedInlineStyle)) {
69 this._nodeForStyle.set(inheritedInlineStyle, parentNode);
70 this._nodeStyles.push(inheritedInlineStyle);
71 this._inheritedStyles.add(inheritedInlineStyle);
72 }
73
74 var inheritedMatchedCSSRules = entryPayload.matchedCSSRules || [];
75 for (var j = inheritedMatchedCSSRules.length - 1; j >= 0; --j) {
76 var inheritedRule = new WebInspector.CSSStyleRule(cssModel, inheritedMatchedCSSRules[j].rule);
77 addMatchingSelectors.call(this, parentNode, inheritedRule, inheritedMatchedCSSRules[j].matchingSelectors);
78 if (!this._containsInherited(inheritedRule.style))
79 continue;
80 this._nodeForStyle.set(inheritedRule.style, parentNode);
81 this._nodeStyles.push(inheritedRule.style);
82 this._inheritedStyles.add(inheritedRule.style);
83 }
84 parentNode = parentNode.parentNode;
85 }
86
87 // Set up pseudo styles map.
88 this._pseudoStyles = new Map();
89 if (pseudoPayload) {
90 for (var i = 0; i < pseudoPayload.length; ++i) {
91 var entryPayload = pseudoPayload[i];
92 // PseudoElement nodes are not created unless "content" css property is set.
93 var pseudoElement = this._node.pseudoElements().get(entryPayload.pseudoType) || null;
94 var pseudoStyles = [];
95 var rules = entryPayload.matches || [];
96 for (var j = rules.length - 1; j >= 0; --j) {
97 var pseudoRule = new WebInspector.CSSStyleRule(cssModel, rules[j].rule);
98 pseudoStyles.push(pseudoRule.style);
99 this._nodeForStyle.set(pseudoRule.style, pseudoElement);
100 if (pseudoElement)
101 addMatchingSelectors.call(this, pseudoElement, pseudoRule, rules[j].matchingSelectors);
102 }
103 this._pseudoStyles.set(entryPayload.pseudoType, pseudoStyles);
104 }
105 }
106
107 if (animationsPayload)
108 this._keyframes = animationsPayload.map(rule => new WebInspector.CSSKeyframesRule(cssModel, rule));
109
110 this.resetActiveProperties();
111
112 /**
113 * @param {!WebInspector.DOMNode} node
114 * @param {!WebInspector.CSSStyleRule} rule
115 * @param {!Array<number>} matchingSelectorIndices
116 * @this {WebInspector.CSSMatchedStyles}
117 */
118 function addMatchingSelectors(node, rule, matchingSelectorIndices)
119 {
120 for (var matchingSelectorIndex of matchingSelectorIndices) {
121 var selector = rule.selectors[matchingSelectorIndex];
122 this._setSelectorMatches(node, selector.text, true);
123 }
124 }
125}

159 {
160 var node = this.nodeForStyle(rule.style);
161 if (!node)
162 return [];
163 var map = this._matchingSelectors.get(node.id);
164 if (!map)
165 return [];
166 var result = [];

174 /**
175 * @param {!WebInspector.CSSStyleRule} rule
176 * @return {!Promise}
177 */
178 recomputeMatchingSelectors: function(rule)
179 {
180 var node = this.nodeForStyle(rule.style);
181 if (!node)
182 return Promise.resolve();
183 var promises = [];
184 for (var selector of rule.selectors)
185 promises.push(querySelector.call(this, node, selector.text));
186 return Promise.all(promises);

188 /**
189 * @param {!WebInspector.DOMNode} node
190 * @param {string} selectorText
191 * @return {!Promise}
192 * @this {WebInspector.CSSMatchedStyles}
193 */
194 function querySelector(node, selectorText)
195 {
196 var ownerDocument = node.ownerDocument || null;
197 // We assume that "matching" property does not ever change during the
198 // MatchedStyleResult's lifetime.
199 var map = this._matchingSelectors.get(node.id);
200 if ((map && map.has(selectorText)) || !ownerDocument)
201 return Promise.resolve();
202
203 var resolve;
204 var promise = new Promise(fulfill => resolve = fulfill);
205 this._node.domModel().querySelectorAll(ownerDocument.id, selectorText, onQueryComplete.bind(this, node, selectorText, resolve));
206 return promise;
207 }

342 /**
343 * @param {!Array<!WebInspector.CSSStyleDeclaration>} styles
344 * @param {!Map<!WebInspector.CSSProperty, !WebInspector.CSSMatchedStyles.PropertyState>} result
345 */
346 _computeActiveProperties: function(styles, result)
347 {
348 /** @type {!Set.<string>} */
349 var foundImportantProperties = new Set();
350 /** @type {!Map.<string, !Map<string, !WebInspector.CSSProperty>>} */
351 var propertyToEffectiveRule = new Map();
352 /** @type {!Map.<string, !WebInspector.DOMNode>} */
353 var inheritedPropertyToNode = new Map();
354 /** @type {!Set<string>} */
355 var allUsedProperties = new Set();
356 for (var i = 0; i < styles.length; ++i) {
357 var style = styles[i];
358 var rule = style.parentRule;
359 // Compute cascade for CSSStyleRules only.
360 if (rule && !(rule instanceof WebInspector.CSSStyleRule))
361 continue;
362 if (rule && !this.hasMatchingSelectors(rule))
363 continue;
364
365 /** @type {!Map<string, !WebInspector.CSSProperty>} */
366 var styleActiveProperties = new Map();
367 var allProperties = style.allProperties;
368 for (var j = 0; j < allProperties.length; ++j) {
369 var property = allProperties[j];
370
371 // Do not pick non-inherited properties from inherited styles.
372 var inherited = this.isInherited(style);
373 if (inherited && !WebInspector.cssMetadata().isPropertyInherited(property.name))
374 continue;
375
376 if (!property.activeInStyle()) {
377 result.set(property, WebInspector.CSSMatchedStyles.PropertyState.Overloaded);
378 continue;
379 }
380
381 var canonicalName = WebInspector.cssMetadata().canonicalPropertyName(property.name);
382 if (foundImportantProperties.has(canonicalName)) {
383 result.set(property, WebInspector.CSSMatchedStyles.PropertyState.Overloaded);
384 continue;
385 }
386
387 if (!property.important && allUsedProperties.has(canonicalName)) {
388 result.set(property, WebInspector.CSSMatchedStyles.PropertyState.Overloaded);
389 continue;
390 }
391
392 var isKnownProperty = propertyToEffectiveRule.has(canonicalName);
393 var inheritedFromNode = inherited ? this.nodeForStyle(style) : null;
394 if (!isKnownProperty && inheritedFromNode && !inheritedPropertyToNode.has(canonicalName))
395 inheritedPropertyToNode.set(canonicalName, inheritedFromNode);
396
397 if (property.important) {
398 if (inherited && isKnownProperty && inheritedFromNode !== inheritedPropertyToNode.get(canonicalName)) {
399 result.set(property, WebInspector.CSSMatchedStyles.PropertyState.Overloaded);
400 continue;
401 }
402
403 foundImportantProperties.add(canonicalName);
404 if (isKnownProperty) {
405 var overloaded = /** @type {!WebInspector.CSSProperty} */(propertyToEffectiveRule.get(canonicalName).get(canonicalName));
406 result.set(overloaded, WebInspector.CSSMatchedStyles.PropertyState.Overloaded);
407 propertyToEffectiveRule.get(canonicalName).delete(canonicalName);
408 }
409 }
410
411 styleActiveProperties.set(canonicalName, property);
412 allUsedProperties.add(canonicalName);
413 propertyToEffectiveRule.set(canonicalName, styleActiveProperties);
414 result.set(property, WebInspector.CSSMatchedStyles.PropertyState.Active);
415 }
416
417 // If every longhand of the shorthand is not active, then the shorthand is not active too.
418 for (var property of style.leadingProperties()) {
419 var canonicalName = WebInspector.cssMetadata().canonicalPropertyName(property.name);
420 if (!styleActiveProperties.has(canonicalName))
421 continue;
422 var longhands = style.longhandProperties(property.name);
423 if (!longhands.length)
424 continue;
425 var notUsed = true;
426 for (var longhand of longhands) {
427 var longhandCanonicalName = WebInspector.cssMetadata().canonicalPropertyName(longhand.name);
428 notUsed = notUsed && !styleActiveProperties.has(longhandCanonicalName);
429 }
430 if (!notUsed)
431 continue;
432 styleActiveProperties.delete(canonicalName);
433 allUsedProperties.delete(canonicalName);
434 result.set(property, WebInspector.CSSMatchedStyles.PropertyState.Overloaded);
435 }
436 }
437 }
