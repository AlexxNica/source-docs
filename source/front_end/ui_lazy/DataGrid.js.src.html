357 /**
358 * @param {boolean} wasChange
359 * @this {WebInspector.DataGrid}
360 */
361 function moveToNextIfNeeded(wasChange)
362 {
363 if (!moveDirection)
364 return;
365
366 if (moveDirection === "forward") {
367 var firstEditableColumn = this._nextEditableColumn(-1);
368 if (currentEditingNode.isCreationNode && cellIndex === firstEditableColumn && !wasChange)
369 return;
370
371 var nextEditableColumn = this._nextEditableColumn(cellIndex);
372 if (nextEditableColumn !== -1)
373 return this._startEditingColumnOfDataGridNode(currentEditingNode, nextEditableColumn);
374
375 var nextDataGridNode = currentEditingNode.traverseNextNode(true, null, true);
376 if (nextDataGridNode)
377 return this._startEditingColumnOfDataGridNode(nextDataGridNode, firstEditableColumn);
378 if (currentEditingNode.isCreationNode && wasChange) {
379 this.addCreationNode(false);
380 return this._startEditingColumnOfDataGridNode(this.creationNode, firstEditableColumn);
381 }
382 return;
383 }
384
385 if (moveDirection === "backward") {
386 var prevEditableColumn = this._nextEditableColumn(cellIndex, true);
387 if (prevEditableColumn !== -1)
388 return this._startEditingColumnOfDataGridNode(currentEditingNode, prevEditableColumn);
389
390 var lastEditableColumn = this._nextEditableColumn(this._visibleColumnsArray.length, true);
391 var nextDataGridNode = currentEditingNode.traversePreviousNode(true, true);
392 if (nextDataGridNode)
393 return this._startEditingColumnOfDataGridNode(nextDataGridNode, lastEditableColumn);
394 return;
395 }
396 }
397
398 if (textBeforeEditing === newText) {
399 this._editingCancelled(element);
400 moveToNextIfNeeded.call(this, false);
401 return;
402 }
403
404 // Update the text in the datagrid that we typed
405 this._editingNode.data[columnIdentifier] = newText;
406
407 // Make the callback - expects an editing node (table row), the column number that is being edited,
408 // the text that used to be there, and the new text.
409 this._editCallback(this._editingNode, columnIdentifier, textBeforeEditing, newText);
410
411 if (this._editingNode.isCreationNode)
412 this.addCreationNode(false);
413
414 this._editingCancelled(element);
415 moveToNextIfNeeded.call(this, true);

498 /**
499 * @param {!Array.<number>} widths
500 * @param {number} minPercent
501 * @param {number=} maxPercent
502 * @return {!Array.<number>}
503 */
504 _autoSizeWidths: function(widths, minPercent, maxPercent)
505 {
506 if (minPercent)
507 minPercent = Math.min(minPercent, Math.floor(100 / widths.length));
508 var totalWidth = 0;
509 for (var i = 0; i < widths.length; ++i)
510 totalWidth += widths[i];
511 var totalPercentWidth = 0;
512 for (var i = 0; i < widths.length; ++i) {
513 var width = Math.round(100 * widths[i] / totalWidth);
514 if (minPercent && width < minPercent)
515 width = minPercent;
516 else if (maxPercent && width > maxPercent)
517 width = maxPercent;
518 totalPercentWidth += width;
519 widths[i] = width;
520 }
521 var recoupPercent = totalPercentWidth - 100;
522
523 while (minPercent && recoupPercent > 0) {
524 for (var i = 0; i < widths.length; ++i) {
525 if (widths[i] > minPercent) {
526 --widths[i];
527 --recoupPercent;
528 if (!recoupPercent)
529 break;
530 }
531 }
532 }
533
534 while (maxPercent && recoupPercent < 0) {
535 for (var i = 0; i < widths.length; ++i) {
536 if (widths[i] < maxPercent) {
537 ++widths[i];
538 ++recoupPercent;
539 if (!recoupPercent)
540 break;
541 }
542 }
543 }
544
545 return widths;

594 // Updates the widths of the table, including the positions of the column
595 // resizers.
596 //
597 // IMPORTANT: This function MUST be called once after the element of the
598 // DataGrid is attached to its parent element and every subsequent time the
599 // width of the parent element is changed in order to make it possible to
600 // resize the columns.
601 //
602 // If this function is not called after the DataGrid is attached to its
603 // parent element, then the DataGrid's columns will not be resizable.
604 updateWidths: function()
605 {
606 var headerTableColumns = this._headerTableColumnGroup.children;
607
608 // Use container size to avoid changes of table width caused by change of column widths.
609 var tableWidth = this.element.offsetWidth - this._cornerWidth;
610 var numColumns = headerTableColumns.length - 1; // Do not process corner column.
611
612 // Do not attempt to use offsetes if we're not attached to the document tree yet.
613 if (!this._columnWidthsInitialized && this.element.offsetWidth) {
614 // Give all the columns initial widths now so that during a resize,
615 // when the two columns that get resized get a percent value for
616 // their widths, all the other columns already have percent values
617 // for their widths.
618 for (var i = 0; i < numColumns; i++) {
619 var columnWidth = this.headerTableBody.rows[0].cells[i].offsetWidth;
620 var column = this._visibleColumnsArray[i];
621 if (!column.weight)
622 column.weight = 100 * columnWidth / tableWidth;
623 }
624 this._columnWidthsInitialized = true;
625 }
626 this._applyColumnWeights();

673 _applyColumnWeights: function()
674 {
675 var tableWidth = this.element.offsetWidth - this._cornerWidth;
676 if (tableWidth <= 0)
677 return;
678
679 var sumOfWeights = 0.0;
680 var fixedColumnWidths = [];
681 for (var i = 0; i < this._visibleColumnsArray.length; ++i) {
682 var column = this._visibleColumnsArray[i];
683 if (column.fixedWidth) {
684 var width = this._headerTableColumnGroup.children[i][WebInspector.DataGrid._preferredWidthSymbol] || this.headerTableBody.rows[0].cells[i].offsetWidth;
685 fixedColumnWidths[i] = width;
686 tableWidth -= width;
687 } else {
688 sumOfWeights += this._visibleColumnsArray[i].weight;
689 }
690 }
691 var sum = 0;
692 var lastOffset = 0;
693
694 for (var i = 0; i < this._visibleColumnsArray.length; ++i) {
695 var column = this._visibleColumnsArray[i];
696 var width;
697 if (column.fixedWidth) {
698 width = fixedColumnWidths[i];
699 } else {
700 sum += column.weight;
701 var offset = (sum * tableWidth / sumOfWeights) | 0;
702 width = offset - lastOffset;
703 lastOffset = offset;
704 }
705 this._setPreferredWidth(i, width);
706 }

709 this.dispatchEventToListeners(WebInspector.DataGrid.Events.ColumnsResized);
