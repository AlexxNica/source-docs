336 */
337WebInspector.FlameChart.ColorGenerator = function(hueSpace, satSpace, lightnessSpace, alphaSpace)
338{
339 this._hueSpace = hueSpace || { min: 0, max: 360 };
340 this._satSpace = satSpace || 67;
341 this._lightnessSpace = lightnessSpace || 80;
342 this._alphaSpace = alphaSpace || 1;
343 /** @type {!Map<string, string>} */
344 this._colors = new Map();
345}

347WebInspector.FlameChart.ColorGenerator.prototype = {
348 /**
349 * @param {string} id
350 * @param {string} color
351 */
352 setColorForID: function(id, color)
353 {
354 this._colors.set(id, color);
355 },
356
357 /**
358 * @param {string} id
359 * @return {string}
360 */
361 colorForID: function(id)
362 {
363 var color = this._colors.get(id);
364 if (!color) {
365 color = this._generateColorForID(id);
366 this._colors.set(id, color);
367 }
368 return color;
369 },
370
371 /**
372 * @param {string} id
373 * @return {string}
374 */
375 _generateColorForID: function(id)
376 {
377 var hash = String.hashCode(id);
378 var h = this._indexToValueInSpace(hash, this._hueSpace);
379 var s = this._indexToValueInSpace(hash >> 8, this._satSpace);
380 var l = this._indexToValueInSpace(hash >> 16, this._lightnessSpace);
381 var a = this._indexToValueInSpace(hash >> 24, this._alphaSpace);
382 return "hsla(" + h + ", " + s + "%, " + l + "%, " + a + ")";
383 },
384
385 /**
386 * @param {number} index
387 * @param {!{min: number, max: number, count: (number|undefined)}|number} space
388 * @return {number}
389 */
390 _indexToValueInSpace: function(index, space)
391 {
392 if (typeof space === "number")
393 return space;
394 var count = space.count || space.max - space.min;
395 index %= count;
396 return space.min + Math.floor(index / (count - 1) * (space.max - space.min));
397 }

585 this._flameChartDelegate.requestWindowTimes(timeLeft - delta, timeRight - delta);
