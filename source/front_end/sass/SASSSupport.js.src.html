7/**
8 * @param {string} url
9 * @param {string} content
10 * @return {!Promise<!WebInspector.SASSSupport.AST>}
11 */
12WebInspector.SASSSupport.parseSCSS = function(url, content)
13{
14 var text = new WebInspector.Text(content);
15 var document = new WebInspector.SASSSupport.ASTDocument(url, text);
16
17 return WebInspector.formatterWorkerPool.runTask("parseSCSS", {content: content}).then(onParsed);
18
19 /**
20 * @param {?MessageEvent} event
21 * @return {!WebInspector.SASSSupport.AST}
22 */
23 function onParsed(event)
24 {
25 if (!event)
26 return new WebInspector.SASSSupport.AST(document, []);
27 var data = /** @type {!Array<!Object>} */(event.data);
28 var rules = [];
29 for (var i = 0; i < data.length; ++i) {
30 var rulePayload = data[i];
31 var selectors = rulePayload.selectors.map(createTextNode);
32 var properties = rulePayload.properties.map(createProperty);
33 var range = WebInspector.TextRange.fromObject(rulePayload.styleRange);
34 var rule = new WebInspector.SASSSupport.Rule(document, selectors, range, properties);
35 rules.push(rule);
36 }
37 return new WebInspector.SASSSupport.AST(document, rules);
38 }
39
40 /**
41 * @param {!Object} payload
42 */
43 function createTextNode(payload)
44 {
45 var range = WebInspector.TextRange.fromObject(payload);
46 var value = text.extract(range);
47 return new WebInspector.SASSSupport.TextNode(document, text.extract(range), range);
48 }
49
50 /**
51 * @param {!Object} payload
52 */
53 function createProperty(payload)
54 {
55 var name = createTextNode(payload.name);
56 var value = createTextNode(payload.value);
57 return new WebInspector.SASSSupport.Property(document, name, value, WebInspector.TextRange.fromObject(payload.range), payload.disabled);
58 }

85 hasChanged: function()

87 return !!this.edits.length;

90 /**
91 * @return {!WebInspector.Text}
92 */
93 newText: function()
94 {
95 this.edits.stableSort(sequentialOrder);
96 var text = this.text;
97 for (var i = this.edits.length - 1; i >= 0; --i) {
98 var range = this.edits[i].oldRange;
99 var newText = this.edits[i].newText;
100 text = new WebInspector.Text(text.replaceRange(range, newText));
101 }
102 return text;
103
104 /**
105 * @param {!WebInspector.SourceEdit} edit1
106 * @param {!WebInspector.SourceEdit} edit2
107 * @return {number}
108 */
109 function sequentialOrder(edit1, edit2)
110 {
111 var range1 = edit1.oldRange.collapseToStart();
112 var range2 = edit2.oldRange.collapseToStart();
113 if (range1.equal(range2))
114 return 0;
115 return range1.follows(range2) ? 1 : -1;
116 }

282 * @constructor
283 * @extends {WebInspector.SASSSupport.Node}

285 * @param {!Array<!WebInspector.SASSSupport.TextNode>} selectors

289WebInspector.SASSSupport.Rule = function(document, selectors, styleRange, properties)
290{
291 WebInspector.SASSSupport.Node.call(this, document);
292 this.selectors = selectors;
293 this.properties = properties;
294 this.styleRange = styleRange;
295
296 var blockStartRange = styleRange.collapseToStart();
297 blockStartRange.startColumn -= 1;
298 this.blockStart = new WebInspector.SASSSupport.TextNode(document, this.document.text.extract(blockStartRange), blockStartRange);
299 this.blockStart.parent = this;
300
301 for (var i = 0; i < this.properties.length; ++i)
302 this.properties[i].parent = this;
303
304 this._hasTrailingSemicolon = !this.properties.length || this.document.text.extract(this.properties.peekLast().range).endsWith(";");

339 * @return {boolean}

342 {
343 if (this.selectors.length !== other.selectors.length)

365 /**
366 * @param {?WebInspector.SASSSupport.Property} anchorProperty
367 * @param {!Array<string>} nameTexts
368 * @param {!Array<string>} valueTexts
369 * @param {!Array<boolean>} disabledStates
370 * @return {!Array<!WebInspector.SASSSupport.Property>}
371 */
372 insertProperties: function(anchorProperty, nameTexts, valueTexts, disabledStates)
373 {
374 console.assert(nameTexts.length === valueTexts.length && valueTexts.length === disabledStates.length, "Input array should be of the same size.");
375
376 this._addTrailingSemicolon();
377 var newProperties = [];
378 var index = anchorProperty ? this.properties.indexOf(anchorProperty) : -1;
379 for (var i = 0; i < nameTexts.length; ++i) {
380 var nameText = nameTexts[i];
381 var valueText = valueTexts[i];
382 var disabled = disabledStates[i];

386 var value = new WebInspector.SASSSupport.TextNode(this.document, valueText, WebInspector.TextRange.createFromLocation(0, 0));
387 var newProperty = new WebInspector.SASSSupport.Property(this.document, name, value, WebInspector.TextRange.createFromLocation(0, 0), disabled);

389 this.properties.splice(index + i + 1, 0, newProperty);
390 newProperty.parent = this;
391 newProperties.push(newProperty);
392 }

609/**
610 * @param {!WebInspector.SASSSupport.AST} oldAST
611 * @param {!WebInspector.SASSSupport.AST} newAST
612 * @return {!WebInspector.SASSSupport.ASTDiff}
613 */
614WebInspector.SASSSupport.diffModels = function(oldAST, newAST)
615{
616 console.assert(oldAST.rules.length === newAST.rules.length, "Not implemented for rule diff.");
617 console.assert(oldAST.document.url === newAST.document.url, "Diff makes sense for models with the same url.");
618 var T = WebInspector.SASSSupport.PropertyChangeType;
619 var changes = [];
620 /** @type {!Map<!WebInspector.SASSSupport.TextNode, !WebInspector.SASSSupport.TextNode>} */
621 var mapping = new Map();
622 for (var i = 0; i < oldAST.rules.length; ++i) {
623 var oldRule = oldAST.rules[i];
624 var newRule = newAST.rules[i];
625 computeRuleDiff(mapping, oldRule, newRule);
626 }
627 return new WebInspector.SASSSupport.ASTDiff(oldAST.document.url, oldAST, newAST, mapping, changes);
628
629 /**
630 * @param {!WebInspector.SASSSupport.PropertyChangeType} type
631 * @param {!WebInspector.SASSSupport.Rule} oldRule
632 * @param {!WebInspector.SASSSupport.Rule} newRule
633 * @param {number} oldPropertyIndex
634 * @param {number} newPropertyIndex
635 */
636 function addChange(type, oldRule, newRule, oldPropertyIndex, newPropertyIndex)
637 {
638 changes.push(new WebInspector.SASSSupport.PropertyChange(type, oldRule, newRule, oldPropertyIndex, newPropertyIndex));
639 }
640
641 /**
642 * @param {!Map<!WebInspector.SASSSupport.TextNode, !WebInspector.SASSSupport.TextNode>} mapping
643 * @param {!WebInspector.SASSSupport.Rule} oldRule
644 * @param {!WebInspector.SASSSupport.Rule} newRule
645 */
646 function computeRuleDiff(mapping, oldRule, newRule)
647 {
648 var oldLines = [];
649 for (var i = 0; i < oldRule.properties.length; ++i)
650 oldLines.push(oldRule.properties[i].name.text.trim() + ":" + oldRule.properties[i].value.text.trim());
651 var newLines = [];
652 for (var i = 0; i < newRule.properties.length; ++i)
653 newLines.push(newRule.properties[i].name.text.trim() + ":" + newRule.properties[i].value.text.trim());
654 var diff = WebInspector.Diff.lineDiff(oldLines, newLines);
655 diff = WebInspector.Diff.convertToEditDiff(diff);
656
657 var p1 = 0, p2 = 0;
658 for (var i = 0; i < diff.length; ++i) {
659 var token = diff[i];
660 if (token[0] === WebInspector.Diff.Operation.Delete) {
661 for (var j = 0; j < token[1]; ++j)
662 addChange(T.PropertyRemoved, oldRule, newRule, p1++, p2);
