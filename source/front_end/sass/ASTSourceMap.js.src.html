179 /**
180 * @param {!Array<!WebInspector.SASSSupport.AST>} updated
181 * @param {!Map<!WebInspector.SASSSupport.Node, !WebInspector.SASSSupport.Node>=} outNodeMapping
182 * @return {?WebInspector.ASTSourceMap}
183 */
184 rebase: function(updated, outNodeMapping)
185 {
186 outNodeMapping = outNodeMapping || new Map();
187 outNodeMapping.clear();
188
189 var models = /** @type {!Map<string, !WebInspector.SASSSupport.AST>} */(new Map(this._models));
190 for (var newAST of updated) {
191 var oldAST = models.get(newAST.document.url);
192 if (!oldAST.match(newAST, outNodeMapping))
193 return null;
194 models.set(newAST.document.url, newAST);
195 }
196
197 var newMap = new WebInspector.ASTSourceMap(this._compiledURL, this._sourceMapURL, models, this._editCallback);
198 var compiledNodes = this._compiledToSource.keysArray();
199 for (var i = 0; i < compiledNodes.length; ++i) {
200 var compiledNode = compiledNodes[i];
201 var sourceNode = /** @type {!WebInspector.SASSSupport.TextNode} */(this._compiledToSource.get(compiledNode));
202 var mappedCompiledNode = /** @type {!WebInspector.SASSSupport.TextNode} */(outNodeMapping.get(compiledNode) || compiledNode);
203 var mappedSourceNode = /** @type {!WebInspector.SASSSupport.TextNode} */(outNodeMapping.get(sourceNode) || sourceNode);
204 newMap.addMapping(mappedCompiledNode, mappedSourceNode);
205 }
206 return newMap;
207 }
