10WebInspector.AnimationTimeline = function()
11{

363 /**
364 * @param {!WebInspector.AnimationModel.AnimationGroup} group
365 */
366 _addAnimationGroup: function(group)
367 {
368 /**
369 * @param {!WebInspector.AnimationModel.AnimationGroup} left
370 * @param {!WebInspector.AnimationModel.AnimationGroup} right
371 */
372 function startTimeComparator(left, right)
373 {
374 return left.startTime() > right.startTime();
375 }
376
377 if (this._previewMap.get(group)) {
378 if (this._selectedGroup === group)
379 this._syncScrubber();
380 else
381 this._previewMap.get(group).replay();
382 return;
383 }
384 this._groupBuffer.sort(startTimeComparator);
385 // Discard oldest groups from buffer if necessary
386 var groupsToDiscard = [];
387 var bufferSize = this.width() / 50;
388 while (this._groupBuffer.length > bufferSize) {
389 var toDiscard = this._groupBuffer.splice(this._groupBuffer[0] === this._selectedGroup ? 1 : 0, 1);
390 groupsToDiscard.push(toDiscard[0]);
391 }
392 for (var g of groupsToDiscard) {
393 this._previewMap.get(g).element.remove();
394 this._previewMap.delete(g);
395 g.release();
396 }
397 // Generate preview
398 var preview = new WebInspector.AnimationGroupPreviewUI(group);
399 this._groupBuffer.push(group);
400 this._previewMap.set(group, preview);
401 this._previewContainer.appendChild(preview.element);
402 preview.removeButton().addEventListener("click", this._removeAnimationGroup.bind(this, group));
403 preview.element.addEventListener("click", this._selectAnimationGroup.bind(this, group));
404 },

429 /**
430 * @param {!WebInspector.AnimationGroupPreviewUI} ui
431 * @param {!WebInspector.AnimationModel.AnimationGroup} group
432 * @this {!WebInspector.AnimationTimeline}
433 */
434 function applySelectionClass(ui, group)
435 {
436 ui.element.classList.toggle("selected", this._selectedGroup === group);
437 }
438
439 if (this._selectedGroup === group) {
440 this._togglePause(false);
441 this._replay();
442 return;
443 }
444 this._clearTimeline();
445 this._selectedGroup = group;
446 this._previewMap.forEach(applySelectionClass, this);
447 this.setDuration(Math.max(500, group.finiteDuration() + 100));
448 for (var anim of group.animations())
449 this._addAnimation(anim);
450 this.scheduleRedraw();
451 this._timelineScrubber.classList.remove("hidden");
452 this._togglePause(false);
453 this._replay();

496 _renderGrid: function()
497 {
498 /** @const */ var gridSize = 250;
499 this._grid.setAttribute("width", this.width() + 10);
500 this._grid.setAttribute("height", this._cachedTimelineHeight + 30);
501 this._grid.setAttribute("shape-rendering", "crispEdges");
502 this._grid.removeChildren();
503 var lastDraw = undefined;
504 for (var time = 0; time < this.duration(); time += gridSize) {
505 var line = this._grid.createSVGChild("rect", "animation-timeline-grid-line");
506 line.setAttribute("x", time * this.pixelMsRatio() + 10);
507 line.setAttribute("y", 23);
508 line.setAttribute("height", "100%");
509 line.setAttribute("width", 1);
510 }
511 for (var time = 0; time < this.duration(); time += gridSize) {
512 var gridWidth = time * this.pixelMsRatio();
513 if (lastDraw === undefined || gridWidth - lastDraw > 50) {
514 lastDraw = gridWidth;
515 var label = this._grid.createSVGChild("text", "animation-timeline-grid-label");
516 label.textContent = WebInspector.UIString(Number.millisToString(time));
517 label.setAttribute("x", gridWidth + 10);
518 label.setAttribute("y", 16);
519 }
520 }

566 /**

570 _resizeWindow: function(animation)
571 {
572 var resized = false;
573
574 // This shows at most 3 iterations
575 var duration = animation.source().duration() * Math.min(2, animation.source().iterations());
576 var requiredDuration = animation.source().delay() + duration + animation.source().endDelay();
577 if (requiredDuration > this._duration) {
578 resized = true;

597 {
598 if (this._scrubberPlayer)
599 this._scrubberPlayer.cancel();
600
601 this._scrubberPlayer = this._timelineScrubber.animate([
602 { transform: "translateX(0px)" },
603 { transform: "translateX(" + this.width() + "px)" }
604 ], { duration: this.duration(), fill: "forwards" });
605 this._scrubberPlayer.playbackRate = this._effectivePlaybackRate();
606 this._scrubberPlayer.onfinish = this._updateControlButton.bind(this);
607 this._scrubberPlayer.currentTime = currentTime;
608 this.element.window().requestAnimationFrame(this._updateScrubber.bind(this));

619 /**
620 * @param {number} timestamp
621 */
622 _updateScrubber: function(timestamp)
623 {
624 if (!this._scrubberPlayer)
625 return;
626 this._currentTime.textContent = WebInspector.UIString(Number.millisToString(this._scrubberPlayer.currentTime));
627 if (this._scrubberPlayer.playState === "pending" || this._scrubberPlayer.playState === "running") {
628 this.element.window().requestAnimationFrame(this._updateScrubber.bind(this));
629 } else if (this._scrubberPlayer.playState === "finished") {
630 this._currentTime.textContent = "";
631 }

634 /**
635 * @param {!Event} event
636 * @return {boolean}
637 */
638 _repositionScrubber: function(event)
639 {
640 if (!this._selectedGroup)
641 return false;
642
643 // Seek to current mouse position.
644 if (!this._gridOffsetLeft)
645 this._gridOffsetLeft = this._grid.totalOffsetLeft() + 10;
646 var seekTime = Math.max(0, event.x - this._gridOffsetLeft) / this.pixelMsRatio();
647 this._selectedGroup.seekTo(seekTime);
648 this._togglePause(true);
649 this._animateTime(seekTime);
650
651 // Interface with scrubber drag.
652 this._originalScrubberTime = seekTime;
653 this._originalMousePosition = event.x;
654 return true;
655 },
656
657 /**
658 * @param {!Event} event
659 * @return {boolean}
660 */
661 _scrubberDragStart: function(event)
662 {
663 if (!this._scrubberPlayer || !this._selectedGroup)
664 return false;
665
666 this._originalScrubberTime = this._scrubberPlayer.currentTime;
667 this._timelineScrubber.classList.remove("animation-timeline-end");
668 this._scrubberPlayer.pause();
669 this._originalMousePosition = event.x;
670
671 this._togglePause(true);
672 return true;
673 },
674
675 /**
676 * @param {!Event} event
677 */
678 _scrubberDragMove: function(event)
679 {
680 var delta = event.x - this._originalMousePosition;
681 var currentTime = Math.max(0, Math.min(this._originalScrubberTime + delta / this.pixelMsRatio(), this.duration()));
682 this._scrubberPlayer.currentTime = currentTime;
683 this._currentTime.textContent = WebInspector.UIString(Number.millisToString(Math.round(currentTime)));
684 this._selectedGroup.seekTo(currentTime);
685 },
686
687 /**
688 * @param {!Event} event
689 */
690 _scrubberDragEnd: function(event)
691 {
692 var currentTime = Math.max(0, this._scrubberPlayer.currentTime);
693 this._scrubberPlayer.play();
694 this._scrubberPlayer.currentTime = currentTime;
695 this._currentTime.window().requestAnimationFrame(this._updateScrubber.bind(this));
