134 /**
135 * @param {!Runtime.Extension} extension
136 * @this {!WebInspector.InspectorView}
137 */
138 function processPanelExtensions(extension)
139 {
140 var descriptor = new WebInspector.ExtensionPanelDescriptor(extension);
141 var weight = this._tabOrderSetting.get()[descriptor.name()];
142 if (weight === undefined)
143 weight = extension.descriptor()["order"];
144 if (weight === undefined)

149 /**
150 * @param {!WebInspector.PanelDescriptor} left
151 * @param {!WebInspector.PanelDescriptor} right
152 */
153 function orderComparator(left, right)
154 {
155 return panelWeights.get(left) > panelWeights.get(right);
156 }
157
158 WebInspector.startBatchUpdate();
159 /** @type {!Map<!WebInspector.PanelDescriptor, number>} */
160 var panelWeights = new Map();
161 self.runtime.extensions(WebInspector.Panel).forEach(processPanelExtensions.bind(this));
162 var sortedPanels = panelWeights.keysArray().sort(orderComparator);
163 for (var panelDescriptor of sortedPanels)
164 this._innerAddPanel(panelDescriptor);
165 WebInspector.endBatchUpdate();

263 /**
264 * The returned Promise is resolved with null if another showPanel()
265 * gets called while this.panel(panelName) Promise is in flight.
266 *
267 * @param {string} panelName
268 * @return {!Promise.<?WebInspector.Panel>}
269 */
270 showPanel: function(panelName)
271 {
272 if (this._currentPanelLocked) {
273 if (this._currentPanel !== this._panels[panelName])
274 return Promise.reject(new Error("Current panel locked"));
275 return Promise.resolve(this._currentPanel);
276 }
277
278 this._panelForShowPromise = this.panel(panelName);
279 return this._panelForShowPromise.then(setCurrentPanelIfNecessary.bind(this, this._panelForShowPromise));

420 _keyPress: function(event)
421 {
422 // BUG 104250: Windows 7 posts a WM_CHAR message upon the Ctrl+']' keypress.
423 // Any charCode < 32 is not going to be a valid keypress.
424 if (event.charCode < 32 && WebInspector.isWin())
425 return;
426 clearTimeout(this._keyDownTimer);
427 delete this._keyDownTimer;
428 },
429
430 _keyDown: function(event)
431 {
432 if (!WebInspector.KeyboardShortcut.eventHasCtrlOrMeta(event))
433 return;
434
435 var keyboardEvent = /** @type {!KeyboardEvent} */ (event);
436 // Ctrl/Cmd + 1-9 should show corresponding panel.
437 var panelShortcutEnabled = WebInspector.moduleSetting("shortcutPanelSwitch").get();
438 if (panelShortcutEnabled && !event.shiftKey && !event.altKey) {
439 var panelIndex = -1;
440 if (event.keyCode > 0x30 && event.keyCode < 0x3A)
441 panelIndex = event.keyCode - 0x31;
442 else if (event.keyCode > 0x60 && event.keyCode < 0x6A && keyboardEvent.location === KeyboardEvent.DOM_KEY_LOCATION_NUMPAD)
443 panelIndex = event.keyCode - 0x61;
444 if (panelIndex !== -1) {
445 var panelName = this._tabbedPane.allTabs()[panelIndex];
446 if (panelName) {
447 if (!WebInspector.Dialog.hasInstance() && !this._currentPanelLocked)
448 this.showPanel(panelName);
449 event.consume(true);
450 }
451 return;
452 }
453 }
454
455 // BUG85312: On French AZERTY keyboards, AltGr-]/[ combinations (synonymous to Ctrl-Alt-]/[ on Windows) are used to enter ]/[,
456 // so for a ]/[-related keydown we delay the panel switch using a timer, to see if there is a keypress event following this one.
457 // If there is, we cancel the timer and do not consider this a panel switch.
458 if (!WebInspector.isWin() || (event.key !== "[" && event.key !== "]")) {
459 this._keyDownInternal(event);
460 return;
461 }
462
463 this._keyDownTimer = setTimeout(this._keyDownInternal.bind(this, event), 0);
464 },
465
466 _keyDownInternal: function(event)
467 {
468 if (this._currentPanelLocked)
469 return;
470
471 var direction = 0;
472
473 if (event.key === "[")
474 direction = -1;
475
476 if (event.key === "]")
477 direction = 1;
478
479 if (!direction)
480 return;
481
482 if (!event.shiftKey && !event.altKey) {
483 if (!WebInspector.Dialog.hasInstance())
484 this._changePanelInDirection(direction);
485 event.consume(true);
486 return;
487 }
488
489 if (event.altKey && this._moveInHistory(direction))
490 event.consume(true);
491 },
492
493 /**
494 * @param {number} direction
495 */
496 _changePanelInDirection: function(direction)
497 {
498 var panelOrder = this._tabbedPane.allTabs();
499 var index = panelOrder.indexOf(this.currentPanel().name);
500 index = (index + panelOrder.length + direction) % panelOrder.length;
501 this.showPanel(panelOrder[index]);
502 },
503
504 /**
505 * @param {number} move
506 */
507 _moveInHistory: function(move)
508 {
509 var newIndex = this._historyIterator + move;
510 if (newIndex >= this._history.length || newIndex < 0)
511 return false;
512
513 this._inHistory = true;
514 this._historyIterator = newIndex;
515 if (!WebInspector.Dialog.hasInstance())
516 this.setCurrentPanel(this._panels[this._history[this._historyIterator]]);
517 delete this._inHistory;
518
519 return true;
520 },
521
522 _pushToHistory: function(panelName)
523 {
524 if (this._inHistory)
525 return;
526
527 this._history.splice(this._historyIterator + 1, this._history.length - this._historyIterator - 1);
528 if (!this._history.length || this._history[this._history.length - 1] !== panelName)
529 this._history.push(panelName);
530 this._historyIterator = this._history.length - 1;
