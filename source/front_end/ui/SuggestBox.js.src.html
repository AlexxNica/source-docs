96 /**
97 * @param {!AnchorBox} anchorBox
98 */
99 _updateBoxPosition: function(anchorBox)
100 {
101 console.assert(this._overlay);
102 if (this._lastAnchorBox && this._lastAnchorBox.equals(anchorBox))
103 return;
104 this._lastAnchorBox = anchorBox;
105
106 // Position relative to main DevTools element.
107 var container = WebInspector.Dialog.modalHostView().element;
108 anchorBox = anchorBox.relativeToElement(container);
109 var totalHeight = container.offsetHeight;
110 var aboveHeight = anchorBox.y;
111 var underHeight = totalHeight - anchorBox.y - anchorBox.height;
112
113 this._overlay.setLeftOffset(anchorBox.x);
114
115 var under = underHeight >= aboveHeight;
116 if (under)
117 this._overlay.setVerticalOffset(anchorBox.y + anchorBox.height, true);
118 else
119 this._overlay.setVerticalOffset(totalHeight - anchorBox.y, false);
120
121 /** const */ var rowHeight = 17;
122 /** const */ var spacer = 6;
123 var maxHeight = this._maxItemsHeight ? this._maxItemsHeight * rowHeight : Math.max(underHeight, aboveHeight) - spacer;
124 this._element.style.maxHeight = maxHeight + "px";

180 * @param {boolean=} isIntermediateSuggestion
181 */
182 _applySuggestion: function(isIntermediateSuggestion)
183 {
184 if (this._onlyCompletion) {
185 this._suggestBoxDelegate.applySuggestion(this._onlyCompletion, isIntermediateSuggestion);

190 return false;

192 var suggestion = this._selectedElement.__fullValue;
193 if (!suggestion)
194 return false;

196 this._suggestBoxDelegate.applySuggestion(suggestion, isIntermediateSuggestion);
197 return true;

215 /**
216 * @param {number} shift
217 * @param {boolean=} isCircular
218 * @return {boolean} is changed
219 */
220 _selectClosest: function(shift, isCircular)
221 {
222 if (!this._length)
223 return false;
224
225 if (this._selectedIndex === -1 && shift < 0)
226 shift += 1;
227
228 var index = this._selectedIndex + shift;
229
230 if (isCircular)
231 index = (this._length + index) % this._length;
232 else
233 index = Number.constrain(index, 0, this._length - 1);
234
235 this._selectItem(index, true);
236 this._applySuggestion(true);
237 return true;

272 /**
273 * @param {!WebInspector.SuggestBox.Suggestions} items
274 * @param {string} userEnteredText
275 * @param {function(number): !Promise<{detail:string, description:string}>=} asyncDetails
276 */
277 _updateItems: function(items, userEnteredText, asyncDetails)
278 {
279 this._length = items.length;
280 this._asyncDetailsPromises.clear();
281 this._asyncDetailsCallback = asyncDetails;
282 this._element.removeChildren();
283 delete this._selectedElement;
284
285 for (var i = 0; i < items.length; ++i) {
286 var item = items[i];
287 var currentItemElement = this._createItemElement(userEnteredText, item.title, item.className, i);
288 this._element.appendChild(currentItemElement);
289 }

292 /**
293 * @param {number} index
294 * @return {!Promise<?{detail: string, description: string}>}

296 _asyncDetails: function(index)
297 {
298 if (!this._asyncDetailsCallback)

301 this._asyncDetailsPromises.set(index, this._asyncDetailsCallback(index));
302 return /** @type {!Promise<?{detail: string, description: string}>} */(this._asyncDetailsPromises.get(index));

318 /**
319 * @param {number} index
320 * @param {boolean} scrollIntoView
321 */
322 _selectItem: function(index, scrollIntoView)
323 {
324 if (this._selectedElement)
325 this._selectedElement.classList.remove("selected");
326
327 this._selectedIndex = index;
328 if (index < 0)
329 return;
330
331 this._selectedElement = this._element.children[index];
332 this._selectedElement.classList.add("selected");
333 this._detailsPopup.classList.add("hidden");
334 var elem = this._selectedElement;
335 this._asyncDetails(index).then(showDetails.bind(this), function(){});
336
337 if (scrollIntoView)
338 this._selectedElement.scrollIntoViewIfNeeded(false);

342 * @this {WebInspector.SuggestBox}

351 /**
352 * @param {!WebInspector.SuggestBox.Suggestions} completions
353 * @param {boolean} canShowForSingleItem
354 * @param {string} userEnteredText
355 */
356 _canShowBox: function(completions, canShowForSingleItem, userEnteredText)
357 {
358 if (!completions || !completions.length)
359 return false;
360
361 if (completions.length > 1)
362 return true;
363
364 // Do not show a single suggestion if it is the same as user-entered prefix, even if allowed to show single-item suggest boxes.
365 return canShowForSingleItem && completions[0].title !== userEnteredText;
366 },
367
368 _ensureRowCountPerViewport: function()
369 {
370 if (this._rowCountPerViewport)
371 return;
372 if (!this._element.firstChild)
373 return;
374
375 this._rowCountPerViewport = Math.floor(this._element.offsetHeight / this._element.firstChild.offsetHeight);
376 },
377
378 /**
379 * @param {!AnchorBox} anchorBox
380 * @param {!WebInspector.SuggestBox.Suggestions} completions
381 * @param {number} selectedIndex
382 * @param {boolean} canShowForSingleItem
383 * @param {string} userEnteredText
384 * @param {function(number): !Promise<{detail:string, description:string}>=} asyncDetails
385 */
386 updateSuggestions: function(anchorBox, completions, selectedIndex, canShowForSingleItem, userEnteredText, asyncDetails)
387 {
388 delete this._onlyCompletion;
389 if (this._canShowBox(completions, canShowForSingleItem, userEnteredText)) {
390 this._updateItems(completions, userEnteredText, asyncDetails);
391 this._show();
392 this._updateBoxPosition(anchorBox);
393 this._selectItem(selectedIndex, selectedIndex > 0);
394 delete this._rowCountPerViewport;
395 } else {
396 if (completions.length === 1)
397 this._onlyCompletion = completions[0].title;
398 this.hide();
399 }

476WebInspector.SuggestBox.Overlay = function()
477{
478 this.element = createElementWithClass("div", "suggest-box-overlay");

480 this._leftSpacerElement = root.createChild("div", "suggest-box-left-spacer");
481 this._horizontalElement = root.createChild("div", "suggest-box-horizontal");
