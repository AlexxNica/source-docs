125 * @constructor

127WebInspector.PanelWithSidebar = function(name, defaultWidth)
128{
129 WebInspector.Panel.call(this, name);
130
131 this._panelSplitWidget = new WebInspector.SplitWidget(true, false, this._panelName + "PanelSplitViewState", defaultWidth || 200);
132 this._panelSplitWidget.show(this.element);
133
134 this._mainWidget = new WebInspector.VBox();
135 this._panelSplitWidget.setMainWidget(this._mainWidget);
136
137 this._sidebarWidget = new WebInspector.VBox();
138 this._sidebarWidget.setMinimumSize(100, 25);
139 this._panelSplitWidget.setSidebarWidget(this._sidebarWidget);
140
141 this._sidebarWidget.element.classList.add("sidebar");
142}
