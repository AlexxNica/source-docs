65WebInspector.ResourceLoader.loadAsStream = function(url, headers, stream, callback)
66{
67 var streamId = WebInspector.ResourceLoader._bindOutputStream(stream);
68 var parsedURL = new WebInspector.ParsedURL(url);
69 if (parsedURL.isDataURL()) {
70 loadXHR(url)

74 }

80 }
