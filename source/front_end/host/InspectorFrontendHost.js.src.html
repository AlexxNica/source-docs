488 /**
489 * @constructor
490 */
491 function InspectorFrontendAPIImpl()
492 {
493 this._debugFrontend = !!Runtime.queryParam("debugFrontend") || (window["InspectorTest"] && window["InspectorTest"]["debugTest"]);
494
495 var descriptors = InspectorFrontendHostAPI.EventDescriptors;
496 for (var i = 0; i < descriptors.length; ++i)
497 this[descriptors[i][0]] = this._dispatch.bind(this, descriptors[i][0], descriptors[i][1], descriptors[i][2]);
498 }

500 InspectorFrontendAPIImpl.prototype = {
501 /**
502 * @param {string} name
503 * @param {!Array.<string>} signature
504 * @param {boolean} runOnceLoaded
505 */
506 _dispatch: function(name, signature, runOnceLoaded)
507 {
508 var params = Array.prototype.slice.call(arguments, 3);
509
510 if (this._debugFrontend)
511 setImmediate(innerDispatch);
512 else
513 innerDispatch();
514
515 function innerDispatch()
516 {
517 // Single argument methods get dispatched with the param.
518 if (signature.length < 2) {
519 try {
520 InspectorFrontendHost.events.dispatchEventToListeners(name, params[0]);
521 } catch (e) {
522 console.error(e + " " + e.stack);
523 }
524 return;
525 }
526 var data = {};
527 for (var i = 0; i < signature.length; ++i)
528 data[signature[i]] = params[i];
529 try {
530 InspectorFrontendHost.events.dispatchEventToListeners(name, data);
531 } catch (e) {
532 console.error(e + " " + e.stack);
533 }
