13WebInspector.SCSSParser.prototype = {
14 /**
15 * @override
16 * @param {string} content
17 * @return {!Array<!WebInspector.SCSSParser.Rule>}
18 */
19 parse: function(content)
20 {
21 var ast = null;
22 try {
23 ast = gonzales.parse(content, {syntax: "scss"});
24 } catch (e) {
25 return [];
26 }
27
28 /** @type {!{properties: !Array<!Gonzales.Node>, node: !Gonzales.Node}} */
29 var rootBlock = {
30 properties: [],
31 node: ast
32 };
33 /** @type {!Array<!{properties: !Array<!Gonzales.Node>, node: !Gonzales.Node}>} */
34 var blocks = [rootBlock];
35 ast.selectors = [];
36 WebInspector.SCSSParser.extractNodes(ast, blocks, rootBlock);
37
38 var rules = [];
39 for (var block of blocks)
40 this._handleBlock(block, rules);
41 return rules;
42 },
43
44 /**
45 * @param {!{node: !Gonzales.Node, properties: !Array<!Gonzales.Node>}} block
46 * @param {!Array<!WebInspector.SCSSParser.Rule>} output
47 */
48 _handleBlock: function(block, output)
49 {
50 var selectors = block.node.selectors.map(WebInspector.SCSSParser.rangeFromNode);
51 var properties = [];
52 var styleRange = WebInspector.SCSSParser.rangeFromNode(block.node);
53 styleRange.startColumn += 1;
54 styleRange.endColumn -= 1;
55 for (var node of block.properties) {
56 if (node.type === "declaration")
57 this._handleDeclaration(node, properties);
58 else if (node.type === "include")
59 this._handleInclude(node, properties);
60 else if (node.type === "multilineComment" && node.start.line === node.end.line)
61 this._handleComment(node, properties);
62 }
63 if (!selectors.length && !properties.length)
64 return;
65 var rule = new WebInspector.SCSSParser.Rule(selectors, properties, styleRange);
66 output.push(rule);
67 },
68
69 /**
70 * @param {!Gonzales.Node} node
71 * @param {!Array<!WebInspector.SCSSParser.Property>} output
72 */
73 _handleDeclaration: function(node, output)
74 {
75 var propertyNode = node.content.find(node => node.type === "property");
76 var valueNode = node.content.find(node => node.type === "value");
77 if (!propertyNode || !valueNode)
78 return;
79
80 var nameRange = WebInspector.SCSSParser.rangeFromNode(propertyNode);
81 var valueRange = WebInspector.SCSSParser.rangeFromNode(valueNode);
82 var range = /** @type {!WebInspector.TextRange} */(node.declarationRange);
83
84 var property = new WebInspector.SCSSParser.Property(range, nameRange, valueRange, false);
85 output.push(property);
86 },
87
88 /**
89 * @param {!Gonzales.Node} node
90 * @param {!Array<!WebInspector.SCSSParser.Property>} output
91 */
92 _handleInclude: function(node, output)
93 {
94 var mixinName = node.content.find(node => node.type === "ident");
95 if (!mixinName)
96 return;
97 var nameRange = WebInspector.SCSSParser.rangeFromNode(mixinName);
98 var mixinArguments = node.content.find(node => node.type === "arguments");
99 if (!mixinArguments)
100 return;
101 var parameters = mixinArguments.content.filter(node => node.type !== "delimiter" && node.type !== "space");
102 for (var parameter of parameters) {
103 var range = WebInspector.SCSSParser.rangeFromNode(node);
104 var valueRange = WebInspector.SCSSParser.rangeFromNode(parameter);
105 var property = new WebInspector.SCSSParser.Property(range, nameRange, valueRange, false);
106 output.push(property);
107 }
108 },
109
110 /**
111 * @param {!Gonzales.Node} node
112 * @param {!Array<!WebInspector.SCSSParser.Property>} output
113 */
114 _handleComment: function(node, output)
115 {
116 if (node.start.line !== node.end.line)
117 return;
118 var innerText = /** @type {string} */(node.content);
119 var innerResult = this.parse(innerText);
120 if (innerResult.length !== 1 || innerResult[0].properties.length !== 1)
121 return;
122 var property = innerResult[0].properties[0];
123 var disabledProperty = property.rebaseInsideOneLineComment(node);
124 output.push(disabledProperty);
125 },
126}

134 return new WebInspector.TextRange(node.start.line - 1, node.start.column - 1, node.end.line - 1, node.end.column);
135}

153 /**
154 * @param {!Gonzales.Node} commentNode
155 * @return {!WebInspector.SCSSParser.Property}
156 */
157 rebaseInsideOneLineComment: function(commentNode)
158 {
159 var lineOffset = commentNode.start.line - 1;
160 // Account for the "/*".
161 var columnOffset = commentNode.start.column - 1 + 2;
162 var range = WebInspector.SCSSParser.rangeFromNode(commentNode);
163 var name = rebaseRange(this.name, lineOffset, columnOffset);
164 var value = rebaseRange(this.value, lineOffset, columnOffset);
165 return new WebInspector.SCSSParser.Property(range, name, value, true);

167 /**
168 * @param {!WebInspector.TextRange} range
169 * @param {number} lineOffset
170 * @param {number} columnOffset
171 * @return {!WebInspector.TextRange}
172 */
173 function rebaseRange(range, lineOffset, columnOffset)
174 {
175 return new WebInspector.TextRange(range.startLine + lineOffset, range.startColumn + columnOffset, range.endLine + lineOffset, range.endColumn + columnOffset);
176 }

193/**
194 * @param {!Gonzales.Node} node
195 * @param {!Array<{node: !Gonzales.Node, properties: !Array<!Gonzales.Node>}>} blocks
196 * @param {!{node: !Gonzales.Node, properties: !Array<!Gonzales.Node>}} lastBlock
197 */
198WebInspector.SCSSParser.extractNodes = function(node, blocks, lastBlock)
199{
200 if (!Array.isArray(node.content))
201 return;
202 if (node.type === "block") {
203 lastBlock = {
204 node: node,
205 properties: []
206 };
207 blocks.push(lastBlock);
208 }
209 var lastDeclaration = null;
210 var selectors = [];
211 for (var i = 0; i < node.content.length; ++i) {
212 var child = node.content[i];
213 if (child.type === "declarationDelimiter" && lastDeclaration) {
214 lastDeclaration.declarationRange.endLine = child.end.line - 1;
215 lastDeclaration.declarationRange.endColumn = child.end.column;
216 lastDeclaration = null;
217 } else if (child.type === "selector") {
218 selectors.push(child);
219 } else if (child.type === "block") {
220 child.selectors = selectors;
221 selectors = [];
222 }
223 if (child.type === "include" || child.type === "declaration" || child.type === "multilineComment")
224 lastBlock.properties.push(child);
225 if (child.type === "declaration") {
226 lastDeclaration = child;
227 lastDeclaration.declarationRange = WebInspector.TextRange.createFromLocation(child.start.line - 1, child.start.column - 1);
228 }
229 WebInspector.SCSSParser.extractNodes(child, blocks, lastBlock);
230 }
231 if (lastDeclaration) {
232 lastDeclaration.declarationRange.endLine = node.end.line - 1;
233 lastDeclaration.declarationRange.endColumn = node.end.column - 1;
234 }
