70/**
71 * @param {string} fileSystemPath
72 * @return {string}
73 */
74WebInspector.IsolatedFileSystemManager.normalizePath = function(fileSystemPath)
75{
76 fileSystemPath = fileSystemPath.replace(/\\/g, "/");
77 if (!fileSystemPath.startsWith("file://")) {
78 if (fileSystemPath.startsWith("/"))
79 fileSystemPath = "file://" + fileSystemPath;
80 else
81 fileSystemPath = "file:///" + fileSystemPath;
82 }
83 return fileSystemPath;
84}
