134 /**
135 * @param {!WebInspector.HeapSnapshotNode} node
136 * @param {!WebInspector.HeapSnapshotEdge} edge
137 * @return {boolean}
138 */
139 function filter(node, edge)
140 {
141 if (node.isHidden())
142 return edge.name() !== "sloppy_function_map" || node.rawName() !== "system / NativeContext";
143 if (node.isArray()) {
144 // DescriptorArrays are fixed arrays used to hold instance descriptors.
145 // The format of the these objects is:
146 // [0]: Number of descriptors
147 // [1]: Either Smi(0) if uninitialized, or a pointer to small fixed array:
148 // [0]: pointer to fixed array with enum cache
149 // [1]: either Smi(0) or pointer to fixed array with indices
150 // [i*3+2]: i-th key
151 // [i*3+3]: i-th type
152 // [i*3+4]: i-th descriptor
153 // As long as maps may share descriptor arrays some of the descriptor
154 // links may not be valid for all the maps. We just skip
155 // all the descriptor links when calculating distances.
156 // For more details see http://crbug.com/413608
157 if (node.rawName() !== "(map descriptors)")
158 return true;
159 var index = edge.name();
160 return index < 2 || (index % 3) !== 1;
161 }
162 return true;
163 }
164 WebInspector.HeapSnapshot.prototype.calculateDistances.call(this, filter);

202 * @param {!WebInspector.HeapSnapshotNode} node
203 */
204 function doAction(node)
205 {
206 var ordinal = node.ordinal();
207 if (!visitedNodes[ordinal]) {
208 action(node);
209 visitedNodes[ordinal] = true;

227 doAction(iter2.edge.node());

229 }

280 _markQueriableHeapObjects: function()
281 {
282 // Allow runtime properties query for objects accessible from Window objects
283 // via regular properties, and for DOM wrappers. Trying to access random objects
284 // can cause a crash due to insonsistent state of internal properties of wrappers.
285 var flag = this._nodeFlags.canBeQueried;
286 var hiddenEdgeType = this._edgeHiddenType;
287 var internalEdgeType = this._edgeInternalType;
288 var invisibleEdgeType = this._edgeInvisibleType;
289 var weakEdgeType = this._edgeWeakType;
290 var edgeToNodeOffset = this._edgeToNodeOffset;
291 var edgeTypeOffset = this._edgeTypeOffset;
292 var edgeFieldsCount = this._edgeFieldsCount;
293 var containmentEdges = this.containmentEdges;
294 var nodeFieldCount = this._nodeFieldCount;
295 var firstEdgeIndexes = this._firstEdgeIndexes;
296
297 var flags = this._flags;
298 var list = [];
299
300 for (var iter = this.rootNode().edges(); iter.hasNext(); iter.next()) {
301 if (iter.edge.node().isUserRoot())
302 list.push(iter.edge.node().nodeIndex / nodeFieldCount);
303 }
304
305 while (list.length) {
306 var nodeOrdinal = list.pop();
307 if (flags[nodeOrdinal] & flag)
308 continue;
309 flags[nodeOrdinal] |= flag;
310 var beginEdgeIndex = firstEdgeIndexes[nodeOrdinal];
311 var endEdgeIndex = firstEdgeIndexes[nodeOrdinal + 1];
312 for (var edgeIndex = beginEdgeIndex; edgeIndex < endEdgeIndex; edgeIndex += edgeFieldsCount) {
313 var childNodeIndex = containmentEdges[edgeIndex + edgeToNodeOffset];
314 var childNodeOrdinal = childNodeIndex / nodeFieldCount;
315 if (flags[childNodeOrdinal] & flag)
316 continue;
317 var type = containmentEdges[edgeIndex + edgeTypeOffset];
318 if (type === hiddenEdgeType || type === invisibleEdgeType || type === internalEdgeType || type === weakEdgeType)
319 continue;
320 list.push(childNodeOrdinal);
321 }
322 }

325 _markPageOwnedNodes: function()
326 {
327 var edgeShortcutType = this._edgeShortcutType;
328 var edgeElementType = this._edgeElementType;
329 var edgeToNodeOffset = this._edgeToNodeOffset;
330 var edgeTypeOffset = this._edgeTypeOffset;
331 var edgeFieldsCount = this._edgeFieldsCount;
332 var edgeWeakType = this._edgeWeakType;
333 var firstEdgeIndexes = this._firstEdgeIndexes;
334 var containmentEdges = this.containmentEdges;
335 var nodeFieldCount = this._nodeFieldCount;
336 var nodesCount = this.nodeCount;
337
338 var flags = this._flags;
339 var pageObjectFlag = this._nodeFlags.pageObject;
340
341 var nodesToVisit = new Uint32Array(nodesCount);
342 var nodesToVisitLength = 0;
343
344 var rootNodeOrdinal = this._rootNodeIndex / nodeFieldCount;
345 var node = this.rootNode();
346
347 // Populate the entry points. They are Window objects and DOM Tree Roots.
348 for (var edgeIndex = firstEdgeIndexes[rootNodeOrdinal], endEdgeIndex = firstEdgeIndexes[rootNodeOrdinal + 1];
349 edgeIndex < endEdgeIndex;
350 edgeIndex += edgeFieldsCount) {
351 var edgeType = containmentEdges[edgeIndex + edgeTypeOffset];
352 var nodeIndex = containmentEdges[edgeIndex + edgeToNodeOffset];
353 if (edgeType === edgeElementType) {
354 node.nodeIndex = nodeIndex;
355 if (!node.isDocumentDOMTreesRoot())
356 continue;
357 } else if (edgeType !== edgeShortcutType)
358 continue;
359 var nodeOrdinal = nodeIndex / nodeFieldCount;
360 nodesToVisit[nodesToVisitLength++] = nodeOrdinal;
361 flags[nodeOrdinal] |= pageObjectFlag;
362 }
363
364 // Mark everything reachable with the pageObject flag.
365 while (nodesToVisitLength) {
366 var nodeOrdinal = nodesToVisit[--nodesToVisitLength];
367 var beginEdgeIndex = firstEdgeIndexes[nodeOrdinal];
368 var endEdgeIndex = firstEdgeIndexes[nodeOrdinal + 1];
369 for (var edgeIndex = beginEdgeIndex; edgeIndex < endEdgeIndex; edgeIndex += edgeFieldsCount) {
370 var childNodeIndex = containmentEdges[edgeIndex + edgeToNodeOffset];
371 var childNodeOrdinal = childNodeIndex / nodeFieldCount;
372 if (flags[childNodeOrdinal] & pageObjectFlag)
373 continue;
374 var type = containmentEdges[edgeIndex + edgeTypeOffset];
375 if (type === edgeWeakType)
376 continue;
377 nodesToVisit[nodesToVisitLength++] = childNodeOrdinal;
378 flags[childNodeOrdinal] |= pageObjectFlag;
379 }
380 }

383 /**
384 * @override
385 */
386 _calculateStatistics: function()
387 {
388 var nodeFieldCount = this._nodeFieldCount;
389 var nodes = this.nodes;
390 var nodesLength = nodes.length;
391 var nodeTypeOffset = this._nodeTypeOffset;
392 var nodeSizeOffset = this._nodeSelfSizeOffset;
393 var nodeNativeType = this._nodeNativeType;
394 var nodeCodeType = this._nodeCodeType;
395 var nodeConsStringType = this._nodeConsStringType;
396 var nodeSlicedStringType = this._nodeSlicedStringType;
397 var distances = this._nodeDistances;
398 var sizeNative = 0;
399 var sizeCode = 0;
400 var sizeStrings = 0;
401 var sizeJSArrays = 0;
402 var sizeSystem = 0;
403 var node = this.rootNode();
404 for (var nodeIndex = 0; nodeIndex < nodesLength; nodeIndex += nodeFieldCount) {
405 var nodeSize = nodes[nodeIndex + nodeSizeOffset];
406 var ordinal = nodeIndex / nodeFieldCount;
407 if (distances[ordinal] >= WebInspector.HeapSnapshotCommon.baseSystemDistance) {
408 sizeSystem += nodeSize;
409 continue;
410 }
411 var nodeType = nodes[nodeIndex + nodeTypeOffset];
412 node.nodeIndex = nodeIndex;
413 if (nodeType === nodeNativeType)
414 sizeNative += nodeSize;
415 else if (nodeType === nodeCodeType)
416 sizeCode += nodeSize;
417 else if (nodeType === nodeConsStringType || nodeType === nodeSlicedStringType || node.type() === "string")
418 sizeStrings += nodeSize;
419 else if (node.name() === "Array")
420 sizeJSArrays += this._calculateArraySize(node);
421 }
422 this._statistics = new WebInspector.HeapSnapshotCommon.Statistics();
423 this._statistics.total = this.totalSize;
424 this._statistics.v8heap = this.totalSize - sizeNative;
425 this._statistics.native = sizeNative;
426 this._statistics.code = sizeCode;
427 this._statistics.jsArrays = sizeJSArrays;
428 this._statistics.strings = sizeStrings;
429 this._statistics.system = sizeSystem;
430 },

432 /**
433 * @param {!WebInspector.HeapSnapshotNode} node
434 * @return {number}
435 */
436 _calculateArraySize: function(node)
437 {
438 var size = node.selfSize();
439 var beginEdgeIndex = node.edgeIndexesStart();
440 var endEdgeIndex = node.edgeIndexesEnd();
441 var containmentEdges = this.containmentEdges;
442 var strings = this.strings;
443 var edgeToNodeOffset = this._edgeToNodeOffset;
444 var edgeTypeOffset = this._edgeTypeOffset;
445 var edgeNameOffset = this._edgeNameOffset;
446 var edgeFieldsCount = this._edgeFieldsCount;
447 var edgeInternalType = this._edgeInternalType;
448 for (var edgeIndex = beginEdgeIndex; edgeIndex < endEdgeIndex; edgeIndex += edgeFieldsCount) {
449 var edgeType = containmentEdges[edgeIndex + edgeTypeOffset];
450 if (edgeType !== edgeInternalType)
451 continue;
452 var edgeName = strings[containmentEdges[edgeIndex + edgeNameOffset]];
453 if (edgeName !== "elements")
454 continue;
455 var elementsNodeIndex = containmentEdges[edgeIndex + edgeToNodeOffset];
456 node.nodeIndex = elementsNodeIndex;
457 if (node.retainersCount() === 1)
458 size += node.selfSize();
459 break;
460 }
461 return size;

522 _consStringName: function()
523 {
524 var snapshot = this._snapshot;
525 var consStringType = snapshot._nodeConsStringType;
526 var edgeInternalType = snapshot._edgeInternalType;
527 var edgeFieldsCount = snapshot._edgeFieldsCount;
528 var edgeToNodeOffset = snapshot._edgeToNodeOffset;

540 var name = "";
541
542 while (nodesStack.length && name.length < 1024) {
543 var nodeIndex = nodesStack.pop();
