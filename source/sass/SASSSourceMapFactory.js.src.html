9WebInspector.SASSSourceMapFactory = function()
10{
11 this._astService = new WebInspector.ASTService();
12}
13
14WebInspector.SASSSourceMapFactory.prototype = {
15 /**
16 * @override
17 * @param {!WebInspector.Target} target
18 * @param {!WebInspector.SourceMap} sourceMap
19 * @return {!Promise<?WebInspector.SourceMap>}
20 */
21 editableSourceMap: function(target, sourceMap)
22 {
23 var cssModel = WebInspector.CSSModel.fromTarget(target);
24 if (!cssModel)
25 return Promise.resolve(/** @type {?WebInspector.SourceMap} */(null));
26
27 var header = cssModel.styleSheetHeaders().find(styleSheetHeader => styleSheetHeader.sourceMapURL === sourceMap.url());
28 if (!header)
29 return Promise.resolve(/** @type {?WebInspector.SourceMap} */(null));
30
31 /** @type {!Map<string, !WebInspector.SASSSupport.AST>} */
32 var models = new Map();
33 var promises = [];
34 for (let url of sourceMap.sourceURLs()) {
35 var contentProvider = sourceMap.sourceContentProvider(url, WebInspector.resourceTypes.SourceMapStyleSheet);
36 var sassPromise = contentProvider.requestContent()
37 .then(text => this._astService.parseSCSS(url, text || ""))
38 .then(ast => models.set(ast.document.url, ast));
39 promises.push(sassPromise);
40 }
41 var cssURL = sourceMap.compiledURL();
42 var cssPromise = header.originalContentProvider().requestContent()
43 .then(text => this._astService.parseCSS(cssURL, text || ""))
44 .then(ast => models.set(ast.document.url, ast));
45 promises.push(cssPromise);
46
47 return Promise.all(promises)
48 .then(this._onSourcesParsed.bind(this, sourceMap, models))
49 .catchException(/** @type {?WebInspector.SourceMap} */(null));
50 },
51
52 /**
53 * @param {!WebInspector.SourceMap} sourceMap
54 * @param {!Map<string, !WebInspector.SASSSupport.AST>} models
55 * @return {?WebInspector.SourceMap}
56 */
57 _onSourcesParsed: function(sourceMap, models)
58 {
59 var editCallback = WebInspector.SASSProcessor.processCSSEdits.bind(WebInspector.SASSProcessor, this._astService);
60 var map = new WebInspector.ASTSourceMap(sourceMap.compiledURL(), sourceMap.url(), models, editCallback);
61 var valid = true;
62 map.compiledModel().visit(onNode);
63 return valid ? map : null;
64
65 /**
66 * @param {!WebInspector.SASSSupport.Node} cssNode
67 */
68 function onNode(cssNode)
69 {
70 if (!valid)
71 return;
72 if (!(cssNode instanceof WebInspector.SASSSupport.TextNode))
73 return;
74 var entry = sourceMap.findEntry(cssNode.range.startLine, cssNode.range.startColumn);
75 if (!entry || !entry.sourceURL || typeof entry.sourceLineNumber === "undefined" || typeof entry.sourceColumnNumber === "undefined")
76 return;
77 var sassAST = models.get(entry.sourceURL);
78 if (!sassAST)
79 return;
80 var sassNode = sassAST.findNodeForPosition(entry.sourceLineNumber, entry.sourceColumnNumber);
81 if (!sassNode)
82 return;
83 if (cssNode.parent && (cssNode.parent instanceof WebInspector.SASSSupport.Property) && cssNode === cssNode.parent.name && cssNode.text.trim() !== sassNode.text.trim()) {
84 valid = false;
85 reportError(cssNode, sassNode);
86 return;
87 }
88 map.addMapping(cssNode, sassNode);
89 }
90
91 /**
92 * @param {!WebInspector.SASSSupport.TextNode} cssNode
93 * @param {!WebInspector.SASSSupport.TextNode} sassNode
94 */
95 function reportError(cssNode, sassNode)
96 {
97 var text = WebInspector.UIString("LiveSASS failed to start: %s", sourceMap.url());
98 text += WebInspector.UIString("\nSourceMap is misaligned: %s != %s", cssNode.text.trim(), sassNode.text.trim());
99 text += "\ncompiled: " + cssNode.document.url + ":" + (cssNode.range.startLine + 1) + ":" + (cssNode.range.startColumn + 1);
100 text += "\nsource: " + sassNode.document.url + ":" + (sassNode.range.startLine + 1) + ":" + (sassNode.range.startColumn + 1);
101 WebInspector.console.error(text);
102 }
