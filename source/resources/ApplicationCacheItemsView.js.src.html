30WebInspector.ApplicationCacheItemsView = function(model, frameId)
31{
32 WebInspector.SimpleView.call(this, WebInspector.UIString("AppCache"));
33
34 this._model = model;
35
36 this.element.classList.add("storage-view", "table");
37
38 this._deleteButton = new WebInspector.ToolbarButton(WebInspector.UIString("Delete"), "delete-toolbar-item");
39 this._deleteButton.setVisible(false);
40 this._deleteButton.addEventListener("click", this._deleteButtonClicked, this);
41
42 this._connectivityIcon = createElement("label", "dt-icon-label");
43 this._connectivityIcon.style.margin = "0 2px 0 5px";
44 this._statusIcon = createElement("label", "dt-icon-label");
45 this._statusIcon.style.margin = "0 2px 0 5px";
46
47 this._frameId = frameId;
48
49 this._emptyWidget = new WebInspector.EmptyWidget(WebInspector.UIString("No Application Cache information available."));
50 this._emptyWidget.show(this.element);
51
52 this._markDirty();
53
54 var status = this._model.frameManifestStatus(frameId);
55 this.updateStatus(status);
56 this.updateNetworkState(this._model.onLine);
57
58 // FIXME: Status bar items don't work well enough yet, so they are being hidden.
59 // http://webkit.org/b/41637 Web Inspector: Give Semantics to "Refresh" and "Delete" Buttons in ApplicationCache DataGrid
60 this._deleteButton.element.style.display = "none";
61}

107 var oldStatus = this._status;

110 var statusInformation = {};
111 // We should never have UNCACHED status, since we remove frames with UNCACHED application cache status from the tree.
112 statusInformation[applicationCache.UNCACHED] = { type: "red-ball", text: "UNCACHED" };
113 statusInformation[applicationCache.IDLE] = { type: "green-ball", text: "IDLE" };
114 statusInformation[applicationCache.CHECKING] = { type: "orange-ball", text: "CHECKING" };
115 statusInformation[applicationCache.DOWNLOADING] = { type: "orange-ball", text: "DOWNLOADING" };
116 statusInformation[applicationCache.UPDATEREADY] = { type: "green-ball", text: "UPDATEREADY" };
117 statusInformation[applicationCache.OBSOLETE] = { type: "red-ball", text: "OBSOLETE" };
118
119 var info = statusInformation[status] || statusInformation[applicationCache.UNCACHED];
120
121 this._statusIcon.type = info.type;
122 this._statusIcon.textContent = info.text;

124 if (this.isShowing() && this._status === applicationCache.IDLE && (oldStatus === applicationCache.UPDATEREADY || !this._resources))

199 _populateDataGrid: function()
200 {
201 var selectedResource = this._dataGrid.selectedNode ? this._dataGrid.selectedNode.resource : null;
202 var sortDirection = this._dataGrid.isSortOrderAscending() ? 1 : -1;
203
204 function numberCompare(field, resource1, resource2)
205 {
206 return sortDirection * (resource1[field] - resource2[field]);
207 }
208 function localeCompare(field, resource1, resource2)
209 {
210 return sortDirection * (resource1[field] + "").localeCompare(resource2[field] + "");
211 }
212
213 var comparator;
214 switch (parseInt(this._dataGrid.sortColumnIdentifier(), 10)) {
215 case 0: comparator = localeCompare.bind(null, "name"); break;
216 case 1: comparator = localeCompare.bind(null, "type"); break;
217 case 2: comparator = numberCompare.bind(null, "size"); break;
218 default: localeCompare.bind(null, "resource"); // FIXME: comparator = ?
219 }
220
221 this._resources.sort(comparator);
222 this._dataGrid.rootNode().removeChildren();
223
224 var nodeToSelect;
225 for (var i = 0; i < this._resources.length; ++i) {
226 var data = {};
227 var resource = this._resources[i];
228 data[0] = resource.url;
229 data[1] = resource.type;
230 data[2] = Number.bytesToString(resource.size);
231 var node = new WebInspector.DataGridNode(data);
232 node.resource = resource;
233 node.selectable = true;
234 this._dataGrid.rootNode().appendChild(node);
235 if (resource === selectedResource) {
236 nodeToSelect = node;
237 nodeToSelect.selected = true;
238 }
239 }
240
241 if (!nodeToSelect && this._dataGrid.rootNode().children.length)
242 this._dataGrid.rootNode().children[0].selected = true;
