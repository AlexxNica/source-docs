39 this._promptElement = createElement("div");

43 this.element.appendChild(this._promptElement);

56 _messagesClicked: function()

62 /**
63 * @param {!Element} proxyElement
64 * @param {!Range} wordRange
65 * @param {boolean} force
66 * @param {function(!Array.<string>, number=)} completionsReadyCallback
67 */
68 completions: function(proxyElement, wordRange, force, completionsReadyCallback)
69 {
70 var prefix = wordRange.toString().toLowerCase();
71 if (!prefix)
72 return;
73 var results = [];
74
75 function accumulateMatches(textArray)
76 {
77 for (var i = 0; i < textArray.length; ++i) {
78 var text = textArray[i].toLowerCase();
79 if (text.length < prefix.length)
80 continue;
81 if (!text.startsWith(prefix))
82 continue;
83 results.push(textArray[i]);
84 }
85 }
86
87 function tableNamesCallback(tableNames)
88 {
89 accumulateMatches(tableNames.map(function(name) { return name + " "; }));
90 accumulateMatches(["SELECT ", "FROM ", "WHERE ", "LIMIT ", "DELETE FROM ", "CREATE ", "DROP ", "TABLE ", "INDEX ", "UPDATE ", "INSERT INTO ", "VALUES ("]);
91
92 completionsReadyCallback(results);
93 }
94 this.database.getTableNames(tableNamesCallback);
95 },

142 _queryFinished: function(query, columnNames, values)
143 {
144 var dataGrid = WebInspector.SortableDataGrid.create(columnNames, values);
145 var trimmedQuery = query.trim();
146
147 if (dataGrid) {

149 this._appendViewQueryResult(trimmedQuery, dataGrid.asWidget());
150 dataGrid.autoSizeColumns(5);

152
153 if (trimmedQuery.match(/^create /i) || trimmedQuery.match(/^drop table /i))
154 this.dispatchEventToListeners(WebInspector.DatabaseQueryView.Events.SchemaUpdated, this.database);
