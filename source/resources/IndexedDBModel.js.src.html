62/**
63 * @param {*} idbKey
64 * @return {({
65 * array: (!Array<?>|undefined),
66 * date: (number|undefined),
67 * number: (number|undefined),
68 * string: (string|undefined),
69 * type: !IndexedDBAgent.KeyType<string>
70 * }|undefined)}
71 */
72WebInspector.IndexedDBModel.keyFromIDBKey = function(idbKey)
73{
74 if (typeof(idbKey) === "undefined" || idbKey === null)
75 return undefined;
76
77 var type;
78 var key = {};
79 switch (typeof(idbKey)) {
80 case "number":
81 key.number = idbKey;
82 type = WebInspector.IndexedDBModel.KeyTypes.NumberType;
83 break;
84 case "string":
85 key.string = idbKey;
86 type = WebInspector.IndexedDBModel.KeyTypes.StringType;
87 break;
88 case "object":
89 if (idbKey instanceof Date) {
90 key.date = idbKey.getTime();
91 type = WebInspector.IndexedDBModel.KeyTypes.DateType;
92 } else if (Array.isArray(idbKey)) {
93 key.array = [];
94 for (var i = 0; i < idbKey.length; ++i)
95 key.array.push(WebInspector.IndexedDBModel.keyFromIDBKey(idbKey[i]));
96 type = WebInspector.IndexedDBModel.KeyTypes.ArrayType;
97 }
98 break;
99 default:
100 return undefined;
101 }
102 key.type = /** @type {!IndexedDBAgent.KeyType<string>} */ (type);
103 return key;
104}

106/**
107 * @param {?IDBKeyRange=} idbKeyRange
108 * @return {?IndexedDBAgent.KeyRange}
109 * eturn {?{lower: ?Object, upper: ?Object, lowerOpen: *, upperOpen: *}}
110 */
111WebInspector.IndexedDBModel.keyRangeFromIDBKeyRange = function(idbKeyRange)
112{
113 if (typeof idbKeyRange === "undefined" || idbKeyRange === null)
114 return null;
115
116 var keyRange = {};
117 keyRange.lower = WebInspector.IndexedDBModel.keyFromIDBKey(idbKeyRange.lower);
118 keyRange.upper = WebInspector.IndexedDBModel.keyFromIDBKey(idbKeyRange.upper);
119 keyRange.lowerOpen = !!idbKeyRange.lowerOpen;
120 keyRange.upperOpen = !!idbKeyRange.upperOpen;
121 return keyRange;
122}

132 case WebInspector.IndexedDBModel.KeyPathTypes.NullType:

336 /**
337 * @param {!WebInspector.IndexedDBModel.DatabaseId} databaseId
338 */
339 _loadDatabase: function(databaseId)
340 {
341 /**
342 * @param {?Protocol.Error} error
343 * @param {!IndexedDBAgent.DatabaseWithObjectStores} databaseWithObjectStores
344 * @this {WebInspector.IndexedDBModel}
345 */
346 function callback(error, databaseWithObjectStores)
347 {
348 if (error) {
349 console.error("IndexedDBAgent error: " + error);
350 return;
351 }
352
353 if (!this._databaseNamesBySecurityOrigin[databaseId.securityOrigin])
354 return;
355 var databaseModel = new WebInspector.IndexedDBModel.Database(databaseId, databaseWithObjectStores.version);
356 this._databases.set(databaseId, databaseModel);
357 for (var i = 0; i < databaseWithObjectStores.objectStores.length; ++i) {
358 var objectStore = databaseWithObjectStores.objectStores[i];
359 var objectStoreIDBKeyPath = WebInspector.IndexedDBModel.idbKeyPathFromKeyPath(objectStore.keyPath);
360 var objectStoreModel = new WebInspector.IndexedDBModel.ObjectStore(objectStore.name, objectStoreIDBKeyPath, objectStore.autoIncrement);
361 for (var j = 0; j < objectStore.indexes.length; ++j) {
362 var index = objectStore.indexes[j];
363 var indexIDBKeyPath = WebInspector.IndexedDBModel.idbKeyPathFromKeyPath(index.keyPath);
364 var indexModel = new WebInspector.IndexedDBModel.Index(index.name, indexIDBKeyPath, index.unique, index.multiEntry);
365 objectStoreModel.indexes[indexModel.name] = indexModel;
366 }
367 databaseModel.objectStores[objectStoreModel.name] = objectStoreModel;
368 }
369
370 this.dispatchEventToListeners(WebInspector.IndexedDBModel.EventTypes.DatabaseLoaded, databaseModel);
371 }
372
373 this._agent.requestDatabase(databaseId.securityOrigin, databaseId.name, callback.bind(this));
374 },

413 _requestData: function(databaseId, databaseName, objectStoreName, indexName, idbKeyRange, skipCount, pageSize, callback)
414 {
415 /**
416 * @param {?Protocol.Error} error
417 * @param {!Array.<!IndexedDBAgent.DataEntry>} dataEntries
418 * @param {boolean} hasMore
419 * @this {WebInspector.IndexedDBModel}
420 */
421 function innerCallback(error, dataEntries, hasMore)
422 {
423 if (error) {
424 console.error("IndexedDBAgent error: " + error);
425 return;
426 }
427
428 if (!this._databaseNamesBySecurityOrigin[databaseId.securityOrigin])
429 return;
430 var entries = [];
431 for (var i = 0; i < dataEntries.length; ++i) {
432 var key = this.target().runtimeModel.createRemoteObject(dataEntries[i].key);
433 var primaryKey = this.target().runtimeModel.createRemoteObject(dataEntries[i].primaryKey);
434 var value = this.target().runtimeModel.createRemoteObject(dataEntries[i].value);
435 entries.push(new WebInspector.IndexedDBModel.Entry(key, primaryKey, value));
436 }
437 callback(entries, hasMore);
438 }
439
440 var keyRange = WebInspector.IndexedDBModel.keyRangeFromIDBKeyRange(idbKeyRange);
441 this._agent.requestData(databaseId.securityOrigin, databaseName, objectStoreName, indexName, skipCount, pageSize, keyRange ? keyRange : undefined, innerCallback.bind(this));
