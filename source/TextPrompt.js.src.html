301 /**
302 * @param {!Event} event
303 */
304 onKeyDown: function(event)
305 {
306 if (isEnterKey(event))
307 return;
308
309 var handled = false;
310 delete this._needUpdateAutocomplete;
311
312 switch (event.key) {
313 case "Tab":
314 handled = this.tabKeyPressed(event);
315 break;
316 case "ArrowLeft":
317 case "Home":
318 this._removeSuggestionAids();
319 break;
320 case "ArrowRight":
321 case "End":
322 if (this.isCaretAtEndOfPrompt())
323 handled = this.acceptAutoComplete();
324 else
325 this._removeSuggestionAids();
326 break;
327 case "Escape":
328 if (this.isSuggestBoxVisible()) {
329 this._removeSuggestionAids();
330 handled = true;
331 }
332 break;
333 case " ": // Space
334 if (event.ctrlKey && !event.metaKey && !event.altKey && !event.shiftKey) {
335 this._updateAutoComplete(true);
336 handled = true;
337 }
338 break;
339 case "Alt":
340 case "Meta":
341 case "Shift":
342 case "Control":
343 break;
344 }
345
346 if (!handled && this.isSuggestBoxVisible())
347 handled = this._suggestBox.keyPressed(event);
348
349 if (!handled)
350 this._needUpdateAutocomplete = true;
351
352 if (handled)
353 event.consume(true);

409 /**
410 * @param {boolean=} force
411 * @param {boolean=} reverse
412 */
413 complete: function(force, reverse)
414 {
415 this.clearAutoComplete(true);
416 var selection = this._element.getComponentSelection();
417 var selectionRange = selection && selection.rangeCount ? selection.getRangeAt(0) : null;
418 if (!selectionRange)
419 return;
420
421 var shouldExit;
422
423 if (!force && !this.isCaretAtEndOfPrompt() && !this.isSuggestBoxVisible())
424 shouldExit = true;
425 else if (!selection.isCollapsed)
426 shouldExit = true;
427 else if (!force) {
428 // BUG72018: Do not show suggest box if caret is followed by a non-stop character.
429 var wordSuffixRange = selectionRange.startContainer.rangeOfWord(selectionRange.endOffset, this._completionStopCharacters, this._element, "forward");
430 if (wordSuffixRange.toString().length)
431 shouldExit = true;
432 }
433 if (shouldExit) {
434 this.hideSuggestBox();
435 return;
436 }
437
438 var wordPrefixRange = selectionRange.startContainer.rangeOfWord(selectionRange.startOffset, this._completionStopCharacters, this._element, "backward");
439 this._waitingForCompletions = true;
440 this._loadCompletions(/** @type {!Element} */ (this._proxyElement), wordPrefixRange, force || false, this._completionsReady.bind(this, selection, wordPrefixRange, !!reverse, !!force));

465 /**
466 * @param {!Array.<string>} completions
467 * @param {number} wordPrefixLength
468 */
469 _buildCommonPrefix: function(completions, wordPrefixLength)
470 {
471 var commonPrefix = completions[0];
472 for (var i = 0; i < completions.length; ++i) {
473 var completion = completions[i];
474 var lastIndex = Math.min(commonPrefix.length, completion.length);
475 for (var j = wordPrefixLength; j < lastIndex; ++j) {
476 if (commonPrefix[j] !== completion[j]) {
477 commonPrefix = commonPrefix.substr(0, j);
478 break;
479 }
480 }
481 }
482 return commonPrefix;

503 /**
504 * @param {!Selection} selection
505 * @param {!Range} originalWordPrefixRange
506 * @param {boolean} reverse
507 * @param {boolean} force
508 * @param {!Array.<string>} completions
509 * @param {number=} selectedIndex
510 */
511 _completionsReady: function(selection, originalWordPrefixRange, reverse, force, completions, selectedIndex)
512 {
513 var prefix = originalWordPrefixRange.toString();
514
515 // Filter out dupes.
516 var store = new Set();
517 completions = completions.filter(item => !store.has(item) && !!store.add(item));
518 var annotatedCompletions = completions.map(item => ({title: item}));
519
520 if (prefix || force) {
521 if (prefix)
522 annotatedCompletions = annotatedCompletions.concat(this.additionalCompletions(prefix));
523 else
524 annotatedCompletions = this.additionalCompletions(prefix).concat(annotatedCompletions);
525 }
526
527 if (!this._waitingForCompletions || !annotatedCompletions.length) {
528 this.hideSuggestBox();
529 return;
530 }
531
532 delete this._waitingForCompletions;
533
534 var selectionRange = selection.getRangeAt(0);
535
536 var fullWordRange = this._createRange();
537 fullWordRange.setStart(originalWordPrefixRange.startContainer, originalWordPrefixRange.startOffset);
538 fullWordRange.setEnd(selectionRange.endContainer, selectionRange.endOffset);
539
540 if (prefix + selectionRange.toString() !== fullWordRange.toString())
541 return;
542
543 selectedIndex = (this._disableDefaultSuggestionForEmptyInput && !this.text()) ? -1 : (selectedIndex || 0);
544
545 this._userEnteredRange = fullWordRange;
546 this._userEnteredText = fullWordRange.toString();
547
548 if (this._suggestBox)
549 this._suggestBox.updateSuggestions(this._boxForAnchorAtStart(selection, fullWordRange), annotatedCompletions, selectedIndex, !this.isCaretAtEndOfPrompt(), this._userEnteredText);
550
551 if (selectedIndex === -1)
552 return;
553
554 var wordPrefixLength = originalWordPrefixRange.toString().length;
555 this._commonPrefix = this._buildCommonPrefix(completions, wordPrefixLength);
556
557 if (this.isCaretAtEndOfPrompt()) {
558 var completionText = annotatedCompletions[selectedIndex].title;
559 var prefixText = this._userEnteredRange.toString();
560 var suffixText = completionText.substring(wordPrefixLength);
561 this._userEnteredRange.deleteContents();
562 this._element.normalize();
563 var finalSelectionRange = this._createRange();
564
565 var prefixTextNode = createTextNode(prefixText);
566 fullWordRange.insertNode(prefixTextNode);
567
568 this.autoCompleteElement = createElementWithClass("span", "auto-complete-text");
569 this.autoCompleteElement.textContent = suffixText;
570
571 prefixTextNode.parentNode.insertBefore(this.autoCompleteElement, prefixTextNode.nextSibling);
572
573 finalSelectionRange.setStart(prefixTextNode, wordPrefixLength);
574 finalSelectionRange.setEnd(prefixTextNode, wordPrefixLength);
575 selection.removeAllRanges();
576 selection.addRange(finalSelectionRange);
577 this.dispatchEventToListeners(WebInspector.TextPrompt.Events.ItemApplied);
578 }

605 /**

607 * @param {boolean=} isIntermediateSuggestion
608 */
609 _applySuggestion: function(completionText, isIntermediateSuggestion)
610 {
611 if (!this._userEnteredRange) {
612 // We could have already cleared autocompletion range by the time this is called. (crbug.com/587683)
613 return;
614 }
615
616 var wordPrefixLength = this._userEnteredText ? this._userEnteredText.length : 0;
617

620 var finalSelectionRange = this._createRange();

624 this.autoCompleteElement.remove();

626 }
627
628 if (isIntermediateSuggestion)
629 finalSelectionRange.setStart(completionTextNode, wordPrefixLength);
630 else
631 finalSelectionRange.setStart(completionTextNode, completionText.length);
632
633 finalSelectionRange.setEnd(completionTextNode, completionText.length);
634
635 var selection = this._element.getComponentSelection();
636 selection.removeAllRanges();
637 selection.addRange(finalSelectionRange);
638 if (isIntermediateSuggestion)
639 this.dispatchEventToListeners(WebInspector.TextPrompt.Events.ItemApplied, { itemText: completionText });

703 /**
704 * @return {boolean}
705 */
706 isCaretAtEndOfPrompt: function()
707 {
708 var selection = this._element.getComponentSelection();
709 var selectionRange = selection && selection.rangeCount ? selection.getRangeAt(0) : null;
710 if (!selectionRange || !selection.isCollapsed)
711 return false;
712
713 var node = selectionRange.startContainer;
714 if (!node.isSelfOrDescendant(this._element))
715 return false;
716
717 if (node.nodeType === Node.TEXT_NODE && selectionRange.startOffset < node.nodeValue.length)
718 return false;
719
720 var foundNextText = false;
721 while (node) {
722 if (node.nodeType === Node.TEXT_NODE && node.nodeValue.length) {
723 if (foundNextText && (!this.autoCompleteElement || !this.autoCompleteElement.isAncestor(node)))
724 return false;
725 foundNextText = true;
726 }
727
728 node = node.traverseNextNode(this._element);
729 }
730
731 return true;

734 /**
735 * @return {boolean}
736 */
737 isCaretOnFirstLine: function()
738 {
739 var selection = this._element.getComponentSelection();
740 var focusNode = selection.focusNode;
741 if (!focusNode || focusNode.nodeType !== Node.TEXT_NODE || focusNode.parentNode !== this._element)
742 return true;
743
744 if (focusNode.textContent.substring(0, selection.focusOffset).indexOf("\n") !== -1)
745 return false;
746 focusNode = focusNode.previousSibling;
747
748 while (focusNode) {
749 if (focusNode.nodeType !== Node.TEXT_NODE)
750 return true;
751 if (focusNode.textContent.indexOf("\n") !== -1)
752 return false;
753 focusNode = focusNode.previousSibling;
754 }
755
756 return true;
