522 /**
523 * @param {number} lineNumber
524 * @param {number} columnNumber
525 */
526 _updateWavePosition: function(lineNumber, columnNumber)
527 {
528 lineNumber = Math.min(lineNumber, this._textEditor.linesCount - 1);
529 var lineText = this._textEditor.line(lineNumber);
530 columnNumber = Math.min(columnNumber, lineText.length);
531 var lineIndent = WebInspector.TextUtils.lineIndent(lineText).length;
532 var base = this._textEditor.cursorPositionToCoordinates(lineNumber, 0);
533
534 var start = this._textEditor.cursorPositionToCoordinates(lineNumber, Math.max(columnNumber - 1, lineIndent));
535 var end = this._textEditor.cursorPositionToCoordinates(lineNumber, lineText.length);
536 /** @const */
537 var codeMirrorLinesLeftPadding = 4;
538 this._wave.style.left = (start.x - base.x + codeMirrorLinesLeftPadding) + "px";
539 this._wave.style.width = (end.x - start.x) + "px";

579 var rowMessage = this._messages[i];

595 {
596 for (var i = 0; i < this._messages.length; ++i) {
597 var rowMessage = this._messages[i];
598 if (!rowMessage.message().isEqual(message))
599 continue;
600 rowMessage.setRepeatCount(rowMessage.repeatCount() - 1);
601 if (!rowMessage.repeatCount())
602 this._messages.splice(i, 1);
603 this._updateDecoration();
604 return;
605 }

608 _updateDecoration: function()
609 {
610 if (!this._sourceFrame.isEditorShowing())
611 return;
612 if (!this._messages.length)
613 return;
614 var position = this._lineHandle.resolve();
615 if (!position)
616 return;
617
618 var lineNumber = position.lineNumber;
619 var columnNumber = Number.MAX_VALUE;
620 var maxMessage = null;
621 for (var i = 0; i < this._messages.length; ++i) {
622 var message = this._messages[i].message();
623 columnNumber = Math.min(columnNumber, message.columnNumber());
624 if (!maxMessage || WebInspector.UISourceCode.Message.messageLevelComparator(maxMessage, message) < 0)
625 maxMessage = message;
626 }
627 this._updateWavePosition(lineNumber, columnNumber);
628
629 if (this._level) {
630 this._textEditor.toggleLineClass(lineNumber, WebInspector.UISourceCodeFrame._lineClassPerLevel[this._level], false);
631 this._icon.type = "";
632 }
633 this._level = maxMessage.level();
634 if (!this._level)
635 return;
636 this._textEditor.toggleLineClass(lineNumber, WebInspector.UISourceCodeFrame._lineClassPerLevel[this._level], true);
637 this._icon.type = WebInspector.UISourceCodeFrame._iconClassPerLevel[this._level];
638 }
