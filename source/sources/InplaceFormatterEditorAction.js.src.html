15 /**

18 _editorSelected: function(event)
19 {
20 var uiSourceCode = /** @type {!WebInspector.UISourceCode} */ (event.data);

24 /**
25 * @param {!WebInspector.Event} event

27 _editorClosed: function(event)
28 {
29 var wasSelected = /** @type {boolean} */ (event.data.wasSelected);
30 if (wasSelected)
31 this._updateButton(null);
32 },
33
34 /**
35 * @param {?WebInspector.UISourceCode} uiSourceCode
36 */
37 _updateButton: function(uiSourceCode)
38 {
39 this._button.element.classList.toggle("hidden", !this._isFormattable(uiSourceCode));
40 },
41
42 /**
43 * @override
44 * @param {!WebInspector.SourcesView} sourcesView
45 * @return {!WebInspector.ToolbarButton}
46 */
47 button: function(sourcesView)
48 {
49 if (this._button)
50 return this._button;
51
52 this._sourcesView = sourcesView;
53 this._sourcesView.addEventListener(WebInspector.SourcesView.Events.EditorSelected, this._editorSelected.bind(this));
54 this._sourcesView.addEventListener(WebInspector.SourcesView.Events.EditorClosed, this._editorClosed.bind(this));
55
56 this._button = new WebInspector.ToolbarButton(WebInspector.UIString("Format"), "format-toolbar-item");
57 this._button.addEventListener("click", this._formatSourceInPlace, this);
58 this._updateButton(sourcesView.currentUISourceCode());
59
60 return this._button;
61 },
62
63 /**
64 * @param {?WebInspector.UISourceCode} uiSourceCode
65 * @return {boolean}
66 */
67 _isFormattable: function(uiSourceCode)
68 {
69 if (!uiSourceCode)
70 return false;
71 if (uiSourceCode.project().type() === WebInspector.projectTypes.FileSystem)
72 return true;
73 return uiSourceCode.contentType().isStyleSheet()
74 || uiSourceCode.project().type() === WebInspector.projectTypes.Snippets;
75 },
76
77 _formatSourceInPlace: function()
78 {
79 var uiSourceCode = this._sourcesView.currentUISourceCode();
80 if (!this._isFormattable(uiSourceCode))
81 return;
82
83 if (uiSourceCode.isDirty())
84 contentLoaded.call(this, uiSourceCode.workingCopy());
85 else
86 uiSourceCode.requestContent().then(contentLoaded.bind(this));
87
88 /**
89 * @this {WebInspector.InplaceFormatterEditorAction}
90 * @param {?string} content
91 */
92 function contentLoaded(content)
93 {
94 var highlighterType = WebInspector.NetworkProject.uiSourceCodeMimeType(uiSourceCode);
95 WebInspector.Formatter.format(uiSourceCode.contentType(), highlighterType, content || "", innerCallback.bind(this));
96 }
97
98 /**
99 * @this {WebInspector.InplaceFormatterEditorAction}
100 * @param {string} formattedContent
101 * @param {!WebInspector.FormatterSourceMapping} formatterMapping
102 */
103 function innerCallback(formattedContent, formatterMapping)
104 {
105 if (uiSourceCode.workingCopy() === formattedContent)
106 return;
107 var sourceFrame = this._sourcesView.viewForFile(uiSourceCode);
108 var start = [0, 0];
109 if (sourceFrame) {
110 var selection = sourceFrame.selection();
111 start = formatterMapping.originalToFormatted(selection.startLine, selection.startColumn);
112 }
113 uiSourceCode.setWorkingCopy(formattedContent);
114 this._sourcesView.showSourceLocation(uiSourceCode, start[0], start[1]);
115 }
