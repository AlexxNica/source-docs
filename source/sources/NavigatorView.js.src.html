29/**
30 * @constructor
31 * @extends {WebInspector.VBox}
32 * @implements {WebInspector.TargetManager.Observer}
33 */
34WebInspector.NavigatorView = function()
35{
36 WebInspector.VBox.call(this);
37
38 this._scriptsTree = new TreeOutlineInShadow();
39 this._scriptsTree.registerRequiredCSS("sources/navigatorView.css");
40 this._scriptsTree.setComparator(WebInspector.NavigatorView._treeElementsCompare);
41 this.element.appendChild(this._scriptsTree.element);
42 this.setDefaultFocusedElement(this._scriptsTree.element);
43
44 /** @type {!Map.<!WebInspector.UISourceCode, !WebInspector.NavigatorUISourceCodeTreeNode>} */
45 this._uiSourceCodeNodes = new Map();
46 /** @type {!Map.<string, !WebInspector.NavigatorFolderTreeNode>} */
47 this._subfolderNodes = new Map();
48
49 this._rootNode = new WebInspector.NavigatorRootTreeNode(this);
50 this._rootNode.populate();
51
52 /** @type {!Map.<!WebInspector.ResourceTreeFrame, !WebInspector.NavigatorGroupTreeNode>} */
53 this._frameNodes = new Map();
54
55 this.element.addEventListener("contextmenu", this.handleContextMenu.bind(this), false);
56
57 this._navigatorGroupByFolderSetting = WebInspector.moduleSetting("navigatorGroupByFolder");
58 this._navigatorGroupByFolderSetting.addChangeListener(this._groupingChanged.bind(this));
59
60 this._initGrouping();
61 WebInspector.targetManager.addModelListener(WebInspector.ResourceTreeModel, WebInspector.ResourceTreeModel.EventTypes.FrameNavigated, this._frameNavigated, this);
62 WebInspector.targetManager.addModelListener(WebInspector.ResourceTreeModel, WebInspector.ResourceTreeModel.EventTypes.FrameDetached, this._frameDetached, this);
63 WebInspector.targetManager.observeTargets(this);
64 this._resetWorkspace(WebInspector.workspace);
65}

80/**
81 * @param {!TreeElement} treeElement
82 */
83WebInspector.NavigatorView._treeElementOrder = function(treeElement)
84{
85 if (treeElement._boostOrder)
86 return 0;
87
88 if (!WebInspector.NavigatorView._typeOrders) {
89 var weights = {};
90 var types = WebInspector.NavigatorView.Types;
91 weights[types.Root] = 1;
92 weights[types.Category] = 1;
93 weights[types.Domain] = 10;
94 weights[types.FileSystemFolder] = 1;
95 weights[types.NetworkFolder] = 1;
96 weights[types.SourceMapFolder] = 2;
97 weights[types.File] = 10;
98 weights[types.Frame] = 70;
99 weights[types.Worker] = 90;
100 weights[types.FileSystem] = 100;
101 WebInspector.NavigatorView._typeOrders = weights;
102 }
103
104 var order = WebInspector.NavigatorView._typeOrders[treeElement._nodeType];
105 if (treeElement._uiSourceCode) {
106 var contentType = treeElement._uiSourceCode.contentType();
107 if (contentType.isDocument())
108 order += 3;
109 else if (contentType.isScript())
110 order += 5;
111 else if (contentType.isStyleSheet())
112 order += 10;
113 else
114 order += 15;
115 }
116
117 return order;
118}

285 /**
286 * @param {!WebInspector.UISourceCode} uiSourceCode
287 * @param {!WebInspector.Project} project
288 * @param {?WebInspector.Target} target
289 * @param {?WebInspector.ResourceTreeFrame} frame
290 * @param {string} projectOrigin
291 * @param {!Array<string>} path
292 * @param {boolean} fromSourceMap
293 * @return {!WebInspector.NavigatorTreeNode}
294 */
295 _folderNode: function(uiSourceCode, project, target, frame, projectOrigin, path, fromSourceMap)
296 {
297 if (project.type() === WebInspector.projectTypes.Snippets)
298 return this._rootNode;
299
300 if (target && !this._groupByFolder && !fromSourceMap)
301 return this._domainNode(uiSourceCode, project, target, frame, projectOrigin);
302
303 var folderPath = path.join("/");
304 var folderId = this._folderNodeId(project, target, frame, projectOrigin, folderPath);
305 var folderNode = this._subfolderNodes.get(folderId);
306 if (folderNode)
307 return folderNode;
308
309 if (!path.length) {
310 if (target)
311 return this._domainNode(uiSourceCode, project, target, frame, projectOrigin);
312 var fileSystemNode = this._rootNode.child(project.id());
313 if (!fileSystemNode) {
314 fileSystemNode = new WebInspector.NavigatorGroupTreeNode(this, project, project.id(), WebInspector.NavigatorView.Types.FileSystem, project.displayName());
315 this._rootNode.appendChild(fileSystemNode);
316 }
317 return fileSystemNode;
318 }
319
320 var parentNode = this._folderNode(uiSourceCode, project, target, frame, projectOrigin, path.slice(0, -1), fromSourceMap);
321 var type = fromSourceMap ? WebInspector.NavigatorView.Types.SourceMapFolder : WebInspector.NavigatorView.Types.NetworkFolder;
322 if (project.type() === WebInspector.projectTypes.FileSystem)
323 type = WebInspector.NavigatorView.Types.FileSystemFolder;
324 var name = path[path.length - 1];
325
326 folderNode = new WebInspector.NavigatorFolderTreeNode(this, project, folderId, type, folderPath, name);
327 this._subfolderNodes.set(folderId, folderNode);
328 parentNode.appendChild(folderNode);
329 return folderNode;

412 /**
413 * @param {!WebInspector.Target} target
414 * @param {string} projectOrigin
415 * @return {string}
416 */
417 _computeProjectDisplayName: function(target, projectOrigin)
418 {
419 for (var context of target.runtimeModel.executionContexts()) {
420 if (context.name && context.origin && projectOrigin.startsWith(context.origin))
421 return context.name;
422 }
423
424 if (!projectOrigin)
425 return WebInspector.UIString("(no domain)");
426
427 var parsedURL = new WebInspector.ParsedURL(projectOrigin);
428 var prettyURL = parsedURL.isValid ? parsedURL.host + (parsedURL.port ? (":" + parsedURL.port) : "") : "";
429
430 return (prettyURL || projectOrigin);

465 /**
466 * @param {!WebInspector.UISourceCode} uiSourceCode
467 */
468 _removeUISourceCode: function(uiSourceCode)
469 {
470 var node = this._uiSourceCodeNodes.get(uiSourceCode);
471 if (!node)
472 return;
473
474 var project = uiSourceCode.project();
475 var target = WebInspector.NetworkProject.targetForUISourceCode(uiSourceCode);
476 var frame = this._uiSourceCodeFrame(uiSourceCode);
477
478 var parentNode = node.parent;
479 this._uiSourceCodeNodes.delete(uiSourceCode);
480 parentNode.removeChild(node);
481 node = parentNode;
482
483 while (node) {
484 parentNode = node.parent;
485 if (!parentNode || !node.isEmpty())
486 break;
487 if (!(node instanceof WebInspector.NavigatorGroupTreeNode || node instanceof WebInspector.NavigatorFolderTreeNode))
488 break;
489 if (node._type === WebInspector.NavigatorView.Types.Frame)
490 break;
491
492 var folderId = this._folderNodeId(project, target, frame, uiSourceCode.origin(), node._folderPath);
493 this._subfolderNodes.delete(folderId);
494 parentNode.removeChild(node);
495 node = parentNode;
496 }
