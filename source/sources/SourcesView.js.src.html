60 function handleBeforeUnload(event)
61 {
62 if (event.returnValue)
63 return;
64 var unsavedSourceCodes = WebInspector.workspace.unsavedSourceCodes();
65 if (!unsavedSourceCodes.length)
66 return;
67
68 event.returnValue = WebInspector.UIString("DevTools have unsaved changes that will be permanently lost.");
69 WebInspector.inspectorView.setCurrentPanel(WebInspector.SourcesPanel.instance());
70 for (var i = 0; i < unsavedSourceCodes.length; ++i)
71 WebInspector.Revealer.reveal(unsavedSourceCodes[i]);
72 }
73 if (!window.opener)
74 window.addEventListener("beforeunload", handleBeforeUnload, true);
75
76 this._shortcuts = {};
77 this.element.addEventListener("keydown", this._handleKeyDown.bind(this), false);
78}

298 /**
299 * @param {!WebInspector.UISourceCode} uiSourceCode
300 * @return {!WebInspector.Widget}
301 */
302 _createSourceView: function(uiSourceCode)
303 {
304 var sourceFrame;
305 var sourceView;
306 var contentType = uiSourceCode.contentType();
307
308 if (contentType.hasScripts())
309 sourceFrame = new WebInspector.JavaScriptSourceFrame(uiSourceCode);
310 else if (contentType.isStyleSheet())
311 sourceFrame = new WebInspector.CSSSourceFrame(uiSourceCode);
312 else if (contentType === WebInspector.resourceTypes.Image)
313 sourceView = new WebInspector.ImageView(WebInspector.NetworkProject.uiSourceCodeMimeType(uiSourceCode), uiSourceCode);
314 else if (contentType === WebInspector.resourceTypes.Font)
315 sourceView = new WebInspector.FontView(WebInspector.NetworkProject.uiSourceCodeMimeType(uiSourceCode), uiSourceCode);
316 else
317 sourceFrame = new WebInspector.UISourceCodeFrame(uiSourceCode);
318
319 if (sourceFrame) {
320 sourceFrame.setHighlighterType(WebInspector.NetworkProject.uiSourceCodeMimeType(uiSourceCode));
321 this._historyManager.trackSourceFrameCursorJumps(sourceFrame);
322 }
323 var widget = /** @type {!WebInspector.Widget} */(sourceFrame || sourceView);
324 this._sourceViewByUISourceCode.set(uiSourceCode, widget);
325 uiSourceCode.addEventListener(WebInspector.UISourceCode.Events.TitleChanged, this._uiSourceCodeTitleChanged, this);
326 return widget;

703/**
704 * @param {!WebInspector.UISourceCode} currentUISourceCode
705 * @return {?WebInspector.UISourceCode}
706 */
707WebInspector.SourcesView.SwitchFileActionDelegate._nextFile = function(currentUISourceCode)
708{
709 /**
710 * @param {string} name
711 * @return {string}
712 */
713 function fileNamePrefix(name)
714 {
715 var lastDotIndex = name.lastIndexOf(".");
716 var namePrefix = name.substr(0, lastDotIndex !== -1 ? lastDotIndex : name.length);
717 return namePrefix.toLowerCase();
718 }
719
720 var uiSourceCodes = currentUISourceCode.project().uiSourceCodes();
721 var candidates = [];
722 var url = currentUISourceCode.parentURL();
723 var name = currentUISourceCode.name();
724 var namePrefix = fileNamePrefix(name);
725 for (var i = 0; i < uiSourceCodes.length; ++i) {
726 var uiSourceCode = uiSourceCodes[i];
727 if (url !== uiSourceCode.parentURL())
728 continue;
729 if (fileNamePrefix(uiSourceCode.name()) === namePrefix)
730 candidates.push(uiSourceCode.name());
731 }
732 candidates.sort(String.naturalOrderComparator);
733 var index = mod(candidates.indexOf(name) + 1, candidates.length);
734 var fullURL = (url ? url + "/" : "") + candidates[index];
735 var nextUISourceCode = currentUISourceCode.project().uiSourceCodeForURL(fullURL);
736 return nextUISourceCode !== currentUISourceCode ? nextUISourceCode : null;
737}
