278 WebInspector.debuggerWorkspaceBinding.setSourceMapping(target, uiSourceCodes[i], null);

280 var scripts = [];
281 for (var j = 0; j < formatData.scripts.length; ++j) {
282 if (formatData.scripts[j].target() === target)
283 this._uiSourceCodes.remove(formatData.scripts[j]);
284 else
285 scripts.push(formatData.scripts[j]);
286 }
287
288 if (scripts.length)
289 formatData.scripts = scripts;
290 else {
291 this._formattedPaths.remove(formatData.projectId + ":" + formatData.path);
292 this._formatData.remove(uiSourceCodes[i]);
293 this._project.removeFile(uiSourceCodes[i].url());

307 /**
308 * @param {!WebInspector.UISourceCode} uiSourceCode
309 * @return {!Array.<!WebInspector.Script>}
310 */
311 _scriptsForUISourceCode: function(uiSourceCode)
312 {
313 /**
314 * @param {!WebInspector.Script} script
315 * @return {boolean}
316 */
317 function isInlineScript(script)
318 {
319 return script.isInlineScript() && !script.hasSourceURL;
320 }
321
322 if (uiSourceCode.contentType() === WebInspector.resourceTypes.Document) {
323 var scripts = [];
324 var debuggerModels = WebInspector.DebuggerModel.instances();
325 for (var i = 0; i < debuggerModels.length; ++i) {
326 var networkURL = WebInspector.networkMapping.networkURL(uiSourceCode);
327 scripts.pushAll(debuggerModels[i].scriptsForSourceURL(networkURL));
328 }
329 return scripts.filter(isInlineScript);
330 }
331 if (uiSourceCode.contentType().isScript()) {
332 var rawLocations = WebInspector.debuggerWorkspaceBinding.uiLocationToRawLocations(uiSourceCode, 0, 0);
333 return rawLocations.map(function(rawLocation) { return rawLocation.script(); });
334 }
335 return [];
336 },

338 /**
339 * @param {!WebInspector.UISourceCode} uiSourceCode
340 */
341 _formatUISourceCodeScript: function(uiSourceCode)
342 {
343 var formattedPath = this._formattedPaths.get(uiSourceCode.project().id() + ":" + uiSourceCode.url());
344 if (formattedPath) {
345 var uiSourceCodePath = formattedPath;
346 var formattedUISourceCode = this._workspace.uiSourceCode(this._projectId, uiSourceCodePath);
347 var formatData = formattedUISourceCode ? this._formatData.get(formattedUISourceCode) : null;
348 if (formatData)
349 this._showIfNeeded(uiSourceCode, /** @type {!WebInspector.UISourceCode} */ (formattedUISourceCode), formatData.mapping);
350 return;
351 }

365 /**
366 * @this {WebInspector.ScriptFormatterEditorAction}
367 * @param {string} formattedContent
368 * @param {!WebInspector.FormatterSourceMapping} formatterMapping
369 */
370 function innerCallback(formattedContent, formatterMapping)
371 {
372 var scripts = this._scriptsForUISourceCode(uiSourceCode);
373 var formattedURL = uiSourceCode.url() + ":formatted";
374 var contentProvider = WebInspector.StaticContentProvider.fromString(formattedURL, uiSourceCode.contentType(), formattedContent);
375 var formattedUISourceCode = this._project.addContentProvider(formattedURL, contentProvider);
376 var formattedPath = formattedUISourceCode.url();
377 var formatData = new WebInspector.FormatterScriptMapping.FormatData(uiSourceCode.project().id(), uiSourceCode.url(), formatterMapping, scripts);
378 this._formatData.set(formattedUISourceCode, formatData);
379 var path = uiSourceCode.project().id() + ":" + uiSourceCode.url();
380 this._formattedPaths.set(path, formattedPath);
381 this._pathsToFormatOnLoad.add(path);
382 for (var i = 0; i < scripts.length; ++i) {
383 this._uiSourceCodes.set(scripts[i], formattedUISourceCode);
384 var scriptMapping = /** @type {!WebInspector.FormatterScriptMapping} */(this._scriptMappingByTarget.get(scripts[i].target()));
385 WebInspector.debuggerWorkspaceBinding.pushSourceMapping(scripts[i], scriptMapping);
386 }
387
388 var targets = WebInspector.targetManager.targets();
389 for (var i = 0; i < targets.length; ++i) {
390 var scriptMapping = /** @type {!WebInspector.FormatterScriptMapping} */(this._scriptMappingByTarget.get(targets[i]));
391 WebInspector.debuggerWorkspaceBinding.setSourceMapping(targets[i], formattedUISourceCode, scriptMapping);
392 }
393 this._showIfNeeded(uiSourceCode, formattedUISourceCode, formatterMapping);
394 }
