10WebInspector.ObjectEventListenersSidebarPane = function()
11{
12 WebInspector.VBox.call(this);

14
15 this._refreshButton = new WebInspector.ToolbarButton(WebInspector.UIString("Refresh"), "refresh-toolbar-item");
16 this._refreshButton.addEventListener("click", this._refreshClick.bind(this));
17 this._refreshButton.setEnabled(false);

34 update: function()
35 {
36 if (this._lastRequestedContext) {
37 this._lastRequestedContext.target().runtimeAgent().releaseObjectGroup(WebInspector.ObjectEventListenersSidebarPane._objectGroupName);
38 delete this._lastRequestedContext;
39 }
40 var executionContext = WebInspector.context.flavor(WebInspector.ExecutionContext);
41 if (!executionContext) {
42 this._eventListenersView.reset();
43 this._eventListenersView.addEmptyHolderIfNeeded();
44 return;
45 }
46 this._lastRequestedContext = executionContext;
47 Promise.all([this._windowObjectInContext(executionContext)]).then(this._eventListenersView.addObjects.bind(this._eventListenersView));
48 },
49

65 /**

76 function windowObjectInContext(fulfill, reject)

82 function mycallback(object)

84 if (object)
85 fulfill(object);
86 else
87 reject(null);
88 }
89 }
