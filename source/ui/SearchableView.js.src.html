157WebInspector.SearchableView.fromElement = function(element)
158{
159 var view = null;
160 while (element && !view) {

634WebInspector.SearchableView.SearchConfig.prototype = {
635 /**
636 * @param {boolean=} global
637 * @return {!RegExp}
638 */
639 toSearchRegex: function(global)
640 {
641 var modifiers = this.caseSensitive ? "" : "i";
642 if (global)
643 modifiers += "g";
644 var query = this.isRegex ? "/" + this.query + "/" : this.query;
645
646 var regex;
647
648 // First try creating regex if user knows the / / hint.
649 try {
650 if (/^\/.+\/$/.test(query)) {
651 regex = new RegExp(query.substring(1, query.length - 1), modifiers);
652 regex.__fromRegExpQuery = true;
653 }
654 } catch (e) {
655 // Silent catch.
656 }
657
658 // Otherwise just do a plain text search.
659 if (!regex)
660 regex = createPlainTextSearchRegex(query, modifiers);
661
662 return regex;
663 }
