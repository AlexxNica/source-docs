58 this._contextMenuElement = root.createChild("div");
59 this.element.style.top = y + "px";
60 var subMenuOverlap = 3;

63 this._contextMenuElement.tabIndex = 0;

65 this._contextMenuElement.addEventListener("keydown", this._menuKeyDown.bind(this), false);

67 for (var i = 0; i < this._items.length; ++i)

77 this._discardMenuOnResizeListener = this._discardMenu.bind(this, true);

83 // Re-position menu in case it does not fit.

86 ? this._parentMenu.element.offsetLeft - this.element.offsetWidth + subMenuOverlap
87 : document.body.offsetWidth - this.element.offsetWidth) + "px";
88 }
89
90 // Move submenus upwards if it does not fit.
91 if (this._parentMenu && document.body.offsetHeight < this.element.offsetTop + this.element.offsetHeight) {
92 y = Math.max(WebInspector.Dialog.modalHostView().element.totalOffsetTop(), document.body.offsetHeight - this.element.offsetHeight);

94 }

301 _menuKeyDown: function(event)
302 {
303 switch (event.key) {
304 case "ArrowUp":
305 this._highlightPrevious(); break;
306 case "ArrowDown":
307 this._highlightNext(); break;
308 case "ArrowLeft":
309 if (this._parentMenu) {
310 this._highlightMenuItem(null, false);
311 this._parentMenu._hideSubMenu();
312 }
313 break;
314 case "ArrowRight":
315 if (!this._highlightedMenuItemElement)
316 break;
317 if (this._highlightedMenuItemElement._subItems) {
318 this._showSubMenu(this._highlightedMenuItemElement);
319 this._subMenu._focus();
320 this._subMenu._highlightNext();
321 }
322 break;
323 case "Escape":
324 this._discardMenu(false, event); break;
325 case "Enter":
326 if (!isEnterKey(event))
327 break;
328 // Fall through
329 case " ": // Space
330 if (this._highlightedMenuItemElement)
331 this._triggerAction(this._highlightedMenuItemElement, event);
332 if (this._highlightedMenuItemElement._subItems) {
333 this._subMenu._focus();
334 this._subMenu._highlightNext();
335 }
336 break;
337 }
338 event.consume(true);

352 /**
353 * @param {boolean} closeParentMenus
354 * @param {!Event=} event
355 */
356 _discardMenu: function(closeParentMenus, event)
357 {
358 if (this._subMenu && !closeParentMenus)
359 return;
360 if (this._glassPaneElement) {
361 var glassPane = this._glassPaneElement;
362 delete this._glassPaneElement;
363 // This can re-enter discardMenu due to blur.
364 this._document.body.removeChild(glassPane);
365 if (this._parentMenu) {
366 delete this._parentMenu._subMenu;
367 if (closeParentMenus)
368 this._parentMenu._discardMenu(closeParentMenus, event);
369 else
370 this._parentMenu._focus();
371 }
372
373 if (event)
374 event.consume(true);
375 } else if (this._parentMenu && this._contextMenuElement.parentElementOrShadowHost()) {
376 this._discardSubMenus();
377 if (closeParentMenus)
378 this._parentMenu._discardMenu(closeParentMenus, event);
379 else
380 this._parentMenu._focus();
381 if (event)
382 event.consume(true);
383 }
384 if (this._discardMenuOnResizeListener) {
385 this._document.defaultView.removeEventListener("resize", this._discardMenuOnResizeListener, false);
386 delete this._discardMenuOnResizeListener;
387 }
388 },
389
390 _discardSubMenus: function()
391 {
392 if (this._subMenu)
393 this._subMenu._discardSubMenus();
394 if (this.element)
395 this.element.remove();
396 if (this._parentMenu)
397 delete this._parentMenu._subMenu;
