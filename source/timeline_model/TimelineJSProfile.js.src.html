13WebInspector.TimelineJSProfileProcessor.generateTracingEventsFromCpuProfile = function(jsProfileModel, thread)
14{
15 var idleNode = jsProfileModel.idleNode;
16 var programNode = jsProfileModel.programNode;
17 var gcNode = jsProfileModel.gcNode;
18 var samples = jsProfileModel.samples;
19 var timestamps = jsProfileModel.timestamps;
20 var jsEvents = [];
21 /** @type {!Map<!Object, !Array<!RuntimeAgent.CallFrame>>} */
22 var nodeToStackMap = new Map();
23 nodeToStackMap.set(programNode, []);
24 for (var i = 0; i < samples.length; ++i) {
25 var node = jsProfileModel.nodeByIndex(i);
26 if (!node) {
27 console.error(`Node with unknown id ${samples[i]} at index ${i}`);
28 continue;
29 }
30 if (node === gcNode || node === idleNode)
31 continue;
32 var callFrames = nodeToStackMap.get(node);
33 if (!callFrames) {
34 callFrames = /** @type {!Array<!RuntimeAgent.CallFrame>} */ (new Array(node.depth + 1));
35 nodeToStackMap.set(node, callFrames);
36 for (var j = 0; node.parent; node = node.parent)
37 callFrames[j++] = /** @type {!RuntimeAgent.CallFrame} */ (node);
38 }
39 var jsSampleEvent = new WebInspector.TracingModel.Event(WebInspector.TracingModel.DevToolsTimelineEventCategory,
40 WebInspector.TimelineModel.RecordType.JSSample,
41 WebInspector.TracingModel.Phase.Instant, timestamps[i], thread);
42 jsSampleEvent.args["data"] = { stackTrace: callFrames };
43 jsEvents.push(jsSampleEvent);
44 }
45 return jsEvents;
46}

54 /**
55 * @param {!RuntimeAgent.CallFrame} frame1
56 * @param {!RuntimeAgent.CallFrame} frame2
57 * @return {boolean}
58 */
59 function equalFrames(frame1, frame2)
60 {
61 return frame1.scriptId === frame2.scriptId && frame1.functionName === frame2.functionName;
62 }

64 /**
65 * @param {!WebInspector.TracingModel.Event} e
66 * @return {number}
67 */
68 function eventEndTime(e)
69 {
70 return e.endTime || e.startTime;
71 }

73 /**
74 * @param {!WebInspector.TracingModel.Event} e
75 * @return {boolean}
76 */
77 function isJSInvocationEvent(e)
78 {
79 switch (e.name) {
80 case WebInspector.TimelineModel.RecordType.FunctionCall:
81 case WebInspector.TimelineModel.RecordType.EvaluateScript:
82 return true;
83 }
84 return false;
85 }
86
87 var jsFrameEvents = [];
88 var jsFramesStack = [];
89 var lockedJsStackDepth = [];
90 var ordinal = 0;
91 var filterNativeFunctions = !WebInspector.moduleSetting("showNativeFunctionsInJSProfile").get();
92
93 /**
94 * @param {!WebInspector.TracingModel.Event} e
95 */
96 function onStartEvent(e)
97 {
98 e.ordinal = ++ordinal;
99 extractStackTrace(e);
100 // For the duration of the event we cannot go beyond the stack associated with it.
101 lockedJsStackDepth.push(jsFramesStack.length);
102 }
103
104 /**
105 * @param {!WebInspector.TracingModel.Event} e
106 * @param {?WebInspector.TracingModel.Event} parent
107 */
108 function onInstantEvent(e, parent)
109 {
110 e.ordinal = ++ordinal;
111 if (parent && isJSInvocationEvent(parent))
112 extractStackTrace(e);
113 }
114
115 /**
116 * @param {!WebInspector.TracingModel.Event} e
117 */
118 function onEndEvent(e)
119 {
120 truncateJSStack(lockedJsStackDepth.pop(), e.endTime);
121 }
122
123 /**
124 * @param {number} depth
125 * @param {number} time
126 */
127 function truncateJSStack(depth, time)
128 {
129 if (lockedJsStackDepth.length) {
130 var lockedDepth = lockedJsStackDepth.peekLast();
131 if (depth < lockedDepth) {
132 console.error("Child stack is shallower (" + depth + ") than the parent stack (" + lockedDepth + ") at " + time);
133 depth = lockedDepth;
134 }
135 }
136 if (jsFramesStack.length < depth) {
137 console.error("Trying to truncate higher than the current stack size at " + time);
138 depth = jsFramesStack.length;
139 }
140 for (var k = 0; k < jsFramesStack.length; ++k)
141 jsFramesStack[k].setEndTime(time);
142 jsFramesStack.length = depth;
143 }
144
145 /**
146 * @param {!Array<!RuntimeAgent.CallFrame>} stack
147 */
148 function filterStackFrames(stack)
149 {
150 for (var i = 0, j = 0; i < stack.length; ++i) {
151 var url = stack[i].url;
152 if (url && url.startsWith("native "))
153 continue;
154 stack[j++] = stack[i];
155 }
156 stack.length = j;
157 }
158
159 /**
160 * @param {!WebInspector.TracingModel.Event} e
161 */
162 function extractStackTrace(e)
163 {
164 var recordTypes = WebInspector.TimelineModel.RecordType;
165 var callFrames;
166 if (e.name === recordTypes.JSSample) {
167 var eventData = e.args["data"] || e.args["beginData"];
168 callFrames = /** @type {!Array<!RuntimeAgent.CallFrame>} */ (eventData && eventData["stackTrace"]);
169 } else {
170 callFrames = /** @type {!Array<!RuntimeAgent.CallFrame>} */ (jsFramesStack.map(frameEvent => frameEvent.args["data"]).reverse());
171 }
172 if (filterNativeFunctions)
173 filterStackFrames(callFrames);
174 var endTime = eventEndTime(e);
175 var numFrames = callFrames.length;
176 var minFrames = Math.min(numFrames, jsFramesStack.length);
177 var i;
178 for (i = lockedJsStackDepth.peekLast() || 0; i < minFrames; ++i) {
179 var newFrame = callFrames[numFrames - 1 - i];
180 var oldFrame = jsFramesStack[i].args["data"];
181 if (!equalFrames(newFrame, oldFrame))
182 break;
183 jsFramesStack[i].setEndTime(Math.max(jsFramesStack[i].endTime, endTime));
184 }
185 truncateJSStack(i, e.startTime);
186 for (; i < numFrames; ++i) {
187 var frame = callFrames[numFrames - 1 - i];
188 var jsFrameEvent = new WebInspector.TracingModel.Event(WebInspector.TracingModel.DevToolsTimelineEventCategory, recordTypes.JSFrame,
189 WebInspector.TracingModel.Phase.Complete, e.startTime, e.thread);
190 jsFrameEvent.ordinal = e.ordinal;
191 jsFrameEvent.addArgs({ data: frame });
192 jsFrameEvent.setEndTime(endTime);
193 jsFramesStack.push(jsFrameEvent);
194 jsFrameEvents.push(jsFrameEvent);
195 }
196 }

198 /**
199 * @param {!Array<!WebInspector.TracingModel.Event>} events
200 * @return {?WebInspector.TracingModel.Event}
201 */
202 function findFirstTopLevelEvent(events)
203 {
204 for (var i = 0; i < events.length; ++i) {
205 if (WebInspector.TracingModel.isTopLevelEvent(events[i]))
206 return events[i];
207 }
208 return null;
209 }
210
211 var firstTopLevelEvent = findFirstTopLevelEvent(events);
212 if (firstTopLevelEvent)
213 WebInspector.TimelineModel.forEachEvent(events, onStartEvent, onEndEvent, onInstantEvent, firstTopLevelEvent.startTime);
214 return jsFrameEvents;
215}

299 * @return {!WebInspector.TimelineJSProfileProcessor.CodeMap.Bank}

301 _getBank: function(addressHex)
302 {
303 addressHex = addressHex.slice(2); // cut 0x prefix.
304 // 13 hex digits == 52 bits, double mantissa fits 53 bits.
305 var /** @const */ bankSizeHexDigits = 13;
306 var /** @const */ maxHexDigits = 16;
307 var bankName = addressHex.slice(-maxHexDigits, -bankSizeHexDigits);
308 var bank = this._banks.get(bankName);
309 if (!bank) {
310 bank = new WebInspector.TimelineJSProfileProcessor.CodeMap.Bank();
311 this._banks.set(bankName, bank);
312 }
313 return bank;
314 },

321 {
322 // 13 hex digits == 52 bits, double mantissa fits 53 bits.
323 var /** @const */ bankSizeHexDigits = 13;
324 addressHex = addressHex.slice(2); // cut 0x prefix.
325 return parseInt(addressHex.slice(-bankSizeHexDigits), 16);

358 {
359 var index = this._entries.upperBound(address, WebInspector.TimelineJSProfileProcessor.CodeMap.comparator) - 1;
360 var entry = this._entries[index];
361 return entry && address < entry.address + entry.size ? entry.callFrame : null;

364 /**
365 * @param {!WebInspector.TimelineJSProfileProcessor.CodeMap.Entry} newEntry
366 */
367 addEntry: function(newEntry)
368 {
369 var endAddress = newEntry.address + newEntry.size;
370 var lastIndex = this._entries.lowerBound(endAddress, WebInspector.TimelineJSProfileProcessor.CodeMap.comparator);
371 var index;
372 for (index = lastIndex - 1; index >= 0; --index) {
373 var entry = this._entries[index];
374 var entryEndAddress = entry.address + entry.size;
375 if (entryEndAddress <= newEntry.address)
376 break;
377 }
378 ++index;
379 this._entries.splice(index, lastIndex - index, newEntry);
380 }

383/**
384 * @param {string} name
385 * @param {number} scriptId
386 * @return {!RuntimeAgent.CallFrame}
387 */
388WebInspector.TimelineJSProfileProcessor._buildCallFrame = function(name, scriptId)
389{
390 /**
391 * @param {string} functionName
392 * @param {string=} url
393 * @param {string=} scriptId
394 * @param {number=} line
395 * @param {number=} column
396 * @param {boolean=} isNative
397 * @return {!RuntimeAgent.CallFrame}
398 */
399 function createFrame(functionName, url, scriptId, line, column, isNative)
400 {
401 return /** @type {!RuntimeAgent.CallFrame} */ ({
402 "functionName": functionName,
403 "url": url || "",
404 "scriptId": scriptId || "0",
405 "lineNumber": line || 0,
406 "columnNumber": column || 0,
407 "isNative": isNative || false
408 });
409 }
410
411 // Code states:
412 // (empty) -> compiled
413 // ~ -> optimizable
414 // * -> optimized
415 var rePrefix = /^(\w*:)?[*~]?(.*)$/m;
416 var tokens = rePrefix.exec(name);
417 var prefix = tokens[1];
418 var body = tokens[2];
419 var rawName;
420 var rawUrl;
421 if (prefix === "Script:") {
422 rawName = "";
423 rawUrl = body;
424 } else {
425 var spacePos = body.lastIndexOf(" ");
426 rawName = spacePos !== -1 ? body.substr(0, spacePos) : body;
427 rawUrl = spacePos !== -1 ? body.substr(spacePos + 1) : "";
428 }
429 var nativeSuffix = " native";
430 var isNative = rawName.endsWith(nativeSuffix);
431 var functionName = isNative ? rawName.slice(0, -nativeSuffix.length) : rawName;
432 var urlData = WebInspector.ParsedURL.splitLineAndColumn(rawUrl);
433 var url = urlData.url || "";
434 var line = urlData.lineNumber || 0;
435 var column = urlData.columnNumber || 0;
436 return createFrame(functionName, url, String(scriptId), line, column, isNative);
437}

439/**
440 * @param {!Array<!WebInspector.TracingModel.Event>} events
441 * @return {!Array<!WebInspector.TracingModel.Event>}
442 */
443WebInspector.TimelineJSProfileProcessor.processRawV8Samples = function(events)
444{
445 var missingAddesses = new Set();
446
447 /**
448 * @param {string} address
449 * @return {?RuntimeAgent.CallFrame}
450 */
451 function convertRawFrame(address)
452 {
453 var entry = codeMap.lookupEntry(address);
454 if (entry)
455 return entry.isNative ? null : entry;
456 if (!missingAddesses.has(address)) {
457 missingAddesses.add(address);
458 console.error("Address " + address + " has missing code entry");
459 }
460 return null;
461 }
462
463 var recordTypes = WebInspector.TimelineModel.RecordType;
464 var samples = [];
465 var codeMap = new WebInspector.TimelineJSProfileProcessor.CodeMap();
466 for (var i = 0; i < events.length; ++i) {
467 var e = events[i];
468 var data = e.args["data"];
469 switch (e.name) {
470 case recordTypes.JitCodeAdded:
471 var frame = WebInspector.TimelineJSProfileProcessor._buildCallFrame(data["name"], data["script_id"]);
472 codeMap.addEntry(data["code_start"], data["code_len"], frame);
473 break;
474 case recordTypes.JitCodeMoved:
475 codeMap.moveEntry(data["code_start"], data["new_code_start"], data["code_len"]);
476 break;
477 case recordTypes.V8Sample:
478 var rawStack = data["stack"];
479 // Sometimes backend fails to collect a stack and returns an empty stack.
480 // Skip these bogus samples.
481 if (data["vm_state"] === "js" && !rawStack.length)
482 break;
483 var stack = rawStack.map(convertRawFrame);
484 stack.remove(null);
485 var sampleEvent = new WebInspector.TracingModel.Event(
486 WebInspector.TracingModel.DevToolsTimelineEventCategory,
487 WebInspector.TimelineModel.RecordType.JSSample,
488 WebInspector.TracingModel.Phase.Instant, e.startTime, e.thread);
489 sampleEvent.ordinal = e.ordinal;
490 sampleEvent.args = {"data": {"stackTrace": stack }};
491 samples.push(sampleEvent);
492 break;
493 }
494 }
495
496 return samples;
