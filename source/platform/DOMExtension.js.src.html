33/**
34 * @param {number} offset
35 * @param {string} stopCharacters
36 * @param {!Node} stayWithinNode
37 * @param {string=} direction
38 * @return {!Range}
39 */
40Node.prototype.rangeOfWord = function(offset, stopCharacters, stayWithinNode, direction)
41{
42 var startNode;
43 var startOffset = 0;
44 var endNode;
45 var endOffset = 0;
46
47 if (!stayWithinNode)
48 stayWithinNode = this;
49
50 if (!direction || direction === "backward" || direction === "both") {
51 var node = this;
52 while (node) {
53 if (node === stayWithinNode) {
54 if (!startNode)
55 startNode = stayWithinNode;
56 break;
57 }
58
59 if (node.nodeType === Node.TEXT_NODE) {
60 var start = (node === this ? (offset - 1) : (node.nodeValue.length - 1));
61 for (var i = start; i >= 0; --i) {
62 if (stopCharacters.indexOf(node.nodeValue[i]) !== -1) {
63 startNode = node;
64 startOffset = i + 1;
65 break;
66 }
67 }
68 }
69
70 if (startNode)
71 break;
72
73 node = node.traversePreviousNode(stayWithinNode);
74 }
75
76 if (!startNode) {
77 startNode = stayWithinNode;
78 startOffset = 0;
79 }
80 } else {
81 startNode = this;
82 startOffset = offset;
83 }
84
85 if (!direction || direction === "forward" || direction === "both") {
86 node = this;
87 while (node) {
88 if (node === stayWithinNode) {
89 if (!endNode)
90 endNode = stayWithinNode;
91 break;
92 }
93
94 if (node.nodeType === Node.TEXT_NODE) {
95 var start = (node === this ? offset : 0);
96 for (var i = start; i < node.nodeValue.length; ++i) {
97 if (stopCharacters.indexOf(node.nodeValue[i]) !== -1) {
98 endNode = node;
99 endOffset = i;
100 break;
101 }
102 }
103 }
104
105 if (endNode)
106 break;
107
108 node = node.traverseNextNode(stayWithinNode);
109 }
110
111 if (!endNode) {
112 endNode = stayWithinNode;
113 endOffset = stayWithinNode.nodeType === Node.TEXT_NODE ? stayWithinNode.nodeValue.length : stayWithinNode.childNodes.length;
114 }
115 } else {
116 endNode = this;
117 endOffset = offset;
118 }
119
120 var result = this.ownerDocument.createRange();
121 result.setStart(startNode, startOffset);
122 result.setEnd(endNode, endOffset);
123
124 return result;
125}

127/**
128 * @param {!Node=} stayWithin
129 * @return {?Node}
130 */
131Node.prototype.traverseNextTextNode = function(stayWithin)
132{
133 var node = this.traverseNextNode(stayWithin);
134 if (!node)
135 return null;
136 var nonTextTags = { "STYLE": 1, "SCRIPT": 1 };
137 while (node && (node.nodeType !== Node.TEXT_NODE || nonTextTags[node.parentElement.nodeName]))
138 node = node.traverseNextNode(stayWithin);
139
140 return node;
141}

144 * @param {number|undefined} x
145 * @param {number|undefined} y
146 * @param {!Element=} relativeTo

148Element.prototype.positionAt = function(x, y, relativeTo)
149{
150 var shift = {x: 0, y: 0};
151 if (relativeTo)
152 shift = relativeTo.boxInWindow(this.ownerDocument.defaultView);
153
154 if (typeof x === "number")
155 this.style.setProperty("left", (shift.x + x) + "px");
156 else
157 this.style.removeProperty("left");
158
159 if (typeof y === "number")
160 this.style.setProperty("top", (shift.y + y) + "px");
161 else
162 this.style.removeProperty("top");
163
164 if (typeof x === "number" || typeof y === "number")
165 this.style.setProperty("position", "absolute");
166 else
167 this.style.removeProperty("position");
168}

170/**
171 * @return {boolean}
172 */
173Element.prototype.isScrolledToBottom = function()
174{
175 // This code works only for 0-width border.
176 // The scrollTop, clientHeight and scrollHeight are computed in double values internally.
177 // However, they are exposed to javascript differently, each being either rounded (via
178 // round, ceil or floor functions) or left intouch.
179 // This adds up a total error up to 2.
180 return Math.abs(this.scrollTop + this.clientHeight - this.scrollHeight) <= 2;
181}

586/**
587 * @param {!Window} targetWindow
588 * @return {!AnchorBox}
589 */
590Element.prototype.offsetRelativeToWindow = function(targetWindow)
591{
592 var elementOffset = new AnchorBox();
593 var curElement = this;
594 var curWindow = this.ownerDocument.defaultView;
595 while (curWindow && curElement) {
596 elementOffset.x += curElement.totalOffsetLeft();
597 elementOffset.y += curElement.totalOffsetTop();
598 if (curWindow === targetWindow)
599 break;
600
601 curElement = curWindow.frameElement;
602 curWindow = curWindow.parent;
603 }
604
605 return elementOffset;
606}

613{
614 targetWindow = targetWindow || this.ownerDocument.defaultView;

618 anchorBox.height = Math.min(this.offsetHeight, window.innerHeight - anchorBox.y);

684/**
685 * @return {?number}
686 */
687Element.prototype.selectionLeftOffset = function()
688{
689 // Calculate selection offset relative to the current element.
690
691 var selection = this.getComponentSelection();
692 if (!selection.containsNode(this, true))
693 return null;
694
695 var leftOffset = selection.anchorOffset;
696 var node = selection.anchorNode;
697
698 while (node !== this) {
699 while (node.previousSibling) {
700 node = node.previousSibling;
701 leftOffset += node.textContent.length;
702 }
703 node = node.parentNodeOrShadowHost();
704 }
705
706 return leftOffset;
707}

813/**
814 * @param {!Node=} stayWithin
815 * @return {?Node}
816 */
817Node.prototype.traverseNextNode = function(stayWithin)
818{
819 if (this.shadowRoot)
820 return this.shadowRoot;
821
822 var distributedNodes = this.getDistributedNodes ? this.getDistributedNodes() : [];
823
824 if (distributedNodes.length)
825 return distributedNodes[0];
826
827 if (this.firstChild)
828 return this.firstChild;
829
830 var node = this;
831 while (node) {
832 if (stayWithin && node === stayWithin)
833 return null;
834
835 var sibling = nextSibling(node);
836 if (sibling)
837 return sibling;
838
839 node = insertionPoint(node) || node.parentNodeOrShadowHost();
840 }

842 /**
843 * @param {!Node} node
844 * @return {?Node}
845 */
846 function nextSibling(node)
847 {
848 var parent = insertionPoint(node);
849 if (!parent)
850 return node.nextSibling;
851 var distributedNodes = parent.getDistributedNodes ? parent.getDistributedNodes() : [];
852
853 var position = Array.prototype.indexOf.call(distributedNodes, node);
854 if (position + 1 < distributedNodes.length)
855 return distributedNodes[position + 1];
856 return null;
857 }

859 /**
860 * @param {!Node} node
861 * @return {?Node}
862 */
863 function insertionPoint(node)
864 {
865 var insertionPoints = node.getDestinationInsertionPoints ? node.getDestinationInsertionPoints() : [];
866 return insertionPoints.length > 0 ? insertionPoints[insertionPoints.length - 1] : null;
867 }
868
869 return null;
870}

872/**
873 * @param {!Node=} stayWithin
874 * @return {?Node}
875 */
876Node.prototype.traversePreviousNode = function(stayWithin)
877{
878 if (stayWithin && this === stayWithin)
879 return null;
880 var node = this.previousSibling;
881 while (node && node.lastChild)
882 node = node.lastChild;
883 if (node)

893Node.prototype.setTextContentTruncatedIfNeeded = function(text, placeholder)
894{
895 // Huge texts in the UI reduce rendering performance drastically.
896 // Moreover, Blink/WebKit uses <unsigned short> internally for storing text content

898 const maxTextContentLength = 10000;
899
