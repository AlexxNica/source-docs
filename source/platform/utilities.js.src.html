97/**
98 * @param {string} chars
99 * @return {string}
100 */
101String.prototype.escapeCharacters = function(chars)
102{
103 var foundChar = false;
104 for (var i = 0; i < chars.length; ++i) {
105 if (this.indexOf(chars.charAt(i)) !== -1) {
106 foundChar = true;
107 break;
108 }
109 }
110
111 if (!foundChar)
112 return String(this);
113
114 var result = "";
115 for (var i = 0; i < this.length; ++i) {
116 if (chars.indexOf(this.charAt(i)) !== -1)
117 result += "\\";
118 result += this.charAt(i);
119 }
120
121 return result;
122}

240/**
241 * @param {string|undefined} string
242 * @return {number}
243 */
244String.hashCode = function(string)
245{
246 if (!string)
247 return 0;
248 // Hash algorithm for substrings is described in "Ãber die KomplexitÃ¤t der Multiplikation in
249 // eingeschrÃ¤nkten Branchingprogrammmodellen" by Woelfe.
250 // http://opendatastructures.org/versions/edition-0.1d/ods-java/node33.html#SECTION00832000000000000000
251 var p = ((1 << 30) * 4 - 5); // prime: 2^32 - 5
252 var z = 0x5033d967; // 32 bits from random.org
253 var z2 = 0x59d2f15d; // random odd 32 bit number
254 var s = 0;
255 var zi = 1;
256 for (var i = 0; i < string.length; i++) {
257 var xi = string.charCodeAt(i) * z2;
258 s = (s + zi * xi) % p;
259 zi = (zi * z) % p;
260 }
261 s = (s + zi * (p - 1)) % p;
262 return Math.abs(s | 0);
263}

279String.prototype.toBase64 = function()
280{
281 /**
282 * @param {number} b
283 * @return {number}
284 */
285 function encodeBits(b)
286 {
287 return b < 26 ? b + 65 : b < 52 ? b + 71 : b < 62 ? b - 4 : b === 62 ? 43 : b === 63 ? 47 : 65;
288 }
289 var encoder = new TextEncoder();
290 var data = encoder.encode(this.toString());
291 var n = data.length;
292 var encoded = "";
293 if (n === 0)
294 return encoded;
295 var shift;
296 var v = 0;
297 for (var i = 0; i < n; i++) {
298 shift = i % 3;
299 v |= data[i] << (16 >>> shift & 24);
300 if (shift === 2) {
301 encoded += String.fromCharCode(encodeBits(v >>> 18 & 63), encodeBits(v >>> 12 & 63), encodeBits(v >>> 6 & 63), encodeBits(v & 63));
302 v = 0;
303 }
304 }
305 if (shift === 0)
306 encoded += String.fromCharCode(encodeBits(v >>> 18 & 63), encodeBits(v >>> 12 & 63), 61, 61);
307 else if (shift === 1)
308 encoded += String.fromCharCode(encodeBits(v >>> 18 & 63), encodeBits(v >>> 12 & 63), encodeBits(v >>> 6 & 63), 61);
309 return encoded;
310}

312/**
313 * @param {string} a
314 * @param {string} b
315 * @return {number}
316 */
317String.naturalOrderComparator = function(a, b)
318{
319 var chunk = /^\d+|^\D+/;
320 var chunka, chunkb, anum, bnum;
321 while (1) {
322 if (a) {
323 if (!b)
324 return 1;
325 } else {
326 if (b)
327 return -1;
328 else
329 return 0;
330 }
331 chunka = a.match(chunk)[0];
332 chunkb = b.match(chunk)[0];
333 anum = !isNaN(chunka);
334 bnum = !isNaN(chunkb);
335 if (anum && !bnum)
336 return -1;
337 if (bnum && !anum)
338 return 1;
339 if (anum && bnum) {
340 var diff = chunka - chunkb;
341 if (diff)
342 return diff;
343 if (chunka.length !== chunkb.length) {
344 if (!+chunka && !+chunkb) // chunks are strings of all 0s (special case)
345 return chunka.length - chunkb.length;
346 else
347 return chunkb.length - chunka.length;
348 }
349 } else if (chunka !== chunkb)
350 return (chunka < chunkb) ? -1 : 1;
351 a = a.substring(chunka.length);
352 b = b.substring(chunkb.length);
353 }
354}

543 /**
544 * @this {Array.<number>}

546 * @param {number} left
547 * @param {number} right

564 swap(this, storeIndex, i);

566 }
567 }
568 swap(this, right, storeIndex);

570 }

820 * @param {!Array.<T>} array1
821 * @param {!Array.<T>} array2
822 * @param {function(T,T):number} comparator
823 * @param {boolean} mergeNotIntersect
824 * @return {!Array.<T>}
825 * @template T
826 */
827 function mergeOrIntersect(array1, array2, comparator, mergeNotIntersect)
828 {
829 var result = [];
830 var i = 0;
831 var j = 0;
832 while (i < array1.length && j < array2.length) {
833 var compareValue = comparator(array1[i], array2[j]);
834 if (mergeNotIntersect || !compareValue)
835 result.push(compareValue <= 0 ? array1[i] : array2[j]);
836 if (compareValue <= 0)
837 i++;
838 if (compareValue >= 0)
839 j++;

894String.tokenizeFormatString = function(format, formatters)
895{
896 var tokens = [];
897 var substitutionIndex = 0;
898
899 function addStringToken(str)
900 {
901 if (tokens.length && tokens[tokens.length - 1].type === "string")
902 tokens[tokens.length - 1].value += str;
903 else
904 tokens.push({ type: "string", value: str });
905 }
906
907 function addSpecifierToken(specifier, precision, substitutionIndex)
908 {
909 tokens.push({ type: "specifier", specifier: specifier, precision: precision, substitutionIndex: substitutionIndex });
910 }
911
912 var index = 0;
913 for (var precentIndex = format.indexOf("%", index); precentIndex !== -1; precentIndex = format.indexOf("%", index)) {
914 if (format.length === index) // unescaped % sign at the end of the format string.
915 break;
916 addStringToken(format.substring(index, precentIndex));
917 index = precentIndex + 1;
918
919 if (format[index] === "%") {
920 // %% escape sequence.
921 addStringToken("%");
922 ++index;
923 continue;
924 }
925
926 if (String.isDigitAt(format, index)) {
927 // The first character is a number, it might be a substitution index.
928 var number = parseInt(format.substring(index), 10);
929 while (String.isDigitAt(format, index))
930 ++index;
931
932 // If the number is greater than zero and ends with a "$",
933 // then this is a substitution index.
934 if (number > 0 && format[index] === "$") {
935 substitutionIndex = (number - 1);
936 ++index;
937 }
938 }
939
940 var precision = -1;
941 if (format[index] === ".") {
942 // This is a precision specifier. If no digit follows the ".",
943 // then the precision should be zero.
944 ++index;
945 precision = parseInt(format.substring(index), 10);
946 if (isNaN(precision))
947 precision = 0;
948
949 while (String.isDigitAt(format, index))
950 ++index;
951 }
952
953 if (!(format[index] in formatters)) {
954 addStringToken(format.substring(precentIndex, index + 1));
955 ++index;
956 continue;
957 }
958
959 addSpecifierToken(format[index], precision, substitutionIndex);
960
961 ++substitutionIndex;
962 ++index;
963 }
964
965 addStringToken(format.substring(index));
966
967 return tokens;
968}

1073 }

1077 if (i in usedSubstitutionIndexes)
