6/**
7 * @constructor
8 * @extends {WebInspector.SDKModel}
9 * @param {!WebInspector.Target} target
10 */
11WebInspector.AnimationModel = function(target)
12{
13 WebInspector.SDKModel.call(this, WebInspector.AnimationModel, target);
14 this._agent = target.animationAgent();
15 target.registerAnimationDispatcher(new WebInspector.AnimationDispatcher(this));
16 /** @type {!Map.<string, !WebInspector.AnimationModel.Animation>} */
17 this._animationsById = new Map();
18 /** @type {!Map.<string, !WebInspector.AnimationModel.AnimationGroup>} */
19 this._animationGroups = new Map();
20 /** @type {!Array.<string>} */
21 this._pendingAnimations = [];
22 this._playbackRate = 1;
23 target.resourceTreeModel.addEventListener(WebInspector.ResourceTreeModel.EventTypes.MainFrameNavigated, this._reset, this);
24 this._screenshotCapture = new WebInspector.AnimationModel.ScreenshotCapture(target, this);
25}

33 _reset: function()
34 {

46 this._pendingAnimations.push(id);

49 /**
50 * @param {string} id
51 */
52 _animationCanceled: function(id)
53 {
54 this._pendingAnimations.remove(id);
55 this._flushPendingAnimationsIfNeeded();
56 },
57
58 /**
59 * @param {!AnimationAgent.Animation} payload
60 */
61 animationStarted: function(payload)
62 {
63 var animation = WebInspector.AnimationModel.Animation.parsePayload(this.target(), payload);
64
65 // Ignore Web Animations custom effects & groups.
66 if (animation.type() === "WebAnimation" && animation.source().keyframesRule().keyframes().length === 0) {
67 this._pendingAnimations.remove(animation.id());
68 } else {
69 this._animationsById.set(animation.id(), animation);
70 if (this._pendingAnimations.indexOf(animation.id()) === -1)
71 this._pendingAnimations.push(animation.id());
72 }
73
74 this._flushPendingAnimationsIfNeeded();
75 },
76
77 _flushPendingAnimationsIfNeeded: function()
78 {
79 for (var id of this._pendingAnimations) {
80 if (!this._animationsById.get(id))
81 return;
82 }
83
84 while (this._pendingAnimations.length)
85 this._matchExistingGroups(this._createGroupFromPendingAnimations());
86 },
87
88 /**
89 * @param {!WebInspector.AnimationModel.AnimationGroup} incomingGroup
90 * @return {boolean}
91 */
92 _matchExistingGroups: function(incomingGroup)
93 {
94 var matchedGroup = null;
95 for (var group of this._animationGroups.values()) {
96 if (group._matches(incomingGroup)) {
97 matchedGroup = group;
98 group._update(incomingGroup);
99 break;
100 }
101 }
102
103 if (!matchedGroup) {
104 this._animationGroups.set(incomingGroup.id(), incomingGroup);
105 this._screenshotCapture.captureScreenshots(incomingGroup.finiteDuration(), incomingGroup._screenshots);
106 }
107 this.dispatchEventToListeners(WebInspector.AnimationModel.Events.AnimationGroupStarted, matchedGroup || incomingGroup);
108 return !!matchedGroup;
109 },
110
111 /**
112 * @return {!WebInspector.AnimationModel.AnimationGroup}
113 */
114 _createGroupFromPendingAnimations: function()
115 {
116 console.assert(this._pendingAnimations.length);
117 var groupedAnimations = [this._animationsById.get(this._pendingAnimations.shift())];
118 var remainingAnimations = [];
119 for (var id of this._pendingAnimations) {
120 var anim = this._animationsById.get(id);
121 if (anim.startTime() === groupedAnimations[0].startTime())
122 groupedAnimations.push(anim);
123 else
124 remainingAnimations.push(id);
125 }
126 this._pendingAnimations = remainingAnimations;
127 return new WebInspector.AnimationModel.AnimationGroup(this, groupedAnimations[0].id(), groupedAnimations);
128 },

888WebInspector.AnimationModel.ScreenshotCapture.prototype = {
889 /**
890 * @param {number} duration
891 * @param {!Array<string>} screenshots
892 */
893 captureScreenshots: function(duration, screenshots)
894 {
895 var screencastDuration = Math.min(duration / this._model._playbackRate, 3000);
896 var endTime = screencastDuration + window.performance.now();
897 this._requests.push({ endTime: endTime, screenshots: screenshots });
898
899 if (!this._endTime || endTime > this._endTime) {
900 clearTimeout(this._stopTimer);
901 this._stopTimer = setTimeout(this._stopScreencast.bind(this), screencastDuration);
902 this._endTime = endTime;
903 }
904
905 if (this._capturing)
906 return;
907 this._capturing = true;
908 this._target.pageAgent().startScreencast("jpeg", 80, undefined, 300, 2);
909 },
910
911 /**
912 * @param {!WebInspector.Event} event
913 */
914 _screencastFrame: function(event)
915 {
916 /**
917 * @param {!WebInspector.AnimationModel.ScreenshotCapture.Request} request
918 * @return {boolean}
919 */
920 function isAnimating(request)
921 {
922 return request.endTime >= now;
923 }
924
925 if (!this._capturing)
926 return;
927
928 var base64Data = /** type {string} */(event.data["data"]);
929 var now = window.performance.now();
930 this._requests = this._requests.filter(isAnimating);
931 for (var request of this._requests)
932 request.screenshots.push(base64Data);
933 },
934
935 _stopScreencast: function()
