31/**
32 * @constructor
33 * @extends {WebInspector.SDKModel}
34 * @param {!WebInspector.Target} target
35 * @param {?WebInspector.NetworkManager} networkManager
36 * @param {!WebInspector.SecurityOriginManager} securityOriginManager
37 */
38WebInspector.ResourceTreeModel = function(target, networkManager, securityOriginManager)
39{
40 WebInspector.SDKModel.call(this, WebInspector.ResourceTreeModel, target);
41 if (networkManager) {
42 networkManager.addEventListener(WebInspector.NetworkManager.EventTypes.RequestFinished,
43 this._onRequestFinished, this);
44 networkManager.addEventListener(WebInspector.NetworkManager.EventTypes.RequestUpdateDropped,
45 this._onRequestUpdateDropped, this);
46 }
47
48 this._agent = target.pageAgent();
49 this._agent.enable();
50 this._securityOriginManager = securityOriginManager;
51
52 this._fetchResourceTree();
53
54 target.registerPageDispatcher(new WebInspector.PageDispatcher(this));
55
56 this._pendingReloadOptions = null;
57 this._reloadSuspensionCount = 0;
58
59 target.runtimeModel.setExecutionContextComparator(this._executionContextComparator.bind(this));
60}

119 _fetchResourceTree: function()
120 {
121 /** @type {!Map<string, !WebInspector.ResourceTreeFrame>} */
122 this._frames = new Map();
123 this._cachedResourcesProcessed = false;
124 this._agent.getResourceTree(this._processCachedResources.bind(this));

127 _processCachedResources: function(error, mainFramePayload)
128 {
129 if (!error) {
130 this.dispatchEventToListeners(WebInspector.ResourceTreeModel.EventTypes.WillLoadCachedResources);
131 this._addFramesRecursively(null, mainFramePayload);
132 this.target().setInspectedURL(mainFramePayload.frame.url);
133 }
134 this._cachedResourcesProcessed = true;
135 this.dispatchEventToListeners(WebInspector.ResourceTreeModel.EventTypes.CachedResourcesLoaded);
136 },

146 /**
147 * @param {!WebInspector.ResourceTreeFrame} frame
148 * @param {boolean=} aboutToNavigate
149 */
150 _addFrame: function(frame, aboutToNavigate)
151 {
152 this._frames.set(frame.id, frame);
153 if (frame.isMainFrame()) {
154 this.mainFrame = frame;
155 this._securityOriginManager.setMainSecurityOrigin(frame.url);
156 }
157 this.dispatchEventToListeners(WebInspector.ResourceTreeModel.EventTypes.FrameAdded, frame);
158 if (!aboutToNavigate)
159 this._securityOriginManager.addSecurityOrigin(frame.securityOrigin);
160 },
161
162 /**
163 * @param {!WebInspector.ResourceTreeFrame} mainFrame
164 */
165 _handleMainFrameDetached: function(mainFrame)
166 {
167 /**
168 * @param {!WebInspector.ResourceTreeFrame} frame
169 * @this {WebInspector.ResourceTreeModel}
170 */
171 function removeOriginForFrame(frame)
172 {
173 for (var i = 0; i < frame.childFrames.length; ++i)
174 removeOriginForFrame.call(this, frame.childFrames[i]);
175 if (!frame.isMainFrame())
176 this._securityOriginManager.removeSecurityOrigin(frame.securityOrigin);
177 }
178 removeOriginForFrame.call(this, mainFrame);
179 },

181 /**
182 * @param {!PageAgent.FrameId} frameId
183 * @param {?PageAgent.FrameId} parentFrameId
184 * @return {?WebInspector.ResourceTreeFrame}
185 */
186 _frameAttached: function(frameId, parentFrameId)
187 {
188 // Do nothing unless cached resource tree is processed - it will overwrite everything.
189 if (!this._cachedResourcesProcessed && parentFrameId)
190 return null;
191 if (this._frames.has(frameId))
192 return null;
193
194 var parentFrame = parentFrameId ? (this._frames.get(parentFrameId) || null) : null;
195 var frame = new WebInspector.ResourceTreeFrame(this, parentFrame, frameId);
196 if (frame.isMainFrame() && this.mainFrame) {
197 this._handleMainFrameDetached(this.mainFrame);
198 // Navigation to the new backend process.
199 this._frameDetached(this.mainFrame.id);
200 }
201 this._addFrame(frame, true);
202 return frame;
203 },
204
205 /**
206 * @param {!PageAgent.Frame} framePayload
207 */
208 _frameNavigated: function(framePayload)
209 {
210 // Do nothing unless cached resource tree is processed - it will overwrite everything.
211 if (!this._cachedResourcesProcessed && framePayload.parentId)
212 return;
213 var frame = this._frames.get(framePayload.id);
214 if (!frame) {
215 // Simulate missed "frameAttached" for a main frame navigation to the new backend process.
216 console.assert(!framePayload.parentId, "Main frame shouldn't have parent frame id.");
217 frame = this._frameAttached(framePayload.id, framePayload.parentId || "");
218 console.assert(frame);
219 }
220
221 this.dispatchEventToListeners(WebInspector.ResourceTreeModel.EventTypes.FrameWillNavigate, frame);
222
223 this._securityOriginManager.removeSecurityOrigin(frame.securityOrigin);
224 frame._navigate(framePayload);
225 var addedOrigin = frame.securityOrigin;
226
227 this.dispatchEventToListeners(WebInspector.ResourceTreeModel.EventTypes.FrameNavigated, frame);
228 if (frame.isMainFrame()) {
229 this.dispatchEventToListeners(WebInspector.ResourceTreeModel.EventTypes.MainFrameNavigated, frame);
230 if (WebInspector.moduleSetting("preserveConsoleLog").get())
231 WebInspector.console.log(WebInspector.UIString("Navigated to %s", frame.url));
232 else
233 this.target().consoleModel.clear();
234 }
235 if (addedOrigin)
236 this._securityOriginManager.addSecurityOrigin(addedOrigin);
237
238 // Fill frame with retained resources (the ones loaded using new loader).
239 var resources = frame.resources();
240 for (var i = 0; i < resources.length; ++i)
241 this.dispatchEventToListeners(WebInspector.ResourceTreeModel.EventTypes.ResourceAdded, resources[i]);
242
243 if (frame.isMainFrame())
244 this.target().setInspectedURL(frame.url);
245 },
246
247 /**
248 * @param {!PageAgent.FrameId} frameId
249 */
250 _frameDetached: function(frameId)
251 {
252 // Do nothing unless cached resource tree is processed - it will overwrite everything.
253 if (!this._cachedResourcesProcessed)
254 return;
255
256 var frame = this._frames.get(frameId);
257 if (!frame)
258 return;
259
260 this._securityOriginManager.removeSecurityOrigin(frame.securityOrigin);
261 if (frame.parentFrame)
262 frame.parentFrame._removeChildFrame(frame);
263 else
264 frame._remove();
265 },

267 /**
268 * @param {!WebInspector.Event} event
269 */

271 {
272 if (!this._cachedResourcesProcessed)
273 return;
274
275 var request = /** @type {!WebInspector.NetworkRequest} */ (event.data);
276 if (request.failed || request.resourceType() === WebInspector.resourceTypes.XHR)
277 return;
278
279 var frame = this._frames.get(request.frameId);
280 if (frame)
281 frame._addRequest(request);
282 },

285 * @param {!WebInspector.Event} event
286 */
287 _onRequestUpdateDropped: function(event)
288 {
289 if (!this._cachedResourcesProcessed)
290 return;
291
292 var frameId = event.data.frameId;
293 var frame = this._frames.get(frameId);
294 if (!frame)
295 return;
296
297 var url = event.data.url;
298 if (frame._resourcesMap[url])
299 return;
300
301 var resource = new WebInspector.Resource(this.target(), null, url, frame.url, frameId, event.data.loaderId, WebInspector.resourceTypes[event.data.resourceType], event.data.mimeType);
302 frame.addResource(resource);

343 /**
344 * @param {?WebInspector.ResourceTreeFrame} parentFrame
345 * @param {!PageAgent.FrameResourceTree} frameTreePayload
346 */
347 _addFramesRecursively: function(parentFrame, frameTreePayload)
348 {
349 var framePayload = frameTreePayload.frame;
350 var frame = new WebInspector.ResourceTreeFrame(this, parentFrame, framePayload.id, framePayload);
351 this._addFrame(frame);
352
353 var frameResource = this._createResourceFromFramePayload(framePayload, framePayload.url, WebInspector.resourceTypes.Document, framePayload.mimeType);
354 frame.addResource(frameResource);
355
356 for (var i = 0; frameTreePayload.childFrames && i < frameTreePayload.childFrames.length; ++i)
357 this._addFramesRecursively(frame, frameTreePayload.childFrames[i]);
358
359 for (var i = 0; i < frameTreePayload.resources.length; ++i) {
360 var subresource = frameTreePayload.resources[i];
361 var resource = this._createResourceFromFramePayload(framePayload, subresource.url, WebInspector.resourceTypes[subresource.type], subresource.mimeType);
362 frame.addResource(resource);
363 }

393 * @param {string=} scriptToEvaluateOnLoad
394 */

396 {
397 // Only dispatch PageReloadRequested upon first reload request to simplify client logic.
398 if (!this._pendingReloadOptions)
399 this.dispatchEventToListeners(WebInspector.ResourceTreeModel.EventTypes.PageReloadRequested);

430 /**
431 * @param {!WebInspector.ExecutionContext} a
432 * @param {!WebInspector.ExecutionContext} b
433 * @return {number}
434 */
435 _executionContextComparator: function(a,b)
436 {
437 /**
438 * @param {!WebInspector.ResourceTreeFrame} frame
439 */
440 function framePath(frame)
441 {
442 var currentFrame = frame;
443 var parents = [];
444 while (currentFrame) {
445 parents.push(currentFrame);
446 currentFrame = currentFrame.parentFrame;
447 }
448 return parents.reverse();
449 }
450
451 var framesA = a.frameId ? framePath(this.frameForId(a.frameId)) : [];
452 var framesB = b.frameId ? framePath(this.frameForId(b.frameId)) : [];
453 var frameA;
454 var frameB;
455 for (var i = 0; ; i++) {
456 if (!framesA[i] || !framesB[i] || (framesA[i] !== framesB[i])) {
457 frameA = framesA[i];
458 frameB = framesB[i];
459 break;
460 }
461 }
462 if (!frameA && frameB)
463 return -1;
464
465 if (!frameB && frameA)
466 return 1;
467
468 if (frameA && frameB) {
469 return frameA.id.localeCompare(frameB.id);
470 }
471 return WebInspector.ExecutionContext.comparator(a,b);
472 },

484WebInspector.ResourceTreeFrame = function(model, parentFrame, frameId, payload)
485{
486 this._model = model;
487 this._parentFrame = parentFrame;
488 this._id = frameId;
489 this._url = "";
490
491 if (payload) {
492 this._loaderId = payload.loaderId;
493 this._name = payload.name;
494 this._url = payload.url;

505 * @type {!Object.<string, !WebInspector.Resource>}
506 */
507 this._resourcesMap = {};
508
509 if (this._parentFrame)
510 this._parentFrame._childFrames.push(this);
511}

701 return resource;

704 this._resourcesMap[resource.url] = resource;
705 this._model.dispatchEventToListeners(WebInspector.ResourceTreeModel.EventTypes.ResourceAdded, resource);
706 return resource;

723 */
724 resourceForURL: function(url)
725 {
726 var result;
727 function filter(resource)
728 {
729 if (resource.url === url) {
730 result = resource;
731 return true;
732 }
733 }
734 this._callForFrameResources(filter);
735 return result || null;
736 },
737
738 /**
739 * @param {function(!WebInspector.Resource)} callback
740 * @return {boolean}
741 */
742 _callForFrameResources: function(callback)
743 {
744 for (var url in this._resourcesMap) {
745 if (callback(this._resourcesMap[url]))
746 return true;
747 }
748
749 for (var i = 0; i < this._childFrames.length; ++i) {
750 if (this._childFrames[i]._callForFrameResources(callback))
751 return true;
752 }
753 return false;
754 },
755

757 * @return {string}
758 */
759 displayName: function()
760 {
761 if (!this._parentFrame)
762 return WebInspector.UIString("top");
763 var subtitle = new WebInspector.ParsedURL(this._url).displayName;
764 if (subtitle) {
765 if (!this._name)
766 return subtitle;
767 return this._name + " (" + subtitle + ")";
768 }
769 return WebInspector.UIString("<iframe>");
770 }
