80WebInspector.ProfileTreeModel.prototype = {
81 _assignDepthsAndParents: function()
82 {
83 var root = this.root;
84 root.depth = -1;
85 root.parent = null;
86 this.maxDepth = 0;
87 var nodesToTraverse = [root];
88 while (nodesToTraverse.length) {
89 var parent = nodesToTraverse.pop();
90 var depth = parent.depth + 1;
91 if (depth > this.maxDepth)
92 this.maxDepth = depth;
93 var children = parent.children;
94 var length = children.length;
95 for (var i = 0; i < length; ++i) {
96 var child = children[i];
97 child.depth = depth;
98 child.parent = parent;
99 if (child.children.length)
100 nodesToTraverse.push(child);
101 }
102 }
103 },
104
105 /**
106 * @param {!WebInspector.ProfileNode} node
107 * @return {number}

110 {
111 node.total = node.children.reduce((acc, child) => acc + this._calculateTotals(child), node.self);
112 return node.total;
