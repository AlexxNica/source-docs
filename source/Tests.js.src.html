44/**
45 * Test suite for interactive UI tests.
46 * @constructor
47 * @param {Object} domAutomationController DomAutomationController instance.
48 */
49function TestSuite(domAutomationController)
50{
51 this.domAutomationController_ = domAutomationController;
52 this.controlTaken_ = false;
53 this.timerId_ = -1;
54 this._asyncInvocationId = 0;
55}
56
57/**
58 * Reports test failure.
59 * @param {string} message Failure description.
60 */
61TestSuite.prototype.fail = function(message)
62{
63 if (this.controlTaken_)
64 this.reportFailure_(message);
65 else
66 throw message;
67};
68
69
70/**
71 * Equals assertion tests that expected === actual.
72 * @param {!Object|boolean} expected Expected object.
73 * @param {!Object|boolean} actual Actual object.
74 * @param {string} opt_message User message to print if the test fails.
75 */
76TestSuite.prototype.assertEquals = function(expected, actual, opt_message)
77{
78 if (expected !== actual) {
79 var message = "Expected: '" + expected + "', but was '" + actual + "'";
80 if (opt_message)
81 message = opt_message + "(" + message + ")";
82 this.fail(message);
83 }
84};
85
86
87/**
88 * True assertion tests that value == true.
89 * @param {!Object} value Actual object.
90 * @param {string} opt_message User message to print if the test fails.
91 */
92TestSuite.prototype.assertTrue = function(value, opt_message)
93{
94 this.assertEquals(true, !!value, opt_message);
95};

98/**
99 * Takes control over execution.
100 */
101TestSuite.prototype.takeControl = function()
102{
103 this.controlTaken_ = true;
104 // Set up guard timer.
105 var self = this;
106 this.timerId_ = setTimeout(function() {
107 self.reportFailure_("Timeout exceeded: 20 sec");
108 }, 20000);
109};
110
111
112/**
113 * Releases control over execution.
114 */
115TestSuite.prototype.releaseControl = function()
116{
117 if (this.timerId_ !== -1) {
118 clearTimeout(this.timerId_);
119 this.timerId_ = -1;
120 }
121 this.controlTaken_ = false;
122 this.reportOk_();
123};
124
125
126/**
127 * Async tests use this one to report that they are completed.
128 */
129TestSuite.prototype.reportOk_ = function()
130{
131 this.domAutomationController_.send("[OK]");
132};
133
134
135/**
136 * Async tests use this one to report failures.
137 */
138TestSuite.prototype.reportFailure_ = function(error)
139{
140 if (this.timerId_ !== -1) {
141 clearTimeout(this.timerId_);
142 this.timerId_ = -1;
143 }
144 this.domAutomationController_.send("[FAILED] " + error);
145};

148/**
149 * Run specified test on a fresh instance of the test suite.
150 * @param {Array<string>} args method name followed by its parameters.
151 */
152TestSuite.prototype.dispatchOnTestSuite = function(args)
153{
154 var methodName = args.shift();
155 try {
156 this[methodName].apply(this, args);
157 if (!this.controlTaken_)
158 this.reportOk_();
159 } catch (e) {
160 this.reportFailure_(e);
161 }
162};
163
164
165/**
166 * Wrap an async method with TestSuite.{takeControl(), releaseControl()}
167 * and invoke TestSuite.reportOk_ upon completion.
168 * @param {Array<string>} args method name followed by its parameters.
169 */
170TestSuite.prototype.waitForAsync = function(var_args)
171{
172 var args = Array.prototype.slice.call(arguments);
173 this.takeControl();
174 args.push(this.releaseControl.bind(this));
175 this.dispatchOnTestSuite(args);
176};

178/**
179 * Overrides the method with specified name until it's called first time.
180 * @param {!Object} receiver An object whose method to override.
181 * @param {string} methodName Name of the method to override.
182 * @param {!Function} override A function that should be called right after the
183 * overridden method returns.
184 * @param {?boolean} opt_sticky Whether restore original method after first run
185 * or not.
186 */
187TestSuite.prototype.addSniffer = function(receiver, methodName, override, opt_sticky)
188{
189 var orig = receiver[methodName];
190 if (typeof orig !== "function")
191 this.fail("Cannot find method to override: " + methodName);
192 var test = this;
193 receiver[methodName] = function(var_args) {
194 try {
195 var result = orig.apply(this, arguments);
196 } finally {
197 if (!opt_sticky)
198 receiver[methodName] = orig;
199 }
200 // In case of exception the override won't be called.
201 try {
202 override.apply(this, arguments);
203 } catch (e) {
204 test.fail("Exception in overriden method '" + methodName + "': " + e);
205 }
206 return result;
207 };
208};

210/**
211 * Waits for current throttler invocations, if any.
212 * @param {!WebInspector.Throttler} throttler
213 * @param {function()} callback
214 */
215TestSuite.prototype.waitForThrottler = function(throttler, callback)
216{
217 var test = this;
218 var scheduleShouldFail = true;
219 test.addSniffer(throttler, "schedule", onSchedule);
220
221 function hasSomethingScheduled()
222 {
223 return throttler._isRunningProcess || throttler._process;
224 }
225
226 function checkState()
227 {
228 if (!hasSomethingScheduled()) {
229 scheduleShouldFail = false;
230 callback();
231 return;
232 }
233
234 test.addSniffer(throttler, "_processCompletedForTests", checkState);
235 }
236
237 function onSchedule()
238 {
239 if (scheduleShouldFail)
240 test.fail("Unexpected Throttler.schedule");
241 }
242
243 checkState();
244};

546 test.assertTrue(resource.startTime < resource.timing.pushEnd, "pushEnd should be after startTime");
