342 /**
343 * @override
344 * @param {!WebInspector.SearchableView.SearchConfig} searchConfig
345 * @param {boolean} shouldJump
346 * @param {boolean=} jumpBackwards
347 */
348 performSearch: function(searchConfig, shouldJump, jumpBackwards)
349 {
350 var matcher = createPlainTextSearchRegex(searchConfig.query, searchConfig.caseSensitive ? "" : "i");
351
352 var selectedEntryIndex = this._searchResultIndex !== -1 ? this._searchResults[this._searchResultIndex] : -1;
353 this._searchResults = [];
354 var entriesCount = this._dataProvider._entryNodes.length;
355 for (var index = 0; index < entriesCount; ++index) {
356 if (this._dataProvider.entryTitle(index).match(matcher))
357 this._searchResults.push(index);
358 }
359
360 if (this._searchResults.length) {
361 this._searchResultIndex = this._searchResults.indexOf(selectedEntryIndex);
362 if (this._searchResultIndex === -1)
363 this._searchResultIndex = jumpBackwards ? this._searchResults.length - 1 : 0;
364 this._mainPane.setSelectedEntry(this._searchResults[this._searchResultIndex]);
365 } else {
366 this.searchCanceled();
367 }
368 this._searchableView.updateSearchMatchesCount(this._searchResults.length);
369 this._searchableView.updateCurrentMatchIndex(this._searchResultIndex);
370 },
371
372 /**
373 * @override
374 */
375 searchCanceled: function()
376 {
377 this._mainPane.setSelectedEntry(-1);
378 this._searchResults = [];
379 this._searchResultIndex = -1;

608 _drawOverviewCanvas: function()
609 {
610 var canvasWidth = this._overviewCanvas.width;
611 var canvasHeight = this._overviewCanvas.height;
612 var drawData = this._calculateDrawData(canvasWidth);
613 var context = this._overviewCanvas.getContext("2d");
614 var ratio = window.devicePixelRatio;
615 var offsetFromBottom = ratio;
616 var lineWidth = 1;
617 var yScaleFactor = canvasHeight / (this._dataProvider.maxStackDepth() * 1.1);
618 context.lineWidth = lineWidth;
619 context.translate(0.5, 0.5);
620 context.strokeStyle = "rgba(20,0,0,0.4)";
621 context.fillStyle = "rgba(214,225,254,0.8)";
622 context.moveTo(-lineWidth, canvasHeight + lineWidth);
623 context.lineTo(-lineWidth, Math.round(canvasHeight - drawData[0] * yScaleFactor - offsetFromBottom));
624 var value;
625 for (var x = 0; x < canvasWidth; ++x) {
626 value = Math.round(canvasHeight - drawData[x] * yScaleFactor - offsetFromBottom);
627 context.lineTo(x, value);
628 }
629 context.lineTo(canvasWidth + lineWidth, value);
630 context.lineTo(canvasWidth + lineWidth, canvasHeight + lineWidth);
631 context.fill();
632 context.stroke();
633 context.closePath();

636 /**
637 * @param {number} width
638 * @return {!Uint8Array}
639 */
640 _calculateDrawData: function(width)
641 {
642 var dataProvider = this._dataProvider;
643 var timelineData = this._timelineData();
644 var entryStartTimes = timelineData.entryStartTimes;
645 var entryTotalTimes = timelineData.entryTotalTimes;
646 var entryLevels = timelineData.entryLevels;
647 var length = entryStartTimes.length;
648 var minimumBoundary = this._dataProvider.minimumBoundary();
649
650 var drawData = new Uint8Array(width);
651 var scaleFactor = width / dataProvider.totalTime();
652
653 for (var entryIndex = 0; entryIndex < length; ++entryIndex) {
654 var start = Math.floor((entryStartTimes[entryIndex] - minimumBoundary) * scaleFactor);
655 var finish = Math.floor((entryStartTimes[entryIndex] - minimumBoundary + entryTotalTimes[entryIndex]) * scaleFactor);
656 for (var x = start; x <= finish; ++x)
657 drawData[x] = Math.max(drawData[x], entryLevels[entryIndex] + 1);
658 }
659 return drawData;
