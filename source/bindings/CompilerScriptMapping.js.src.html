194 /**
195 * @param {!WebInspector.Script} script
196 */
197 _processScript: function(script)
198 {
199 if (WebInspector.blackboxManager.isBlackboxedURL(script.sourceURL, script.isContentScript()))
200 return;
201 // Create stub UISourceCode for the time source mapping is being loaded.
202 var stubUISourceCode = this._stubProject.addContentProvider(script.sourceURL, WebInspector.StaticContentProvider.fromString(script.sourceURL, WebInspector.resourceTypes.Script, "\n\n\n\n\n// Please wait a bit.\n// Compiled script is not shown while source map is being loaded!"));
203 this._stubUISourceCodes.set(script.scriptId, stubUISourceCode);
204
205 this._debuggerWorkspaceBinding.pushSourceMapping(script, this);
206 this._loadSourceMapForScript(script).then(this._sourceMapLoaded.bind(this, script, stubUISourceCode.url()));

209 /**
210 * @param {!WebInspector.Script} script
211 * @param {string} uiSourceCodePath
212 * @param {?WebInspector.TextSourceMap} sourceMap
213 */
214 _sourceMapLoaded: function(script, uiSourceCodePath, sourceMap)
215 {
216 WebInspector.blackboxManager.sourceMapLoaded(script, sourceMap);
217
218 this._stubUISourceCodes.delete(script.scriptId);
219 this._stubProject.removeFile(uiSourceCodePath);
220
221 if (!sourceMap) {
222 this._debuggerWorkspaceBinding.updateLocations(script);
223 return;
224 }
225
226 if (this._scriptForSourceMap.get(sourceMap)) {
227 this._sourceMapForScriptId.set(script.scriptId, sourceMap);
228 this._debuggerWorkspaceBinding.updateLocations(script);
229 return;
230 }
231
232 this._sourceMapForScriptId.set(script.scriptId, sourceMap);
233 this._scriptForSourceMap.set(sourceMap, script);
234
235 // Report sources.
236 var sourceURLs = sourceMap.sourceURLs();
237 var missingSources = [];
238 for (var i = 0; i < sourceURLs.length; ++i) {
239 var sourceURL = sourceURLs[i];
240 if (this._sourceMapForURL.get(sourceURL))
241 continue;
242 this._sourceMapForURL.set(sourceURL, sourceMap);
243 var uiSourceCode = this._networkMapping.uiSourceCodeForScriptURL(sourceURL, script);
244 if (!uiSourceCode && !this._networkMapping.hasMappingForNetworkURL(sourceURL)) {
245 var contentProvider = sourceMap.sourceContentProvider(sourceURL, WebInspector.resourceTypes.SourceMapScript);
246 uiSourceCode = this._networkProject.addFile(contentProvider, WebInspector.ResourceTreeFrame.fromScript(script), script.isContentScript());
247 uiSourceCode[WebInspector.CompilerScriptMapping._originSymbol] = script.sourceURL;
248 }
249 if (uiSourceCode) {
250 this._bindUISourceCode(uiSourceCode);
251 } else {
252 if (missingSources.length < 3)
253 missingSources.push(sourceURL);
254 else if (missingSources.peekLast() !== "\u2026")
255 missingSources.push("\u2026");
256 }
257 }
258 if (missingSources.length) {
259 WebInspector.console.warn(
260 WebInspector.UIString("Source map %s points to the files missing from the workspace: [%s]",
261 sourceMap.url(), missingSources.join(", ")));
262 }
263
264 this._debuggerWorkspaceBinding.updateLocations(script);

321 /**
322 * @param {!WebInspector.Script} script
323 * @return {!Promise<?WebInspector.TextSourceMap>}
324 */
325 _loadSourceMapForScript: function(script)
326 {
327 // script.sourceURL can be a random string, but is generally an absolute path -> complete it to inspected page url for
328 // relative links.
329 var scriptURL = WebInspector.ParsedURL.completeURL(this._target.inspectedURL(), script.sourceURL);
330 if (!scriptURL)
331 return Promise.resolve(/** @type {?WebInspector.TextSourceMap} */(null));
332
333 console.assert(script.sourceMapURL);
334 var scriptSourceMapURL = /** @type {string} */ (script.sourceMapURL);
335
336 var sourceMapURL = WebInspector.ParsedURL.completeURL(scriptURL, scriptSourceMapURL);
337 if (!sourceMapURL)
338 return Promise.resolve(/** @type {?WebInspector.TextSourceMap} */(null));
339
340 var loadingPromise = this._sourceMapLoadingPromises.get(sourceMapURL);
341 if (!loadingPromise) {
342 loadingPromise = WebInspector.TextSourceMap.load(sourceMapURL, scriptURL).then(sourceMapLoaded.bind(this, sourceMapURL));
343 this._sourceMapLoadingPromises.set(sourceMapURL, loadingPromise);
344 }
345 return loadingPromise;

365 /**
366 * @param {!WebInspector.TextSourceMap} sourceMap
367 * @this {WebInspector.CompilerScriptMapping}
368 */
369 function unbindSourceMapSources(sourceMap)
370 {
371 var script = this._scriptForSourceMap.get(sourceMap);
372 if (!script)
373 return;
374 var sourceURLs = sourceMap.sourceURLs();
375 for (var i = 0; i < sourceURLs.length; ++i) {
376 var uiSourceCode = this._networkMapping.uiSourceCodeForScriptURL(sourceURLs[i], script);
377 if (uiSourceCode)
378 this._unbindUISourceCode(uiSourceCode);
379 }
380 }
381
382 this._sourceMapForURL.valuesArray().forEach(unbindSourceMapSources.bind(this));
383
384 this._sourceMapLoadingPromises.clear();
385 this._sourceMapForScriptId.clear()
386 this._scriptForSourceMap.clear();
387 this._sourceMapForURL.clear();
