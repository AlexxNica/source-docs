31/**
32 * @constructor
33 * @implements {WebInspector.DebuggerSourceMapping}
34 * @param {!WebInspector.DebuggerModel} debuggerModel
35 * @param {!WebInspector.Workspace} workspace
36 * @param {!WebInspector.DebuggerWorkspaceBinding} debuggerWorkspaceBinding
37 */
38WebInspector.DefaultScriptMapping = function(debuggerModel, workspace, debuggerWorkspaceBinding)
39{
40 this._debuggerModel = debuggerModel;
41 this._debuggerWorkspaceBinding = debuggerWorkspaceBinding;
42 this._workspace = workspace;
43 this._projectId = WebInspector.DefaultScriptMapping.projectIdForTarget(debuggerModel.target());
44 this._project = new WebInspector.ContentProviderBasedProject(this._workspace, this._projectId, WebInspector.projectTypes.Debugger, "");
45 debuggerModel.addEventListener(WebInspector.DebuggerModel.Events.GlobalObjectCleared, this._debuggerReset, this);
46 this._debuggerReset();
47}

61 /**
62 * @override
63 * @param {!WebInspector.DebuggerModel.Location} rawLocation
64 * @return {!WebInspector.UILocation}
65 */
66 rawLocationToUILocation: function(rawLocation)
67 {
68 var debuggerModelLocation = /** @type {!WebInspector.DebuggerModel.Location} */ (rawLocation);
69 var script = debuggerModelLocation.script();
70 var uiSourceCode = this._uiSourceCodeForScriptId.get(script.scriptId);
71 var lineNumber = debuggerModelLocation.lineNumber - (script.isInlineScriptWithSourceURL() ? script.lineOffset : 0);
72 var columnNumber = debuggerModelLocation.columnNumber || 0;
73 if (script.isInlineScriptWithSourceURL() && !lineNumber && columnNumber)
74 columnNumber -= script.columnOffset;
75 return uiSourceCode.uiLocation(lineNumber, columnNumber);
76 },
77
78 /**
79 * @override
80 * @param {!WebInspector.UISourceCode} uiSourceCode
81 * @param {number} lineNumber
82 * @param {number} columnNumber
83 * @return {?WebInspector.DebuggerModel.Location}
84 */
85 uiLocationToRawLocation: function(uiSourceCode, lineNumber, columnNumber)
86 {
87 var scriptId = this._scriptIdForUISourceCode.get(uiSourceCode);
88 var script = this._debuggerModel.scriptForId(scriptId);
89 if (script.isInlineScriptWithSourceURL())
90 return this._debuggerModel.createRawLocation(script, lineNumber + script.lineOffset, lineNumber ? columnNumber : columnNumber + script.columnOffset);
91 return this._debuggerModel.createRawLocation(script, lineNumber, columnNumber);
92 },

94 /**
95 * @param {!WebInspector.Script} script
96 */
97 addScript: function(script)
98 {
99 var name = WebInspector.ParsedURL.extractName(script.sourceURL);
100 var url = "debugger:///VM" + script.scriptId + (name ? " " + name : "");
101
102 var uiSourceCode = this._project.createUISourceCode(url, WebInspector.resourceTypes.Script);
103 uiSourceCode[WebInspector.DefaultScriptMapping._scriptSymbol] = script;
104 this._uiSourceCodeForScriptId.set(script.scriptId, uiSourceCode);
105 this._scriptIdForUISourceCode.set(uiSourceCode, script.scriptId);
106 this._project.addUISourceCodeWithProvider(uiSourceCode, script);
107
108 this._debuggerWorkspaceBinding.setSourceMapping(this._debuggerModel.target(), uiSourceCode, this);
109 this._debuggerWorkspaceBinding.pushSourceMapping(script, this);
110 },
